/* PROGRAM    : PUE8UPC                                              */
/* TITLE      : GENERIC PROGRAM TO EXECUTE THE ESNDMAIL COMMAND      */
/* DATE       : 09/30/2017                                           */
/* BY         : DANNY NGUYEN                                         */
/* PURPOSE    : EXECUTE THE ESNDNMAIL COMMAND.                       */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &EMAIL      EMAIL ADDRESS TO SEND TO                     */
/*    I     &SUBJECT    EMAIL SUBJECT                                */
/*    I     &MESSAGE    MESSAGE TO SEND                              */
/*********************************************************************/

             PGM        PARM(&EMAIL &SUBJECT &MESSAGE)

             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)
             DCL        VAR(&SUBJECT) TYPE(*CHAR) LEN(50)
             DCL        VAR(&MESSAGE) TYPE(*CHAR) LEN(256)

/*-------------------------------------------------------------------*/
/* EXECUTE COMMAND TO SEND EMAIL                                     */
/*-------------------------------------------------------------------*/
             ADDLIBLE   LIB(ESEND) POSITION(*LAST)
             MONMSG     MSGID(CPF2103 CPF9999)

             ESNDMAIL   RECIPIENT(&EMAIL) SUBJECT(&SUBJECT) +
                          MSG(&MESSAGE)

             ENDPGM
