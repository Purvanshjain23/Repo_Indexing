/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION USED TO PROCESS SHIPPING */
/*              SCHEDULE.  CAUSES PRINTER OVERRIDE PROMPTING AND    */
/*              PASSES COMPANY, WAREHOUSE, AND SHIP DATE.           */
/*              ALSO USED FOR RE-PRINT PICK SLIPS.                  */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRT SHIPPING SCHEDULE                               */
/*  DATE      - 05/15/95                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &PGM        - CL PROGRAM TO SUBMIT                      */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*     B    &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     B    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     B    &REQDATN    - REQ. SHIP DATE (PASSED)                   */
/*     B    &SOFLAG     - SELECT/OMIT    (PASSED)                   */
/*     B    &UOM        - UNIT OF MEASURE(PASSED)                   */
/*     B    &SHPUNSHP   - LOADS SHIPPED OR UNSHIPPED (PASSED)       */
/*     B    &ORDCOMN    - Order Co Number(Passed)                   */
/*     B    &WHSE1      - WAREHOUSE 1    (PASSED)                   */
/*     B    &WHSE2      - WAREHOUSE 2    (PASSED)                   */
/*     B    &WHSE3      - WAREHOUSE 3    (PASSED)                   */
/*     B    &WHSE4      - WAREHOUSE 4    (PASSED)                   */
/*     B    &WHSE5      - WAREHOUSE 5    (PASSED)                   */
/*     B    &WHSE6      - WAREHOUSE 6    (PASSED)                   */
/*     B    &WHSE7      - WAREHOUSE 7    (PASSED)                   */
/*     B    &WHSE8      - WAREHOUSE 8    (PASSED)                   */
/*     B    &WHSE9      - WAREHOUSE 9    (PASSED)                   */
/*     B    &WHSE10     - WAREHOUSE 10   (PASSED)                   */
/*     B    &LDTYP      - LOAD TYPE      (PASSED)                   */
/********************************************************************/
/*** CHANGES                                                      ***/
/*   08/28/2008  BLG  ADDED REPORT TYPE TO SORT REPORT BY EITHER    */
/*   SCHEDULED SHIP DATE OR DDT PER REQUEST E00213.                 */
/********************************************************************/
 PDOIUPR:    PGM        PARM(&RETURN &APPLIC &PGM &DESC &OUTQ &HOLD +
                          &SAVE &NQSTAT &NIGHTQ &COMPNYN &REQDATN +
                          &COPYN &UOM &SHPUNSHP &ORDCOMN &WHSE1 +
                          &WHSE2 &WHSE3 &WHSE4 &WHSE5 &WHSE6 &WHSE7 +
                          &WHSE8 &WHSE9 &WHSE10 &SOFLAG &LDTYP &RPTTYP)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &PGM        *CHAR   (10)          /* CL to Submit              */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &REQDATN    *DEC                  /* REQ SHIP DATE  (Passed)   */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &SOFLAG     *CHAR   (1)           /* SOFLAG                    */
        DCL    &UOM        *CHAR   (2)           /* UNIT OF MEASURE           */
        DCL    &SHPUNSHP   *CHAR   (1)           /* SHIP OR UNSHIPPED         */
        DCL    &ORDCOMN    *DEC                  /* Order Co Number(Passed)   */
        DCL    &WHSE1      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE2      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE3      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE4      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE5      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE6      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE7      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE8      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE9      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE10     *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &LDTYP      *CHAR   (1)           /* LOAD TYPE                 */
        DCL    &RPTTYP     *CHAR   (1)           /* REPORT TYPE               */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/
             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PROMPT PROGRAM                                            */
/********************************************************************/
             CALL       PGM(PDG0PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM                                        */
/********************************************************************/
             IF         COND(&NIGHTQ *EQ '*JOBD') THEN(DO)

 SUBMIT:     SBMJOB     CMD(CALL PGM(PDOIPCLP) PARM(&COMPNYN &REQDATN +
                          &UOM &SHPUNSHP &ORDCOMN &WHSE1 &WHSE2 +
                          &WHSE3 &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 +
                          &WHSE9 &WHSE10 &SOFLAG &LDTYP &RPTTYP +
                          &DESC &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)) +
                          JOB(&PGM) JOBQ(PRIORITY)

 /******** FOR TESTING                                                 +
             CALL       PGM(PDOIPCLP) PARM(&COMPNYN &REQDATN &UOM +
                          &SHPUNSHP &ORDCOMN &WHSE1 &WHSE2 &WHSE3 +
                          &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 &WHSE9 +
                          &WHSE10 &SOFLAG &LDTYP &RPTTYP &DESC +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)     */
             ENDDO

             IF         COND(&NIGHTQ *NE '*JOBD') THEN(DO)
 SUBMITN:    SBMJOB     CMD(CALL PGM(PDOIPCLP) PARM(&COMPNYN +
                          &REQDATN &UOM &SHPUNSHP &ORDCOMN &WHSE1 +
                          &WHSE2 &WHSE3 &WHSE4 &WHSE5 &WHSE6 &WHSE7 +
                          &WHSE8 &WHSE9 &WHSE10 &SOFLAG &LDTYP +
                          &RPTTYP &DESC &OUTQ &HOLD &SAVE &COPYN +
                          &NIGHTQ)) JOB(&PGM) JOBQ(&NIGHTQ)
             ENDDO

             GOTO       CMDLBL(END)

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
