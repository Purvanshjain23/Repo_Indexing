EXECUTE FUNCTION(Exc Total Pallet Count RT) TYPE(RTVOBJ) FILE(OMFNCPP)          AC2195468;
PARAMETER(PAR.Company_Number);
PARAMETER(PAR.Order_Number);
PARAMETER(1);
PARAMETER(PAR.Total_Pallet_Count_Hdr_U);
{
 //?USER: Initialize routine

 // PAR.Total Pallet Count Hdr U = CON.*ZERO
 PAR.Total_Pallet_Count_Hdr_U = *ZERO;

 //?USER: Process Data record

 //?Counting Only Unique Pallet Numbers > 0 & Case QTY > 0.
 CASE;

 // IF DB1.Case Quantity is GT *ZERO
 IF DB1.Case_Quantity > *ZERO;

 // AND DB1.Manifest Pallet Number is Entered
 AND DB1.Manifest_Pallet_Number <> *ZERO;

 CASE;

 // IF DB1.Manifest Pallet Number NE LCL.Manifest Pallet Number
 IF DB1.Manifest_Pallet_Number <> LCL.Manifest_Pallet_Number;

 // LCL.Manifest Pallet Number = DB1.Manifest Pallet Number
 LCL.Manifest_Pallet_Number = DB1.Manifest_Pallet_Number;

 // PAR.Total Pallet Count Hdr U = PAR.Total Pallet Count Hdr U + CON.1
 PAR.Total_Pallet_Count_Hdr_U = PAR.Total_Pallet_Count_Hdr_U + 1;

 ENDIF;

 ENDIF;

}
