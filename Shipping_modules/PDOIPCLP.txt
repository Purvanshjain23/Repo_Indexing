/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM WITH SELECTIONS FOR LOAD LIST          */
/*             COMPANY, WAREHOUSES, REQUESTED SHIP DATE             */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT SHIPPING SCHEDULE                             */
/*  DATE      -  5/15/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPNYN    - COMPANY NUMBER - BLANK FOR ALL            */
/*          &REQDATN    - REQUESTED SHIP DATE                       */
/*          &UOM        - UNIT OF MEASURE                           */
/*          &SHPUNSHP   - SHIPPED OR UNSHIPPED                      */
/*          &ORDCOMN    - ORDER COMPANY NUMBER                      */
/*          &WHSE1      - WAREHOUSE 1                               */
/*          &WHSE2      - WAREHOUSE 2                               */
/*          &WHSE3      - WAREHOUSE 3                               */
/*          &WHSE4      - WAREHOUSE 4                               */
/*          &WHSE5      - WAREHOUSE 5                               */
/*          &WHSE6      - WAREHOUSE 6                               */
/*          &WHSE7      - WAREHOUSE 7                               */
/*          &WHSE8      - WAREHOUSE 8                               */
/*          &WHSE9      - WAREHOUSE 9                               */
/*          &WHSE10     - WAREHOUSE 10                              */
/*          &SOFLAG     - SELECT/OMIT FLAG                          */
/*          &LDTYP      - LOAD TYPE                                 */
/*          &RPTTYP     - REPORT TYPE                               */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/*          &NIGHTQ     - NIGHT OUTPUT QUEUE                        */
/********************************************************************/
/* LJB 12/22/08 E00294 ADD REPORT TYPE TO SORT BY SSD AND           */
/*     SEQUENCE NUMBER                                              */
/* BLG 08/28/08 E00213 ADDED REPORT TYPE TO SORT REPORT BY EITHER   */
/*     SCHEDULED SHIP DATE OR DEADLINE DEPARTURE DATE               */
/* SLM  12/31/04  PRTF01 - PASS IN THE ORDER COMPANY NUMBER         */
/* RMC 03/08/12 E01505 ADD REPORT TYPE 5 - BY LOAD PLAN START       */
/********************************************************************/
 PDOIPCLP:   PGM        PARM(&COMPNYN &REQDATN &UOM &SHPUNSHP +
                          &ORDCOMN &WHSE1 &WHSE2 &WHSE3 &WHSE4 +
                          &WHSE5 &WHSE6 &WHSE7 &WHSE8 &WHSE9 +
                          &WHSE10 &SOFLAG &LDTYP &RPTTYP &DESC +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)

/* PARMS                                                                      */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &REQDATN    *DEC                  /* REQ SHIP DATE  (Passed)   */
        DCL    &UOM        *CHAR   (2)           /* UNIT OF MEASURE           */
        DCL    &SHPUNSHP   *CHAR   (1)           /* SHIP OR UNSHIPPED         */
        DCL    &SOFLAG     *CHAR   (1)           /* WAREHOUSE                 */
        DCL    &ORDCOMN    *DEC                  /* Order Comp Numb(Passed)   */
        DCL    &WHSE1      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE2      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE3      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE4      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE5      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE6      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE7      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE8      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE9      *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &WHSE10     *CHAR   (3)           /* WAREHOUSE                 */
        DCL    &LDTYP      *CHAR   (1)           /* LOAD TYPE                 */
        DCL    &RPTTYP     *CHAR   (1)           /* REPORT TYPE               */
        DCL    &DESC       *CHAR   (25)          /* DESCRIPTION               */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */
        DCL    &ORDERCO    *DEC    (3 0)         /* Order Company Number      */
        DCL    &REQDAT     *DEC    (7 0)         /* REQ. SHIP DATE            */

/* OPENQRYF FIELDS                                                            */
        DCL    &COMPA      *CHAR    3            /* ACK COMPLETE              */
        DCL    &REQDATA    *CHAR    7            /* REQ SHIP DATE  (Passed)   */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/
             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&ORDERCO) VALUE(&ORDCOMN)
             CHGVAR     VAR(&COMPA) VALUE(&COMPANY)
             CHGVAR     VAR(&REQDAT) VALUE(&REQDATN)
             CHGVAR     VAR(&REQDATA) VALUE(&REQDAT)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/********************************************************************/
/* PROCESS THE REPORT BASED UPON THE REPORT TYPE PASSED IN.         */
/********************************************************************/
/*  BY SCHEDULED SHIP DATE    */
             IF         COND(&RPTTYP *EQ '1') THEN(DO)

/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(OMD6PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*   CALL PRINT PROGRAM -  BY SCHEDULED SHIP DATE                    */
             CALL       PGM(OMD6PFR) PARM(' ' &COMPANY &REQDAT +
                          &LDTYP &WHSE1 &WHSE2 &WHSE3 &WHSE4 &WHSE5 +
                          &WHSE6 &WHSE7 &WHSE8 &WHSE9 &WHSE10 +
                          &SOFLAG &UOM &SHPUNSHP &ORDERCO)
             ENDDO

/********************************************************************/
/*  BY DEADLINE DEPARTURE DATE  */
             IF         COND(&RPTTYP *EQ '2') THEN(DO)

/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMXUPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*   CALL PRINT PROGRAM - BY DEADLINE DEPARTURE DATE                 */
             CALL       PGM(PMXUPFR) PARM(' ' &COMPANY  &REQDAT +
                          &LDTYP &WHSE1 &WHSE2 &WHSE3 &WHSE4 &WHSE5 +
                          &WHSE6 &WHSE7 &WHSE8 &WHSE9 &WHSE10 +
                          &SOFLAG &UOM &SHPUNSHP &ORDERCO)
             ENDDO

/********************************************************************/
/*  BY SEQUENCE NUMBER                                              */
             IF         COND(&RPTTYP *EQ '4') THEN(DO)

/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMCCPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*   CALL PRINT PROGRAM -  BY SSD AND SEQUENCE NUMBER                */
             CALL       PGM(PMCCPFR) PARM(' ' &COMPANY &REQDAT +
                          &LDTYP &WHSE1 &WHSE2 &WHSE3 &WHSE4 &WHSE5 +
                          &WHSE6 &WHSE7 &WHSE8 &WHSE9 &WHSE10 +
                          &SOFLAG &UOM &SHPUNSHP &ORDERCO)
             ENDDO

/********************************************************************/
/*  BY LOAD PLAN START DATE                                         */
             IF         COND(&RPTTYP *EQ '5') THEN(DO)

/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PBCOPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*   CALL PRINT PROGRAM -  BY SSD AND SEQUENCE NUMBER                */
             CALL       PGM(PBCOPFR) PARM(' ' &COMPANY &REQDAT +
                          &LDTYP &WHSE1 &WHSE2 &WHSE3 &WHSE4 &WHSE5 +
                          &WHSE6 &WHSE7 &WHSE8 &WHSE9 &WHSE10 +
                          &SOFLAG &UOM &SHPUNSHP &ORDERCO)
             ENDDO

 END:        ENDPGM
