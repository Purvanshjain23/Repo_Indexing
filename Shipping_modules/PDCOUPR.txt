/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION. CAUSE PRINTER OVERRIDE  */
/*              PROMPTING AND PASSES COMPANY, ORDER, CUSTOMER, ORD  */
/*              STATUS, EVENT CODE, CREDIT HOLD, PROD. HOLD, LOAD   */
/*              PLAN FLAG, LOAD TYPE, SCHEDULED SHIP DATE.          */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT LIST UNBOOKED ORDERS                          */
/*  DATE      - 11/07/95                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &CLP        - CL PROGRAM TO SUBMIT                      */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*                         IF THIS IS BLANK AT START, PROMPT        */
/*                         PGM IS CALLED; OTHERWISE, THE PRINT CL   */
/*                         IS SUBMITTED WITHOUT PROMPTING           */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*     B    &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*     B    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     B    &CUSTN      - CUSTOMER NUMBER (PASSED)                  */
/*     B    &FRMDATN    - FROM DATE      (PASSED)                   */
/*     B    &DLVDATN    - DELIVERY DATE  (PASSED)                   */
/*     B    &ORDERN     - ORDER NUMBER   (PASSED)                   */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     B    &ORDSTS     - ORDER STATUS   (PASSED)                   */
/*     B    &CRDHLD     - CREDIT HOLDS STS (PASSED)                 */
/*     B    &PRDHLD     - PRODUCTION HOLD STS (PASSED)              */
/*     B    &LTLFLG     - LTL FLAG       (PASSED)                   */
/*     B    &LOADTYP    - LOAD TYPE      (PASSED)                   */
/*     B    &LPFLAG     - LOAD PLAN FLAG (PASSED)                   */
/*     B    &EVENT      - EVENT CODE     (PASSED)                   */
/*     I    &LOADSUM    - U/M - LOAD SCHEDULING (PASSED)            */
/*                                                                  */
/********************************************************************/
/* RMC 01/09/08 REARRANGE PARMS TO ACCOM CALL TO PDCNPCLP      */
/*              FROM JOB SCHED PGM                              */
/* DN  08/21/19 S15343 - Added &LOADSUM to entry parm & pass it to  */
/*              PDCNPCLP program.                                   */
/********************************************************************/

 PDOWUPR:    PGM        PARM(&RETURN &APPLIC &CLP &DESC &OUTQ &HOLD +
                          &SAVE &NQSTAT &NIGHTQ &COMPNYN &CUSTN +
                          &FRMDATN &DLVDATN &ORDERN &COPYN &ORDSTS +
                          &CRDHLD &PRDHLD &LTLFLG &LOADTYP &LPFLAG +
                          &EVENT &LOADSUM)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &CLP        *CHAR   (10)          /* CL to Submit              */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &CUSTN      *DEC                  /* CUSTOMER NO.   (Passed)   */
        DCL    &FRMDATN    *DEC                  /* SCHED SHIP DT  (Passed)   */
        DCL    &DLVDATN    *DEC                  /* DELIVERY DATE  (Passed)   */
        DCL    &ORDERN     *DEC                  /* ORDER NUMBER   (Passed)   */
        DCL    &ORDSTS     *CHAR   (1)           /* ORDER STATUS              */
        DCL    &CRDHLD     *CHAR   (1)           /* CREDIT HOLD FLAG          */
        DCL    &PRDHLD     *CHAR   (1)           /* PROD   HOLD FLAG          */
        DCL    &LTLFLG     *CHAR   (1)           /* LTL FLAG                  */
        DCL    &LOADTYP    *CHAR   (1)           /* LOAD TYPE                 */
        DCL    &LPFLAG     *CHAR   (1)           /* LOAD PLAN FLAG            */
        DCL    &EVENT      *CHAR   (2)           /* EVENT NUMBER              */
        DCL    &LOADSUM    *CHAR   (2)           /* U/M - LOAD SCHEDULING     */

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PROMPT PROGRAM                                            */
/********************************************************************/

             CALL       PGM(PDG0PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM 'PDCNPCLP'                             */
/********************************************************************/

 /* MIT:     SBMJOB     CMD(CALL PGM(&CLP) PARM(&COMPNYN &CUSTN +
                          &FRMDATN &ORDERN &ORDSTS &CRDHLD &PRDHLD +
                          &DLVDATN &LTLFLG &LOADTYP &LPFLAG &EVENT +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)) +
                          JOB(&CLP) JOBQ(&NIGHTQ)                */
 /* S15343 - Added &LOADSUM to PARM */
             SBMJOB     CMD(CALL PGM(&CLP) PARM(&OUTQ &HOLD &SAVE +
                          &COMPNYN &CUSTN &FRMDATN &DLVDATN &ORDERN +
                          &COPYN &ORDSTS &CRDHLD &PRDHLD &LTLFLG +
                          &LOADTYP &LPFLAG &EVENT &LOADSUM)) JOB(&CLP) +
                          JOBQ(&NIGHTQ)
             GOTO       CMDLBL(END)

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
