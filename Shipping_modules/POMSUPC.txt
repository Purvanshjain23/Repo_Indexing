/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDHPUPR                                       */
/*  PROGRAM NAME:    Send Pager Message (MPLUS)                    */
/*  PROGRAMMER:      Purva Droge                                   */
/*  CREATE DATE:     08/13/2001                                    */
/*                                                                 */
/*  FUNCTION:        This CLP will send a message to the users'    */
/*                   in the distribution list through MPLUS.       */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/* 08/17/16  E6917 - Danny Nguyen                                  */
/* - Commented out ADDLIBLE to MPLUS. MPLUS is now replaced by     */
/*   HALCYON. The SNDPGRMSG command will continue to work with     */
/*   HALCYON in QGPL library. Ops will remove MPLUS library from   */
/*   the iSeries.                                                  */
/* - Removed MPLUS verbiage from the SNDPGMMSG command.            */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&DSTLST &MSG)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Input                                                           */
             DCL        VAR(&DSTLST) TYPE(*CHAR) LEN(10)
             DCL        VAR(&MSG) TYPE(*CHAR) LEN(132)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

/* E7075     ADDLIBLE   LIB(MPLUS) POSITION(*LAST)         */
/* E7075     MONMSG     MSGID(CPF0000)                     */

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             SNDPGRMSG  TOPGR(&DSTLST) MSG(&MSG) CONFMSG(*NO)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(CPF0000))
             GOTO       CMDLBL(ENDPGM)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 CPF0000:    SNDPGMMSG  MSG('Send Pager Message failed for +
                          List' *BCAT &DSTLST *BCAT 'and message' +
                          *BCAT &MSG)
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
