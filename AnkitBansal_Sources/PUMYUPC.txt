/**************************************************************************/
/*  PROGRAM:       PUMYUPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO MANIFEST DETAIL (OMFNCPP) FILE.       */
/*  PROGRAMMER:    BRAD BADEN                                             */
/*  CREATE DATE:   06/19/2019                                             */
/*  PROJECT:       S15143                                                 */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY A PHYSICAL FILE TRIGGER TO THE         */
/*                 MANIFEST DETAIL (OMFNCPP) FILE FOR ALL TRIGGER         */
/*                 DATABASE EVENTS:                                       */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 03/03/23 DN   W122889 ADDED MONITOR MESSAGE FOR 'RMVPFTRG' TO PREVENT  */
/*                       ERROR.                                           */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/OMFNCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PUM0UPR) +
                          RPLTRG(*YES) TRG(SEB_OMFNCPP_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/OMFNCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PUM0UPR) +
                          RPLTRG(*YES) TRG(SEB_OMFNCPP_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/OMFNCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PUM0UPR) +
                          RPLTRG(*YES) TRG(SEB_OMFNCPP_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/OMFNCPP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
