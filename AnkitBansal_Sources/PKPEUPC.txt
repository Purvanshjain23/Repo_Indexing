             PGM        PARM(&PDEV)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4) VALUE('*YES')
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4) VALUE('*NO')
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0) VALUE(1)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKPDPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKPDPFR) PARM(' ')

             DLTOVR *ALL
             ENDPGM
