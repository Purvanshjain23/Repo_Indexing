/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDM9UPC                                       */
/*  PROGRAM NAME:    Print INVENTORY/ORDERS                        */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     05/20/98                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from THE PRINT WAREHOUSE INVENTORY/ORDERS     */
/*                   COMPARISON PROMPT.                            */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC 01.29.07 ADDED REPORT TYPE(DAILY/WEKLY) PARM- PASS TO PDRPPFR  */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &PGM +
                          &PGMPRTF &APP &AVLG &RPTTYP &XCOMP &WHS1  +
                          &WHS2 &XFDATE &XTDATE &SINGLE &DAYWK)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGMPRTF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(6)
             DCL        VAR(&AVLG) TYPE(*CHAR) LEN(6)
             DCL        VAR(&SINGLE) TYPE(*CHAR) LEN(1)

/* Function Specific                                               */
             DCL        VAR(&RPTTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&WHS1) TYPE(*CHAR) LEN(3)
             DCL        VAR(&WHS2) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XFDATE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTDATE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DAYWK) TYPE(*CHAR) LEN(1)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FDATE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDATE) TYPE(*DEC) LEN(7 0)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&FDATE) VALUE(&XFDATE)
             CHGVAR     VAR(&TDATE) VALUE(&XTDATE)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGMPRTF) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(&PGM) PARM(' ' &COMP &AVLG &RPTTYP &WHS1 +
                          &WHS2 &FDATE &TDATE &SINGLE &DAYWK)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
