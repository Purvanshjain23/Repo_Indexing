/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  THIS CL IS CALLED FROM THE SCHEDULER TO SEND THE    */
/*              DISTRIBUTION MESSAGE FOR ACCUSORT DATA COLLECTIONS  */
/*              NOT RUN IN THE SPECIFIED TIME.                      */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PROCESS ACCUSORT DATA COLLECTION MESSAGE            */
/*  DATE      - 09/23/98                                            */
/*                                                                  */
/*  DESCRIPTION - THIS FUNCTION STARTS RUNNING ON MONDAY MORNING,  */
/*                RUNS HOURLY, AND ENDS ON SUNDAY MORNING.         */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*                                                                  */
/********************************************************************/
/*  REVISIONS                                                       */
/*                                                                  */
/*                                                                  */
/********************************************************************/

 PDVMUPC:    PGM        PARM(&RETURN)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &DAYOFWK    *CHAR   (4)

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

CHECKIT:
             CALL       PGM(PDTDXFR) PARM(&RETURN)
             IF COND(&RETURN *NE '      ') THEN(GOTO USERQUIT)

  /*                                                            */
  /*  DELAY THE JOB 1 HOUR                                      */
  /*                                                            */
 DELAY:      DLYJOB     DLY(3600) /* delay for 1 hr */

  /*                                                            */
  /*  END THE JOB ON SUNDAY  -- IT WILL RESTART MONDAY          */
  /*                                                            */
             RTVSYSVAL  SYSVAL(QDAYOFWEEK) RTNVAR(&DAYOFWK)
             IF COND(&DAYOFWK *EQ '*SUN') THEN(GOTO MSGTRAP)
  /*                                                            */
             GOTO CHECKIT

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
