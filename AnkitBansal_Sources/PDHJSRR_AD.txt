//?USER: Initialize program

// PAR.USR Active Flag = CND.Record Not Found
PAR.USR_Active_Flag = 'N';

// PAR.USR Inactive Flag = CND.Record Not Found
PAR.USR_Inactive_Flag = 'N';

//?USER: Load subfile record from DBF record

// PGM.*Record selected = CND.*NO
PGM.*Record_selected = 'N';

CASE;

// IF PAR.USR Active Flag is Record Found
IF PAR.USR_Active_Flag = 'Y';

// AND PAR.USR Inactive Flag is Record Found
AND PAR.USR_Inactive_Flag = 'Y';

QUIT;

ENDIF;

CASE;

// IF DB1.Record Status is Active
IF DB1.Record_Status = 'A';

// PAR.USR Active Flag = CND.Record Found
PAR.USR_Active_Flag = 'Y';

// PGM.*Record selected = CND.*YES
PGM.*Record_selected = 'Y';

ENDIF;

CASE;

// IF DB1.Record Status is Inactive
IF DB1.Record_Status = 'I';

// PAR.USR Inactive Flag = CND.Record Found
PAR.USR_Inactive_Flag = 'Y';

// PGM.*Record selected = CND.*YES
PGM.*Record_selected = 'Y';

ENDIF;
