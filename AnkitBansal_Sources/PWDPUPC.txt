             PGM

             DCL        VAR(&FROMLIB)  TYPE(*CHAR) LEN(10)
             DCL        VAR(&TOLIB)    TYPE(*CHAR) LEN(10)
             DCL        VAR(&COMPANY)  TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&FROMDATE) TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&TODATE)   TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&FLIBSTS)  TYPE(*CHAR) LEN(1)
             DCL        VAR(&TLIBSTS)  TYPE(*CHAR) LEN(1)
             DCL        VAR(&RETURN)   TYPE(*CHAR) LEN(7)

             /* Call screen for user selection */
 SCREEN:     CALL       PGM(PWDOPVR) PARM(&RETURN &FROMLIB &TOLIB +
                          &COMPANY &FROMDATE &TODATE &FLIBSTS &TLIBSTS)
             IF         COND(&RETURN = 'Y2U9999') THEN(GOTO +
                          CMDLBL(ENDIT))

             /* Validate that From and To libraries exist */
             CHKOBJ     OBJ(&FROMLIB) OBJTYPE(*LIB)
             MONMSG     MSGID(CPF9801) EXEC(DO)
             CHGVAR     VAR(&FLIBSTS) VALUE('Y')
             GOTO       CMDLBL(SCREEN)
             ENDDO

             CHKOBJ     OBJ(&TOLIB) OBJTYPE(*LIB)
             MONMSG     MSGID(CPF9801) EXEC(DO)
             CHGVAR     VAR(&FLIBSTS) VALUE('Y')
             GOTO       CMDLBL(SCREEN)
             ENDDO

             /* Validate that files exist in From and To Libraries */
             CHKOBJ     OBJ(&FROMLIB/PKBLCPP) OBJTYPE(*FILE)
             MONMSG     MSGID(CPF9801) EXEC(DO)
             CHGVAR     VAR(&MESSAGE) VALUE('''Hog Voucher Detail +
                          file does not exist in library '' *CAT +
                          &FROMLIB')
