/*                                                                 */
/*  FUNCTION:        This CL creates a Directory and Object Link   */
/*                   that is used to generate BrickFTP documents.  */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&EDITRDPTN)

/* INPUT PARMS                                                     */
             DCL        VAR(&EDITRDPTN) TYPE(*CHAR) LEN(6)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&DIRPATH) TYPE(*CHAR) LEN(30)

/* CREATE DIRECTORY WITH OBJECT LINK FOR BRICKFTP                 */

             CHGVAR     VAR(&DIRPATH) VALUE('/PORK/SFTP/OUT/' +
                          *TCAT &EDITRDPTN)

             CRTDIR     DIR(&DIRPATH)

             MONMSG     MSGID(CPFA0A0) /* Object already exists. */

             ENDPGM
