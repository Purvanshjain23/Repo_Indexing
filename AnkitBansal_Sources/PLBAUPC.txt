/*********************************************************************/
/*  PROGRAM  - PLBAUPC                                               */
/*  SYNOPSIS - Print Internal Company PRICE PELIMANARY REPORT.       */
/*             THIS CL WILL WIL BE ON THE JOB SCHEDULER.             */
/*  DEVELOPER- PURVA DROGE                                           */
/*  SYSTEM   - ORDER MANAGEMENT SYSTEM                               */
/*  DATE     - 08/31/06                                              */
/*                                                                   */
/*********************************************************************/
/*                 MAINTENANCE LOG                                   */
/*-------------------------------------------------------------------*/
/* DATE     BY  CHANGE DESCRIPTION                                   */
/* -------- --- ---------------------------------------------------- */
/* 05/08/17 DN  R9700 - STF Inter Company & Affiliated Sales Pricing.*/
/*                Hard coded 'G' (Guymon & Triumph) as Inter Company */
/*                Group when calling PPUCPFR.                        */
/*                                                                   */
/*********************************************************************/
 PLBAUPC:    PGM        PARM(&OUTQ &HOLD &SAVE &COPYN)
/* PARMS                                                                      */
             DCL        VAR(&OUTQ) TYPE(*CHAR) LEN(10) /* Output +
                          Queue              */
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4) /* Hold +
                          Status               */
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4) /* Save +
                          Status               */
             DCL        VAR(&COPYN) TYPE(*DEC) /* No. of +
                          Copies             */
             DCL        VAR(&ERROR1) TYPE(*CHAR) LEN(1)
             DCL        VAR(&TYPRUN) TYPE(*CHAR) LEN(1)


/* INTERNAL FIELDS                                                            */
             DCL        VAR(&FDATE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDATE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&COPY) TYPE(*DEC) LEN(2 0)

/* ERROR MESSAGE FIELDS                                                       */
             DCL        VAR(&MSG) TYPE(*CHAR) LEN(512)
             DCL        VAR(&MSGID) TYPE(*CHAR) LEN(7)
             DCL        VAR(&MSGL) TYPE(*DEC) LEN(5 0)

/*------------------------------------------------------------------*/
/*   INITIALIZE VARIABLES                                           */
/*------------------------------------------------------------------*/
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/*------------------------------------------------------------------*/
/* RETRIEVE THE REVENUE CLOSE FROM AND TO DATES                     */
/*------------------------------------------------------------------*/
             CALL       PGM(TF817) PARM(&FDATE &TDATE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/*------------------------------------------------------------------*/
/* PRINT REPORT OF TRANSFER PRICE UPDATE RECORDS AND ERORORS        */
/*------------------------------------------------------------------*/
/*           OVRPRTF    FILE(PPUBPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE) */
/*           MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP)) */

/*           CALL       PGM(PPUBPFR) PARM(' ' &TDATE &FDATE &TDATE +
                          &ERROR1 &TYPRUN) */
/*           MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP)) */

             OVRPRTF    FILE(PPUCPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

    /* R9700 - Added 'G' to Parm as the Inter Company Group. */
             CALL       PGM(PPUCPFR) PARM(' ' &TDATE &FDATE &TDATE +
                          'Y' 'P' 'G')
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
/*------------------------------------------------------------------*/
 END:        ENDPGM
             DLTOVR     FILE(*ALL)
