/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     12/29/98                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL CAN BE CALLED FROM AN RPG PROGRAM     */
/*                   FOR CONVERTING A DATE.                        */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*******************************************************************/
             PGM        PARM(&FDTE &TDTE &FDTEF &TDTEF &DTES)

             DCL        VAR(&FDTE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TDTE) TYPE(*CHAR) LEN(11)
             DCL        VAR(&FDTEF) TYPE(*CHAR) LEN(8)
             DCL        VAR(&TDTEF) TYPE(*CHAR) LEN(8)
             DCL        VAR(&DTES) TYPE(*CHAR) LEN(7)

             CVTDAT     DATE(&FDTE) TOVAR(&TDTE) FROMFMT(&FDTEF) +
                          TOFMT(&TDTEF) TOSEP(&DTES)

             ENDPGM
