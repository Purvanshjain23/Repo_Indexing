/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING CALENDAR                     */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  3/07/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &FRMDTEN       -  PASSED FROM DATE                       */
/*         &TODTEN        -  PASSED THRU DATE                       */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 CACLCLP:    PGM        PARM(&COMPNYN &FRMDTEN &TODTEN &APPLIC &OUTQ +
                          &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &COMPNYN     *DEC
        DCL    &FRMDTEN     *DEC
        DCL    &TODTEN      *DEC
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND    *CHAR   (512)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &FRDATE      *DEC   (7 0)
        DCL    &TODATE      *DEC   (7 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&FRDATE)   VALUE(&FRMDTEN)
             CHGVAR     VAR(&TODATE)   VALUE(&TODTEN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(CACLPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(CACLPFR) PARM(' ' &COMPANY &FRDATE &TODATE)

 ENDPGM:     ENDPGM
