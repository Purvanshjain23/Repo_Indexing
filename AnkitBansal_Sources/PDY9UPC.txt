/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDY9UPC                                       */
/*  PROGRAM NAME:    UPDATE CUTOUT MATRIX WITH PC UPLOAD           */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     05/13/97                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from A PROMPT.                                */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&XCOMP &MTX &XMODY &PGM)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)

/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&MTX) TYPE(*CHAR) LEN(6)
             DCL        VAR(&XMODY) TYPE(*CHAR) LEN(15)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&MODY) TYPE(*DEC) LEN(4 0)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&MODY) VALUE(&XMODY)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             CALL       PGM(&PGM) PARM(' ' &COMP &MTX &MODY)
             RGZPFM     FILE(PDI4REP)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
