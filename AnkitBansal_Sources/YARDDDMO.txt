SELECT
  ALL       A2B0DT/100 AS "MONTH", (T01.A2C0NB), (T01.A2D8NB),
            ((A2C0NB-A2D8NB))
  FROM      PRKFLIB/PKA2CPP T01 INNER JOIN
            PRKFLIB/PKB3CPP T02
  ON        A2HMNB = B3HMNB
    AND     A2BNNB = B3BNNB
    AND     A2BONB = B3BONB
    AND     A2CVNB = B3CVNB
    AND     A2B0DT = B3B0DT
  WHERE     A2B0DT >= 980101
    AND     A2B0DT <= 981231
    AND     B3FKNB = 1
  ORDER BY  001 ASC
