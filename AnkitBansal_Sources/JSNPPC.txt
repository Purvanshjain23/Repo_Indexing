SELECT
  ALL       T01.CDBQCD, T02.A1AECD, T02.A1OXNB, T01.CDBNNB, T01.CDBONB,
            T01.CDCVNB, T01.CDB0DT, T02.A1NZTX
  FROM      PRKFLIB/PKCDCPP T01 LEFT OUTER JOIN
            PRKFLIB/PKA1CPP T02
  ON        CDCVNB = A1CVNB
    AND     CDB0DT = A1B0DT
  WHERE     A1LKST = 'SBD'
    AND     CDB0DT >= 990301
    AND     CDB0DT < 990401
    AND     CDBQCD = 'NPPCI'
  GROUP BY  T01.CDB0DT, T02.A1OXNB, T01.CDBQCD, T02.A1AECD, T01.CDBNNB,
            T01.CDBONB, T01.CDCVNB, T02.A1NZTX
