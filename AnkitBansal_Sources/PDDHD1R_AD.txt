- DB1.PS_Pieces_Converted ) - DB1.PS_Pieces_Whs_Tfr_Adj;

// Weight Scheduled
DTL.Weight_Scheduled = ( ( DB1.PS_Weight_Standard + DB1.PS_Weight_Prod_Adj )                       +
- DB1.PS_Weight_Converted ) - DB1.PS_Weight_Whs_Tfr_Adj;

* PAR.Bags_Per_Unit );

// Quantity Prod Adj
DTL.Quantity_Adjustment_USR = DTL.PS_Pieces_Prod_Adj / ( PAR.Pieces_Per_Bag                        +
* PAR.Bags_Per_Unit );

// Quantity Converted

* PAR.Bags_Per_Unit );

// Quantity Non Prod Whse

* PAR.Bags_Per_Unit );
