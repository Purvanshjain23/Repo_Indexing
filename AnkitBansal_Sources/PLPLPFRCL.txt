/*                                                                 */
/*  PROGRAM NUMBER:  PLPLPFRCL                                     */
/*  PROGRAM NAME:    PRINT EVAL TRANS UPL COMMENT                  */
/*  PROGRAMMER:      SUSAN L. MASON                                */
/*  CREATE DATE:     10/13/2010                                    */
/*                                                                 */
/*  FUNCTION:        THIS CL PROCESS DURING THE EVAL TRANS UPL     */
/*                   MOVE TATTOO/REMOVE TAOO PROCESS               */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&XCOMP &XDATE &PDEV &HOLD &SAVE +
                          &XCOPIES &APP)

/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&APP) TYPE(*CHAR) LEN(40)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)
/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XDATE) TYPE(*CHAR) LEN(7)

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&DATE) TYPE(*DEC) LEN(7 0)



/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&APP)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&DATE) VALUE(&XDATE)
/* *********************************************************/

/* Create Printer File Override  Code   Eval Trans Upl Comment  */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLPLPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

             CALL       PGM(PLPLPFR) PARM(' ' &COMP &DATE)

             DLTOVR *ALL
             ENDPGM
