/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDMPUPR                                       */
/*  PROGRAM NAME:    Apply Production Adjustments                  */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     05/02/95                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from Apply Adjustments prompt.                */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &XCOMP &XPRDRGF &XPRDRGT &PGM)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)

/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XPRDRGF) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XPRDRGT) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&PRDRGF) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&PRDRGT) TYPE(*DEC) LEN(7 0)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&PRDRGF) VALUE(&XPRDRGF)
             CHGVAR     VAR(&PRDRGT) VALUE(&XPRDRGT)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             CALL       PGM(&PGM) PARM(' ' &COMP &PRDRGF &PRDRGT)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
