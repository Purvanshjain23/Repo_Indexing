             PGM        PARM(&ERROR &COMPIN &CUSTIN &ORDIN &PROC)

             DCL        VAR(&ERROR)  TYPE(*CHAR) LEN(1)
             DCL        VAR(&COMPIN) TYPE(*DEC)
             DCL        VAR(&CUSTIN) TYPE(*DEC)
             DCL        VAR(&ORDIN)  TYPE(*DEC)
             DCL        VAR(&PROC)   TYPE(*CHAR) LEN(1)

             DCL        VAR(&COMP)   TYPE(*DEC)  LEN(3)
             DCL        VAR(&CUST)   TYPE(*DEC)  LEN(7)
             DCL        VAR(&ORDR)   TYPE(*DEC)  LEN(7)
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(7)

             CHGVAR     VAR(&COMP) VALUE(&COMPIN)
             CHGVAR     VAR(&CUST) VALUE(&CUSTIN)
             CHGVAR     VAR(&ORDR) VALUE(&ORDIN)
/*                                                              */
/* MOVE QTEMP TO LAST ON THE LIBRARY LIST SO THE ORDER ENTRY    */
/* PROGRAMS WILL USE THE PRODUCTION FILES NOT FILES IN QTEMP    */
/*                                                              */
             RMVLIBLE   LIB(QTEMP)
             MONMSG     MSGID(CPF0000)

             ADDLIBLE   LIB(QTEMP) POSITION(*LAST)
             MONMSG     MSGID(CPF0000)

/** ** ** ** ** ** ** CALL ORDER ENTRY  ** ** ** ** ** ** ** **/
             CALL       PGM(PDSQXFR) PARM(' ' &COMP &ORDR &CUST &PROC)

/*                                                              */
/* MOVE QTEMP TO FIRST ON THE LIBRARY LIST SO THE BLOCK ORDER   */
/* TEMPLATE PROGRAM WILL USE THE FILES IN QTEMP NOT PRODUCTION  */
/*                                                              */
             RMVLIBLE   LIB(QTEMP)
             MONMSG     MSGID(CPF0000)

             ADDLIBLE   LIB(QTEMP) POSITION(*FIRST)
             MONMSG     MSGID(CPF0000)

 END:        ENDPGM
