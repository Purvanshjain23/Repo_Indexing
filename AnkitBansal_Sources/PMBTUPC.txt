/*********************************************************************/
/*                                                                   */
/* PROGRAM:     PMBTUPC                                              */
/* FUNCTION:    RETRIEVE FILE DESCRIPTION & RETURN TO CALLER         */
/*              SET RETURN CODE TO "NOT FOUND" VALUE IF INVALID      */
/*********************************************************************/
 PMBTUPC:    PGM        PARM(&FILE &DESC &RECFOUND)

             DCL        VAR(&FILE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&DESC) TYPE(*CHAR) LEN(50)
             DCL        VAR(&RECFOUND) TYPE(*CHAR) LEN(1)

             RTVMBRD    FILE(&FILE) TEXT(&DESC)
             MONMSG     MSGID(CPF0000) EXEC(CHGVAR VAR(&RECFOUND) +
                          VALUE('N'))

 ENDPGM:     ENDPGM
