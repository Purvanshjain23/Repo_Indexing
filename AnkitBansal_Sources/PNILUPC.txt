/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PNILUPC                                       */
/*  PROGRAM NAME:    Print Schedule ship off cut                */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     05/04/09                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from Display SCHEDULE VARIANCE                */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC 3/05/13 E2490 ADD EMAIL DIST LIST CALLED FROM INVNIGHTRN,  */
/*                    INVNIGHTRS                                   */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES  +
                  &COMP   &XDATE &DISTSCH)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&DISTSCH) TYPE(*CHAR) LEN(30)

/* Function Specific                                               */
             DCL        VAR(&COMP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XDATE) TYPE(*CHAR) LEN(7)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

             DCL        VAR(&COMPN) TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&PDATE) TYPE(*DEC) LEN(7 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&PDATE) VALUE(&XDATE)
             CHGVAR     VAR(&COMPN) VALUE(&COMP)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PNIJPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PNIJPFR) PARM(' ' &COMPN &PDATE)

             IF         COND(&DISTSCH *GT '     ') THEN(DO)
             ESNDMAIL   RECIPIENT(&DISTSCH) SUBJECT('Ship Off Cut +
                          Report') ATTLIST((* *PDF *N PNIJPFR$))
             ENDDO
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
