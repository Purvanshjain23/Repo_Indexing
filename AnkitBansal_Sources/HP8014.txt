      * TITLE:       Retrieve Beginning/Ending Dates for a Period
      * PROGRAMMER:  LeAnne Fedor
      * CREATED:     06/26/03
      *
      * FUNCTION:  Given an incoming year/period, this program:
      *                   1) calls a SYNON program to retrieve the beginning/ending period dates
      *                   2) calls an HPS program to format the returned SYNON dates into
      *                      CCYYMMDD format
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      * 99/99/99
      *
      /EJECT
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * CONSTANTS
      ****************************************************************
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      *
      ****************************************************************
      * ARRAYS AND TABLES
      ****************************************************************
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      * Parms
      *
     D xxreturn        s              7a
     D xxcocd          s              3  0
      *
     D xxacyr          s              4  0
     D xxacpe          s              2  0
      *
     D xxsynbegdt      s              7  0
     D xxsynenddt      s              7  0
      *
     D xxbpdt          s              8  0
     D xxepdt          s              8  0
      *
      *
      *
      *
      ****************************************************************

      ****************************************************************

     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************
      *
      * Retrieve the beginning/ending dates for the period.
      *
     C                   call      'POWCXFR'
     C                   parm      *blank        xxreturn
     C                   parm      350           xxcocd
     C                   parm                    xxacyr
     C                   parm                    xxacpe
     C                   parm      0             xxsynbegdt
     C                   parm      0             xxsynenddt
      *
      *
      * Get Synon beginning date into CCYYMMDD format
      *
     C                   call      'HP8011'
     C                   parm                    xxsynbegdt
     C                   parm      0             xxbpdt
      *
      *
      * Get Synon ending date into CCYYMMDD format
      *
     C                   call      'HP8011'
     C                   parm                    xxsynenddt
     C                   parm      0             xxepdt
      *
     C                   seton                                        lr
      /EJECT

      *
     C                   endsr
