PDGFUPC:    PGM  PARM(&CURDATE &CURTIME)

/* PARMS */
            DCL        VAR(&CURDATE) TYPE(*DEC)
            DCL        VAR(&CURTIME) TYPE(*DEC)
/* INTERNAL FIELDS */
            DCL        VAR(&CURDAT)  TYPE(*CHAR) LEN(6)
            DCL        VAR(&CURDATA) TYPE(*CHAR) LEN(7)
            DCL        VAR(&CURYYA)  TYPE(*CHAR) LEN(2)
            DCL        VAR(&CURTIM)  TYPE(*CHAR) LEN(8)
            DCL        VAR(&CURTIMA) TYPE(*CHAR) LEN(6)
/*‚JM   06/05/25 SDN1295 */
             DCL        VAR(&SYSNAME) TYPE(*CHAR) LEN(10)
/*‚JM   06/05/25 SDN1295 */

            RTVSYSVAL  SYSVAL(QDATE) RTNVAR(&CURDAT)
/*‚JM   06/05/25 SDN1295 */
             RTVNETA    SYSNAME(&SYSNAME)
             IF         COND(&SYSNAME *EQ 'SBFDEV') THEN(DO)
/*           CHGVAR     VAR(&CURDAT) VALUE('061825') */
             ENDDO
/*‚JM   06/05/25 SDN1295 */
            CVTDAT     DATE(&CURDAT) TOVAR(&CURDAT) FROMFMT(*MDY) +
                        TOFMT(*YMD) TOSEP(*NONE)
            CHGVAR VAR(&CURYYA) VALUE(%SUBSTRING(&CURDAT 1 2))
            IF (&CURYYA *LT '40') +
               CHGVAR VAR(&CURDATA) VALUE('1' *CAT &CURDAT)
            ELSE +
               CHGVAR VAR(&CURDATA) VALUE('0' *CAT &CURDAT)
            CHGVAR VAR(&CURDATE) VALUE(&CURDATA)

            RTVSYSVAL  SYSVAL(QTIME) RTNVAR(&CURTIM)
            CHGVAR VAR(&CURTIMA) VALUE(%SST(&CURTIM 1 6))
            CHGVAR VAR(&CURTIME) VALUE(&CURTIMA)

ENDPGM
