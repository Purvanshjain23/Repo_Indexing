/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKGNUPC                                       */
/*  PROGRAM NAME:    PRINT BALANCE TO KILL REPORT                  */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     12/31/96                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM WORK WITH BAL TO KILL.                   */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  R. CENTONZE  04/16/97  ADD PARM FOR SCHEDULE KILL SHIFT SELECT */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &XCO &XKILL &PDEV &HOLD &SAVE +
                          &XCOPIES &OMS &KSHIFT)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&NME) TYPE(*CHAR) LEN(40)
             DCL        VAR(&XKILL) TYPE(*CHAR) LEN(7)
             DCL        VAR(&KSHIFT) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&Y) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&Z) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&DKILL) TYPE(*DEC) LEN(7 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
 LOOP:          CHGVAR &Y VALUE(&Y + 1)
             IF         COND(&Y = 40) THEN(GOTO CMDLBL(GOON))
             IF         COND((%SST(&NME &Y 2)) = '  ') THEN(DO)
             CHGVAR &Z VALUE(40 - &Y)
             CHGVAR     VAR(%SST(&NME &Y &Z)) VALUE(' ')
             GOTO       CMDLBL(GOON)
             ENDDO
             GOTO       CMDLBL(LOOP)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
GOON:        CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&CO) VALUE(&XCO)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&DKILL) VALUE(&XKILL)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKGMPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKGMPFR) PARM(' ' &CO &DKILL &KSHIFT &LPP)

             DLTOVR *ALL
             ENDPGM
