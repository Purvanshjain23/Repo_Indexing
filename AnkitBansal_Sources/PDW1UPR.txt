/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  THIS CL WILL CREATE AN EMAIL RECORD FOR THE CUSTOMER*/
/*              USER ID PASSED IN. THE RECORD WILL BE CREATED IN    */
/*              AN IBM FILE BY USING THE PRKEMAIL COMMAND.          */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  DATE      - 03/08/99                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     I    &EMAILADDR  - EMAIL ADDRESS                             */
/*     I    &EMAILUSR   - EMAIL CUSTOMER USER ID (A + CUSTOMER #)   */
/********************************************************************/
/* 04/11/2012 LJB E000900  MONMSG FOR BLANK EMAIL ADDRESS           */
/********************************************************************/

 PDW1UPR:    PGM        PARM(&EMAILADDR &EMAILUSR)

/* PARMS                                                                      */
        DCL    &EMAILADDR  *CHAR   (50)          /* EMAIL ADDRESS             */
        DCL    &EMAILUSR   *CHAR   (8)           /* EMAIL USER ID             */

             MONMSG     MSGID(CPF0000)

             PRKEMAIL   USER(&EMAILUSR) ROUTE(&EMAILADDR)

 END:        ENDPGM
