/********************************************************************/
/*  SYNOPSIS -  RECEIVE PARAMETERS AND PASS TO SUBMITTED            */
/*              PROGRAM THAT WILL CREATE THE REPORT                 */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - SUBMITS PROGRAM TO PRINT THE FUEL SURCHARGE REPORT  */
/*  DATE      - 05/18/2011                                          */
/*  PROGRAMMER- LARA BUSER                                          */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*   &DESC       - PRINT PROGRAM DESCRIPTION                        */
/*   &OUTQ       - OUTPUT QUEUE                                     */
/*   &HOLD       - HOLD STATUS                                      */
/*   &SAVE       - SAVE STATUS                                      */
/*   &BHSTS      - BACKHAUL STATUS                                  */
/*   &BHCARR     - BACKHAUL CARRIER CODE                            */
/*   &BHCOMPN    - BACKHAUL COMPANY NUMBER 15.5                     */
/*   &FRMDATN    - FROM ACTUAL SHIPPED DATE 15.5                    */
/*   &THRDATN    - THRU ACTUAL SHIPPED DATE 15.5                    */
/*   &COPYN      - NO. OF COPIES  15.5                              */
/*   &EMAIL      - EMAIL ADDRESS                                    */
/*   &EMAILSEL   - EMAIL SELECT FLAG                                */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/* XXX 99/99/9999 X999999                                           */
/********************************************************************/
 PNXPUPC:    PGM        PARM(&BHSTS &BHCARR &RPTTYPE &DESC &OUTQ +
                          &HOLD &SAVE &BHCOMPN &FRMDATN &THRDATN +
                          &COPYN &EMAILSEL &EMAIL)

             DCL        VAR(&BHSTS)     TYPE(*CHAR) LEN(1)
             DCL        VAR(&BHCARR)    TYPE(*CHAR) LEN(3)
             DCL        VAR(&RPTTYPE)   TYPE(*CHAR) LEN(1)
             DCL        VAR(&DESC)      TYPE(*CHAR) LEN(25)
             DCL        VAR(&OUTQ)      TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD)      TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE)      TYPE(*CHAR) LEN(4)
             DCL        VAR(&BHCOMPN)   TYPE(*DEC)
             DCL        VAR(&FRMDATN)   TYPE(*DEC)
             DCL        VAR(&THRDATN)   TYPE(*DEC)
             DCL        VAR(&COPYN)     TYPE(*DEC)
             DCL        VAR(&EMAIL)     TYPE(*CHAR) LEN(50)
             DCL        VAR(&EMAILSEL)  TYPE(*CHAR) LEN(1)
/* ERROR MESSAGE FIELDS                                             */
             DCL        VAR(&MSG)       TYPE(*CHAR) LEN(512)
             DCL        VAR(&MSGID)     TYPE(*CHAR) LEN(7)
             DCL        VAR(&MSGL)      TYPE(*DEC) LEN(5 0)

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/*-------------------------------------------------------------------*/
/* SUBMIT FUEL SURCHARGE REPORT PRINT CL - PNWUUPC                   */
/*-------------------------------------------------------------------*/
 SUBMIT:     SBMJOB     CMD(CALL PGM(PNWUUPC) PARM(&BHSTS &BHCARR +
                          &RPTTYPE &DESC &OUTQ &HOLD &SAVE &BHCOMPN +
                          &FRMDATN &THRDATN &COPYN &EMAILSEL &EMAIL))

             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/
 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
