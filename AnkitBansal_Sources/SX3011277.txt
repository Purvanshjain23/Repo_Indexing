/******************************************************************************/
             PGM

             DCL        VAR(&SYSNAME) TYPE(*CHAR) LEN(10)

             RTVNETA    SYSNAME(&SYSNAME)

             IF         COND(&SYSNAME *NE 'SBFPROD') THEN(GOTO +
                          CMDLBL(END))

             CRTDDMF    FILE(QGPL/PVBVREPDDM) +
                          RMTFILE(OMSDEVSQC/PVBVREP) +
                          RMTLOCNAME(SBFDEV.SEABOARDFOODS.COM *IP)
             MONMSG     MSGID(CPF0000)

             CPYF       FROMFILE(QGPL/PVBVREPDDM) +
                          TOFILE(PRKFLIB/PVBVREP) MBROPT(*REPLACE) +
                          CRTFILE(*NO)
             MONMSG     MSGID(CPF0000)
END:
             ENDPGM
