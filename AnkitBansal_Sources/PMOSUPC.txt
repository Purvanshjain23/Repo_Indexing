/*  PROGRAMMER:      LARA BUSER  10/12/2007                        */
/*  FUNCTION:        REMOVE 'Z' MEMBERS FROM PURGE TAPE FILE       */
/*******************************************************************/
/*                         MODIFICATIONS                           */
/*******************************************************************/
/* 05/19/2009 LJB E00383                                           */
/*    REMOVE FROM PURGE LIB: PURGE LOGICAL, PHYSICAL, TAPE FILE    */
/*    INSTEAD OF JUST CLEARING THEM. THEY WILL BE RECREATED        */
/*    DURING THE NEXT PURGE. ELIMINATES THE NEED TO REMEMBER       */
/*    TO REPLACE THEM WHEN A FILE CHANGES.                         */
/* 05/12/2008 LJB FP1009                                           */
/*    REMOVE 'Z' MEMBERS, CLEAR THE REST                           */
/*******************************************************************/
/***SUPC:    PGM        PARM(&PCGCODE &PCFTAPEF &TOLIB)         ****/
 PMOSUPC:    PGM        PARM(&PCGCODE &FILE &LOGICAL &PCFTAPEF &TOLIB)

             DCL        VAR(&PCGCODE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&FILE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&LOGICAL) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PCFTAPEF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TOLIB) TYPE(*CHAR) LEN(10)

             MONMSG     MSGID(CPF0000)
             DLTF       FILE(&TOLIB/&LOGICAL)
             MONMSG     MSGID(CPF2105)
             DLTF       FILE(&TOLIB/&FILE)
             MONMSG     MSGID(CPF2105)
             DLTF       FILE(&TOLIB/&PCFTAPEF)
             MONMSG     MSGID(CPF2105)

/***         RMVM       FILE(&TOLIB/&PCFTAPEF) MBR(Z*)            ***/
/***         CLRPFM     FILE(&TOLIB/&PCFTAPEF) MBR(*ALL)          ***/

 END:        ENDPGM
