/*                                                                  */
/*  SYNOPSIS -  CALL PGM TO SUBMIT UPDATE ACCU-SORT SALES ORDER     */
/*              HEADER & DETAIL FILES.                              */
/*                                                                  */
/*  USER NAME - DON MCINTYRE                                        */
/*  DATE      -  2/09/96                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPL      - COMPANY NUMBER                            */
/*                                                                  */
/********************************************************************/

 PDQGLCLP:   PGM        PARM(&COMP &LOAD)

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3)
             DCL        VAR(&COMPL) TYPE(*DEC)
             DCL        VAR(&LOAD) TYPE(*DEC) LEN(7)
             DCL        VAR(&LOADL) TYPE(*DEC)

/********************************************************************/
/*   SUBMIT JOB TO BATCH                                            */
/********************************************************************/
             CHGVAR     VAR(&COMPL) VALUE(&COMP)
             CHGVAR     VAR(&LOADL) VALUE(&LOAD)

/*           SBMJOB     CMD(CALL PGM(PDSOPCLP) PARM(&COMPL &LOADL)) +
                          JOB(BLDORDERS)                            */
             CALL       PGM(PDSOPCLP) PARM(&COMPL &LOADL)

 END:        ENDPGM
