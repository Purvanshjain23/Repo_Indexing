 PPPNUPC:    PGM        PARM(&RTN &WEEKA &RESTART &IVG)

             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7) /*  RETURN +
                          CODE   */
             DCL        VAR(&WEEKA) TYPE(*CHAR) LEN(8) /*  ISC WEEK +
                          END DATE */
             DCL        VAR(&WEEKN) TYPE(*DEC) LEN(8 0) /*  ISC WEEK +
                          END DATE */
             DCL        VAR(&RESTART) TYPE(*CHAR) LEN(1) /*  RESTART */
             DCL        VAR(&IVG) TYPE(*CHAR) LEN(1) /* Inv Valuation Group */

             CHGVAR    VAR(&WEEKN) VALUE(&WEEKA) /* Change to CHAR  */
 /* ------------------------------------------------------------------------*/
 /* DELETE INV VAL FILE FOR A WEEK IF RESTARTING THE VALUATION PROCESS      */
 /* ------------------------------------------------------------------------*/
    /* R9083-Added Inv Valuation Grouping as parm. */
          /* IF         COND(&RESTART *EQ 'Y') THEN(CALL +
                          PGM(PPPOXFR) PARM(' ' &WEEKN)) */
             IF         COND(&RESTART *EQ 'Y') THEN(CALL +
                          PGM(PPPOXFR) PARM(' ' &IVG &WEEKN))

 /* ------------------------------------------------------------------------*/
 /* CALL THE INVENTORY VALUATION PROCESS TO CREATE THE IVF FILE          */
 /* ------------------------------------------------------------------------*/
    /* R9083-Added Inv Valuation Grouping as parm. */
          /* CALL       PGM(PPPKXFR) PARM('' &WEEKN) */
             CALL       PGM(PPPKXFR) PARM('' &IVG &WEEKN)
 /* ------------------------------------------------------------------------*/

 END:        ENDPGM
