/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PMFYUPC                                       */
/*  PROGRAM NAME:    Build Production Schedule/Plan FROM MENU   */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     03/04/09                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from Build Production Schedule/Plan prompt.   */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC 02/05/09 FP1054 ADDED NEW ADJUSTMENT TYPE "OV" REPORT    */
/*                      TO OVERRIED STD PROD IN PDI8CPP->PMEFPFR */
/*                      REMOVE OVERRIDE FOR POTYPFR$ - NOT CALLED ATALL */
/*                                                                 */
/*  SLM 02/27/09 FP1054 ADD OVERRIDE PRINT FILE PMFPPFR$         */
/*******************************************************************/
             PGM        PARM(&RTN &XCOMP &XKILLDTE &XKILLRGF +
                          &XKILLRGT &XPRDDTE &XPRDRGF &XPRDRGT +
                          &FSTLVL &OVRLAY &PDEV &HOLD &SAVE &XCOPIES)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* Function Specific                                               */
             DCL        VAR(&PGMRUN) TYPE(*CHAR) LEN(10) VALUE('MENU  ')
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XKILLDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XPRDDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XPRDRGF) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XPRDRGT) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XKILLRGF) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XKILLRGT) TYPE(*CHAR) LEN(7)
             DCL        VAR(&FSTLVL) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OVRLAY) TYPE(*CHAR) LEN(1)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KILLDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&PRDDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&PRDRGF) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&PRDRGT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&KILLRGF) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&KILLRGT) TYPE(*DEC) LEN(7 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&HOLD) VALUE('*YES')
             CHGVAR     VAR(&SAVE) VALUE('*YES')

             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&KILLDTE) VALUE(&XKILLDTE)
             CHGVAR     VAR(&PRDDTE) VALUE(&XPRDDTE)
             CHGVAR     VAR(&PRDRGF) VALUE(&XPRDRGF)
             CHGVAR     VAR(&PRDRGT) VALUE(&XPRDRGT)
             CHGVAR     VAR(&KILLRGF) VALUE(&XKILLRGF)
             CHGVAR     VAR(&KILLRGT) VALUE(&XKILLRGT)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PKY9PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDJZPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDBHPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDAEPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDZ9PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDMAPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDVZPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDXYPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMEFPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMFPPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PKY2PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)
/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             CALL       PGM(PMFVXFR) PARM(' ' &COMP &KILLDTE +
                          &KILLRGF &KILLRGT &PRDDTE &PRDRGF &PRDRGT +
                          &FSTLVL &OVRLAY &PGMRUN)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
