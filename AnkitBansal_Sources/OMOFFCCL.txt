/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  OFFICE  DOCUMENTS ARE USED FOR INPUTING ITEM SPEC-  */
/*              IFICATIONS                                          */
/*              THIS PROGRAM WILL CREATE A FOLDER WITH A PREFIX     */
/*              OF FS + COMPANY, AND A DOCUMENT WITHIN THE FOLDER   */
/*              PREFIXED WITH AN S + ITEM NUMBER IF ONE DOES  NOT   */
/*              ALREADY EXIST.                                      */
/*              *** THIS IS FOOD SERVICE GROUP                      */
/*  COMPANY   - SEABOARD CORPORATION                                */
/*  SYSTEM    - IDEAS/400 CROSS APPLICATION                         */
/*  PROGRAMMER- DON MCINTYRE                                        */
/*  PROGRAM   - SYSTEM CONTROL SELECTIONS                           */
/*  DATE      - 11/29/93                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &CO         - COMPANY NUMBER ALPHA                      */
/*          &ITEM       - ITEM CODE ALPHA                           */
/*          &CODE       - SPECIFICATION TYPE = S                    */
/*                                                                  */
/********************************************************************/

             PGM        PARM(&CO &ITEM &CODE)
             DCL        VAR(&CO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&ITEM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&CODE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&FOLDER) TYPE(*CHAR) LEN(8)
             DCL        VAR(&DOC) TYPE(*CHAR) LEN(8)
      /*     MONMSG     MSGID(CPF0000)     */

      /*  BUILD FOLDER NAME (&FOLDER)                                     */
      CHGVAR VAR(&FOLDER) VALUE('FS' *CAT &CO)

      /*  BUILD DOCUMENE NAME (&DOC)                                      */
      CHGVAR VAR(&DOC   ) VALUE(&CODE *CAT &ITEM)

             CHKDLO     DLO(&FOLDER) OBJTYPE(*FLR)
             MONMSG     MSGID(CPF8A77) EXEC(+
             CRTFLR     FLR(&FOLDER) TEXT('Folder for Company'))

             CHKDLO     DLO(&DOC) FLR(&FOLDER) OBJTYPE(*DOC)
             MONMSG     MSGID(CPF8A82) EXEC(CRTDOC DOC(&DOC) +
                          FLR(&FOLDER) TEXT('Document for +
                          Company/Item') DETAILS(*NO) EDIT(*NO) +
                          EXITPNL(*NO))

             EDTDOC     DOC(&DOC) FLR(&FOLDER) EXITPNL(*NO)

             ENDPGM
