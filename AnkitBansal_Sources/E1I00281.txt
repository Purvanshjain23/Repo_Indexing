     C*    MAINLINE PROGRAM
     C*    ----------------
     C*
     C     *ENTRY        PLIST
     C                   PARM                    D00921                         Data Str
     C*
     C*    Open file - User Display Preference File
     C*
     C*                  OPEN      F00921                               81
     C                   OPEN      E100921B                             81
     C     *IN81         IFEQ      '1'
     C                   GOTO      END
     C*                    ---- ---
     C                   END
     C*
     C*    Load User Display information into Data Structure
     C*
     C                   MOVE      *BLANKS       jLUSER
     C                   MOVEL     ##USER        jLUSER
     C     jLUSER        CHAIN     I00921b                            81
     C     *IN81         IFEQ      '0'
     C     jLFRMT        IFNE      *BLANKS
     C                   MOVE      jLFRMT        #9FRMT
     C                   END
     C*
     C     jLDSEP        IFNE      *BLANKS
     C     jLDSEP        IFEQ      '*'
     C                   MOVE      *BLANKS       #9DSEP
     C                   ELSE
     C*
     C                   MOVE      jLDSEP        #9DSEP
     C                   END
     C                   END
     C*
     C     jLLNGP        IFNE      *BLANKS
     C                   MOVE      jLLNGP        #9LNGP
     C                   END
     C*
     C     jLCO          IFNE      *BLANKS
     C                   MOVE      jLCO          #9CO
     C                   END
     C*
     C     jLDECF        IFNE      *BLANKS
     C     jLDECF        IFEQ      '.'
     C                   MOVE      *BLANKS       #9DECF
     C                   END
     C*
     C     jLDECF        IFEQ      ','
     C                   MOVE      'I'           #9DECF
     C                   END
     C                   END
     C*
     C     jLVRSP        IFNE      *BLANKS
     C                   MOVE      jLVRSP        #9VRSP
     C                   END
     C*
     C     jLCUR         IFNE      *BLANKS
     C                   MOVE      jLCUR         #9CUR
     C                   END
     C*
     C     jLCTR         IFNE      *BLANKS
     C                   MOVE      jLCTR         #9CTR
     C                   END
     C                   END
     C*
     C*                  CLOSE     F00921
     C                   CLOSE     E100921B
     C*
     C*    Return to calling program, no LR.
