/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM FOR SHIPPING LINE/SAIL PORT REORT      */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT SHIPPING LINE REPORT                          */
/*  DATE      - 12/31/96                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &SHPLIN     - SHIPPING LINE                             */
/*          &SAILP      - SAIL PORT CITY                            */
/*          &STATUS     - RECORD STATUS                             */
/*          &APPLIC     - APPLICATION CODE                          */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/*          &NIGHTQ     - NIGHT OUTPUT QUEUE                        */
/********************************************************************/

 PDUUPCLP:   PGM        PARM(&SHPLIN &SAILP &STATUS &APPLIC &OUTQ +
                          &HOLD &SAVE &COPYN &NIGHTQ)

/* PARMS                                                                      */
        DCL    &SHPLIN     *CHAR   (3)           /* SHIPPING LINE             */
        DCL    &SAILP      *CHAR   (20)          /* SAIL PORT CITY            */
        DCL    &STATUS     *CHAR   (1)           /* RECORD STATUS             */
        DCL    &DESC       *CHAR   (25)          /* DESCRIPTION               */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &APPLIC     *CHAR  (06)
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDUUPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)


/*********************************************************************/
/*   CALL PRINT PROGRAM -
/*********************************************************************/

             CALL       PGM(PDUUPFR) PARM(' ' &SHPLIN &SAILP +
                          &STATUS)

 END:        ENDPGM
