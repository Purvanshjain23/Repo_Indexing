/********************************************************************/
/*  SYNOPSIS -  RECEIVE 2 PARAMETER STRINGS. CONCATONATE AND        */
/*              TRIM OUT EXTRA BLANKS. RETURN AS ONE STRING         */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - *BCAT 2 STRINGS                                     */
/*  DATE      - 05/16/2011                                          */
/*                                                                  */
/*  INPUT PARAMETERS                                                */
/*  DESCRIPTON - 30 CHARACTER STRING                                */
/*  TEXT 10A   - 10 CHARACTER STRING                                */
/*  OUTPUT PARAMETER                                                */
/*  TEXT 40A   - 40 CHARACTER STRING RETURNED TO CALLING PROGRAM    */
/*                                                                  */
/***** MODIFICATION LOG *********************************************/
/*  9/99/99 XXX                                                     */
/*                                                                  */
/********************************************************************/
 PDOCUPR:    PGM PARM(&PARM1 &PARM2 &PARM3)

/* PARMS */
        DCL    &PARM1    *CHAR  (30)
        DCL    &PARM2    *CHAR  (10)
        DCL    &PARM3    *CHAR  (40)

/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
             MONMSG     MSGID(CPF0000)

/*   INITIALIZE FIELDS                                              */
             CHGVAR     VAR(&PARM3) VALUE(&PARM1 *BCAT &PARM2)

 END:        ENDPGM
