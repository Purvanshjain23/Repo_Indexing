/*                                                                  */
/*  SYNOPSIS - PROGRAM FOR INSTALLING DWI SYSTEM APPLICATIONS       */
/*             AND PTF TAPES.  THIS PROGRAM IS RUN BY THE USER      */
/*             USING THE LODRUN COMMAND                             */
/*                                                                  */
/*  COMPANY   - DATAWRIGHT INC.                                     */
/*  SYSTEM    - PROGRAM INSTALLATION                                */
/*  USER NAME - DEBBIE HAMNING                                      */
/*  DATE      - 11/18/92                                            */
/*  COPYRIGHT 1992 DATAWRIGHT INC.                                  */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*             &DEV         - MEDIA DEVICE                          */
/*                                                                  */
/********************************************************************/
 QINSTAPP:     PGM       PARM(&DEV)
        DCL    &DEV          *CHAR   10          /*  DEVICE         */

 RESTORE:    RSTLIB     SAVLIB(DWINSTAL) DEV(&DEV) ENDOPT(*LEAVE)

             ADDLIBLE   DWINSTAL *FIRST
             MONMSG     MSGID(CPF2103)
             CALL       PGM(JCYZCLP) PARM(&DEV)

             RMVLIBLE   DWINSTAL
             SNDPGMMSG  MSGID(CPF9898) MSGF(QCPFMSG) MSGDTA('DWS PTF +
                        installation is now complete') MSGTYPE(*COMP)
 END:        ENDPGM
