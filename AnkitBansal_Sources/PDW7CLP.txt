/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR POSTING PHYSICAL COUNT                */
/*                                                                  */
/*  SYSTEM    - INVENTORY MANAGEMENT                                */
/*  DATE      -  8/18/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  COMPANY                                */
/*         &DATE          -  DATE                                   */
/*         &WHSE          -  WAREHOUSE                              */
/*         &CYCLE         -  CYCLE                                  */
/*                                                                  */
/* RMC  06/09/08  RUN PGM TO DEL ITEM BAL DETAIL WITH -0- QTY BEFORE */
/*                PDW7XFR       - E00168                            */
/********************************************************************/
 ICPOCLP:    PGM        PARM(&COMPNYN &DATE &WHSE &CYCLE)

        DCL    &COMPNYN     *DEC
        DCL    &DATE        *DEC
        DCL    &WHSE        *CHAR  (03)
        DCL    &CYCLE       *CHAR  (03)
        DCL    &EXCEP       *CHAR  (01)
        DCL    &POSTEXC     *CHAR  (01)
        DCL    &RETURN      *CHAR  (7)

        DCL    &COMP        *DEC   (3 0)
        DCL    &DATEL       *DEC   (7 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)
             CHGVAR     VAR(&DATEL)  VALUE(&DATE)
             CHGVAR     VAR(&POSTEXC) VALUE(Y)

 E00168:    CALL       PGM(PMVHXFR) PARM(&RETURN &COMP &WHSE)

 CALL:       CALL       PGM(PDW7XFR) PARM(&RETURN &COMP &WHSE &CYCLE +
                          &DATEL &POSTEXC &EXCEP)

/* THIS PROGRAM WILL ADD THE SPECIAL ADJUSTMENTS BACK INTO THE             */
/* INVENTORY                                                               */

             CALL       PGM(PDLYXFR) PARM(&RETURN &COMP &WHSE &DATEL +
                                          &CYCLE)

 ENDPGM:     ENDPGM
