/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING CUSTOMER TRAFFIC INFO LIST   */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      - 12/31/96                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*         &SLSRTE        -  SALES ROUTE                            */
/*                                                                  */
/********************************************************************/
 PDGBPBLP:   PGM        PARM(&COMPNYN &APPLIC &OUTQ &HOLD &SAVE +
                             &COPYN &NIGHTQ &SLSRTE)
/* NOTE:  SLSRTE NOT USED HERE; PASSED IN SO COULD USE SAME CALLING  +
          CL                                                        */

        DCL    &COMPNYN     *DEC
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &SLSRTE      *CHAR  (3)

        DCL    &COMP        *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDUVPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDUVPFR) PARM(' ' &COMP)

 ENDPGM:     ENDPGM
