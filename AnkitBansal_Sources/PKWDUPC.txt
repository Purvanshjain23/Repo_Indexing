/*  PROGRAM NAME:    RETRIEVE SYSTEM DATE AND TIME                 */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     01/19/01                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL RETRIEVES THE SYSTEM DATE AND TIME.   */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&DTE &TIME)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */

/* Function Specific                                               */

/*                         */
             DCL        VAR(&DTE) TYPE(*DEC) LEN(7)
             DCL        VAR(&SYSDTE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&XDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&SYSTIM) TYPE(*CHAR) LEN(8)
             DCL        VAR(&ATIM) TYPE(*CHAR) LEN(6)
             DCL        VAR(&TIME) TYPE(*DEC) LEN(6 0)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/


/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             RTVSYSVAL  SYSVAL(QDATE) RTNVAR(&SYSDTE)

             CVTDAT     DATE(&SYSDTE) TOVAR(&XDTE) FROMFMT(*MDY) +
                          TOFMT(*CYMD) TOSEP(*NONE)

             CHGVAR     VAR(&DTE) VALUE(&XDTE)

             RTVSYSVAL  SYSVAL(QTIME) RTNVAR(&SYSTIM)
             CHGVAR     VAR(&ATIM) VALUE(%SST(&SYSTIM 1 6))
             CHGVAR     VAR(&TIME) VALUE(&ATIM)

/*-----------------------------------------------------------------*/
/*                     ERROR MESSAGES                              */
/*-----------------------------------------------------------------*/
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
