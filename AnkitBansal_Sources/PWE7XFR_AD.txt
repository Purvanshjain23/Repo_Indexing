//?Execute user function

//?This function will Convent a TME field that is 4.0 Packed into the
//?format of HH:MM
//?Convert TME to 5-Character Alpha
// LCL.Time Alpha USR 5 = CVTVAR(PAR.Time Usr 4.0)
LCL.Time_Alpha_USR_5 = CVTVAR(PAR.Time_Usr_4_0);

//?Substring HH and MM
// LCL.HH Alpha USR = SUBSTRING(LCL.Time Alpha USR 5,CON.2,CON.2)
LCL.HH_Alpha_USR = SUBSTRING(LCL.Time_Alpha_USR_5:2:2);

// LCL.MM Alpha USR = SUBSTRING(LCL.Time Alpha USR 5,CON.4,CON.2)
LCL.MM_Alpha_USR = SUBSTRING(LCL.Time_Alpha_USR_5:4:2);

//?Concatenate HH:MM
// LCL.Time HH:MM Alpha USR = CONCAT(LCL.HH Alpha USR,CON.:,CON.*ZERO)
LCL.Time_HHMM_Alpha_USR = LCL.HH_Alpha_USR + ':' (*ZERO);

// LCL.Time HH:MM Alpha USR = CONCAT(LCL.Time HH:MM Alpha USR,LCL.MM Alpha USR,CON.*ZERO)
LCL.Time_HHMM_Alpha_USR = LCL.Time_HHMM_Alpha_USR + LCL.MM_Alpha_USR (*ZERO);

//?Output Time
// PAR.Time HH:MM Alpha USR = LCL.Time HH:MM Alpha USR
PAR.Time_HHMM_Alpha_USR = LCL.Time_HHMM_Alpha_USR;
