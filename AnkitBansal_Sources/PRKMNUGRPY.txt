SELECT
  ALL       T01.GRPID#, T02.MNAME, T02.MOPTN, T03.MDESC
  FROM      WM400/WMGROUP T01 INNER JOIN
            WM400/WMAUTH T02
  ON        T01.GRPID# = T02."USERID" INNER JOIN
            WM400/WMOPTN T03
  ON        T02.MCLASS = T03.MCLASS
    AND     T02.MAPLCN = T03.MAPLCN
    AND     T02.MNAME = T03.MNAME
    AND     T02.MOPTN = T03.MOPTN
  WHERE     T01.COID = 'GUY'
    AND     T02.COID = 'GUY'
    AND     T03.COID = 'GUY'
    AND     T02.MNAME BETWEEN 'HPSMNU01' AND 'HPSMNU99'
  ORDER BY  T01.GRPID# ASC, T02.MNAME ASC, T02.MOPTN ASC
