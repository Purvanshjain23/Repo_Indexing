/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PMRKUPR                                       */
/*  PROGRAM NAME:    Print EMTS TRANS PER DATE                     */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     07/07/10                                      */
/*                                                                 */
/*  FUNCTION:        THIS CLP IS SUBMITTED FROM A WW               */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
      PGM        PARM(&RTN  &PDEV &HOLD &SAVE &TRDATE +
                          &COPYN &GENEMTS &RINEMTS)
                                                                       /**/
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPYN) TYPE(*DEC)
             DCL        VAR(&TRDATE) TYPE(*DEC)
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&COPY) TYPE(*DEC) LEN(2 0)

/* Function Specific                                               */
               DCL        VAR(&DATE) TYPE(*DEC) LEN(7 0)
               DCL        VAR(&GENEMTS) TYPE(*CHAR)  LEN(1)
               DCL        VAR(&RINEMTS) TYPE(*CHAR) LEN(1)


/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&DATE) VALUE(&TRDATE)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             OVRPRTF    FILE(PNTEPFR$) OUTQ(&PDEV) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

             CALL      PGM(PNTEPFR) PARM(' '  &DATE &GENEMTS &RINEMTS)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
