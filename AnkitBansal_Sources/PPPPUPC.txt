/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION. CAUSE PRINTER OVERRIDE  */
/*              PROMPTING AND PASSES SELECTED FIELDS FOR THE        */
/*              CUSTOMER CLAIM REPORT.                              */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  DATE      - 06/28/06                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*                         IF THIS IS BLANK AT START, PROMPT        */
/*                         PGM IS CALLED; OTHERWISE, THE PRINT CL   */
/*                         IS SUBMITTED WITHOUT PROMPTING           */
/*          &HOLD       -  HOLD STATUS                              */
/*          &SAVE       -  SAVE STATUS                              */
/*          &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*          &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*          &COPYN      - NO. OF COPIES  (PASSED)                   */
/*          &LOAD       - LOAD ID                                   */
/*          &CLM        - CLAIM NUMBER                              */
/*          &CLMT       - CLAIM TYPE                                */
/*          &CLMST      - CLAIM STATUS                              */
/*          &CLMDT      - CLAIM ENTRY DAT                           */
/*          &CLMCO      - CLAIM COMPANY N                           */
/*          &CLMSP      - CLAIM SHIP TO C                           */
/*          &ORDNO      - CLAIM ORDER NUM                           */
/*          &RECST      - RECORD STATUS                             */
/*          &STPCL      - CLM DTA STEPS-CLM                         */
/*          &STPCM      - CLM DTA STEPS-CMT                         */
/*          &STPRS      - CLM DTA STEPS-RSL                         */
/*          &STPEX      - CLM DTA STEPS-EXP                         */
/*                                                                  */
/********************************************************************/

 PPPPUPC:    PGM        PARM(&RETURN &APPLIC &DESC &OUTQ &HOLD &SAVE +
                          &NQSTAT &NIGHTQ &LOAD &CLM &CLMT &CLMST +
                          &CLMDT &CLMNM &CLMSP &CLMCO &ORDNO &STPCL +
                          &STPCM &STPRS &STPEX)

        DCL            &LOAD   *DEC         /*  LOAD ID                  */
        DCL            &CLM    *DEC         /*  CLAIM NUMBER             */
        DCL            &CLMT   *CHAR (02)   /*  CLAIM TYPE               */
        DCL            &CLMST  *CHAR (01)   /*  CLAIM STATUS             */
        DCL            &CLMDT  *DEC         /*  CLAIM ENTRY DAT          */
        DCL            &CLMNM  *CHAR (30)   /*  CLAIM CUSTOMER NAME      */
        DCL            &CLMSP  *DEC         /*  CLAIM SHIP TO C          */
        DCL            &CLMCO  *DEC         /*  CLAIM COMPANY N          */
        DCL            &ORDNO  *DEC         /*  CLAIM ORDER NUM          */
        DCL            &STPCL  *CHAR (01)   /*  CLM DTA STEPS-CLM        */
        DCL            &STPCM  *CHAR (01)   /*  CLM DTA STEPS-CMT        */
        DCL            &STPRS  *CHAR (01)   /*  CLM DTA STEPS-RSL        */
        DCL            &STPEX  *CHAR (01)   /*  CLM DTA STEPS-EXP        */

/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &PGM        *CHAR   (10)          /* CL  TO SUBMIT             */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PROMPT PROGRAM                                            */
/********************************************************************/

             CALL       PGM(PDG0PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM                                        */
/********************************************************************/

 SUBMIT:     SBMJOB     CMD(CALL PGM(PPY5PFRCLP) PARM(&RETURN +
                          &APPLIC &DESC &OUTQ &HOLD &SAVE &COPYN +
                          &NQSTAT &NIGHTQ &LOAD &CLM &CLMT &CLMST +
                          &CLMDT &CLMNM &CLMSP &CLMCO &ORDNO &STPCL +
                          &STPCM &STPRS &STPEX)) JOB(CUSTCLAIMP) +
                          JOBQ(&NIGHTQ)

/*           CALL       PGM(PPY5PFRCLP) PARM(&RETURN &APPLIC &DESC +
                          &OUTQ &HOLD &SAVE &COPYN &NQSTAT &NIGHTQ +
                          &LOAD &CLM &CLMT &CLMST &CLMDT &CLMNM +
                          &CLMSP &CLMCO &ORDNO &STPCL &STPCM &STPRS +
                          &STPEX)          */

             GOTO       CMDLBL(END)

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
