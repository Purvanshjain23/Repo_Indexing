//?Execute user function

// LCL.Send Email USR = CND.No
LCL.Send_Email_USR = 'N';

EXECUTE FUNCTION(RTV Data Libraries     RT) TYPE(RTVOBJ) FILE(CADPREP)          AC1165753;
PARAMETER('DTALIB');
PARAMETER(LCL.Library_List_Code);
PARAMETER(NLL.@Library_List_Code);
{
 //?USER: Process Data record

 CASE;

 // IF DB1.System Value Code is Data File Library
 IF DB1.System_Value_Code = 'DTALIB';

 // PAR.Library List Code = DB1.System Value Alpha
 PAR.Library_List_Code = DB1.System_Value_Alpha;

 // IF DB1.System Value Code is History File Library
 IF DB1.System_Value_Code = 'HISLIB';

 // PAR.@Library List Code = DB1.System Value Alpha
 PAR.@Library_List_Code = DB1.System_Value_Alpha;

 QUIT;

 ENDIF;

}


//?Format Email Subject Line.
CASE;

// IF *OTHERWISE
IF *OTHERWISE;

//?Populate company
CASE;

// IF PAR.Company Number is greater than zero
IF PAR.Company_Number > *ZERO;

// LCL.Send Email USR = CND.Yes
LCL.Send_Email_USR = 'Y';

// LCL.Company Alpha = CVTVAR(PAR.Company Number)
LCL.Company_Alpha = CVTVAR(PAR.Company_Number);

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,LCL.Company Alpha,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + LCL.Company_Alpha (*ZERO);

ENDIF;

//?Populate Order
CASE;

// IF PAR.From Order Number is Greater then zero
IF PAR.From_Order_Number > *ZERO;

// AND PAR.From Order Number EQ PAR.To Order Number
AND PAR.From_Order_Number = PAR.To_Order_Number;

// LCL.Send Email USR = CND.Yes
LCL.Send_Email_USR = 'Y';

// LCL.Order Alpha = CVTVAR(PAR.From Order Number)
LCL.Order_Alpha = CVTVAR(PAR.From_Order_Number);

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,CON.-,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + '-' (*ZERO);

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,LCL.Order Alpha,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + LCL.Order_Alpha (*ZERO);

//?Populate Order
// IF PAR.From Order Number is Greater then zero
IF PAR.From_Order_Number > *ZERO;

// AND PAR.From Order Number LT PAR.To Order Number
AND PAR.From_Order_Number < PAR.To_Order_Number;

// LCL.Send Email USR = CND.Yes
LCL.Send_Email_USR = 'Y';

// LCL.Order Alpha = CVTVAR(PAR.From Order Number)
LCL.Order_Alpha = CVTVAR(PAR.From_Order_Number);

// LCL.From Order Number = PAR.From Order Number
LCL.From_Order_Number = PAR.From_Order_Number;

// LCL.To Order Number = PAR.To Order Number
LCL.To_Order_Number = PAR.To_Order_Number;

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,CON.-,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + '-' (*ZERO);

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,LCL.Order Alpha,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + LCL.Order_Alpha (*ZERO);

//?Populate Range of order numbers
// DOW LCL.From Order Number LT PAR.To Order Number
DOW LCL.From_Order_Number < PAR.To_Order_Number;

// LCL.From Order Number = LCL.From Order Number + CON.1
LCL.From_Order_Number = LCL.From_Order_Number + 1;

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,CON.,,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + ',' (*ZERO);

// LCL.Order Alpha = CVTVAR(LCL.From Order Number)
LCL.Order_Alpha = CVTVAR(LCL.From_Order_Number);

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,LCL.Order Alpha,CON.*ZERO)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + LCL.Order_Alpha (*ZERO);

ENDDO;

ENDIF;

//?If Not PROD Then Append 'TEST' to Email Subject Line.
CASE;

// IF LCL.Library List Code is Not PRKFLIB
IF LCL.Library_List_Code <> 'PRKFLIB';

// LCL.Email Subject 3 USR = CONCAT(LCL.Email Subject 3 USR,CON.- TEST,CON.1)
LCL.Email_Subject_3_USR = LCL.Email_Subject_3_USR + '- TEST' (1);

ENDIF;

ENDIF;

//?Format Email Message.
CASE;

// IF *OTHERWISE
IF *OTHERWISE;

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.itions of Sale which can,CON.*ZERO)

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.be found at https://legal,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'be found at https://legal' (1);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.andcompliance.seaboardfoo,CON.*ZERO)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'andcompliance.seaboardfoo' (0);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.ds.com/#terms-and-conditi,CON.*ZERO)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'ds.com/#terms-and-conditi' (0);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.ons-for-sales,CON.*ZERO)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'ons-for-sales' (0);

//?Force New Line.
// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.&N,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + '&N' (1);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.None of the terms and con,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'None of the terms and con' (1);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.ditions contained herein,CON.*ZERO)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'ditions contained herein' (0);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.may be added to, modified,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'may be added to, modified' (1);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.or superseded, or otherwi,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'or superseded, or otherwi' (1);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.se  altered,,CON.*ZERO)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'se  altered,' (0);

//?Force New Line.
// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.&N,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + '&N' (1);

// LCL.Email Message 2 USR = CONCAT(LCL.Email Message 2 USR,CON.unless in writing and,CON.1)
LCL.Email_Message_2_USR = LCL.Email_Message_2_USR + 'unless in writing and' (1);

ENDIF;

// PAR.Email Subject 3 USR = LCL.Email Subject 3 USR
