PGM PARM(&CRCDE)

DCL &CRCDE *CHAR 3  /* Accrual Code */
DCL &LIBRARY *CHAR 10  /* Override Library  */

/* Override to DFDATA for first run */
CHGVAR &LIBRARY 'DFDATA'
OVRDBF     FILE(ACCRUECU) TOFILE(&LIBRARY/ACCRUECU)
OVRDBF     FILE(ACCRUEPR) TOFILE(&LIBRARY/ACCRUEPR)

/* Call EUP Bld Cust Accrual   */
CALL PMK5UPR PARM(&CRCDE &LIBRARY)

/* Delete overrides */
DLTOVR  ACCRUECU
DLTOVR  ACCRUEPR

/* Override to JRDATA for second run */
CHGVAR &LIBRARY 'JRDATA'
OVRDBF     FILE(ACCRUECU) TOFILE(&LIBRARY/ACCRUECU)
OVRDBF     FILE(ACCRUEPR) TOFILE(&LIBRARY/ACCRUEPR)

/* Call EUP Bld Cust Accrual   */
CALL PMK5UPR PARM(&CRCDE &LIBRARY)

/* Delete overrides */
DLTOVR  ACCRUECU
DLTOVR  ACCRUEPR
ENDPGM
