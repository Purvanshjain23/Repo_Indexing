/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  RECEIVE PARAMETERS AND PASS TO THE PROGRAM          */
/*              THAT WILL CREATE THE JAPANESE CHILLED ORDERS        */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - CALLS PROGRAM TO CREATE THE JAPANESE CHILLED ORDERS */
/*  DATE      - 06/17/2008                                          */
/*  PROGRAMMER- LARA BUSER                                          */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*  O    &RETURN     - RETURN CODE                                  */
/*  I    &COMPYN     - COMPANY NUMBER - PASSED AS 15.5              */
/*  I    &CUSTN      - CUSTOMER NUMBER - PASSED AS 15.5             */
/*  I    &SCHSHDT    - SCHEDULED SHIP DATE - PASSED AS 15.5         */
/*  I    &REQSHDT    - REQUESTED SHIP DATE - PASSED AS 15.5         */
/*  I    &REQDLVDT   - REQUESTED DELIVERY DATE - PASSED AS 15.5     */
/*  B    &ORDERN     - ORDER NUMBER - (RETURNED TO PROMPT SCREEN)   */
/*  B    &LOADIDN    - LOAD ID - (RETURNED TO PROMPT SCREEN)        */
/*  I    &PRICECDN   - PRICE CODE TO USE FOR PRODUCTS - 15.5        */
/*  I    &SHIPINST   - SHIPPING INSTRUCTIONS - CONSTANT             */
/*  I    &CUSTPPO    - CUST PO #                                    */
/*  I    &AGEDAYS    - ITEM AGE DAYS                                */
/*  I    &SHIPMARK   - SHIPPING MARKS - CONSTANT                    */
/********************************************************************/
/* LJB 08/08/2011  ADD EXPORT TYPE. M=MENU, FROM PROMPT             */
/*    SCREEN, U=UPLOADED FROM SPREADSHEET TO WORK FILE. THIS CAN BE */
/*    CALLED FROM THE PROMPT SCREEN OR SUBMITTED FROM SSIS WHICH    */
/*    WILL THEN RUN AGAINST A WORK FILE.                            */
/* SLM 04/26/10 FP1309                                              */
/*    removed the age code, ADD THE ITEM AGE DAYS                   */
/* RMC 05/08/09 ADDED CUST PO PARM, WHICH WAS CALLED SHIPMARK,      */
/*    AND MOVED &SHIPMARK TO THE END                                */
/********************************************************************/
 PMV4UPC:    PGM        PARM(&RETURN &COMPYN &CUSTN &SCHSHDT +
                          &REQSHDT &REQDLVDT &ORDERN &LOADIDN +
                          &WHSEN &PRICECDN &EXPORTTYPE &SHIPINST +
                          &CUSTPO &AGEDAYS &DAYSFROM &DAYSTO &SHIPMARK)

             DCL        VAR(&RETURN)    TYPE(*CHAR) LEN(7)
             DCL        VAR(&COMPYN)    TYPE(*DEC)
             DCL        VAR(&CUSTN)     TYPE(*DEC)
             DCL        VAR(&SCHSHDT)   TYPE(*DEC)
             DCL        VAR(&REQSHDT)   TYPE(*DEC)
             DCL        VAR(&REQDLVDT)  TYPE(*DEC)
             DCL        VAR(&ORDERN)    TYPE(*DEC)
             DCL        VAR(&LOADIDN)   TYPE(*DEC)
             DCL        VAR(&WHSEN)     TYPE(*DEC)
             DCL        VAR(&PRICECDN)  TYPE(*DEC)
             DCL        VAR(&CUSTPO)    TYPE(*CHAR) LEN(15)
             DCL        VAR(&SHIPMARK)  TYPE(*CHAR) LEN(15)
             DCL        VAR(&SHIPINST)  TYPE(*CHAR) LEN(30)
             DCL        VAR(&EXPORTTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&AGEDAYS)   TYPE(*DEC) LEN(4 0)
             DCL        VAR(&DAYSFROM)  TYPE(*DEC)
             DCL        VAR(&DAYSTO)    TYPE(*DEC)

             MONMSG     MSGID(CPF0000)

/*------------------- CALL TRANSFER TEMPLATE ORDER ------------------*/
             CALL       PGM(PMV5XFR) PARM(' ' &COMPYN &CUSTN +
                          &SCHSHDT &REQSHDT &REQDLVDT &ORDERN +
                          &LOADIDN &WHSEN &PRICECDN &EXPORTTYPE +
                          &SHIPINST &CUSTPO &AGEDAYS &DAYSFROM +
                          &DAYSTO &SHIPMARK)

 END:        ENDPGM
