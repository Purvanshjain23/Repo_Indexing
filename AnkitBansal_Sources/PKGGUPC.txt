/*                                                                 */
/*  PROGRAM NUMBER:  PKGGUPC                                       */
/*  PROGRAM NAME:    TRUCKER PAYMENT EDIT                          */
/*  PROGRAMMER:      SUSAN L. MASON                                */
/*  CREATE DATE:     12/19/96                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PMT TRUCKER PAYMENT EDIT                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  CREATED          12/19/96  PROJECT HPE044                      */
/*                   PRINT A TRUCKER PAYMENT EDIT BEFORE POST      */
/*                                                                 */
/*  5/25/2017 JBB E9890 - Don't Allow Trucker Payment Processing   */
/*                        Check Company Values for the Company     */
/*                        and DALWTRKPMT to determine if the       */
/*                        company processes trucker payments.      */
/*  5/29/2017 SLM E7519   ADD VALIDATION OF M3 SUPPLIER TO REPORT  */
/*                        REMOVE THE HOG VOUCHER DATA FROM THIS    */
/*                        PROCESS WHICH WAS PERFORMED IN THE PRINT */
/*                        REMOVE THE PROGRAM PKGEPFR$ FROM THIS    */
/*                        PROCESS WILL NOT USE PROGRAM ANYMORE     */
/*                        THIS PAYMENT EDIT IS A PAYMENT ONLY      */
/*                        NOT A PAYMENT EDIT AND POST BATCHES      */
/*                        PKGEPFR WAS POSTING BATCHES              */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &XCOMP +
                       &APP)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* ALPHA VARIABLES                                                */
             DCL        VAR(&FLAG) TYPE(*CHAR) LEN(1)
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(7)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&LPP) VALUE(&APP)


/* ***********************************************************/
/*  5/25/2017 JBB E9890 Get Don't Allow Trucker Payment flag */
             CALL       PGM(POMTXFR) PARM(&RETURN &COMP +
                           'DALWTRKPMT' &FLAG)
            /* If Trucker Payments not allowed, exit program */
             IF         COND(&FLAG *EQ 'Y') THEN(GOTO CMDLBL(ENDIT))
/* ***********************************************************/

/* CREATE PRINTER FILE OVERRIDE CODE    TRUCKER PAYMENT EDIT    */
/*********   SBDPRTOVR  CMDVAR(&CMD) FILE(PKGEPFR$) OUTQ(&PDEV) + */
/*********                COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) */

             SBDPRTOVR  CMDVAR(&CMD) FILE(PWCDPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* ADD VALIDATION TO NEXT PROGRAM                             */
/* PROCESS TRUCKER PAYMENT EDIT                               */

             CALL       PGM(PKGHXFR) PARM(' ' &COMP &LPP)

             DLTOVR *ALL
 ENDIT:      ENDPGM
