  WRK.Head_Count = WRK.Head_Count + 1;

  //?Min Adj Actual Yield
  CASE;

  // IF WRK.Tattoo Quad Count USR is = 1
  IF WRK.Tattoo_Quad_Count_USR = 1;

  // AND WRK.Head Count is = 1
  AND WRK.Head_Count = 1;

  AND PAR.Tattoo_Count_USR = WRK.Head_Count;
