SELECT
  ALL       T02.MNAME, T02.MOPTN, T01.GRPID#, T01."USERID", T04.UNAME,
            T03.MDESC
  FROM      WM400/WMGROUP T01 INNER JOIN
            WM400/WMAUTH T02
  ON        T01.GRPID# = T02."USERID" INNER JOIN
            WM400/WMOPTN T03
  ON        T02.MCLASS = T03.MCLASS
    AND     T02.MAPLCN = T03.MAPLCN
    AND     T02.MNAME = T03.MNAME
    AND     T02.MOPTN = T03.MOPTN INNER JOIN
            WM400/WMUSER T04
  ON        T01."USERID" = T04."USERID"
  WHERE     T01.COID = 'GUY'
    AND     T02.COID = 'GUY'
    AND     T03.COID = 'GUY'
    AND     T02.MNAME BETWEEN 'PPS000' AND 'PPS999'
    AND     GRPID# NOT IN ('QPGMR', 'GPPRKTST')
    AND     T04.COID = 'GUY'
  ORDER BY  T02.MNAME ASC, T02.MOPTN ASC, T01.GRPID# ASC, T01."USERID" ASC
