PARAMETER(PAR.Oms_SalesPerson_Code);
PARAMETER(PAR.Oms_Allow_Multi_Company);
PARAMETER(PAR.Oms_Default_Printer);
PARAMETER(PAR.Oms_Name);
PARAMETER(PAR.Oms_System_Value_Alpha);
PARAMETER(PAR.Oms_Access_Denied_Y_N);
{
 //?Execute internal function

 // Call program Val Application/User  XF.
 CALL PROGRAM(Val Application/User  XF) ('PDGXXFR');
 PARAMETER(PAR.Oms_Warehouse_Code);
 PARAMETER(PAR.CC_Company_Number);
 PARAMETER(PAR.Oms_Application_Code);
 PARAMETER(PAR.Oms_SalesPerson_Code);
 PARAMETER(PAR.Oms_Allow_Multi_Company);
 PARAMETER(PAR.Oms_Default_Printer);
 PARAMETER(PAR.Oms_Name);
 PARAMETER(PAR.Oms_System_Value_Alpha);
 PARAMETER(PAR.Oms_Access_Denied_Y_N);

 // Execute function Mov ApplN to Company  US.

 PARAMETER(PAR.Oms_System_Value_Alpha);
 {
  C***   MOVE USER SOURCE                      ***
  C                     MOVEL#IEVTX    ##CMP
 }

 ENDIF;

}


// LCL.First Time Flag Usr = CND.Yes
LCL.First_Time_Flag_Usr = 'Y';

// PGM.*Scan limit = CON.5000
PGM.*Scan_limit = 5000;

//?USER: Initialize subfile control

CASE;

// IF LCL.First Time Flag Usr is Yes
IF LCL.First_Time_Flag_Usr = 'Y';

// CTL.CSD Company Number = PAR.CC Company Number
CTL.CSD_Company_Number = PAR.CC_Company_Number;

// CTL.CSD Type = CND.Weekly
CTL.CSD_Type = 'W';

// LCL.First Time Flag Usr = CND.No
LCL.First_Time_Flag_Usr = 'N';

// CTL.CSD Producer Code = CON.*ALL
CTL.CSD_Producer_Code = *ALL;

// CTL.CSD Location Code = CON.*ALL
CTL.CSD_Location_Code = *ALL;

ENDIF;

EXECUTE FUNCTION(Val Company Number    IF) TYPE(EXCINTFUN)                      AB1129077;
PARAMETER(CTL.CSD_Company_Number);
PARAMETER(CTL.Oms_Name);
PARAMETER(LCL.Invalid_Y_N_USR);
{
 //?Execute internal function

 // Call program Val Company Number    XF.
 CALL PROGRAM(Val Company Number    XF) ('PDGZXFR');
 PARAMETER(PAR.CC_Company_Number);
 PARAMETER(PAR.Oms_Name);
 PARAMETER(WRK.Invalid_Y_N_USR);

 CASE;

 // IF WRK.Invalid Y/N USR is Yes
 IF WRK.Invalid_Y_N_USR = 'Y';

 // Send error message - 'Invalid Company Number'
 ERROR(PRK0299);
 MSGPARM(PAR.CC_Company_Number);

 ENDIF;

 // PAR.Invalid Y/N USR = WRK.Invalid Y/N USR
 PAR.Invalid_Y_N_USR = WRK.Invalid_Y_N_USR;

}


// CTL.Condition Name USR = Condition name of CTL.CSD Type
CTL.Condition_Name_USR = RTVCND(CTL.CSD_Type);

CASE;

// IF CTL.CSD Type is Weekly
IF CTL.CSD_Type = 'W';

// CTL.CSD Month = CON.*ZERO
CTL.CSD_Month = *ZERO;

// IF CTL.CSD Type is Monthly
IF CTL.CSD_Type = 'M';

// CTL.CSD Week = CON.*ZERO
CTL.CSD_Week = *ZERO;

ENDIF;

// LCL.CSD Year = CTL.CSD Year
LCL.CSD_Year = CTL.CSD_Year;

//?USER: Initialize subfile record from DBF record

// RCD.CSD Year = RCD.CSD From Date *YEAR
RCD.CSD_Year = DATEDTLS(RCD.CSD_From_Date 'YR' 1111111 'NONE' 'N' 1);

//?SEL:  Company Number
CASE;

// IF DB1.CSD Company Number NE CTL.CSD Company Number
IF DB1.CSD_Company_Number <> CTL.CSD_Company_Number;

// PGM.*Record selected = CND.*NO
PGM.*Record_selected = 'N';

ENDIF;

//?SEL:  Deviation Type
CASE;

// IF DB1.CSD Type NE CTL.CSD Type
IF DB1.CSD_Type <> CTL.CSD_Type;

// PGM.*Record selected = CND.*NO
PGM.*Record_selected = 'N';

ENDIF;

//?SEL:  Year
CASE;

// IF CTL.CSD Year is Entered
IF CTL.CSD_Year <> *ZERO;

// AND CTL.CSD Year NE RCD.CSD Year
AND CTL.CSD_Year <> RCD.CSD_Year;

// PGM.*Record selected = CND.*NO
PGM.*Record_selected = 'N';

ENDIF;

//?CALC: Subfile control function fields

// CTL.Condition Name USR = Condition name of CTL.CSD Type
CTL.Condition_Name_USR = RTVCND(CTL.CSD_Type);

//?USER: Final processing (Pre-confirm)

CASE;

// IF CTL.CSD Year NE LCL.CSD Year
IF CTL.CSD_Year <> LCL.CSD_Year;

// PGM.*Reload subfile = CND.*YES
PGM.*Reload_subfile = 'Y';

ENDIF;

//?USER: Process command keys

CASE;

// IF CTL.*CMD key is Print List
IF CTL.*CMD_key = '21';

// Call program Prt Std Deviation     PV.
CALL PROGRAM(Prt Std Deviation     PV) ('PKU0PVR');

ENDIF;
