/*  SYNOPSIS -  EXECUTE JOB TO DROP PRODUCTION SCHEDULES TO      */
/*              THE PLANTS - RUN FOR EACH PLANT FROM INVBLDALC   */
/*                                                                  */
/*                                                                  */
/*  SYSTEM    - PRODUCTION SCHEDULING AND PLANNING                  */
/*  PROGRAM   - PUT6UPR                                             */
/*  DATE      - 09/20/2021                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     I    &COMP    - COMPANY NUMBER                            */
/********************************************************************/
/* MODIFICATIONS LOG:                                               */
/*------------------------------------------------------------------*/
/* W118693 DN  01/27/23 - IF CO 360 THEN SUBMIT PUL8UPC TO JOBQ     */
/*                        'PRKJS1' TO PREVENT ANY HOLD UP ON        */
/*                        INVBLDALC PROCESS.                        */
/*                                                                  */
/********************************************************************/

             PGM        PARM(&COMP)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)

             CALL       PGM(*LIBL/PBS1XFR) PARM('' &COMP)
             CALL       PGM(*LIBL/PULRXFR) PARM('' &COMP 'Y')

/* SEND TO DW ONCE                              */
             IF         COND(&COMP *EQ 360) THEN(DO)
  /* W118693 - Changed to SBMJOB Program to Jobq PRKJS1 to Prevent Hold Up. */
  /* W118693 CALL       PGM(*LIBL/PUL8UPC) PARM('Y')                 */
             SBMJOB     CMD(CALL PGM(PUL8UPC) PARM('Y')) +
                          JOB(PUL8UPC) JOBQ(PRKJS1)
             ENDDO

ENDPGM
