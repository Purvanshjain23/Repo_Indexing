/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING ACCRUAL CODES LISTING        */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  3/07/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &AUTOACRL      -  PASSED AUTO ACCRUAL                    */
/*         &ADDON         -  PASSED ADD-ON/ALLOW                    */
/*         &STATUS        -  PASSED STATUS                          */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
/* RMC  10/21/04 REMOVE COMPANY FROM PARM AND CALLED PGM-PROJ EXPCOM      */
/********************************************************************/
 PDBXCLP:    PGM        PARM(&APPLIC  &AUTOACRL &ADDON +
                          &STATUS &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &AUTOACRL    *CHAR  (1)
        DCL    &ADDON       *CHAR  (1)
        DCL    &STATUS      *CHAR  (1)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDBXPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDBXPFR) PARM(' '  &AUTOACRL +
                          &ADDON &STATUS)

 ENDPGM:     ENDPGM
