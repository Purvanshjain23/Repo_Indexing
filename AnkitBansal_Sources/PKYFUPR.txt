/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKYFUPR                                       */
/*  PROGRAM NAME:    PURGE CARCASS DETAIL TO HISTORY               */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     03/18/02                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL WILL COPY RECORDS TO PKB1CPPHST.      */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC 8/15/02  USE SNDPGRMSG A LIEU SNDDST                       */
/*******************************************************************/
             PGM        PARM(&PDTE &HDTE)

/* INPUT PARAMETERS */
             DCL        VAR(&PDTE) TYPE(*DEC)  LEN(7 0)   /* PURGE DATE   */
             DCL        VAR(&HDTE) TYPE(*DEC)  LEN(7 0)   /* HISTORY DATE */

/* FUNCTION SPECIFIC PARMS */
             DCL        VAR(&PDTEX) TYPE(*CHAR) LEN(7)    /* PURGE DATE   */
             DCL        VAR(&HDTEX) TYPE(*CHAR) LEN(7)    /* HISTORY DATE */
             DCL        VAR(&SEL)  TYPE(*CHAR) LEN(512)   /* SELECT       */

/*-----------------------------------------------------------------*/
/* INITIALIZE                                                      */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&PDTEX) VALUE(&PDTE)
             CHGVAR     VAR(&HDTEX) VALUE(&HDTE)

/*-----------------------------------------------------------------*/
/* MAINLINE                                                        */
/*-----------------------------------------------------------------*/

/* Insert into HISTORY */
             CHGVAR     VAR(&SEL) VALUE('Select * from ' *CAT +
                          'pKb1cpp where b1b0dt<' *CAT &HDTEX)

             INSERT     INTO(PKB1CPPHST) SQL(&SEL)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(INSERTHST))

/* Delete from PRODUCTION */
             CHGVAR     VAR(&SEL) VALUE('from ' *CAT 'pKb1cpp  where +
                          b1b0dt<' *CAT &HDTEX)

             DELETE     SQL(&SEL)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(DELETEPRD))

/* Delete from HISTORY */
             CHGVAR     VAR(&SEL) VALUE('from ' *CAT 'pKb1cppHST +
                          where b1b0dt<' *CAT &PDTEX)

             DELETE     SQL(&SEL)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(DELETEHST))

/* Successful Completion */
             GOTO       CMDLBL(PGMMSG)
/*-----------------------------------------------------------------*/
/*                     ERROR MESSAGES                              */
/*-----------------------------------------------------------------*/

INSERTHST:
/*           SNDDST     TYPE(*MSG) TOUSRID((PRKPURGE BIGBYTE)) +
                          DSTD('Carcass Detail Purge') MSG('INSERT +
                          ERROR into file PKB1CPPHST')       */
             SNDPGRMSG  TOPGR(PRKPURGE) MSG('Insert error into file +
                          PKB1CPPHST')
             GOTO       CMDLBL(ERROR)

DELETEPRD:
/*           SNDDST     TYPE(*MSG) TOUSRID((PRKPURGE BIGBYTE)) +
                          DSTD('Carcass Detail Purge') MSG('DELETE +
                          ERROR from file PKB1CPP')              */
             SNDPGRMSG  TOPGR(PRKPURGE) MSG('DELETE ERROR from file +
                          PKB1CPP')
             GOTO       CMDLBL(ERROR)

DELETEHST:
/*           SNDDST     TYPE(*MSG) TOUSRID((PRKPURGE BIGBYTE)) +
                          DSTD('Carcass Detail Purge') MSG('DELETE +
                          ERROR from file PKB1CPPHST')            */
             SNDPGRMSG  TOPGR(PRKPURGE) MSG('DELETE ERROR from file +
                          PKB1CPPHST')
             GOTO       CMDLBL(ERROR)

ERROR:
 /*          SNDDST     TYPE(*MSG) TOUSRID((PRKPURGE BIGBYTE)) +
                          DSTD('Carcass Detail Purge') MSG('Carcass +
                   Detail Purge to History Ended Abnormally')      */
             SNDPGRMSG  TOPGR(PRKPURGE) MSG('Carcass Detail Purge to +
                          History Ended Abnormally')

             CHGJOB     LOG(4 00 *SECLVL)

             GOTO       CMDLBL(ENDPGM)
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
PGMMSG:
/*          SNDDST     TYPE(*MSG) TOUSRID((PRKPURGE BIGBYTE)) +
                          DSTD('Carcass Detail Purge') MSG('Carcass +
                     Detail Purge to History Ended Normally.')    */
             SNDPGRMSG  TOPGR(PRKPURGE) MSG('Carcass Detail Purge to +
                          History Ended Normally')

ENDPGM:
             ENDPGM
