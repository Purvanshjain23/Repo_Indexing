/********************************************************************/
/*                                                                  */
/*  PROGRAM   - PMEFUPC - PRT SHIPPED WAREHOUSE TRANSFER ORDERS     */
/*  SYNOPSIS  - SUBMITTED FROM PMEEUPC, PASSES JOB CONTROL          */
/*              INFORMATION                                         */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  DATE      - 03/27/2007                                          */
/*            - LARA BUSER                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION                               */
/*     I    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     I    &FRMDATN    - FROM SHIP DATE (PASSED)                   */
/*     I    &THRDATN    - THRU SHIP DATE (PASSED)                   */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     I    &ORDTYPE    - ORDER TYPE CODE                           */
/*     B    &OUTQ       - OUTPUT QUEUE                              */
/*     B    &HOLD       - HOLD STATUS                               */
/*     B    &SAVE       - SAVE STATUS                               */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     B    &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q     */
/*     B    &NIGHTQ     - NIGHT JOB QUEUE                           */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/*  DATE      - 07/25/2011 PASS IN THE WAREHOUSE FROM AND TO        */
/*------------------------------------------------------------------*/
/*                                                                  */
/********************************************************************/
 PMEFUPC:    PGM        PARM(&RETURN &APPLIC &DESC &OUTQ &HOLD &SAVE +
                          &NQSTAT &NIGHTQ &COMPNYN &FRMDATN +
                          &THRDATN &COPYN &ORDTYPE &WHSFRM &WHSTO)
/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)   /* RETURN CODE              */
        DCL    &APPLIC     *CHAR   (6)   /* APPLICATION              */
        DCL    &COMPNYN    *DEC          /* COMPANY NUMBER (PASSED)  */
        DCL    &FRMDATN    *DEC          /* FROM ACTUAL SHIP DATE    */
        DCL    &THRDATN    *DEC          /* THROUGH ACTUAL SHIP DATE */
        DCL    &DESC       *CHAR   (25)  /* PRT PGM DESCRIPTION      */
        DCL    &OUTQ       *CHAR   (10)  /* OUTPUT QUEUE             */
        DCL    &HOLD       *CHAR   (4)   /* HOLD STATUS              */
        DCL    &SAVE       *CHAR   (4)   /* SAVE STATUS              */
        DCL    &NQSTAT     *CHAR   (01)  /* IF MUST RUN IN NIGHT Q   */
        DCL    &NIGHTQ     *CHAR   (10)  /* NIGHT JOB QUEUE          */
        DCL    &COPYN      *DEC          /* NO. OF COPIES  (PASSED)  */
        DCL    &ORDTYPE    *CHAR   (2)   /* ORDER TYPE CODE          */
        DCL    &WHSFRM     *CHAR   (3)   /* WAREHOUSE FROM           */
        DCL    &WHSTO      *CHAR   (3)   /* WAREHOUSE TO             */

/* INTERNAL FIELDS                                                   */
        DCL    &COMP       *DEC    (3 0) /* COMPANY NUMBER           */
        DCL    &FROMDT     *DEC    (7 0) /* FROM ACTUAL SHIP DATE    */
        DCL    &THRUDT     *DEC    (7 0) /* THRU ACTUAL SHIP DATE    */
        DCL    &COPY       *DEC    (2 0) /* NUMBER OF COPIES         */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/*------------------------------------------------------------------*/
             CHGVAR     VAR(&COMP) VALUE(&COMPNYN)
             CHGVAR     VAR(&FROMDT) VALUE(&FRMDATN)
             CHGVAR     VAR(&THRUDT) VALUE(&THRDATN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/*------------------------------------------------------------------*/
             OVRPRTF    FILE(PMEGPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

/********************************************************************/
/* CALL THE INVOICE EXCEPTION REPORT PROGRAM                        */
/********************************************************************/
 CALL:       CALL       PGM(PMEGPFR) PARM(&RETURN &COMP &ORDTYPE +
                          &FROMDT &THRUDT &WHSFRM &WHSTO)

 END:        ENDPGM
