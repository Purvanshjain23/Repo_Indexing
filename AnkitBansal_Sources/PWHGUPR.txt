/* PROGRAM:      PWHGUPR                                              */
/* PROGRAM NAME: Rtv Trucker Name Ovr  CL                             */
/* PROGRAMMER:   Julius High                                          */
/* CREATE DATE:  09/17/2018                                           */
/*                                                                    */
/* FUNCTION:     This CLP will retrieve the TRK Name from the Trucker */
/*               file in the Override Library.  Override Library is   */
/*               listed in Company Value TRKNAMEOVR.                  */
/**********************************************************************/
/* Modification History                                               */
/*                                                                    */
/**********************************************************************/
             PGM        PARM(&TRKCODE &TRKOVRLIB &TRKNAME)
/*--------------------------------------------------------------------*/
/*                      PARAMETER                                     */
/*--------------------------------------------------------------------*/
/* Function Specific                                                  */
             DCL        VAR(&TRKCODE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&TRKOVRLIB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TRKNAME) TYPE(*CHAR) LEN(40)
/*--------------------------------------------------------------------*/
/* Add Producer Site To Library                                      */
             ADDLIBLE   LIB(&TRKOVRLIB) POSITION(*LAST)
             MONMSG     MSGID(CPF2103) /* LIB ALREADY EXIST */
/*--------------------------------------------------------------------*/
/* Override Access Paths to use Trucker OverrideLibrary               */
/*                      Trucker - Retrieval Index                     */
             OVRDBF     FILE(PKBAREL0) TOFILE(&TRKOVRLIB/PKBAREL0)
/*--------------------------------------------------------------------*/
/* Retrieve Trucker Name from Trucker Override Library                */
             CALL       PGM(PWHHXFR) PARM('' &TRKCODE &TRKNAME)
/*--------------------------------------------------------------------*/
/* Clean Up                                                           */
             DLTOVR     FILE(*ALL)
             RMVLIBLE   LIB(&TRKOVRLIB)
/*--------------------------------------------------------------------*/
/*                      PROGRAM END                                   */
             ENDPGM
