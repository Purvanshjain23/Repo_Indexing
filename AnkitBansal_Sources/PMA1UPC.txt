/*                                                                   */
/*  SYNOPSIS -  THIS PROGRAM IS CALLED USING RMTCMD FROM A           */
/*              PC BATCH FILE. IT SETS THE LIBL THEN CALLS           */
/*              THE PROGRAMS THAT BUILD  PNESCPP FILE                */
/*                                                                   */
/*  SYSTEM    - OMS                                                  */
/*  PROGRAM   - PMA1UPC - SET LIBL AND CALL FILE BUILD PROGRAM       */
/*  DATE      - 11/04/2008                                           */
/*********************************************************************/
/*  MODIFICATIONS:                                                   */
/*                                                                   */
/*********************************************************************/
 PMA1UPC:    PGM

             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(7)

             ADDLIBLE   LIB(PRKPLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF0000)
/***>>>>     ADDLIBLE   LIB(PRKDEVQRY) POSITION(*FIRST) +
             MONMSG     MSGID(CPF0000)                  +
             ADDLIBLE   LIB(PRKTESTF ) POSITION(*FIRST) +
             MONMSG     MSGID(CPF0000)                  <<<<****/
             ADDLIBLE   LIB(QTEMP) POSITION(*FIRST)
             MONMSG     MSGID(CPF0000)

             CLRPFM     FILE(PNESCPP)
             MONMSG     MSGID(CPF0000)
             CHGJOB     LOG(4 00 *SECLVL)

/*-------------------------------------------------------------------*/
/* BUILD THE SSI DOWNLOAD FILE FOR TRANSPORTATION                    */
/*-------------------------------------------------------------------*/
             CALL       PGM(PMAEXFR) PARM(&RETURN)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(ERROR))

             GOTO       CMDLBL(END)

 ERROR:      CHGVAR     VAR(&RETURN) VALUE('Y2U0032')

 END:        ENDPGM
