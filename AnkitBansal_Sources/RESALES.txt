SELECT
  ALL       2000 AS "YEAR", 4 AS QTR, T02.A1THHD, T01.DZBDCD, (T01.DZLJNB),
            (T02.A1F7NB)
  FROM      PRKFLIB/PKDZREP T01 INNER JOIN
            PRKFLIB/PKA1CPP T02
  ON        DZHMNB = A1HMNB
    AND     DZBNNB = A1BNNB
    AND     DZBONB = A1BONB
    AND     DZCVNB = A1CVNB
    AND     DZB0DT = A1B0DT
  WHERE     DZB0DT >= 1001001
    AND     DZB0DT <= 1001231
    AND     A1THHD IN (314, 315)
    AND     A1LLST = 'R'
  ORDER BY  001 ASC, 002 ASC, 003 ASC, 004 ASC
