/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKXTPFR                                       */
/*  PROGRAM NAME:    Print BUY ORDER                               */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     11/15/96                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from YGO PRKRPT                               */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&PDEV &FDATE &TDATE &PROD)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4) VALUE(*YES)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4) VALUE(*YES)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2) VALUE(1)
             DCL        VAR(&FDATE) TYPE(*DEC) LEN(15 5)
             DCL        VAR(&TDATE) TYPE(*DEC) LEN(15 5)
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6)
             DCL        VAR(&FDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDTE) TYPE(*DEC) LEN(7 0)

/* Function Specific                                               */

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&FDTE) VALUE(&FDATE)
             CHGVAR     VAR(&TDTE) VALUE(&TDATE)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PKXTPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PKXTPFR) PARM(' ' &FDTE &TDTE &PROD)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
