//?USER: Initialize program

// Call program Rtv Co for User       XF.
CALL PROGRAM(Rtv Co for User       XF) ('PLF8XFR');

PARAMETER('C/S');

//?USER: Process selected line

MOVE *ALL (To: PAR From: RCD);

//?CALC: Screen function fields

CASE;

// IF CTL.CC Company Number is *ZERO
IF CTL.CC_Company_Number = *ZERO;

// CTL.CC Company Number = PAR.CC Company Number
CTL.CC_Company_Number = PAR.CC_Company_Number;

ENDIF;

//?USER: Process command keys

//?Command Keys
CASE;

// IF CTL.*CMD key is CF04
IF CTL.*CMD_key = '04';

CASE;

// IF PGM.*Cursor field <IS> CTL.CC Company Number
IF PGM.*Cursor_field = CTL.CC_Company_Number;

// Call program PMT Company Number OMS XF.
CALL PROGRAM(PMT Company Number OMS XF) ('PPF2SRR');

ENDIF;

ENDIF;
