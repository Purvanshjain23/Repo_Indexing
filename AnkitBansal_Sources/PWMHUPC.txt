/***************************************************************** */
/*                                                                 */
/*  PROGRAM:         PWMHUPC                                       */
/*  PROGRAM TITLE:   REPAYMENT LISTING                             */
/*  PROGRAMMER:      BRAD BADEN                                    */
/*  CREATE DATE:      5/06/2021                                    */
/*                                                                 */
/*  DESCRIPTION:     This CL program overrides the printer file    */
/*                   for the Repayment Listing report.             */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&CO# &DATEFROM &DATETO &RPTTYPE +
                          &REPAYID &REPAYSTS &INCCOMM &PDEV1 +
                          &HOLD &SAVE &COPIES)

/* Application Variables */
             DCL        VAR(&CO#) TYPE(*DEC)       LEN(3 0)
             DCL        VAR(&DATEFROM) TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&DATETO)   TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&RPTTYPE)  TYPE(*CHAR) LEN(1)
             DCL        VAR(&REPAYID)  TYPE(*DEC)  LEN(5 0)
             DCL        VAR(&REPAYSTS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&INCCOMM)  TYPE(*CHAR) LEN(1)

/* Printer Override Parms */
             DCL        VAR(&PDEV1)    TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD)     TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE)     TYPE(*CHAR) LEN(4)
             DCL        VAR(&COPIES)   TYPE(*DEC)  LEN(2 0)
             DCL        VAR(&CMD)      TYPE(*CHAR) LEN(512)
             DCL        VAR(&RETURN)   TYPE(*CHAR) LEN(7)

/* Override printer file   */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PWL6PFR$) OUTQ(&PDEV1) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* Call report program     */
             CALL       PGM(PWL6PFR) PARM(&RETURN &CO# &REPAYID +
                          &REPAYSTS &DATEFROM &DATETO &RPTTYPE +
                          &INCCOMM)

ENDIT:       ENDPGM
