/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - THIS PROGRAM IS SUBMITTED FROM PNXIPVR WHICH IS      */
/*             CALLED FROM A MENU.                                  */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT EXPORT LOAD VERIFICATION REPORT               */
/*  DATE      - 05/18/2011                                          */
/*                                                                  */
/*  PARAMETERS RECEIVED:                                            */
/*     &COMPNYN - COMPANY NUMBER - BLANK FOR ALL - 15.5             */
/*     &FRMDATN - FROM SHIP DATE - 15.5                             */
/*     &THRDATN - THROUGH SHIP DATE - 15.5                          */
/*     &CARRIER - CARRIER CODE (IN SALESPERSON CODE FIELD)          */
/*     &RATEAMT - FREIGHT CHARGE RATE AMOUNT - REGULAR LOADS        */
/*     &RATEAMT2- FREIGHT CHARGE RATE AMOUNT - BULK LOADS           */
/*     &EMAIL   - EMAIL ADDRESS                                     */
/*     &OUTQ    - OUTPUT QUEUE                                      */
/*     &HOLD    - HOLD PRINT                                        */
/*     &SAVE    - SAVE PRINT                                        */
/*     &RPTTYPE - REPORT TYPE                                       */
/********************************************************************/
/*  XXX 99/99/2099                                                  */
/*                                                                  */
/********************************************************************/
 PNXNUPC:    PGM        PARM(&COMPNYN &FRMDATN &THRDATN &RATEAMTN +
                          &RATEAMTN2 &CARRIER &RPTTYPE &EMAIL)

/* PARMS                                                             */
        DCL    &COMPNYN    *DEC
        DCL    &FRMDATN    *DEC
        DCL    &THRDATN    *DEC
        DCL    &RATEAMTN   *DEC
        DCL    &RATEAMTN2  *DEC
        DCL    &CARRIER    *CHAR  (3)
        DCL    &RPTTYPE    *CHAR  (1)
        DCL    &EMAIL      *CHAR  (50)
/********************************************************************/
/*  SUBMIT DOWNLOAD PROGRAM                                          */
/*********************************************************************/
 /****  GOTO CALL             FOR TESTING    ****/

 SUBMIT:     SBMJOB     CMD(CALL PGM(PNWUUPC) PARM(&COMPNYN &FRMDATN +
                          &THRDATN &RATEAMTN &RATEAMTN2 &CARRIER +
                          &RPTTYPE &EMAIL)) JOB(EXPLDVERFY)
             GOTO       CMDLBL(END)

 CALL:       CALL       PGM(PNWUUPC) PARM(&COMPNYN &FRMDATN &THRDATN +
                          &RATEAMTN &RATEAMTN2 &CARRIER &RPTTYPE +
                          &EMAIL)
             GOTO       CMDLBL(END)


 END:        ENDPGM
