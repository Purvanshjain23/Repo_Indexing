/**************************************************************************/
/*  PROGRAM:       PUPPUPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO ORDER DETAIL SCHED-INV (POB5CPP) FILE.*/
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   02/20/2020                                             */
/*  PROJECT:       S16166                                                 */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY PHYSICAL FILE TRIGGER TO THE ORDER     */
/*                 DETAIL SCHED-INV (POB5CPP) FILE FOR ALL TRIGGER        */
/*                 DATABASE EVENTS:                                       */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 03/03/23 DN   W122889 ADDED MONITOR MESSAGE FOR 'RMVPFTRG' TO PREVENT  */
/*                       ERROR.                                           */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/POB5CPP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PUPRUPR) +
                          RPLTRG(*YES) TRG(SEB_POB5CPP_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/POB5CPP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PUPRUPR) +
                          RPLTRG(*YES) TRG(SEB_POB5CPP_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/POB5CPP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PUPRUPR) +
                          RPLTRG(*YES) TRG(SEB_POB5CPP_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/POB5CPP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
