             PGM        PARM(&OBJECT &TYPE &DESCRIPT)

             DCL        VAR(&OBJECT) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TYPE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DESCRIPT) TYPE(*CHAR) LEN(50)

             RTVOBJD    OBJ(&OBJECT) OBJTYPE(&TYPE) TEXT(&DESCRIPT)

             RETURN
             ENDPGM
