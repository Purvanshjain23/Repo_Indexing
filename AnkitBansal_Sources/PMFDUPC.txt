/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  DELETE THE PRODUCT COST RECORDS IN THE              */
/*              ITEM STANDARD COST FILE                             */
/*                                                                  */
/*  SYSTEM    - DAILY FOODS                                         */
/*  PROGRAM   -                                                     */
/*                                                                  */
/*  DATE      - 02/20/2009                                          */
/*  PROGRAMMER- SHARON ZURN                                         */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &EFTDTE     - EFFECTIVE DATE                            */
/********************************************************************/
 PMFDUPC:    PGM        PARM(&EFTDTE)

/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)  /* RETURN CODE               */
        DCL    &EFTDTE     *DEC    (7 0)/* EFFECTIVE DATE            */
        DCL    &EFTDTA     *CHAR   (7)/* EFFECTIVE DATE            */

/********************************************************************/
             CHGVAR     VAR(&EFTDTA) VALUE(&EFTDTE)

             CALL       PGM(PMFIXFR) PARM( &RETURN &EFTDTE)

 END:        ENDPGM
