/*  FUNCTION:        THIS CL IS SUBMITTED FROM THE JOB SCHEDULER.  */
/*                                                                 */
/*                   SINCE THIS IS ALSO AN ON-DEMAND REPORT THAT   */
/*                   GETS ITS SELECTION CRITERIA FROM THE LDA, WE  */
/*                   POPULATE SOME LDA FIELDS BEFORE GENERATING    */
/*                   THE REPORT.                                   */
/*                                                                 */
/*                   AND, WE RUN IT TWICE--BY CELL MANAGER         */
/*                                         BY PRODUCTION MANGER    */
/*                                                                 */
/*******************************************************************/
/*  MODIFIED:                                                      */
/*******************************************************************/

             PGM

/* Since the print program is also on-demand, we need to populate    */
/* some of the LDA fields that the print program uses.               */

/* Set the 'E-Mail' and 'Page Break' options and initialize numeric */
/* fields to zero. Set the 'first email' flag to YES.               */

             CHGDTAARA  DTAARA(*LDA (1 140)) VALUE(' ')
             CHGDTAARA  DTAARA(*LDA (141 1)) VALUE('Y')     /* First e-mail */
             CHGDTAARA  DTAARA(*LDA (2 1)) VALUE('Y')       /* E-mail    */
             CHGDTAARA  DTAARA(*LDA (3 1)) VALUE('Y')       /* Page break*/
             CHGDTAARA  DTAARA(*LDA (19 2)) VALUE('00')     /* Cell      */
             CHGDTAARA  DTAARA(*LDA (21 5)) VALUE('00000')  /* Farm site */
             CHGDTAARA  DTAARA(*LDA (412 1)) VALUE('1')     /* Copies */

/* Set the 'Version' to "Sort/Print by Cell Manager" and run report   */

             CHGDTAARA  DTAARA(*LDA (1 1)) VALUE('1')      /* Version   */
             CALL       PGM(HP6709CL)

/* Set the 'Version' to "Sort/Print by Production Manager" and run report   */

             CHGDTAARA  DTAARA(*LDA (1 1)) VALUE('2')      /* Version   */
             CHGDTAARA  DTAARA(*LDA (141 1)) VALUE('N')    /* First e-mail */
             CALL       PGM(HP6709CL)

             ENDPGM
