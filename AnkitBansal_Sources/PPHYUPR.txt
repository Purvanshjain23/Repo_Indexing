 PPHYUPR:    PGM        PARM(&RETURN &APPLIC &DTLSUM &ITMS &TFCLS +
                          &STRTYP &STRGRP &STRCLS &CLSGROUP +
                          &MEATCOST &ACTSTS &CRDSTS &OUTQ &HOLD +
                          &SAVE &COPY &CHG &ITEMTYPE)

/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &DTLSUM     *CHAR   (01)          /* DETAIL SUMMARY            */
        DCL    &ITMS       *CHAR   (01)          /* ALL SELECT OMIT ITM       */
        DCL    &TFCLS      *CHAR   (05)          /* TF CLASSIFICATION         */
        DCL    &STRCLS     *DEC   LEN(3 0)       /* ITEM STR CLASS */
        DCL    &STRGRP     *DEC   LEN(3 0)       /* ITEM STR GROUP */
        DCL    &STRTYP     *DEC   LEN(3 0)       /* ITEM STR TYPE  */
        DCL    &CLSGROUP   *CHAR  LEN(02)        /* CLASS GROUP    */
        DCL    &MEATCOST   *CHAR  LEN(04)        /* TF MEAT COST   */
        DCL    &ACTSTS     *CHAR  LEN(01)        /* ACTIVE STS     */
        DCL    &CRDSTS     *CHAR  LEN(01)        /* CREDIT ITEM STS*/
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &CHG        *CHAR   (1)           /* CHANGE FLAG               */
        DCL    &ITEMTYPE   *CHAR   (3)           /* ITEM TYPE CODE            */

/* INTERNAL FIELDS                                                            */
        DCL    &COMMAND    *CHAR   (512)         /* message                   */
/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPHWPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PPHWPFR) PARM(' ' &DTLSUM &ITMS &TFCLS +
                          &STRTYP &STRGRP &STRCLS &CLSGROUP +
                          &MEATCOST &ACTSTS &CRDSTS &CHG &ITEMTYPE)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
 END:        ENDPGM
