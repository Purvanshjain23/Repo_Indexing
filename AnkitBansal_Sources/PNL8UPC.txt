/*                                                                   */
/* PARAMETERS :                                                      */
/*  &FROMDATE - DATE TO BE CONVERTED, CHAR 10                        */
/*  &FROMFMT  - FORMAT OF FROM DATE                                  */
/*  &TODATE   - DATE TO BE CONVERTED TO, CHAR 10                     */
/*  &TOFMT    - FORMAT OF TO DATE                                    */
/*  &SEP      - SEPARATERS, IF ANY. VALID VALUES: *SYSVAL *NONE      */
/*              *JOB *BLANK  /  -  ,  .                              */
/*********************************************************************/
             PGM        PARM(&FROMDATE &TODATE &FROMFMT &TOFMT &SEP)

             DCL        VAR(&FROMDATE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TODATE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&FROMFMT) TYPE(*CHAR) LEN(6)
             DCL        VAR(&TOFMT) TYPE(*CHAR) LEN(6)
             DCL        VAR(&SEP) TYPE(*CHAR) LEN(7)

             CVTDAT     DATE(&FROMDATE) TOVAR(&TODATE) +
                          FROMFMT(&FROMFMT) TOFMT(&TOFMT) TOSEP(&SEP)
             MONMSG     MSGID(CPF9999)             /* DON'T ERR OUT */

             ENDPGM
