/* PURPOSE : Setup the required library list for any Inbound API     */
/*           which is initiated from the P2P WIM application.        */
/*                                                                   */
/* PARMS   : &ECIENV = ERADANI ENVIRONMENT CODE; VALID VALUES:       */
/*                     DEV, QUA, UAT, STG OR PRD.                    */
/*                                                                   */
/*********************************************************************/
/* MAINTENANCE LOG:                                                  */
/*-------------------------------------------------------------------*/
/* DATE     BY  PROJ      CHANGE DESCRIPTION                         */
/* -------- --- --------- ------------------------------------------ */
/*                                                                   */
/*                                                                   */
/*********************************************************************/

             PGM        PARM(&ECIENV)

             DCL        VAR(&ECIENV) TYPE(*CHAR) LEN(3)


    /*ˆIf PROD - Setup Production Environment Libraries */
             IF         COND(&ECIENV *EQ 'PRD') THEN(DO)

             ADDLIBLE LIB(PRKPLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)                /* ALREADY EXIST */
             ADDLIBLE   LIB(PRKFLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)

             ENDDO      /* &ECIENV = 'PRD' */

    /*If DEV - Setup DEV Environment Libraries */
             IF         COND(&ECIENV *EQ 'DEV') THEN(DO)

             ADDLIBLE   LIB(PRKPLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKTESTP) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(OMSDEVSQC) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(OMSDEVGEN) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)

             ENDDO      /* &ECIENV = 'DEV' */

    /*If QA - Setup QUA Environment Libraries */
             IF         COND(&ECIENV *EQ 'QUA') THEN(DO)

             ADDLIBLE   LIB(PRKPLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKTESTP) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKTESTF) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)

             ENDDO      /* &ECIENV = 'QUA' */

    /*If UAT - Setup UAT Environment Libraries */
             IF         COND(&ECIENV *EQ 'UAT') THEN(DO)

             ADDLIBLE   LIB(PRKPLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKTESTP) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKUATF) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)

             ENDDO      /* &ECIENV = 'UAT' */

    /*If STAGE - Setup STG Environment Libraries */
             IF         COND(&ECIENV *EQ 'STG') THEN(DO)

             ADDLIBLE   LIB(PRKPLIB) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKTESTP) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)
             ADDLIBLE   LIB(PRKSTGF) POSITION(*FIRST)
             MONMSG     MSGID(CPF2103)

             ENDDO      /* &ECIENV = 'STG' */

 ENDPGM:
             ENDPGM
