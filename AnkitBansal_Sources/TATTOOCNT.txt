H QM4 05 Q 01 E V W E R 01 03 25/08/14 07:07
V 1001 050 Tattoo Count
V 5001 004 *HEX
SELECT
  ALL       A1HMNB, A1B0DT, (A1CVNB)
  FROM      PRKFLIB/PKA1CPP T01
  WHERE     A1LLST IN ('M', 'R')
    AND     A1B0DT >= 990101
    AND     A1HMNB = 360
    AND     A1KAST = 'E'
    OR      A1LLST = 'R'
    AND     A1B0DT >= 990101
    AND     A1HMNB = 360
    AND     A1KAST = 'I'
  ORDER BY  001 ASC, 002 ASC
