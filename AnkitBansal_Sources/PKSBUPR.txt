/*  Modification History                                           */
/*  ROSE CENTONZE  12/17/98  SEND IN ALPHA CUT DATE FOR Y2K        */
/*  LARA BUSER     02/28/06  MODIFY PARMS                          */
/*  ROSE CENTONZE  05/25/06  REPORT TYPE IS 2A -- NOT 1.           */
/*******************************************************************/
/*           PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES +
                          &APP &XCO  &SHIFT &RPTTYP &XCUTDT)      */

             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES +
                          &XCO  &SHIFT &RPTTYP &ACUTDT &LPP)

/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
/*           DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(7)          */
             DCL        VAR(&XCOPIES) TYPE(*DEC)

/* SYSTEM NAME PARM                                               */
/*           DCL        VAR(&APP) TYPE(*CHAR) LEN(33)             */
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCO) TYPE(*DEC)
             DCL        VAR(&XCUTDT) TYPE(*CHAR) LEN(7)
             DCL        VAR(&ACUTDT) TYPE(*CHAR) LEN(7)
             DCL        VAR(&SHIFT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&RPTTYP) TYPE(*CHAR) LEN(2)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO)    TYPE(*DEC) LEN(3 0)
             DCL        VAR(&CUTDT) TYPE(*DEC) LEN(7 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                         */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&CO) VALUE(&XCO)
/*           CHGVAR     VAR(&CUTDT) VALUE(&XCUTDT)                 */
             CHGVAR     VAR(&CUTDT) VALUE(&ACUTDT)

/* CREATE PRINTER FILE OVERRIDE CODE                              */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKH3PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) +
                          USRDTA(PRMSCLSUM)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
             IF COND(&CO *EQ 360)  THEN(DO)
             OVRDBF     FILE(PKDLCPP) TOFILE(PRKFGUP/PKDLCPP)
             OVRDBF     FILE(PKDLCPL7) TOFILE(PRKFGUP/PKDLCPL7)
             OVRDBF     FILE(PLAUREP) TOFILE(PRKFGUP/PLAUREP)
             OVRDBF     FILE(PLAUREL1) TOFILE(PRKFGUP/PLAUREL1)
             ENDDO
             IF COND(&CO *EQ 960)  THEN(DO)
             OVRDBF     FILE(PKDLCPP) TOFILE(PRKFTFP/PKDLCPP)
             OVRDBF     FILE(PKDLCPL7) TOFILE(PRKFTFP/PKDLCPL7)
             OVRDBF     FILE(PLAUREP) TOFILE(PRKFTFP/PLAUREP)
             OVRDBF     FILE(PLAUREL1) TOFILE(PRKFTFP/PLAUREL1)
             ENDDO

/* PRINT PGM                                                   */
             CALL       PGM(PKH3PFR) PARM(' ' &CO &CUTDT &SHIFT &LPP +
                          &RPTTYP)

             DLTOVR *ALL
             ENDPGM
