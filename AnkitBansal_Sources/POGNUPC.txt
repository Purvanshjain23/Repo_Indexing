/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  POCNUPC                                       */
/*  PROGRAM NAME:    Print LOAD STATUS                             */
/*  PROGRAMMER:      Purva DROGE                                   */
/*  CREATE DATE:     06/22/00                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted FROM WORK WITH LAOD     */
/*                   STATUS.                                       */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&COMP &SDTE &LDSTS &WHS &INCEXC &PDEV +
                          &HOLD &SAVE &XCOPIES &PGM &PGMPRTF)
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGMPRTF) TYPE(*CHAR) LEN(10)

/* Function Specific                                               */
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&SDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&LDSTS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&WHS) TYPE(*CHAR) LEN(3)
             DCL        VAR(&INCEXC) TYPE(*CHAR) LEN(1)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGMPRTF) OUTQ(&PDEV) +
                          DRAWER(*PRTF) COPIES(&COPIES) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(&PGM) PARM(' ' &COMP &SDTE &LDSTS &WHS +
                          &INCEXC)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
