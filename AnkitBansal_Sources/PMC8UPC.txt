/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  CALL AND PASS COMPANY NUMBER                        */
/*                                                                  */
/*  SYSTEM    - INVENTORY MANAGEMENT SYSTEM                         */
/*  PROGRAM   - CALLS PROGRAM TO CREATE TRANSFER TEMPLATE ORDERS    */
/*  DATE      - 12/27/2006                                          */
/*  PROGRAMMER- LARA BUSER                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &COMPYN     - COMPANY NUMBER - PASSED AS 15.5           */
/*     I    &CUSTN      - CUSTOMER NUMBER - PASSED AS 15.5          */
/*     I    &SCHSHDT    - SCHEDULED SHIP DATE - PASSED AS 15.5      */
/*     I    &REQSHDT    - REQUESTED SHIP DATE - PASSED AS 15.5      */
/*     I    &REQDLVDT   - REQUESTED DELIVERY DATE - PASSED AS 15.5  */
/*     B    &ORDFRESHN  - ORDER NUMBER FOR FRESH ORDER 15.5         */
/*     B    &ORDFROZNN  - ORDER NUMBER FOR FROZEN ORDER 15.5        */
/*     B    &LOADIDN    - LOAD ID - PASSED AS 15.5                  */
/*     I    &FRFZ       - FRESH, FROZEN, BOTH FLAG                  */
/********************************************************************/
 PMC8UPC:    PGM        PARM(&RETURN &COMPYN &CUSTN &SCHSHDT +
                          &REQSHDT &REQDLVDT &ORDFRESHN &ORDFROZNN +
                          &LOADIDN &FRFZ)

             DCL        VAR(&RETURN)    TYPE(*CHAR) LEN(7)
             DCL        VAR(&COMPYN)    TYPE(*DEC)
             DCL        VAR(&CUSTN)     TYPE(*DEC)
             DCL        VAR(&SCHSHDT)   TYPE(*DEC)
             DCL        VAR(&REQSHDT)   TYPE(*DEC)
             DCL        VAR(&REQDLVDT)  TYPE(*DEC)
             DCL        VAR(&COMP)      TYPE(*DEC)  LEN(3)
             DCL        VAR(&CUST)      TYPE(*DEC)  LEN(7)
             DCL        VAR(&ORDFRESHN) TYPE(*DEC)
             DCL        VAR(&ORDFROZNN) TYPE(*DEC)
             DCL        VAR(&LOADIDN)   TYPE(*DEC)
             DCL        VAR(&FRFZ)      TYPE(*CHAR) LEN(2)

             CHGVAR     VAR(&COMP) VALUE(&COMPYN)
             CHGVAR     VAR(&CUST) VALUE(&CUSTN)

/*------------------- CALL TRANSFER TEMPLATE ORDER ------------------*/
             CALL       PGM(PMC5XFR) PARM(' ' &COMP &CUST &SCHSHDT +
                          &REQSHDT &REQDLVDT &ORDFRESHN &ORDFROZNN +
                          &LOADIDN &FRFZ)

 END:        ENDPGM
