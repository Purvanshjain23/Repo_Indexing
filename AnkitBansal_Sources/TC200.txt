      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *---------------------------------------------------------------
      *  Named Constants
      *---------------------------------------------------------------
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      *---------------------------------------------------------------
      *  Standalone field
      *---------------------------------------------------------------
      *
     D wkmkcd          s                   like(f0lqcd)
     D wkpgcd          s                   like(f0lrcd)
      *
      *
      * Workfields for date manipulation
      *
     D wkisodate       s               d   datfmt(*iso)
     D wkdt            s              7  0

      ****************************************************************
      * Data Structures
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.

     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ***************************************************************************************
      * Mainline
      ***************************************************************************************
      *
      * Read the WCB/WTDAVG records in the Commodity Market Price file to get
      * Live Hog Prices.
      *
     C                   exsr      $hogprices
      *
     C                   seton                                        lr
      /eject
      *---------------------------------------------------------------
      * Get Live Hog Prices
      *---------------------------------------------------------------
      *
     C     $hogprices    begsr
      *
      * Set up key fields
     C                   move      'WCB'         wkmkcd
     C                   move      'WTDAVG '     wkpgcd
      *
     C     key01         setll     omf0cpl1
      *
     C                   dou       *in90 = *on                                  Do hog prices
     C     key02         reade     omf0cpl1                               90
     C                   if        *in90 = *off                                 If not EOF
     C                   exsr      $wrt200
     C                   endif                                                  If not EOF
     C                   enddo                                                  Do hog prices
      *
     C                   endsr
      /eject
      *---------------------------------------------------------------
      * Write Commodity Price reocrds
      *---------------------------------------------------------------
      *
     C     $wrt200       begsr
      *
      * Market code
      * Price group code
     C                   move      f0lqcd        cpmkcd
     C                   move      f0lrcd        cppgcd
      *
      * Price per UOM
     C                   z-add     f0a5pr        cpmppr
      *
      * Flip Synon date to CCYYMMDD
      *
     C     *cymd         test(d)                 f0eqdt                 92
     C                   if        *in92 = *off
     C     *cymd         move      f0eqdt        wkisodate
     C     *iso          move      wkisodate     cpmpdt
     C                   endif
      *
     C                   move      f0vsst        cpvsst
      *
      * Retrieve the UOM from the Commodity Group file
      *
     C     f0lrcd        chain     omfzrel1                           92
     C                   if        *in92 = *off
     C                   move      fzaicd        cpumcd
     C                   endif
      *
     C                   write     cprec
     C                   clear                   cprec
      *
     C                   endsr
      /EJECT
      *---------------------------------------------------------------
      * Initialization
      *---------------------------------------------------------------
      *
     C     *inzsr        begsr
      *
     C     key01         klist
     C                   kfld                    wkmkcd
     C                   kfld                    wkpgcd
     C                   kfld                    wkdt
      *
     C     key02         klist
     C                   kfld                    wkmkcd
     C                   kfld                    wkpgcd
      *
      * We will start out sending the entire file based on a date of January 1, 2006.
      *
     C                   z-add     1060101       wkdt
      *
     C                   endsr
      /EJECT
