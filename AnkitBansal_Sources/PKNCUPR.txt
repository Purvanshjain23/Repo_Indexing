/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKNCUPR                                       */
/*  PROGRAM NAME:    PRINT CARCASS MERIT PROGRAM PER GRADING TYPE  */
/*  PROGRAMMER:      ROSE MARTIN                                   */
/*  CREATE DATE:     03/07/95                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM WORK WITH CARCASS MERIT PROGRAM.         */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  04/20/95                                                       */
/*  Rose M. Martin   Added SBDPRTOVR code and command.             */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &CMP +
                          &GRADE &OMS)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&CMP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&GRADE) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&OMS)

/*                                   */
/* PRINT REPORT PER GRADE TYPE PARM  */
/*                                   */

 BACKFAT:    IF         COND(&GRADE = 'B') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKKJPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
/* PRINT PROGRAM                                                  */
             CALL       PGM(PKKJPFR) PARM(' ' &CMP &OMS)
             DLTOVR *ALL
             ENDDO

 GRADE:      IF         COND(&GRADE = 'G') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKKIPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
/* PRINT PROGRAM                                                  */
             CALL       PGM(PKKIPFR) PARM(' ' &CMP &LPP)
             DLTOVR *ALL
             ENDDO

 LEANFACT:   IF         COND(&GRADE = 'F') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKKKPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
/* PRINT PROGRAM                                                  */
             CALL       PGM(PKKKPFR) PARM(' ' &CMP &OMS)
             DLTOVR *ALL
             ENDDO

 LEANPCT:    IF         COND(&GRADE = 'L' *OR &GRADE = 'S' *OR +
                          &GRADE = 'D' *OR &GRADE = "M") THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKT6PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
/* PRINT PROGRAM                                                  */
             CALL       PGM(PKT6PFR) PARM(' ' &CMP &OMS)
             DLTOVR *ALL
             ENDDO

 LEANFOM:    IF         COND(&GRADE = 'P') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKT6PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
/* PRINT PROGRAM                                                  */
             CALL       PGM(PKT6PFR) PARM(' ' &CMP &OMS)
             DLTOVR *ALL
             ENDDO

             ENDPGM
