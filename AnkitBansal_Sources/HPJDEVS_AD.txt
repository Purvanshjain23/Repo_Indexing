PARAMETER(LCL.E1_Live);
PARAMETER(LCL.JW_Live);

//?Logic based upon Live In flag (Y/N/P)
CASE;

IF LCL.E1_Live = 'P'/'Y';

//?12/11/2020 JBB E17022 - [ME] HPS Edit on Sales Type error
