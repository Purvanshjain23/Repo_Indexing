/*  TITLE:      POTENTIAL LATE DELIVERIES--SEND E-MAIL MESSAGE      */
/*  PROGRAMMER: LEANNE FEDOR                                        */
/*  DATE:       06/19/01                                            */
/*                                                                  */
/*  SYNOPSIS:                                                       */
/*              SEND E-MAIL TO SALESMEN WHEN THEIR LOAD WILL        */
/*              NOT BE SHIPPED ON TIME.                             */
/*                                                                  */
/********************************************************************/
/* Modifications:                                                   */
/*------------------------------------------------------------------*/
/*   Date     Pgmr      Dscription                                  */
/* --------  ---------  ------------------------------------------- */
/* 11/21/16  D.NGUYEN   E7891 - Added &MESSAGE as input parameter   */
/*                      which will contain Customer PO# & Ship to   */
/*                      Customer Name data to populate the email    */
/*                      body message content.                       */
/********************************************************************/

             PGM        PARM(&ADDRESS &SUBJECT &MESSAGE)

             DCL        VAR(&ADDRESS) TYPE(*CHAR) LEN(150)
             DCL        VAR(&SUBJECT)  TYPE(*CHAR) LEN(45)
             DCL        VAR(&MESSAGE) TYPE(*CHAR) LEN(1500)


/* SEND E-MAIL MESSAGE USING SEQUEL-SUPPLIED COMMAND                 */

             ESNDMAIL   RECIPIENT(&ADDRESS) SUBJECT(&SUBJECT) +
                          MSG(&MESSAGE)

             ENDPGM
