/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING ITEM STRUCTURE               */
/*                                                                  */
/*  SYSTEM    - INVENTORY CONTROL                                   */
/*  DATE      -  2/28/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &ITMTYP        -  ITEM STRUCTURE TYPE                    */
/*         &STAT          -  RECORD STATUS                          */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
/*  RMC 4/30/03  CALL NEW PRINT FILE                                */
/*  RMC 5/07/03  ADDED DTLSUM PARM  TO PRINT WITH/WITHOUT FG       */
/********************************************************************/
 PDMRCLP:    PGM        PARM(&ITMTYP &STAT &APPLIC &OUTQ &HOLD &SAVE +
                             &COPYN &NIGHTQ &DTLSUM)

        DCL    &ITMTYP      *DEC
        DCL    &STAT        *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND     *CHAR  (512)
        DCL    &DTLSUM      *CHAR  (1)

        DCL    &ITEMTYP     *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&ITEMTYP)  VALUE(&ITMTYP)
             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(POU2PFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 /*          CALL       PGM(PDMRPFR) PARM(' ' &ITEMTYP &STAT)  */
 CALL:       CALL       PGM(POU2PFR)  PARM(' ' &DTLSUM)

 ENDPGM:     ENDPGM
