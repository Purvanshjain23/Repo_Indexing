/*  SUSAN L. MASON 01/22/96                                        */
/*                 PLACED OVERRIDER TO CARCASS DETAIL PRINTER      */
/*                 TO BE THE SAME AS THE ONDEMAND SELECTION OF     */
/*                 CARCASS DETAIL                                  */
/*                                                                 */
/*  SUSAN L. MASON 09/11/96                                        */
/*                 CHANGE PDEV TO PDEV1 ADDED PDEV2                */
/*                                                                 */
/*  SUSAN L. MASON 10/04/96                                        */
/*                 REMOVE HARD CODE LIBRARY FROM CL                */
/*                                                                 */
/*  MARK ARENSBERG 03/03/97                                        */
/*                 REMOVE PRINTER OVERRIDES FOR CARCASS DETAIL     */
/*                 AND SUMMARY. THE OVERRIDES WILL BE DONE IN      */
/*                 ANOTHER CL.                                     */
/*                                                                 */
/*  PURVA DROGE    09/05/97                                        */
/*                 ADDED CALL TO CL FOR CREATING EDI FILES.        */
/*                                                                 */
/*  PURVA DROGE    10/15/97                                        */
/*                 ADDED CALL TO CL FOR CREATING EDI FILES INTO    */
/*                 THE PAYMENT POST XF.  REMOVED THE CALL FROM     */
/*                 THIS CL.                                        */
/*                                                                 */
/*  ROSE CENTONZE  10/21/97                                        */
/*                 CHANGED CRTDUP FROM PKC6CP* TO PKD7CP* FAX DRVR */
/*                 BUT ** IT, SO THE PGMS IN THIS JOB WILL USE    */
/*                 PRKFLIB/PKD7CPP, SO THAT THE PRT CRC DTL/SUM &  */
/*                 FAX CRC DTL/SUM CAN BE SUBMITTED AS STAND ALONE */
/*                 JOBS.                                           */
/*                                                                 */
/*  ROSE CENTONZE  12/16/98                                        */
/*                 ADDED PRTOVR FOR COMMISSION PAYM EDIT/POST    */
/*                                                                 */
/*  ROSE CENTONZE  12/28/98                                        */
/*                 ADDED &SAVE2 FOR SAVING PKT1PFR AFTER PRINTING  */
/*  ROSE CENTONZE  03/22/01                                        */
/*                 PRINT MPR COMPARISON REPORT IN PKTFXFR         */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV1 &HOLD &SAVE &XCOPIES &XCOMP +
                          &APP)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV1) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PDEV2) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE2) TYPE(*CHAR) LEN(4) VALUE('*YES')
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&LPP) VALUE(&APP)

/* RTV USERPROFILE                                               */
             RTVUSRPRF  OUTQ(&PDEV2)

   CLRPFM PKD7CPP
   GOTO SKIPDUP
/* CREATE TATTOO HEADER DRIVER FILE IN QTEMP                    */
             CRTDUPOBJL  OBJ(PKD7CPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL3) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL4) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL5) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL6) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKD7CPL7) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 SKIPDUP:
/* CREATE PRINTER FILE OVERRIDE CODE    PAY EDIT/POST           */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKS2PFR$) OUTQ(&PDEV2) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* CREATE PRINTER FILE OVERRIDE CODE  COMMISSION EDIT/POST      */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKPMPFR$) OUTQ(&PDEV2) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* CREATE PRINTER FILE OVERRIDE CODE    PAY CHECK POST          */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKT1PFR$) OUTQ(&PDEV2) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE2)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* CREATE PRINTER FILE OVERRIDE CODE    PUR JOURNAL PRIM/FINAL  */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKTTPFR$) OUTQ(&PDEV2) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* CREATE PRINTER FILE OVERRIDE CODE    PUR JOURNAL RESALE      */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKTZPFR$) OUTQ(&PDEV2) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* CREATE PRINTER FILE OVERRIDE CODE    MPR COMPARISON          */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKWZPFR$) OUTQ(&PDEV2) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE2)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)
/*********************************************************/
/*  FORM CREATED BY PGM PRODUCER SPECIAL PUT ON HOLD     */
/*       PER USERS ON 091996 IN PAYMENTS ONLY            */
/*                                                       */
/*********************************************************/
/* CREATE PRINTER FILE OVERRIDE CODE    CARCASS SUMMARY FORM    */
/*           OVRPRTF    FILE(PKW9PFR$) PAGESIZE(*N 96) CPI(12) +
                          OUTQ(&PDEV2) FORMTYPE(CARCSTDPAP) +
                          COPIES(&COPIES) HOLD(*YES) SAVE(&SAVE) */


/* REMOVE OVERRIDE FROM THIS CL  03/03/97 M ARENSBERG           */
/* CREATE PRINTER FILE OVERRIDE CODE    CARCASS SUMMARY BLK FRM */
/*           OVRPRTF    FILE(PKTBPFR$) OUTQ(&PDEV1) FORMTYPE(*STD) + */
/*                        COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)    */
/*********************************************************/


/* REMOVE OVERRIDE FROM THIS CL  03/03/97 M ARENSBERG           */
/* CREATE PRINTER FILE OVERRIDE CODE   CARCASS DETAIL           */
/*           SBDPRTOVR  CMDVAR(&CMD) FILE(PKTCPFR$) OUTQ(&PDEV1) + */
/*                        COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) +*/
/*                        LETTER(*YES)                             */
/*           CALL       PGM(QCMDEXC) PARM(&CMD 512)                */

/* CREATE PRINTER FILE OVERRIDE CODE    LEAN% DIST INTERNAL RPT */
/* This report will be deleted as soon as it is printed.        */
             OVRPRTF    FILE(PKSKPFR$) HOLD(*YES)

/* PROCESS PAYMENT PRODUCER POST                               */

             CALL       PGM(PKTFXFR) PARM(' ' &COMP &LPP &PDEV1 &HOLD +
                                              &SAVE &COPIES)
             DLTOVR *ALL
             ENDPGM
