/*  PROGRAM NUMBER:  PLA8UPC                                       */
/*  PROGRAM NAME:    PRINT MPR DETAIL                              */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     12/16/02                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM DSP MPR PRIOR DAY SCHED                  */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &IDNUM +
       &RPTDT  &RPTTM &KILLD &APP)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(33)
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(40)


/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&IDNUM) TYPE(*CHAR) LEN(5)
             DCL        VAR(&KILLD) TYPE(*CHAR) LEN(7)
             DCL        VAR(&RPTDT) TYPE(*CHAR) LEN(7)
             DCL        VAR(&RPTTM) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&KILLDN) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&RPTDTN) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&RPTTMN) TYPE(*DEC) LEN(1 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

   CHGVAR     VAR(&OMS) VALUE(&APP)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&KILLDN) VALUE(&KILLD)
             CHGVAR     VAR(&RPTDTN) VALUE(&RPTDT)
             CHGVAR     VAR(&RPTTMN) VALUE(&RPTTM)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLA7PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PLA7PFR) PARM(' ' &IDNUM &RPTDTN &RPTTMN +
                          &KILLDN &OMS)

             DLTOVR *ALL
             ENDPGM
