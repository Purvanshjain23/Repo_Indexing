/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER  -> PDIJUPR                                     */
/*  PROGRAM NAME    -> PROCESS REPORT HELP TEXT                    */
/***************************************************************** */
PDIJUPR:     PGM        PARM(&PGM)
             DCL        VAR(&PGM)    TYPE(*CHAR) LEN(10)

             DCL        VAR(&DOC)    TYPE(*CHAR) LEN(8)
             DCL        VAR(&FLDR)   TYPE(*CHAR) LEN(17)
             DCL        VAR(&LBL)    TYPE(*CHAR) LEN(10)


             CALL       PGM(PKJGXFR) PARM('       ' &PGM &DOC +
                          &FLDR &LBL)

             IF         COND(&DOC *NE ' ' & &FLDR *NE ' ' & &LBL *NE +
                          ' ') THEN(DSPHLPDOC DOC(&DOC) FLR(&FLDR) +
                          HLPLBL(&LBL))
             ELSE       CMD(SNDPGMMSG MSG('Report help does not exist'))


ENDPGM
