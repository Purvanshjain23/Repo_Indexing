/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING SALES ROUTE                  */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  3/01/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*         &SLSP          - SALESPERSON    (PASSED)                 */
/*         &POSLVL        - POSITION LEVEL (PASSED)                 */
/*         &MGRLVL        - MANAGER LEVEL  (PASSED)                 */
/*         &VPLVL         - VP LEVEL       (PASSED)                 */
/*         &STATUS        - STATUS         (PASSED)                 */
/*                                                                  */
/********************************************************************/
 CABQCLP:    PGM        PARM(&APPLIC &SLSP &MGRLVL &VPLVL &POSLVL +
                          &STATUS &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &SLSP       *CHAR   (3)           /* SALES ROUTE/PERSON        */
        DCL    &POSLVL     *CHAR   (1)           /* POSITION LEVEL            */
        DCL    &MGRLVL     *CHAR   (2)           /* MANAGER LEVEL             */
        DCL    &VPLVL      *CHAR   (3)           /* VP LEVEL                  */
        DCL    &STATUS     *CHAR   (1)           /* RECORD STATUS             */

        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDFYPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDFYPFR) PARM(' ' &SLSP &MGRLVL &VPLVL +
                          &POSLVL &STATUS)

 ENDPGM:     ENDPGM
