/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION. CAUSE PRINTER OVERRIDE  */
/*              PROMPTING AND PASSES COMPANY, DATE, LOAD, AND       */
/*              CARRIER.                                            */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT REPORTS                                       */
/*  DATE      - 12/18/96                                            */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*  &RETURN     - RETURN CODE                                       */
/*  &APPLIC     - APPLICATION CODE                                  */
/*  &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT SCREEN)     */
/*  &OUTQ       - OUTPUT QUEUE                                      */
/*  &HOLD       - HOLD STATUS                                       */
/*  &SAVE       - SAVE STATUS                                       */
/*  &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q             */
/*  &NIGHTQ     - NIGHT JOB QUEUE                                   */
/*  &COMPNYN    - LOAD COMPANY NUMBER (PASSED)                      */
/*  &FRMDATN    - FROM DATE                                         */
/*  &THRDATN    - THRU DATE                                         */
/*  &LOADN      - LOAD NUMBER                                       */
/*  &CARR1 - 10 - 10 CARRIER CODES                                  */
/*  &SHIP1 - 5  - 5 SHIPPING LINES                                  */
/*  &COPYN      - NO. OF COPIES                                     */
/*  &CUST       - SHIP TO CUST                                      */
/*  &CNTRY1-10  - 10 COUNTRY CODES                                  */
/*  &WHS 1-9    - 9 WAREHOUSE CODES IN AN ARRAY                     */
/*  &COLLECT    - INCLUDE COLLECT TRUCKS Y OR NO (STATUS - Y OR N)  */
/*  &TRANSTYP1  - CUSTOMER TYPE CODE (ES OR BLANK)                  */
/*  &WHSETYPE   - WAREHOUSE TYPE (FR OR FZ)                         */
/*  &TYPECUST   - TYPE OF CUSTOMER (SHIP TO OR BILL TO)             */
/*  &LC         - INCLUDE OR SKIP IF LETTER OF CREDIT ENTERED       */
/*                IS PASSED IN THE USR CONFIRM STATUS (Y/N) FIELD   */
/********************************************************************/
/*  RMC 1/02/2001 ADD SALESPERSON AS INPUT PARM AND PASS TO CALLED  */
/*                PROGRAMS                                          */
/*  RMC 6/27/2003 ADDED REASON PARM                                 */
/*  SLM 1/06/2005 ADDED ORDER COMPANY, AND CHG DOCO ON COMPANY      */
/*                WILL NOW STATE LOAD COMPANY                       */
/*  LJB 12/04/2005 REMOVED SALESPERSON, REASON, LOADT, RPTSEL AND   */
/*                 UOM - THESE WERE REMOVED FROM THE CALLED PGM     */
/*  LJB 02/10/2006 CHANGED THE ORDER OF THE PARMS ON THE SBMJOB     */
/*  LJB 05/01/2007 THIS PROGRAM NOW RECEIVES PMGDUPC FROM THE       */
/*                 PROMPT PROGRAM                                   */
/* 01/30/2008 PKD E00025 ADDED WAREHOUSE & COUNTRY, PASS TO REPORT  */
/* 08/22/2008 LJB E00210 ADD COLLECT Y/N TO SCREEN AND PASS         */
/* 11/14/2008 LJB E00269 ADD EMAIL ADDRESS                          */
/* 11/17/2008 LJB E00281 ADD MULTIPLE COUNTRY CODES                 */
/* 03/01/2011 RMC C01397 PASS 2 25 CHAR PARMS FOR EMAIL INSTEAD OF  */
/*                 ONE 50                                           */
/* 04/26/2011 LJB P000737 ADD 5 MORE COUNTRY CODES AND INCREASE     */
/*                CARRIERS TO 10. ADD LOAD HAUL TYPE.               */
/*                ADD 5 SHIPPING LINES.                             */
/* 12/29/2011 LJB E001788 ADD BOOKING NUMBER PARM                   */
/* 02/12/2012 LJB E001704 ADD BROKER CODE FOR SUMIE'S REPORTS       */
/* 01/11/2013 LJB E002395 ADD TYPE CODE TO SELECT EXPORT SEABOARD   */
/*                BOOKINGS IF NEEDED                                */
/* 05/08/2013 LJB E002587 ADD WAREHOUSE TYPE AND TYPE OF CUSTOMER   */
/* 11/22/2013 LJB E002768 ADD INCLUDE OR SKIP IF LETTER OF CREDIT   */
/*                HAS BEEN PASSED IN                                */
/* 05/22/2015 LJB E004105 ADD 7 MORE WAREHOUSE CODES                */
/* 07/18/2019 DN  R15161 - EXPORT LOAD AS400 UPDATES.               */
/*                CHANGED &BOOKING PARM FROM 12A TO 15A.            */
/* 01/17/2020 DN  S16050 - PASSED IN 9TH WAREHOUSE CODE PARM &      */
/*                PASS IT TO PDUJPCLP.                              */
/********************************************************************/
 PPICUPC:    PGM        PARM(&RETURN &APPLIC &BOOKING &DESC &OUTQ +
                          &HOLD &SAVE &NQSTAT &NIGHTQ &COMPNYN +
                          &FRMDATN &THRDATN &LOADN &COPYN +
                          &TRANSTYP1 &WHSETYPE &LDHAUL &TYPECUST +
                          &CUST &COLLECT &SELECT &LC &CARR1 &CARR2 +
                          &CARR3 &CARR4 &CARR5 &CARR6 &CARR7 &CARR8 +
                          &CARR9 &CARR10 &SHIP1 &SHIP2 &SHIP3 +
                          &SHIP4 &SHIP5 &CNTRY1 &CNTRY2 &CNTRY3 +
                          &CNTRY4 &CNTRY5 &CNTRY6 &CNTRY7 &CNTRY8 +
                          &CNTRY9 &CNTRY10 &WHS &WHS1 &WHS2 &WHS3 +
                          &WHS4 &WHS5 &WHS6 &WHS7 &WHS8 &BROKER +
                          &EMAILYN &EMAIL1 &EMAIL2)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)  /* RETURN CODE               */
        DCL    &APPLIC     *CHAR   (06) /* APPLICATION CODE          */
        DCL    &DESC       *CHAR   (25) /* PRT PGM DESCRIPTION       */
        DCL    &OUTQ       *CHAR   (10) /* OUTPUT QUEUE              */
        DCL    &HOLD       *CHAR   (4)  /* HOLD STATUS               */
        DCL    &SAVE       *CHAR   (4)  /* SAVE STATUS               */
        DCL    &NQSTAT     *CHAR   (01) /* IF MUST RUN IN NIGHT Q    */
        DCL    &NIGHTQ     *CHAR   (10) /* NIGHT JOB QUEUE           */
        DCL    &COMPNYN    *DEC         /* LOAD COMPANY   (PASSED)   */
        DCL    &FRMDATN    *DEC         /* FROM DATE      (PASSED)   */
        DCL    &THRDATN    *DEC         /* THRU DATE      (PASSED)   */
        DCL    &LOADN      *DEC         /* LOAD NUMBER    (PASSED)   */
        DCL    &COPYN      *DEC         /* NO. OF COPIES             */
        DCL    &LDHAUL     *CHAR   (3)  /* LOAD HAUL TYPE            */
        DCL    &CARR1      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR2      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR3      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR4      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR5      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR6      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR7      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR8      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR9      *CHAR   (3)  /* CARRIER CODE              */
        DCL    &CARR10     *CHAR   (3)  /* CARRIER CODE              */
        DCL    &SHIP1      *CHAR   (3)  /* SHIPPING LINE             */
        DCL    &SHIP2      *CHAR   (3)  /* SHIPPING LINE             */
        DCL    &SHIP3      *CHAR   (3)  /* SHIPPING LINE             */
        DCL    &SHIP4      *CHAR   (3)  /* SHIPPING LINE             */
        DCL    &SHIP5      *CHAR   (3)  /* SHIPPING LINE             */
        DCL    &CNTRY1     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY2     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY3     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY4     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY5     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY6     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY7     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY8     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY9     *CHAR   (6)  /* COUNTRY                   */
        DCL    &CNTRY10    *CHAR   (6)  /* COUNTRY                   */
        DCL    &WHS        *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS1       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS2       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS3       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS4       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS5       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS6       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS7       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &WHS8       *CHAR   (3)  /* WAREHOUSE                 */
        DCL    &BOOKING    *CHAR   (15) /* BOOKING NUMBER            */
        DCL    &EMAIL1     *CHAR   (25)
        DCL    &EMAIL2     *CHAR   (25)
        DCL    &EMAILYN    *CHAR   (1)
        DCL    &CUST       *DEC         /* SHIP TO CUSTOMER          */
        DCL    &COLLECT    *CHAR   (1)  /* COLLECT Y/N               */
        DCL    &SELECT     *CHAR   (1)  /* SELECT REPORT TYPE        */
        DCL    &LC         *CHAR   (1)  /* INCLUDE/SKIP IF L/C ENTRD */
        DCL    &BROKER     *CHAR   (6)  /* BROKER CODE               */
        DCL    &TRANSTYP1  *CHAR   (2)  /* CUSTOMER TYPE CODE        */
        DCL    &WHSETYPE   *CHAR   (2)  /* WAREHOUSE TYPE            */
        DCL    &TYPECUST   *CHAR   (1)  /* TYPE OF CUST (S OR B)     */
        DCL    &COPY       *DEC    (2 0)/* NUMBER OF COPIES          */
        DCL    &MSG        *CHAR   (512)/* MESSAGE                   */
        DCL    &MSGID      *CHAR   (7)  /* MESSAGE ID                */
        DCL    &MSGL       *DEC    (5 0)

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/********************************************************************/
/*   CALL PROMPT PROGRAM-HOLD IF REPORT IS TO BE EMAILED            */
/********************************************************************/
             CALL       PGM(PDG0PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND((&EMAILYN *EQ 'Y') *OR (&EMAILYN *EQ +
                          'E')) THEN(CHGVAR VAR(&HOLD) VALUE(*YES))
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

              GOTO       CMDLBL(TEST) /**    ***/
/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM - SUBMITS PDUJPCLP WHICH CALLS         */
/*   PDUIPFR OR PNW5PFR (ADD 7 WHS CODES  E004105)                  */
/********************************************************************/
/* 01/17/20 DN S16050-ADDED &WHS8 AS PARM. */
 SUBMIT:     SBMJOB     CMD(CALL PGM(PDUJPCLP) PARM(&OUTQ &HOLD +
                          &SAVE &NIGHTQ &COMPNYN &FRMDATN &THRDATN +
                          &LOADN &COPYN &TRANSTYP1 &WHSETYPE +
                          &LDHAUL &TYPECUST &CUST &COLLECT &SELECT +
                          &LC &CARR1 &CARR2 &CARR3 &CARR4 &CARR5 +
                          &CARR6 &CARR7 &CARR8 &CARR9 &CARR10 +
                          &SHIP1 &SHIP2 &SHIP3 &SHIP4 &SHIP5 +
                          &CNTRY1 &CNTRY2 &CNTRY3 &CNTRY4 &CNTRY5 +
                          &CNTRY6 &CNTRY7 &CNTRY8 &CNTRY9 &CNTRY10 +
                          &WHS &WHS1 &WHS2 &WHS3 &WHS4 &WHS5 &WHS6 +
                          &WHS7 &WHS8 &EMAILYN &EMAIL1 &EMAIL2 +
                          &BOOKING &BROKER)) JOB(PDUJPCLP)

              GOTO       CMDLBL(END)
/********************************************************************/
/* FOR TESTING ONLY.                                                */
/********************************************************************/
 TEST:       CALL       PGM(PDUJPCLP) PARM(&OUTQ &HOLD &SAVE &NIGHTQ +
                          &COMPNYN &FRMDATN &THRDATN &LOADN &COPYN +
                          &TRANSTYP1 &WHSETYPE &LDHAUL &TYPECUST +
                          &CUST &COLLECT &SELECT &LC &CARR1 &CARR2 +
                          &CARR3 &CARR4 &CARR5 &CARR6 &CARR7 &CARR8 +
                          &CARR9 &CARR10 &SHIP1 &SHIP2 &SHIP3 +
                          &SHIP4 &SHIP5 &CNTRY1 &CNTRY2 &CNTRY3 +
                          &CNTRY4 &CNTRY5 &CNTRY6 &CNTRY7 &CNTRY8 +
                          &CNTRY9 &CNTRY10 &WHS &WHS1 &WHS2 &WHS3 +
                          &WHS4 &WHS5 &WHS6 &WHS7 &WHS8 &EMAILYN +
                          &EMAIL1 &EMAIL2 &BOOKING &BROKER)

              GOTO       CMDLBL(END)
/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
