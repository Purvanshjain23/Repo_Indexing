      * Check selection fields - if fail, Send back status no
     C           *ENTRY    PLIST
     C                     PARM           PTXT1  80
     C                     PARM           PTXT2  80
     C                     PARM           PTSTS   1
     C                     PARM           PTLNG   2
     C                     MOVE PTLNG     NMLNG   20
     C                     MOVE 'N'       PTSTS
     C                     EXSR A010A
     C                     EXSR E010A
     C           A010A     BEGSR
      * Scan for search string
     C                     CALL 'QCLSCAN'
     C                     PARM           PTXT1            PTXT1
      * Length
     C                     PARM NMLNG     WQA3N   30       LENGTH
      * Start
     C                     PARM 1         WQB3N   30
      * Mask
     C                     PARM           PTXT2            PTXT2
      * Length
     C                     PARM NMLNG     WQC3N   30       LENGTH
      * Translate
     C                     PARM '1'       WQD1    1
      * Trim
     C                     PARM '1'       WQE1    1
      * Wild
     C                     PARM '?'       WQF1    1
      * Result
     C                     PARM           WQG3N   30
     C           WQG3N     CABLT1         A010E
     C                     MOVE 'Y'       PTSTS
     C           A010E     ENDSR
      **********************************************************
     C           E010A     BEGSR
     C                     RETRN
     C           E010E     ENDSR
