//?Execute user function

// LCL.Order Number Alpha = CVTVAR(PAR.Order Number)
LCL.Order_Number_Alpha = CVTVAR(PAR.Order_Number);

// Execute function Eus %Scan for Order # US.
EXECUTE FUNCTION(Eus %Scan for Order # US) TYPE(EXCUSRSRC) ('PUWDUFR');
PARAMETER(PAR.Orders_on_PO_TXT);
PARAMETER(LCL.Order_Number_Alpha);
PARAMETER(PAR.Match_Found_USR);
{
 * Scan for Order Number in field 'Orders on PO - TXT' & Set Match Found Flag.
 d $string_data    s             30a   inz(*blank)
 d $order_number   s              7a   inz(*blank)
 d $pos            s              5u 0 inz(*zero)
 *
 /free
 $string_data = #INVU1;
 $order_number = #IE7TX;
 $pos = %scan ($order_number : $string_data);
 If $pos = 0;
 #OMLSC = 'N';
 Else;
 #OMLSC = 'Y';  // Order Found.
 EndIf;
 /end-free
}
