/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDHPUPR                                       */
/*  PROGRAM NAME:    Send Distribution Message                     */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     02/12/96                                      */
/*                                                                 */
/*  FUNCTION:        This CLP will send a message to the users'    */
/*                   in the distribution list created in Office.   */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&DSTLST &MSG)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DSTLST) TYPE(*CHAR) LEN(8)
             DCL        VAR(&MSG) TYPE(*CHAR) LEN(132)

/* Function Specific                                               */

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             SNDDST     TYPE(*MSG) TOUSRID((&DSTLST BIGBYTE)) +
                          DSTD('OFFICE MESSAGE') MSG(&MSG)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(CPF0000))
             GOTO       CMDLBL(ENDPGM)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 CPF0000:    SNDPGMMSG  MSG('Send Distribution Failed for List' +
                          *BCAT &DSTLST *BCAT 'and message' *BCAT &MSG)
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
