/*********************************************************************/
/* PROGRAM    : PPW8UPC                                              */
/* TITLE      : ADDLIBLE COMMAND                                     */
/* DATE       : 02/23/2006                                           */
/* Description: ADD OR REMOVE LIBRARY, MONITOR FOR ERROR             */
/* NAME       : LARA BUSER                                           */
/*                                                                   */
/* Parameters:                                                       */
/* Usage   Field       Field Description                             */
/* -----   ---------   ----------------------------------------------*/
/*   I     &LIBRARY    LIBRARY TO ADD                                */
/*   I     &ACTION     ACTION: A=ADD, R=REMOVE                       */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*-------------------------------------------------------------------*/
/*   Date     Pgmr     Description                                   */
/* -------- ---------- --------------------------------------------  */
/*                                                                   */
/*********************************************************************/
             PGM  PARM(&LIBRARY &ACTION)

/*-------------------------------------------------------------------*/
/* Declare Variables                                                 */
/*-------------------------------------------------------------------*/

             DCL  VAR(&LIBRARY) TYPE(*CHAR) LEN(10)
             DCL  VAR(&ACTION)  TYPE(*CHAR) LEN(1)

/*-------------------------------------------------------------------*/
/* ADD OR REMOVE LIBRARY                                             */
/*-------------------------------------------------------------------*/
             IF         COND(&ACTION *EQ 'A') THEN(DO)
             ADDLIBLE   LIB(&LIBRARY)
             MONMSG     MSGID(CPF0000)
             GOTO       CMDLBL(ENDPGM)
             ENDDO

             IF         COND(&ACTION *EQ 'R') THEN(DO)
             RMVLIBLE   LIB(&LIBRARY)
             MONMSG     MSGID(CPF0000)
             GOTO       CMDLBL(ENDPGM)
             ENDDO

/*-------------------------------------------------------------------*/
/* End of Program */
/*-------------------------------------------------------------------*/
 ENDPGM:     ENDPGM
