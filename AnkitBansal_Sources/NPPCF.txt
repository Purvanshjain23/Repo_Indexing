SELECT
  ALL       T01.A1BYDT, T01.A1LPNB, SUM(T01.A1CCPR), SUM((CDA0VA*-1))
  FROM      PRKFLIB/PKA1CPP T01 INNER JOIN
            PRKFLIB/PKCDCPP T02
  ON        A1HMNB = CDHMNB
    AND     A1BNNB = CDBNNB
    AND     A1BONB = CDBONB
    AND     A1CVNB = CDCVNB
    AND     A1B0DT = CDB0DT
  WHERE     A1B0DT >= 970923
    AND     A1B0DT <= 971105
    AND     A1KAST = 'I'
    AND     CDBQCD = 'NPPCI'
  GROUP BY  T01.A1BYDT, T01.A1LPNB
  ORDER BY  T01.A1BYDT ASC, T01.A1LPNB ASC
