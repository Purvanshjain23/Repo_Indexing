/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PMRKUPR                                       */
/*  PROGRAM NAME:    Print RAIL CAR RINS                          */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     02/22/08                                      */
/*                                                                 */
/*  FUNCTION:        THIS CLP IS SUBMITTED FROM AN DSP FILE      */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &COMPN  &RAILC  &PDEV &HOLD  +
         &SAVE     &COPYN)
                                                                       /**/
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPYN) TYPE(*DEC)
             DCL        VAR(&COMPN) TYPE(*DEC)
             DCL        VAR(&RAILC) TYPE(*CHAR) LEN(15)
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&COPY) TYPE(*DEC) LEN(2 0)

/* Function Specific                                               */
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)


/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&COMP) VALUE(&COMPN)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             OVRPRTF    FILE(PMR7PFR$) OUTQ(&PDEV) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

     CALL      PGM(PMR7PFR) PARM(' ' &COMP  &RAILC)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
