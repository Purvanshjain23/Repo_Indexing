/*                                                                 */
/*  PROGRAM NUMBER:  POI3UPR                                       */
/*  PROGRAM NAME:    PRINT CUSTOMER MARKETING ACCRUAL              */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     11/07/00                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS CALLED FROM A PROMPT FOR           */
/*                   PRINTING THE CUSTOMER MARKETING ACCRUAL       */
/*                   REPORT.                                       */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC 7/30/02 ADDED ACCRUAL CODE AS A PARM/SELECTOR              */
/*  LJB 6/18/07 ADDED ACCRUAL TYPE AS A PARM/SELECTOR              */
/*              REARRANGED INPUT PARMS                             */
/*  SLM 7/21/08 ADDED PASS IN PARENT # AND A/R CUSTOMER #          */
/*  SLM 7/21/08 ADD NEW WF SALES HISTORY WORK FILE &               */
/*              CREATE OF THE NEW WF SALES HISTORY                 */
/*              PRINT OF THE NEW WF SALES HISTORY REPORT           */
/*              ALSO ADD SENDING OF THE DETAIL BY EMAIL            */
/*              IF EMAIL THEN DO NOT CREATE THE REPORT             */
/*  SLM 9/26/13 TEST REPORT                                        */
/*              COMPLETED TESTING                                  */
/* SLM 12/27/13 TEST REPORT                                        */
/*******************************************************************/
             PGM        PARM(&COMP &FDTE &TDTE &SHPTO &PDEV &HOLD +
                          &SAVE &COPY &ACCRCODE &ACCRTYPE &LDGRACCR +
                          &RPTTYP &PARNUM &ARCNUM &WFSELC +
                          &EMAIL)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&COPY) TYPE(*DEC) LEN(15 5)

/* Function Specific                                               */
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&ADTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&FDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&SHPTO) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&LDGRACCR) TYPE(*CHAR) LEN(1)
             DCL        VAR(&RPTTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&ACCRCODE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&ACCRTYPE) TYPE(*CHAR) LEN(2)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)
             DCL        VAR(&PARNUM) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&ARCNUM) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&WFSELC) TYPE(*CHAR) LEN(1)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)
             DCL        VAR(&EMAILBLK) TYPE(*CHAR) LEN(50) VALUE(' ')
             DCL        VAR(&USRNAME) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TEXT)  TYPE(*CHAR) LEN(50)
             DCL        VAR(&CRTF) TYPE(*CHAR) LEN(1) VALUE('N')

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&COPY)
             CHGVAR     VAR(&ADTE) VALUE(&FDTE)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

/*-----------------------------------------------------------------*/
/*  Create WF Sales History based on selection                     */
/*-----------------------------------------------------------------*/

             CRTDUPOBJL OBJ(PMD1CPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJL OBJ(PMD1CPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJL OBJ(PMD1CPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
            CRTDUPOBJL OBJ(PMD1CPL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)

/*-----------------------------------------------------------------*/
/*  Build  WF Sales History report file                            */
/*-----------------------------------------------------------------*/
             CALL       PGM(PMWRXFR) PARM(' ' &WFSELC &PARNUM +
                          &ARCNUM &SHPTO &ADTE &COMP &FDTE &TDTE +
                          &ACCRCODE &ACCRTYPE &LDGRACCR &RPTTYP)

/* ***********************************************************************/
/* *******                  USED FOR TESTING ONLY                      ***/
/* ***********************************************************************/
GOTO ENDTEST
             CHKOBJ     OBJ(PRKTESTF/PMD1CPPSV) OBJTYPE(*FILE)
             MONMSG     MSGID(CPF9801) EXEC(CHGVAR VAR(&CRTF) +
                          VALUE('N'))

             IF         COND(&CRTF *EQ 'N') THEN(DO)
             CPYF       FROMFILE(PMD1CPP) TOFILE(PRKTESTF/PMD1CPPSV) +
                          MBROPT(*ADD) CRTFILE(*YES) FMTOPT(*MAP *DROP)
             ENDDO
             ELSE       CMD(DO)
             CPYF       FROMFILE(PMD1CPP) TOFILE(PRKTESTF/PMD1CPPSV) +
                          MBROPT(*REPLACE) CRTFILE(*NO) FMTOPT(*MAP +
                          *DROP)
             ENDDO
ENDTEST:
/* *******                  USED FOR TESTING ONLY                      ***/
/* ***********************************************************************/

             RTVUSRPRF  RTNUSRPRF(&USRNAME)

/*-----------------------------------------------------------------*/
/*  Print  WF Sales History report                                 */
/*-----------------------------------------------------------------*/
             IF         COND(&EMAIL = &EMAILBLK) THEN(DO)
/*-----------------------------------------------------------------*/
/*  OverRd WF Sales History report                                 */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMWQPFR$) OUTQ(&PDEV) +
                          DRAWER(*PRTF) COPIES(&COPIES) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)
             CALL       PGM(PMWQPFR) PARM(' ' &WFSELC &PARNUM +
                          &ARCNUM &SHPTO &COMP &FDTE &TDTE +
                          &ACCRCODE &ACCRTYPE &LDGRACCR &RPTTYP)

             SNDMSG     MSG('Print Accrual Listing has completed') +
                          TOUSR(&USRNAME)


             ENDDO
             ELSE       CMD(DO)
/*-----------------------------------------------------------------*/
/*  Send   WF Sales History report file                            */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&TEXT) VALUE('Accrual Listing by Ship +
                          date          ')
             EXECUTE    VIEW(*LIBL/WFSLSHSXLS) PCFMT(*XLS) +
                          REPLACE(*YES) RECIPIENT(&EMAIL) EMLMSG(&TEXT)


             SNDMSG     MSG('Sending of the Accrual List via +
                          Download file has completed') TOUSR(&USRNAME)

             ENDDO
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:     DLTOVR     FILE(*ALL)

             ENDPGM
