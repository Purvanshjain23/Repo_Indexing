/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION. CAUSE PRINTER OVERRIDE  */
/*              PROMPTING AND PASSES COMPANY, FROM DATE, AND THROUGH*/
/*              DATE.                                               */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT REPORTS                                       */
/*  DATE      - 05/17/95                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &PGM        - CL PROGRAM TO SUBMIT                      */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                        SCREEN)                                   */
/*     B    &OUTQ       - OUTPUT QUEUE                              */
/*     B    &HOLD       - HOLD STATUS                               */
/*     B    &SAVE       - SAVE STATUS                               */
/*     B    &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q     */
/*     B    &NIGHTQ     - NIGHT JOB QUEUE                           */
/*     B    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     B    &FRMDATN    - FROM DATE      (PASSED)                   */
/*     B    &THRDATN    - THRU DATE      (PASSED)                   */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     I    &RPTTYP     - REPORT TYPE    (PASSED)                   */
/*     I    10 WAREHOUSES                                           */
/*     I    &CARRIER    - CARRIER CODE                              */
/*     B    &SENDYN     - SEND EMAIL FLAG                           */
/*                                                                  */
/********************************************************************/
/*  MODIFIED:                                                       */
/*   RMC 8/20/01  ADDED FROM AND TO TIME PARMS                      */
/*   LJB 10/21/2008  ADDED 10 WAREHOUSE PARMS AND SELECT/OMIT FLAG  */
/*   LJB 04/04/2014  ADDED REQUESTED DELIVERY DATE                  */
/*   LJB 03/05/2015  ADDED CARRIER CODE                             */
/*   LJB 04/17/2015  E004076 ADDED SEND Y/N TO CONTROL EMAILING     */
/*                           ADDED OPTIONAL EMAIL ADDRESS           */
/********************************************************************/
 PDFJUPC:    PGM        PARM(&RETURN &APPLIC &PGM &DESC &OUTQ &HOLD +
                          &SAVE &NQSTAT &NIGHTQ &COMPNYN &DELIVDTN +
                          &FRMDATN &THRDATN &COPYN &CARRIER &RPTTYP +
                          &FRMTIMN &THRTIMN &WHSE1 &WHSE2 &WHSE3 +
                          &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 &WHSE9 +
                          &WHSE10 &SELOMIT &EMAIL &SENDYN)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &PGM        *CHAR   (10)          /* CL to Submit              */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    VAR(&PAPER) TYPE(*CHAR) LEN(1) VALUE('L')
                                                 /* LEGAL, LETTER */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &FRMDATN    *DEC                  /* FROM DATE      (Passed)   */
        DCL    &THRDATN    *DEC                  /* THROUGH DATE   (Passed)   */
        DCL    &DELIVDTN   *DEC
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &RPTTYP     *CHAR   (1)           /* Report Type               */
        DCL    &FRMTIMN    *DEC                  /* FROM TIME      (Passed)   */
        DCL    &THRTIMN    *DEC                  /* THROUGH TIME   (Passed)   */
        DCL    &SELOMIT    *CHAR   (1)
        DCL    &WHSE1      *CHAR   (3)
        DCL    &WHSE2      *CHAR   (3)
        DCL    &WHSE3      *CHAR   (3)
        DCL    &WHSE4      *CHAR   (3)
        DCL    &WHSE5      *CHAR   (3)
        DCL    &WHSE6      *CHAR   (3)
        DCL    &WHSE7      *CHAR   (3)
        DCL    &WHSE8      *CHAR   (3)
        DCL    &WHSE9      *CHAR   (3)
        DCL    &WHSE10     *CHAR   (3)
        DCL    &CARRIER    *CHAR   (3)           /* CARRIER CODE              */
        DCL    &SENDYN     *CHAR   (1)           /* SEND YES/NO               */
        DCL    &EMAIL      *CHAR   (50)
        DCL    &EMAIL1     *CHAR   (25)
        DCL    &EMAIL2     *CHAR   (25)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/
             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&EMAIL1) VALUE(%SST(&EMAIL 1 25))
             CHGVAR     VAR(&EMAIL2) VALUE(%SST(&EMAIL 26 25))

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

 /*****      GOTO TEST          TEST *****/
/********************************************************************/
/*   CALL PROMPT PROGRAM                                            */
/********************************************************************/
             CALL       PGM(PDV8PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC &PAPER)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM                                        */
/********************************************************************/
 SUBMIT:     SBMJOB     CMD(CALL PGM(PDX9CLP) PARM(&COMPNYN &FRMDATN +
                          &THRDATN &RPTTYP &FRMTIMN &THRTIMN &OUTQ +
                          &HOLD &SAVE &COPYN &NIGHTQ &WHSE1 &WHSE2 +
                          &WHSE3 &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 +
                          &WHSE9 &WHSE10 &SELOMIT &DELIVDTN +
                          &CARRIER &SENDYN &EMAIL1 &EMAIL2)) +
                          JOB(DLYHOLDOVR) JOBQ(&NIGHTQ)
             GOTO       CMDLBL(END)

TEST:
             CHGVAR     VAR(&HOLD) VALUE(*YES)
             CHGVAR     VAR(&SAVE) VALUE(*YES)
             CHGVAR     VAR(&COPY) VALUE(1)
             CALL       PGM(PDV8PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC &PAPER)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

             CALL       PGM(PDX9CLP) PARM(&COMPNYN &FRMDATN &THRDATN +
                          &RPTTYP &FRMTIMN &THRTIMN &OUTQ &HOLD +
                          &SAVE &COPYN &NIGHTQ &WHSE1 &WHSE2 &WHSE3 +
                          &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 &WHSE9 +
                          &WHSE10 &SELOMIT &DELIVDTN &CARRIER +
                          &SENDYN &EMAIL1 &EMAIL2)
             GOTO       CMDLBL(END)

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/
 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
