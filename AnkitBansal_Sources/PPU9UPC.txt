/********************************************************************/
/*                                                                  */
/*  PPU9UPC -   CALL CUSTOMER/COMPANY TRAFFIC INFORMATION REPORT    */
/*                                                                  */
/*  SYNOPSIS -  PASSES THE COMPANY NUMBER                           */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PPU9UPC                                             */
/*  DATE      - 02/07/97                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &PGM        - PROGRAM TO CALL                           */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION                 */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*     B    &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*     B    &COMPNYN    - COMPANY NUMBER (PASSED) AS 15.5           */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*                                                                  */
/********************************************************************/
/* MAINTENANCE                                                      */
/* 07/27/06 LJB L060727A PASS SALESPERSON AS A PARM TO THE PRINT    */
/*                       PROGRAM.                                   */
/* 08/09/06 RMC ADDED RECORD STS PARM                              */
/********************************************************************/

 PPU9UPC:    PGM        PARM(&RETURN &PGM &DESC &OUTQ &HOLD &SAVE +
                          &NQSTAT &NIGHTQ &COMPNYN &COPYN &SALESP +
                          &RCDSTS)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &PGM        *CHAR   (10)          /* CALLING PGM NAME          */
        DCL    &PRTF       *CHAR   (10)          /* CALLING PGM NAME          */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (1)           /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &SALESP     *CHAR   (3)           /* SALESPERSON CODE          */
        DCL    &RCDSTS     *CHAR   (1)           /* RECORD STS                */
/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMP       *DEC    (3 0)         /* COMPANY                   */
        DCL    &COMMAND    *CHAR   (512)

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             CHGVAR     VAR(&PRTF) VALUE(&PGM *TCAT '$')
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PRTF) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
/*           SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPU8PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE) */

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/***** CALL:       CALL       PGM(&PGM) PARM(' ' &COMP)         *****/

      CALL       PGM(&PGM) PARM(' ' &COMP &SALESP &RCDSTS)

 END:        ENDPGM
