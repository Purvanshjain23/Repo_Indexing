/*                                                                 */
/*  FUNCTION:        THIS CL IS CALLED BY THE PAYMENT POST PROGRAM */
/*                   (PKTFXFR) AND THE PRINT CARC DTL & SUM CLP    */
/*                   (PKXCUPR).  THE PURPOSE OF THE CL IS TO       */
/*                   CHANGE THE OVERRIDE ON THE PRINT FILE FOR     */
/*                   FAXING OR FOR REPORTING.                      */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  SLM     03/21/97 DO NOT ALLOW PROGRAMERS TO FAST FAX ON        */
/*          EITHER LITTBIT  OR BIGBYTE                             */
/*                                                                 */
/*  RMC     10/10/97  CHANGED TO JUST FAX THE CARC DTL/SUMM -      */
/*                    A DIFFERENT CL IS USED TO PRINT THE RPTS.    */
/*                    &POST = P IF THIS IS CALLED FROM THE POST,   */
/*                    &POST = S IF THIS IS CALLED FROM THE PROMPT  */
/*  RMC     05/28/03  ADDED PARM TO PRINT SORT LOSS                */
/*  RMC     05/28/03  ADDED PARM TO PRINT SORT LOSS                */
/*  SLM     04/16/04  TFM - REPLACE PKMLPFR WITH PLEMPFR           */
/*  SLM     06/21/07  REMOVE * CODE ON PKMLPFR ALSO                */
/*                    HAVE THE FAXQ USE *LIBL AND DO NOT NEED      */
/*                    THE *QPGMR, BUT WILL GO AHEAD AND PASS ON    */
/*                    UNTIL PKUQUPR.  PKUQU WILL RTV AGAIN         */
/*                    SO WE DO NOT HAVE SOME CALL PKUQUPR FROM     */
/*                    THE COMMAND LINE                             */
/*                    THE FAXING WILL NOT BE DONE FOR A *QPRMR     */
/*                    BUT WE HAVE NEW RECORDS IN THE COMPANY       */
/*                    VALUE THAT WILL ALLOW US TO SEE FAXING       */
/*                    BUY USING THE COMPANY VALUE AND PLACE        */
/*                    THE IS COMPKANY (SEABOARD) TO USE OUR        */
/*                    OWN FAX NUMBERS.                             */
/*                    SEE HPEFAX...                                */
/*                    PRODUCTION THESE RECORDS WILL NOT            */
/*                    BE SLECTED DUE TO THE FACT THE VALUE         */
/*                    TO RUN A FAXING USING OUR NUMBERS WOULD      */
/*                    SET TO THE VALUE OF NO                       */
/*  SLM     06/22/07  CAN NOT USE GRPAUTH USE USRCLS *PGMR         */
/*  SLM     02/22/07  REMOVE DMP OF CL ....NOT NEEDED ANYMORE      */
/*******************************************************************/
/*           PGM        PARM(&RTN &XTATSQ &LIVCRC &SYSVAL +
                             &PDEV1 &HOLD &SAVE &XCOPIES +
                             &XPRDVN# &XPRDLCVN# &FXPROD &FXPRDLC +
                             &IGNFAX)                            */
             PGM        PARM(&RTN &POST &LIVCRC &SORT &SYSVAL)
/* INPUT PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&SYSVAL) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)
             DCL        VAR(&POST) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SORT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&LIVCRC) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
/*******     DCL        VAR(&GRPAUT) TYPE(*CHAR) LEN(10)  ********/
             DCL        VAR(&USRCLS) TYPE(*CHAR) LEN(10)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&LPP) VALUE(&SYSVAL)


/*********   RTVUSRPRF  GRPPRF(&GRPAUT)               ************/
             RTVUSRPRF  USRCLS(&USRCLS)

/* ----------------------------------------------------------------  */
/* WHEN TESTING ON LITTLBIT  OR BIGBYTE PROGRAMERS WILL     */
/* USE THE FAXING VERIFICATION ROUTINE - SEE PKUQUPR CLP    */
/* ----------------------------------------------------------------  */
/*           IF         COND(&GRPAUT *EQ 'QPGMR') THEN(DO)           */
/*           OVRPRTF    FILE(PLEMPFR$) OUTQ(QPRINT2) FORMTYPE(FAX) + */
/*                        COPIES(1) HOLD(*NO) SAVE(*NO)              */
/*           ENDDO                                                   */
/*                                                                   */
/* IF THE REPORT IS GOING TO BE FAXED THEN SETUP THE                 */
/* OVERRIDE TO THE FAST FAX QUEUE                                    */
/*           IF         COND(&GRPAUT *NE 'QPGMR') THEN(DO)           */
/*                                                                   */
/*           OVRPRTF    FILE(PLEMPFR$) OUTQ(PRKFLIB/FAXQ)            */
/*                                                                   */
/*           ENDDO                                                   */
/* ----------------------------------------------------------------  */

             IF         COND(&USRCLS *EQ '*PGMR') THEN(DO)
             OVRPRTF    FILE(PLEMPFR$) OUTQ(*LIBL/FAXQ) FORMTYPE(FAX)
             ENDDO

             IF         COND(&USRCLS *NE '*PGMR') THEN(DO)
             OVRPRTF    FILE(PLEMPFR$) OUTQ(*LIBL/FAXQ)
             ENDDO

/* FAX CARCASS SUMMARY/DETAIL                                */


             CALL       PGM(PKMPXFR) PARM(&RTN &USRCLS &POST &LIVCRC +
                   &SORT  &LPP)


             DLTOVR *ALL

 END:
 /********** DMPCLPGM     *******************/

             ENDPGM
