/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDBOUPR                                       */
/*  PROGRAM NAME:    Print Item Default Listing                    */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     10/30/95                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from Work With Item Defaults                  */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &PGM +
                          &PGMPRTF &APP &CTHWGT &XITMCLS &ITMTYP &ITMBCF +
                          &ITMVAR &CSERDY &CRDSTS &SCDSTS &RCDSTS)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGMPRTF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(3)

/* Function Specific                                               */
             DCL        VAR(&CTHWGT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&XITMCLS) TYPE(*CHAR) LEN(15)
             DCL        VAR(&ITMTYP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&ITMBCF) TYPE(*CHAR) LEN(2)
             DCL        VAR(&ITMVAR) TYPE(*CHAR) LEN(1)
             DCL        VAR(&CSERDY) TYPE(*CHAR) LEN(1)
             DCL        VAR(&CRDSTS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SCDSTS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&RCDSTS) TYPE(*CHAR) LEN(1)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&ITMCLS) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&ITMCLS) VALUE(&XITMCLS)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGMPRTF) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(&PGM) PARM(' ' &CTHWGT &ITMCLS &ITMTYP +
                          &ITMBCF &ITMVAR &CSERDY &CRDSTS &SCDSTS +
                          &RCDSTS)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
