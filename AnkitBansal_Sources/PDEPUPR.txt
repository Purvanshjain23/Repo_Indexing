/*   Download Customer Master Data and Aging information to the CMS */
/*   Credit Management System   */
PGM
          /* Clear and Delete Existing Physical Files   */
             CLRPFM     FILE(PDKDREP)
             MONMSG     MSGID(CPF9999)
             CLRPFM     FILE(PDKCREP)
             MONMSG     MSGID(CPF9999)
             CLRPFM     FILE(PDLAREP)
             MONMSG     MSGID(CPF9999)
          /* Call the Download files from A/R Summary and Customer */
          /*  */
 PROGRAM:    CALL       PGM(PDBXXFR) PARM(' ') /* CALL DOWNLOAD DATA */
          /* Create the files used for the Download  */
          /*  */
             CPYF       FROMFILE(PDKDREP) TOFILE(CMSNUMS) +
                          MBROPT(*REPLACE)
             CPYF       FROMFILE(PDKCREP) TOFILE(CMSCUST) +
                          MBROPT(*REPLACE)
             CPYF       FROMFILE(PDLAREP) TOFILE(CMSCREDIT) +
                          MBROPT(*REPLACE)
             ENDPGM
