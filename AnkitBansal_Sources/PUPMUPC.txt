/**************************************************************************/
/*  PROGRAM:       PUPMUPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO PS PROD ITEM INVENTORY (POCWCPP) FILE.*/
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   02/20/2020                                             */
/*  PROJECT:       S16166                                                 */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY PHYSICAL FILE TRIGGER TO THE PS PROD   */
/*                 ITEM INVENTORY (POCWCPP) FILE FOR ALL TRIGGER          */
/*                 DATABASE EVENTS:                                       */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 03/03/23 DN   W122889 ADDED MONITOR MESSAGE FOR 'RMVPFTRG' TO PREVENT  */
/*                       ERROR.                                           */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/POCWCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PUPOUPR) +
                          RPLTRG(*YES) TRG(SEB_POCWCPP_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/POCWCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PUPOUPR) +
                          RPLTRG(*YES) TRG(SEB_POCWCPP_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/POCWCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PUPOUPR) +
                          RPLTRG(*YES) TRG(SEB_POCWCPP_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/POCWCPP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
