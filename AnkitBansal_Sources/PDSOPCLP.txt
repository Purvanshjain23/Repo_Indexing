 PDRVPCLP:   PGM        PARM(&COMPL &LOADL)

        DCL     &COMPL       *DEC
        DCL     &COMP        *DEC     (3 0)      /*  COMPANY #  */
        DCL     &LOADL       *DEC
        DCL     &LOAD        *DEC     (7 0)      /*  COMPANY #  */
        DCL     &RETURN      *CHAR    7

/********************************************************************/
/* load Company                                                     */
/********************************************************************/
             CHGVAR     VAR(&COMP) VALUE(&COMPL)
             CHGVAR     VAR(&LOAD) VALUE(&LOADL)

/*********************************************************************/
/*   Call DOWNLOAD SALES ORDERS FILES                                */
/*********************************************************************/

             CALL       PGM(PDFKXFR) PARM(&RETURN &COMP &LOAD) /* +
                          &load */

 END:        ENDPGM
