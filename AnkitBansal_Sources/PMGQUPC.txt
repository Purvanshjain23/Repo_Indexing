/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PMGQUPC                                       */
/*  PROGRAM NAME:    Print AGED NOT SOLD REPORT                   */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     05/04/07                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted FROM WW SCHEDULING ORDERS */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &COMPA &WHS &PDEV &HOLD &SAVE &XCOPIES)
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*DEC)

/* Function Specific                                               */
             DCL        VAR(&COMPA)   TYPE(*DEC)
             DCL        VAR(&COMP)   TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&WHS)    TYPE(*CHAR) LEN(3)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&COMPA)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             OVRPRTF    FILE(PMGPPFR$) OUTQ(&PDEV) COPIES(&COPIES) +
                          HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(PMGPPFR) PARM(' ' &COMP &WHS)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
