/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING TF MEAT COST GROUP LISTING   */
/*             THIS PROGRAM IS SUBMITTED FROM A GENERIC SUBMIT      */
/*             PROGRAM: PDALUPR.                                    */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      - 05/03/2007                                          */
/*  PROGRAMMER- LARA BUSER                                          */
/*                                                                  */
/* PARAMETERS RECEIVED:                                             */
/*    &TFMC          -  MEAT COST GROUP                             */
/*    &APPLIC        -  APPLICATION CODE                            */
/*    &STATUS        -  RECORD STATUS                               */
/*    &OUTQ          -  OUTPUT QUEUE                                */
/*    &HOLD          -  HOLD STATUS                                 */
/*    &SAVE          -  SAVE STATUS                                 */
/*    &COPYN         -  PASSED NUMBER OF COPIES                     */
/*    &NIGHTQ        -  NIGHT JOB QUEUE                             */
/*                                                                  */
/********************************************************************/
 PMGNUPC:    PGM        PARM(&TFMC &STATUS &APPLIC &OUTQ &HOLD &SAVE +
                          &COPYN &NIGHTQ)

        DCL    &TFMC        *CHAR  (04)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &STATUS      *CHAR  (1)
        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             OVRPRTF    FILE(PMGLPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

 CALL:       CALL       PGM(PMGLPFR) PARM(' ' &TFMC &STATUS)

 ENDPGM:     ENDPGM
