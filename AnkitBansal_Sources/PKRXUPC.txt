/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKRXUPC                                       */
/*  PROGRAM NAME:    PRINT HOT SCALE UPLOAD VALUE                  */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     07/23/99                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED FROM THE PROMT FOR       */
/*                   PRINTING THE HOT SCALE UPLOAD VALUE           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*******************************************************************/
             PGM        PARM(&CO &PROD &XFDTE &XTDTE &PTYP &INCM +
                          &INCS &INCR &TTOTYPE &TATTOO &EXCEP &RNK +
                          &PGM &PRTF &PDEV &HOLD &SAVE &XCOPIES &OMS)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6)
             DCL        VAR(&PTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&INCM) TYPE(*CHAR) LEN(1)
             DCL        VAR(&INCS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&INCR) TYPE(*CHAR) LEN(1)
             DCL        VAR(&TTOTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&EXCEP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&RNK) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PRTF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&XFDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTDTE) TYPE(*CHAR) LEN(7)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&FDTE) TYPE(*DEC) LEN(7)
             DCL        VAR(&TDTE) TYPE(*DEC) LEN(7)
             DCL        VAR(&TATTOO) TYPE(*DEC) LEN(5)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FDTE) VALUE(&XFDTE)
             CHGVAR     VAR(&TDTE) VALUE(&XTDTE)

/*******************************************************************/
/* PRINT REPORT                                                   */
/*******************************************************************/

             SBDPRTOVR  CMDVAR(&CMD) FILE(&PRTF) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

             CALL       PGM(&PGM) PARM(' ' &CO &FDTE &PROD &TDTE +
                          &PTYP &INCM &INCS &INCR &TATTOO &TTOTYPE +
                          &EXCEP &RNK &LPP)

/*******************************************************************/
ENDPGM:
/*******************************************************************/
             DLTOVR *ALL
             ENDPGM
