//?***********   Send Pen and Tattoo number in big image *********
//?GET DEVICE FOR BIG CHARS
// Call program Rtv Company Values    XF.
CALL PROGRAM(Rtv Company Values    XF) ('PDGKXFR');
PARAMETER(PAR.BOH_Company_Number);
PARAMETER('HPEBIGCHR');
PARAMETER(LCL.System_Value_Alpha_USR);

// LCL.Device Id Usr = LCL.System Value Alpha    USR
LCL.Device_Id_Usr = LCL.System_Value_Alpha_USR;

// LCL.Tattoo Nbr 4          USR = PAR.TH Tattoo Number
LCL.Tattoo_Nbr_4_USR = PAR.TH_Tattoo_Number;

// LCL.Tattoo Nbr Alpha 4    USR = CVTVAR(LCL.Tattoo Nbr 4          USR)
LCL.Tattoo_Nbr_Alpha_4_USR = CVTVAR(LCL.Tattoo_Nbr_4_USR);

//?Send big pen numbers and tattoo numbers
// Execute command - 'Snd Big Pen Big Tattoo'
EXECUTE COMMAND(PRK0718);
PARAMETER(LCL.Device_Id_Usr);
PARAMETER(LCL.Tattoo_Nbr_Alpha_4_USR);
PARAMETER(PAR.Pen_Number);
