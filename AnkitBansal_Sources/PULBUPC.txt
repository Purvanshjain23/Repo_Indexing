/**************************************************************************/
/*  PROGRAM:       PULBUPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO PACKAGING BOM (POAZCPP) FILE.         */
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   01/22/2019                                             */
/*  PROJECT:       R14394 - MRP REPORT                                    */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY/REMOVE PHYSICAL FILE TRIGGER TO THE    */
/*                 PACKAGING BOM (POAZCPP) FILE FOR ALL TRIGGER           */
/*                 DATABASE EVENTS:                                       */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 03/03/23 DN   W122889 ADDED MONITOR MESSAGE FOR 'RMVPFTRG' TO PREVENT  */
/*                       ERROR.                                           */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/POAZCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PULAUPR) +
                          RPLTRG(*YES) TRG(SEB_POAZCPP_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/POAZCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PULAUPR) +
                          RPLTRG(*YES) TRG(SEB_POAZCPP_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/POAZCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PULAUPR) +
                          RPLTRG(*YES) TRG(SEB_POAZCPP_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/POAZCPP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
