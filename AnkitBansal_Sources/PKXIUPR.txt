/*                                                                 */
/*  PROGRAM NUMBER:  PKXIUPR                                       */
/*  PROGRAM NAME:    PRINT DAILY DETAIL                            */
/*  PROGRAMMER:      ROSE MARTIN                                   */
/*  CREATE DATE:     11/09/95                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT PRINT DAILY DETAIL.               */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES +
                          &APP  &XCO &XKDTE &XCUTDT)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&APP) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XKDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XCUTDT) TYPE(*CHAR) LEN(7)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3)
             DCL        VAR(&KDTE) TYPE(*DEC) LEN(7)
             DCL        VAR(&CUTDT) TYPE(*DEC) LEN(7)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&APP)
             CHGVAR     VAR(&CO) VALUE(&XCO)
             CHGVAR     VAR(&KDTE) VALUE(&XKDTE)
             CHGVAR     VAR(&CUTDT) VALUE(&XCUTDT)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKXFPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKXFPFR) PARM(' ' &CO &KDTE &LPP)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PDDAPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PDDAPFR) PARM(' ' &CO &CUTDT &LPP)
             DLTOVR *ALL
             ENDPGM
