/*‚Desc   : PVJSUPC - Driver Program for PFSINVREST Job                       */
/*‚       : Entry is written in PBBWUPC file --> Trigger Calls PBP2UPR Program*/
/*‚       : (PBP2UPR do a 'SBMJOBJS PVJSUPC' with CmpID and PFS_Trans_Dt as   */
/*‚       : Parameters and then call is made to Job scheduler job PFSINVREST  */
/*‚Note   : PFSINVREST job will be in *HELD status all the time but will be   */
/*‚       : triggered from Carlisle/SQL Server side when an entry is written  */
/*‚       : in PBBWCPP file with correct company and transaction date         */
/*‚Caller : PFSINVREST Job Or An entry in PBBWCPP file                        */
/*‚Issue# : 1036470                                                           */
/*‚Date   : 06/02/2025                                                        */
/*‚****************************************************************************/
             PGM        PARM(&CMPID &PFSTRANSDT)

             DCL        VAR(&CMPID) TYPE(*CHAR) LEN(3) /* Company ID*/
             DCL        VAR(&PFSTRANSDT) TYPE(*CHAR) LEN(8) /* PFS Trans Date*/

             /*‚If Job is triggered without correct parameters, then exit*/
             IF         COND(&CMPID *EQ ' ') THEN(RETURN)

             /*‚AUTOMATED INVENTORY RESTORATION  */
             CALL       PGM(PBNMXFR) PARM(' ' &CMPID)

             /*‚UPDATE PFS INVENTORY REST TRG STS TO P=PROCESSED */
             CALL       PGM(PBP3XFR) PARM(' ' &CMPID &PFSTRANSDT)
             MONMSG     MSGID(CPF0000)

             ENDPGM
