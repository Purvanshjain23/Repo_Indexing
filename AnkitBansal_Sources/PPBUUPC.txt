/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PPBUUPC                                      */
/*  PROGRAM NAME:    Print CUSTOMER ACCRUAL LIST BY SALES ROUTE     */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     11/20/03                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is PRINTS THE CUSTOMER ACCRUAL       */
/*                   LIST BY SALES ROUTE                          */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
       PGM PARM(&RTN &COL &SLS &RPTSEQ &RPT &PDEV &HOLD &SAVE &XCOPIES)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&COL) TYPE(*DEC)      /*COMPANY  */
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)     /*COMPANY  */
             DCL        VAR(&SLS) TYPE(*CHAR) LEN(3)    /* SALES ROUTE */
             DCL        VAR(&RPTSEQ) TYPE(*CHAR) LEN(1)    /*REPORT SEQ  */
             DCL        VAR(&RPT) TYPE(*CHAR) LEN(1)    /*DETAIL/SUMM */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*DEC)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&NIGHTQ) TYPE(*CHAR) LEN(1)

/* Function Specific                                               */

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&CO) VALUE(&COL)
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
IF COND(&RPTSEQ *EQ '1') THEN(DO)
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(POLHPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(POLHPFR) PARM(' ')
        ENDDO
IF COND(&RPTSEQ *EQ '2') THEN(DO)
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPBSPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PPBSPFR) PARM(' ' &CO &SLS &RPT)
        ENDDO
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
