/********************************************************************/
/*                                                                  */
/*  PROGRAM   - PPY2UPC - PRT ON TIME REASON LISTUP                 */
/*  SYNOPSIS  - OVERRIDE PRINTER FILE TO PASSED IN PARAMETERS       */
/*              PRINT A LIST OF THE ON TIME REASON CODES            */
/*                                                                  */
/*  SYSTEM    - TRAFFIC                                             */
/*  DATE      - 06/21/2006                                          */
/*            - LARA BUSER                                          */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION                               */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       - OUTPUT QUEUE                              */
/*     B    &HOLD       - HOLD STATUS                               */
/*     B    &SAVE       - SAVE STATUS                               */
/*     B    &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q     */
/*     B    &NIGHTQ     - NIGHT JOB QUEUE                           */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/*------------------------------------------------------------------*/
/*                                                                  */
/********************************************************************/
 PPXZUPC:    PGM        PARM(&RETURN &APPLIC &DESC &OUTQ &HOLD &SAVE +
                          &NQSTAT &NIGHTQ &COPYN)

/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)   /* RETURN CODE              */
        DCL    &APPLIC     *CHAR   (6)   /* APPLICATION              */
        DCL    &DESC       *CHAR   (25)  /* PRT PGM DESCRIPTION      */
        DCL    &OUTQ       *CHAR   (10)  /* OUTPUT QUEUE             */
        DCL    &HOLD       *CHAR   (4)   /* HOLD STATUS              */
        DCL    &SAVE       *CHAR   (4)   /* SAVE STATUS              */
        DCL    &NQSTAT     *CHAR   (1)   /* IF MUST RUN IN NIGHT Q   */
        DCL    &NIGHTQ     *CHAR   (10)  /* NIGHT JOB QUEUE          */
        DCL    &COPYN      *DEC          /* NO. OF COPIES  (PASSED)  */
        DCL    &COPY       *DEC    (2 0) /* NO. OF COPIES            */

/*------------------------------------------------------------------*/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/*------------------------------------------------------------------*/
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/*------------------------------------------------------------------*/
/* OVERRIDE PRINTER FILE TO PARMS PASSED IN FROM CALLING PROGRAM    */
/*------------------------------------------------------------------*/
             OVRPRTF    FILE(PPYTPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

/********************************************************************/
/* CALL THE INVOICE EXCEPTION REPORT PROGRAM                        */
/********************************************************************/
 CALL:       CALL       PGM(PPYTPFR) PARM(&RETURN)

 END:        ENDPGM
