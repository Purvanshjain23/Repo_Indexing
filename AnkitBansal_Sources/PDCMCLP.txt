/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING ITEM ACCRUAL CODES LISTING   */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  4/03/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &ITEML         -  PASSED ITEM CODE                       */
/*         &RCDSTS        -  RECORD STATUS                          */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/********************************************************************/
/* RMC 08/30/07 DONT THINK THIS IS USED -- BUT TOOK COMPANY OFF OF  */
/*              CALL TO PRTF                                       */
/*                                                                  */
/********************************************************************/
 PDCMCLP:    PGM        PARM(&COMPNYN &ITEML &RCDSTS &APPLIC &OUTQ +
                             &HOLD &SAVE &COPYN &NIGHTQ)

        DCL    &COMPNYN     *DEC
        DCL    &ITEML       *DEC
        DCL    &RCDSTS      *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &ITEM        *DEC   (7 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&ITEM)  VALUE(&ITEML)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDCMPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDCMPFR) PARM(' '  &ITEM &RCDSTS)

 ENDPGM:     ENDPGM
