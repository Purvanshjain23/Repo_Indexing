/**************************************************************************/
/*  PROGRAM:       PVHZUPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO CUSTOMER ACCRUAL ITEM (OMHPREP) FILE. */
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   10/28/2024                                             */
/*  PROJECT:       SR 3021581                                             */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY PHYSICAL FILE TRIGGER TO THE CUSTOMER  */
/*                 ACCRUAL ITEM (OMHPREP) FILE FOR ALL TRIGGER            */
/*                 DATABASE EVENTS:                                       */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/*                                                                        */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/OMHPREP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PVH1UPR) +
                          RPLTRG(*YES) +
                          TRG(TRG_CUSTOMER_ACCRUAL_ITEM_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/OMHPREP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PVH1UPR) +
                          RPLTRG(*YES) +
                          TRG(TRG_CUSTOMER_ACCRUAL_ITEM_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/OMHPREP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PVH1UPR) +
                          RPLTRG(*YES) +
                          TRG(TRG_CUSTOMER_ACCRUAL_ITEM_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/OMHPREP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
