/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  ASPURGES                                      */
/*  PROGRAM NAME:    SBM THE CLPS TO PURGE THE ACCUSORT FILES      */
/*  PROGRAMMER:                                                    */
/*  CREATE DATE:     11/  /99                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS CALLED FROM THE JOB SCHEDULE       */
/*                   TO PURGE THE ACCUSORT FILES.  THE CURRENT     */
/*                   FILE IS COPIED TO ITS FIRST MEMBER AND THE    */
/*                   CURRENT FILE IS CLEARED.                      */
/*                                                                 */
/* **** NOTE ****                                                  */
/*  WHEN ADDING A FILE TO THIS ROUTINE, PRIOR TO EXECUTING         */
/*    THE MBR(&FILE1) MUST BE CREATED FOR THE FILE.                */
/*    IE. FOR ASJ9CPP--ADDPFM ASJ9CP1                              */
/*  EG. CRTDUPOBJ  PKDLCPP NEWOBJ(PKDLCPPPRG)   DATA(*NO)          */
/*      CHGPF  PKDLCPPPRG  MAXMBRS(*NOMAX)                         */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&LIB)
             DCL        VAR(&LIB) TYPE(*CHAR) LEN(10)
 /*SHIPPING FILES PURGE*/
             CPYF       FROMFILE(&LIB/ASJSCPP) TOFILE(ASJSCPP) +
                          FROMMBR(PDJSCPP) TOMBR(ASJSCP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASJSCPP) MBR(PDJSCPP)
 /*                                                              */
             CPYF       FROMFILE(&LIB/ASJUCPP) TOFILE(ASJUCPP) +
                          FROMMBR(ASJUCPP) TOMBR(ASJUCP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASJUCPP) MBR(ASJUCPP)
 /*                                                              */
             CPYF       FROMFILE(&LIB/ASJ9CPP) TOFILE(ASJ9CPP) +
                          FROMMBR(ASJ9CPP) TOMBR(ASJ9CP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASJ9CPP) MBR(ASJ9CPP)
 /*                                                              */
 /*INVENTORY FILES PURGE*/
 /*                                                              */
             CPYF       FROMFILE(&LIB/ASJTCPP) TOFILE(ASJTCPP) +
                          FROMMBR(PDJTCPP) TOMBR(ASJTCP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASJTCPP) MBR(PDJTCPP)
 /*                                                              */
             CPYF       FROMFILE(&LIB/ASJVCPP) TOFILE(ASJVCPP) +
                          FROMMBR(PDJVCPP) TOMBR(ASJVCP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASJVCPP) MBR(PDJVCPP)
 /*                                                              */
             CPYF       FROMFILE(&LIB/ASKACPP) TOFILE(ASKACPP) +
                          FROMMBR(PDKACPP) TOMBR(ASKACP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASKACPP) MBR(PDKACPP)
 /*                                                              */
             CPYF       FROMFILE(&LIB/ASKWCPP) TOFILE(ASKWCPP) +
                          FROMMBR(PDKWCPP) TOMBR(ASKWCP1) +
                          MBROPT(*REPLACE)
             MONMSG     MSGID(CPF2817) EXEC(SNDPGMMSG MSG('One of +
                          the Files were not copied because the +
                          source file was empty.'))
             CLRPFM     FILE(ASKWCPP) MBR(PDKWCPP)
             ENDPGM
