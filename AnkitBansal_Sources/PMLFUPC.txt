PGM PARM(&CRCDE)

DCL &CRCDE *CHAR 3  /* Accrual Code */
DCL &LIBRARY *CHAR 10  /* Override Library  */

/* Override to DFDATA for first run */
CHGVAR &LIBRARY 'DFDATA'
OVRDBF     FILE(REBATECU) TOFILE(&LIBRARY/REBATECU)
OVRDBF     FILE(REBATECX) TOFILE(&LIBRARY/REBATECX)

/* Call EUP Bld Customer Rebate  */
CALL PMLEUPR PARM(&CRCDE &LIBRARY)

/* Delete overrides */
DLTOVR  REBATECU
DLTOVR  REBATECX

/* Override to JRDATA for second run */
CHGVAR &LIBRARY 'JRDATA'
OVRDBF     FILE(REBATECU) TOFILE(&LIBRARY/REBATECU)
OVRDBF     FILE(REBATECX) TOFILE(&LIBRARY/REBATECX)

/* Call EUP Bld Customer Rebate   */
CALL PMLEUPR PARM(&CRCDE &LIBRARY)

/* Delete overrides */
DLTOVR  REBATECU
DLTOVR  REBATECX
ENDPGM
