      *----------------------------------------------------------------

     C     $fmdy         begsr

     C                   z-add     0             d1fwemdy
     C                   z-add     0             d1fyr
     C                   z-add     0             d1fwk
     C                   z-add     0             d1fpe
      *
     C                   select
     C                   when      d1fmdy = 0
     C                   seton                                        30
     C                   if        error < maxmsg
     C                   add       1             error
     C                   eval      mgid(e) = 'HS09001'
     C                   endif
      *
     C                   when      d1fmdy < 0
     C                   seton                                        30
     C                   if        error < maxmsg
     C                   add       1             error
     C                   eval      mgid(e) = 'HS09000'
     C                   endif
     C                   other
      *
     C     *mdy          test(d)                 d1fmdy                 92
     C                   if        *in92 = *on                                  If bad date
     C                   seton                                        30
     C                   if        error < maxmsg
     C                   add       1             error
     C                   eval      mgid(e) = 'HS09004'
     C                   endif
     C                   else
      *
      * Flip the date from MMDDYY to CCYYMMDD.
      *
     C     *mdy          move      d1fmdy        wkisodate
     C                   move      wkisodate     wkdt
      *
      * It is an error if the Date is on/before June 28, 2008.
      * (A major rewrite in July 2008 made prior data invalid for this report.)
      *
     C                   if        wkdt <= 20080628                             If too old
     C                   seton                                        30
     C                   if        error < maxmsg
     C                   add       1             error
     C                   eval      mgid(e) = 'HS08148'
     C                   endif
     C                   else
      *
      * Retrieve:
      *  1) the week-ending date for this date
      *  2) the year/period/week for the week-ending date
      *
     C                   z-add     wkdt          wkfcymd
     C                   exsr      $wedt
     C                   z-add     wkcymd        ldfwedt
     C                   z-add     wkmdy         d1fwemdy
     C                   z-add     wkyr          d1fyr
     C                   z-add     wkpe          d1fpe
     C                   z-add     wkwk          d1fwk
      *
     C                   if        *in98 = *on
     C                   seton                                        30
     C                   endif
     C                   endif                                                  If too old
      *
     C                   endif                                                  If bad date
     C                   endsl

     C                   endsr

     C                   move      wkisodate     wkdt
     C                   z-add     wkdt          wktcymd
      *
     C                   exsr      $wedt
     C                   z-add     wkcymd        ldtwedt
     C                   z-add     wkmdy         d1twemdy
     C                   z-add     wkyr          d1tyr
     C                   z-add     wkpe          d1tpe
     C                   z-add     wkwk          d1twk
      *
     C                   if        *in98 = *on
     C                   seton                                        31
     C                   endif
      *
     C                   endif                                                  If bad date
     C                   endsl
      *
      * Make the "TO date cannot be earlier than FROM date" check.
      *
     C                   if        *in30 = *off and                             If OK
     C                             *in31 = *off and
     C                             wktcymd < wkfcymd
     C                   seton                                        31
     C                   if        error < maxmsg
     C                   add       1             error
     C                   eval      mgid(e) = 'HS09009'
     C                   endif
     C                   endif                                                  If OK
      *
     C                   endsr

      *----------------------------------------------------------------
      * Retrieve Week-ending date
      *----------------------------------------------------------------
      *
     C     $wedt         begsr
      *
     C                   z-add     0             wkcymd
     C                   z-add     0             wkmdy
     C                   z-add     0             wkyr
     C                   z-add     0             wkpe
     C                   z-add     0             wkwk

      * Retrieve Saturday Week-Ending Date for the user's date.

     C                   call      'TF811'
     C                   parm      wkdt          xxdt

     C                   parm      0             xxyr
     C                   parm      0             xxpe
     C                   parm      0             xxwk

     C                   add       1             error
     C                   eval      mgid(e) = 'HS09229'

      *
     C                   setoff                                       98

     C                   call      'TF815'
     C     wkcymd        parm                    xxsatdt
     C                   parm      'P'           xxapfl
     C     wkyr          parm      0             xxyr
     C     wkpe          parm      0             xxpe
