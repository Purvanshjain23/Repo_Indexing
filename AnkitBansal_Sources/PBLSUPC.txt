/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION FOR PRINTER OVERRIDE     */
/*              PROMPTING.    PASSES VESSEL AND VOYAGE              */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT REPORTS                                       */
/*  PROGRAMMER- LARA BUSER                                          */
/*  DATE      - 01/27/2014                                          */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*  &RETURN     - RETURN CODE                                       */
/*  &APPLIC     - APPLICATION CODE                                  */
/*  &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT SCREEN)     */
/*  &OUTQ       - OUTPUT QUEUE                                      */
/*  &HOLD       - HOLD STATUS                                       */
/*  &SAVE       - SAVE STATUS                                       */
/*  &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q             */
/*  &NIGHTQ     - NIGHT JOB QUEUE                                   */
/********************************************************************/
/* 99/99/9999 XXX                                                   */
/*                                                                  */
/********************************************************************/
 PBLSUPC:    PGM        PARM(&RETURN &DESC &OUTQ &HOLD &SAVE +
                          &FROMDTN &THRUDTN &COPYN &VOYAGEN &VESSEL)
/* PARMS                                                                      */
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&APPLIC) TYPE(*CHAR) LEN(06) VALUE(OMS)
             DCL        VAR(&DESC) TYPE(*CHAR) LEN(25)
             DCL        VAR(&OUTQ) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&NQSTAT) TYPE(*CHAR) LEN(01) VALUE(N)
             DCL        VAR(&NIGHTQ) TYPE(*CHAR) LEN(10)
             DCL        VAR(&FROMDTN) TYPE(*DEC)
             DCL        VAR(&THRUDTN) TYPE(*DEC)
             DCL        VAR(&FROMDT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&THRUDT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&COPYN) TYPE(*DEC)
             DCL        VAR(&COPY) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&VESSEL) TYPE(*CHAR) LEN(30)
             DCL        VAR(&VOYAGEN) TYPE(*DEC)
             DCL        VAR(&VOYAGE) TYPE(*DEC) LEN(6 0)
             DCL        VAR(&MSG) TYPE(*CHAR) LEN(512)
             DCL        VAR(&MSGID) TYPE(*CHAR) LEN(7)
             DCL        VAR(&MSGL) TYPE(*DEC) LEN(5 0)

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&FROMDT) VALUE(&FROMDTN)
             CHGVAR     VAR(&THRUDT) VALUE(&THRUDTN)
             CHGVAR     VAR(&VOYAGE) VALUE(&VOYAGEN)

/*   CALL PROMPT PROGRAM-HOLD IF REPORT IS TO BE EMAILED            */
             CALL       PGM(PDG0PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))
             CHGVAR     VAR(&COPYN) VALUE(&COPY)

/****   GOTO CALL           *****/
/*   SUBMIT PRINT PRINT PROGRAM - PBLBPFR                           */
 SUBMIT:     SBMJOB     CMD(CALL PGM(PBL2UPC) PARM(' ' &OUTQ &HOLD +
                          &SAVE &FROMDT &THRUDT &VOYAGE &COPY +
                          &VESSEL)) JOB(PRTVESSEL) OUTQ(&OUTQ)

              GOTO       CMDLBL(END)
/********************************************************************/
/* CALL PBL2UPC WHICH CALLS PNWBPFR                                 */
/********************************************************************/
 CALL:       CALL       PGM(PBL2UPC) PARM(' ' &OUTQ &HOLD &SAVE +
                          &FROMDT &THRUDT &VOYAGE &COPY &VESSEL)

              GOTO       CMDLBL(END)
/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
