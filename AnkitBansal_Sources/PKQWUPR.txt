/*  FUNCTION:        THIS CL IS CALLED FROM THE EXTERNAL SCALE     */
/*                   PRINT FUNCTION TO RETRIEVE THE DTAARA THAT    */
/*                   SIGNALS A USER QUIT HAS BEEN REQUESTED        */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&ENDSCL) /* Retrieve dtaara */
             DCL        VAR(&ENDSCL) TYPE(*CHAR) LEN(1)
             RTVDTAARA  DTAARA(ENDSCALE (1 1)) RTNVAR(&ENDSCL)
             CHGDTAARA  DTAARA(ENDSCALE (1 1)) VALUE(' ')
             ENDPGM
