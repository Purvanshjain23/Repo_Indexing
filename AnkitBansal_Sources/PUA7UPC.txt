/* DESCRIPTION: CONVERT DATE FROM *YMD TO *MDY                       */
/* PROGRAMMER : LARA BUSER                                           */
/* DATE       : 03/13/2017                                           */
/*                                                                   */
/* PARAMETERS :                                                      */
/*  &FROMDATE - DATE TO BE CONVERTED, CHAR 10                        */
/*  &TODATE   - DATE TO BE CONVERTED TO, CHAR 10                     */
/*  &FROMFMT  - FORMAT OF FROM DATE                                  */
/*  &TOFMT    - FORMAT OF TO DATE                                    */
/*  &SEP      - SEPARATOR - *NONE IF THERE IS NONE                   */
/*********************************************************************/
             PGM        PARM(&FROMDATE &TODATE &FROMFMT &TOFMT &SEP)

             DCL        VAR(&FROMDATE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&TODATE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&FROMFMT) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TOFMT) TYPE(*CHAR) LEN(10)
             DCL        VAR(&SEP) TYPE(*CHAR) LEN(7)

             IF         COND(&SEP *EQ '       ') THEN(CHGVAR +
                          VAR(&SEP) VALUE('*NONE'))
             CVTDAT     DATE(&FROMDATE) TOVAR(&TODATE) +
                          FROMFMT(&FROMFMT) TOFMT(&TOFMT) TOSEP(*NONE)

             ENDPGM
