/*                                                                  */
/*  SYNOPSIS - EMAIL CARRIER INSURANCE REPORT AS SPREADSHEET        */
/*                                                                  */
/*  COMPANY   - SEABOARD COPRORATION                                */
/*  SYSTEM    - ORDER MANAGEMENT                                    */
/*  USER NAME - LARA BUSER                                          */
/*  DATE      - 01/04/2011                                          */
/*                                                                  */
/*  PARAMETERS RECEIVED:                                            */
/*          &EMAIL      - EMAIL ADDRESS                             */
/********************************************************************/
/* XXX MM/DD/YYYY TASK# DESCRIPTION                                 */
/*                                                                  */
/********************************************************************/
 PNVFUPC:    PGM        PARM(&EMAIL)

             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)
             ADDLIBLE   LIB(PRKDEVQRY) POSITION(*LAST)

/* EXECUTE SEQUEL TO BUILD AND EMAIL THE SPREADSHEET                */

             EXECUTE    VIEW(POCKREP) NBRRCDS(*ALL) PCFMT(*XLS) +
                          RECIPIENT(&EMAIL) TEXT('Carrier Insurance +
                          Report')

 END:        ENDPGM
