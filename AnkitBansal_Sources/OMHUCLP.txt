/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING COMPANY/ITEM LISTING         */
/*                                                                  */
/*  SYSTEM    - ORDER PROCESSING                                    */
/*  DATE      -  5/08/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &ITMTYPE       -  ITEM TYPE(PASSED)                      */
/*         &CNVBASITM     -  CONVERTED/BASE ITEM(PASSED)            */
/*         &ITMCLS        -  ITEM CLASS                             */
/*         &ITMCLSL       -  ITEM CLASS(PASSED)                     */
/*         &CWSTS         -  CATCHWEIGHT STATUS(PASSED)             */
/*         &PRIMUM        -  PRIMARY U/M(PASSED)                    */
/*         &PRDDT         -  PRODUCTION DATE CONTROLLED(PASSED)     */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 OMHUCLP:    PGM        PARM(&APPLIC &COMPNYN &ITMTYPE &CNVBASITM +
                          &ITMCLSL &PRDDT &CWSTS &PRIMUM &RCDSTS +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &COMPNYN     *DEC
        DCL    &RCDSTS      *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &ITMTYPE     *CHAR   (3)
        DCL    &CNVBASITM   *CHAR   (2)
        DCL    &ITMCLS      *DEC    (3 0)
        DCL    &ITMCLSL     *DEC    (15 5)
        DCL    &PRDDT       *CHAR   (1)
        DCL    &CWSTS       *CHAR   (1)
        DCL    &PRIMUM      *CHAR   (2)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND     *CHAR  (512)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)
             CHGVAR     VAR(&ITMCLS) VALUE(&ITMCLSL)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(OMHUPFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(OMHUPFR) PARM(' ' &COMPANY &RCDSTS +
                          &ITMTYPE &CNVBASITM &ITMCLS &CWSTS +
                          &PRIMUM &PRDDT)

 ENDPGM:     ENDPGM
