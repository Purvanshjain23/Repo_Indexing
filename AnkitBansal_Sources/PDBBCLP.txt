/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING ITEM BOM LISTING             */
/*                                                                  */
/*  SYSTEM    - ITEM MANAGEMENT                                     */
/*  DATE      - 10/25/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &RCDSTS        -  PASSED RECORD STATUS                   */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 PDZGCLP:    PGM   PARM(&COMPNYN &ITEMT &BCFST &ITEM &APPLIC &OUTQ +
                   &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &COMPNYN     *DEC
        DCL    &ITEMT       *CHAR  (03)
        DCL    &BCFST       *CHAR  (02)
             DCL        VAR(&ITEM) TYPE(*CHAR) LEN(7)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND     *CHAR  (512)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &ITEMN       *DEC   (7 0)
        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&ITEMN) VALUE(&ITEM)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDBBPFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDBBPFR) PARM(' ' &COMPANY &ITEMT &BCFST +
                          &ITEMN)

 ENDPGM:     ENDPGM
