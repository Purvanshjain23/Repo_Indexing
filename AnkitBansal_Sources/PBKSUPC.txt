/*********************************************************************/
/* PROGRAM    : PBKSUPC                                              */
/* TITLE      : OVRPRTF COMMAND                                      */
/* DATE       : 11/06/2013                                           */
/* Description: OVERRIDE PRINTER FILE                                */
/* NAME       : LARA BUSER                                           */
/*                                                                   */
/* Parameters:                                                       */
/*  &FILE       PRINTER FILE NAME                                    */
/*  &DESC       PRT DESCRIPTION                                      */
/*  &OUTQ       OUTPUT QUEUE                                         */
/*  &HOLD       HOLD ON QUEUE - YES OR NO                            */
/*  &SAVE       SAVE - YES OR NO                                     */
/*  &COPIES     NUMBER OF COPIES, DEFAULT IS 1                       */
/*********************************************************************/
/* Modifications:                                                    */
/*********************************************************************/
             PGM        PARM(&FILE &DESCR &OUTQ &HOLD &SAVE &COPIES)

/* Declare Variables                                                 */
             DCL  VAR(&FILE)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&DESCR)   TYPE(*CHAR) LEN(25)
             DCL  VAR(&OUTQ)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&COPIES)  TYPE(*DEC)
             DCL  VAR(&COPYN)   TYPE(*DEC)  LEN(2 0)

/* OVRPRTF                                                           */

             IF         COND(&COPIES *EQ 0) THEN(CHGVAR VAR(&COPIES) +
                          VALUE(1))
             CHGVAR     VAR(&COPYN) VALUE(&COPIES)

             OVRPRTF    FILE(&FILE) OUTQ(&OUTQ) COPIES(&COPYN) +
                          HOLD(&HOLD) SAVE(&SAVE)

/* End of Program */
 ENDPGM:     ENDPGM
