/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING COMPANY/CUSTOMER ACCRUALS    */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  3/06/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &CUSTN         -  PASSED CUSTOMER NUMBER                 */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*         &CUSTN         -  CUSTOMER NUMBER PASSED                 */
/*         &ACCR          -  ACCRUAL CODE                           */
/*                                                                  */
/********************************************************************/
/* RMC 10/21/04 PROJ EXPCOM - REMOVE COMPANY FROM PARM & CALLED PGM */
/********************************************************************/
 PDBWCLP:    PGM        PARM(&CUSTN &APPLIC &OUTQ &HOLD &SAVE +
                          &COPYN &NIGHTQ &ACCR)
        DCL    &CUSTN       *DEC
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &ACCR        *CHAR  (03)

        DCL    &CUSTOMER    *DEC   (7 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&CUSTOMER)  VALUE(&CUSTN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDDSPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDDSPFR) PARM(' '  &CUSTOMER &ACCR)

 ENDPGM:     ENDPGM
