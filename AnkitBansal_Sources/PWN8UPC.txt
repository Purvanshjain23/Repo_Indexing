 PWN8UPC:    PGM        PARM(&MOVEM &DATEF &DATET &TIME)

             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&MOVEM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DATEF) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DATET) TYPE(*CHAR) LEN(7)
             DCL        VAR(&TIME) TYPE(*CHAR) LEN(4)

             DCL        VAR(&GRPPRF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&JOBD) TYPE(*CHAR) LEN(10)

/*-------------------------------------------------------------------*/
/*  Set Jobd and library list based on user's group profile                   */
/*-------------------------------------------------------------------*/
             RTVUSRPRF  GRPPRF(&GRPPRF)

/* Set library list                                                 */
             IF         COND(&GRPPRF *EQ 'SBDPGMR') THEN(DO)
             SETLIBL    JOBD(TPMTESTJD)
             ENDDO

             IF         COND(&GRPPRF *NE 'SBDPGMR') THEN(DO)
             SETLIBL    JOBD(PRKPRDGUY)
             ENDDO

 /*        CALL EXC CLC WAIT TM F/WEB XF                               */

             CALL       PGM(PLDEXFR) PARM(&RETURN &MOVEM &DATEF &DATET &TIME)


 END:
            ENDPGM
