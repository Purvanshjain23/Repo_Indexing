/* ***************************************************************** */
/* PROGRAM - PODPUPR EXC FAX LOAD CONFIRM UP                         */
/* ***************************************************************** */
/*  04/15/11 LJB P000737  REMOVED UNUSED PARMS, PASS LOAD ID TO      */
/*    PPVVUPC, FAX FLAGS WILL BE ANY OF THE FOLLOWING: CARRIER,      */
/*    TRANSLOAD CARRIER, MEXICAN CARRIER, SHIPPING LINE.             */
/*    DISPATCH WILL BE JUST THE DISPATCHER, EMAIL WILL BE THE        */
/*    EXPORT COORDINATOR PLUS ANY ADDITIONAL EMAIL NAMES OR GROUPS   */
/*    THAT FIT IN THE 50 CHAR FIELD                                  */
/*  MODIFIED:  LJB 5/16/06  CHANGED &FAX# TO 25 CHARACTERS TO        */
/*                            ACCOMMODATE INTERNATIONAL FAX NUMBERS  */
/*  MODIFIED:  RMC 5/24/02  ADDED 'FAX TO CARRIER' PARM &FAXCAR AND  */
/*                            'FAX TO DISPATCH OFFICE' PARM &FAXDIS  */
/* ***************************************************************** */
/******      PGM        PARM(&FAXR &FAXC &FAXA &FAXP &USER &PGM +
                          &COMPL &ORDERL &CARRIER &JOBNM &PRTF +
                          &FAXCAR &FAXDIS)                 ***********/
             PGM        PARM(&COMPN &LOADN &CARRIER &FAXCAR &FAXDIS +
                          &PRINTFAX &EMAIL)

             DCL        VAR(&COMPN) TYPE(*DEC)
             DCL        VAR(&LOADN) TYPE(*DEC)
             DCL        VAR(&CARRIER) TYPE(*CHAR) LEN(3)
             DCL        VAR(&FAXCAR) TYPE(*CHAR) LEN(1)
             DCL        VAR(&FAXDIS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PRINTFAX) TYPE(*CHAR) LEN(1)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)

             CALL       PGM(PDVVUPC) PARM(&COMPN &LOADN &CARRIER +
                          &FAXCAR &FAXDIS &PRINTFAX &EMAIL)

             ENDPGM
