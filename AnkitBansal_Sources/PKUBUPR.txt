/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKUBUPR                                       */
/*  PROGRAM NAME:    PRINT PRODUCER MERIT REPORT                   */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     07/12/95                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT PRINT PRODUCER MERIT REPORT.      */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  R. CENTONZE  11/20/97  ADD INCLUDE YARD DEAD,CONDEMN PARM TO   */
/*                         SELECTION                               */
/*  P. DROGE     12/04/97  ADD PAYMENT TYPE, LOCATION CODE, AND    */
/*                         REPORT TYPE TO THE SELECTION            */
/*  P. LYONS      6/03/98  ADD CARC ANALYSIS FLAG TO THE           */
/*                         SELECTION                               */
/*  R. CENTONZE   2/24/04  ADD FARM SITE PARM                     */
/*******************************************************************/
             PGM        PARM(&RTN &XCO &SRCE &PAYTYPE &PROD &LOC +
                          &XFROM &XTO &RPTTYPE &INCLD &CRCA &ANL +
                          &FARMA  &PDEV &HOLD &SAVE &XCOPIES +
                          &OMS)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCO)     TYPE(*CHAR) LEN(3)
             DCL        VAR(&PROD)    TYPE(*CHAR) LEN(6)
             DCL        VAR(&LOC)     TYPE(*CHAR) LEN(6)
             DCL        VAR(&XFROM)   TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTO)     TYPE(*CHAR) LEN(7)
             DCL        VAR(&SRCE)    TYPE(*CHAR) LEN(1)
             DCL        VAR(&PAYTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&RPTTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&INCLD)   TYPE(*CHAR) LEN(1)
             DCL        VAR(&CRCA)    TYPE(*CHAR) LEN(1)
             DCL        VAR(&ANL)     TYPE(*CHAR) LEN(1)
             DCL        VAR(&FARMA)    TYPE(*CHAR) LEN(5)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FROM) TYPE(*DEC) LEN(7)
             DCL        VAR(&TO) TYPE(*DEC) LEN(7)
             DCL        VAR(&FARM) TYPE(*DEC) LEN(5)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&CO) VALUE(&XCO)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FROM) VALUE(&XFROM)
             CHGVAR     VAR(&TO) VALUE(&XTO)
             IF COND(&FARMA *NE '     ') +
       THEN(CHGVAR     VAR(&FARM) VALUE(&FARMA))

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKT5PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKT5PFR) PARM(' ' &CO &SRCE &PAYTYPE +
                     &FARM &PROD &LOC &FROM &TO &RPTTYPE &INCLD +
                          &CRCA &ANL &LPP)

             DLTOVR *ALL
             ENDPGM
