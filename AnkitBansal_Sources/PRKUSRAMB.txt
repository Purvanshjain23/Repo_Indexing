SELECT
  ALL       GRP.COID, GRP.GRPID#, GRP."USERID", USR.UNAME
  FROM      WM400/WMGROUP GRP LEFT OUTER JOIN
            WM400/WMUSER USR
  ON        GRP.COID = USR.COID
    AND     GRP."USERID" = USR."USERID"
  WHERE     GRP.GRPID# BETWEEN 'QPGMR    ' AND 'QPGMR    '
    AND     GRP.COID = 'GUY'
    AND     USR.COID = 'GUY'
  ORDER BY  GRP.COID ASC, GRP.GRPID# ASC, GRP."USERID" ASC
