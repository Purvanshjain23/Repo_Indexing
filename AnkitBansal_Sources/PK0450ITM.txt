SELECT
  ALL       T01.BZAIC3, T01.BZHGCD, T01.BZQZCD, T02.BZQZCD, T01.BZPPHE,
            T02.BZPPHE, T01.BZPCCT, T02.BZPCCT, T01.BZBDNB, T02.BZBDNB,
            T01.BZBJNB, T02.BZBJNB, T01.BZF6WG, T02.BZF6WG, T01.BZF7WG,
            T02.BZF7WG
  FROM      PRKTESTF/CABZREP T01 INNER JOIN
            PRKFLIB/CABZREP T02
  ON        T01.BZAIC3 = T02.BZAIC3
    AND     T01.BZHGCD = T02.BZHGCD
  WHERE     T01.BZQZCD <> T02.BZQZCD
    OR      T01.BZPPHE <> T02.BZPPHE
    OR      T01.BZPCCT <> T02.BZPCCT
    OR      T01.BZBDNB <> T02.BZBDNB
    OR      T01.BZBJNB <> T02.BZBJNB
    OR      T01.BZF6WG <> T02.BZF6WG
    OR      T01.BZF7WG <> T02.BZF7WG
  ORDER BY  T01.BZAIC3 ASC, T01.BZHGCD ASC
