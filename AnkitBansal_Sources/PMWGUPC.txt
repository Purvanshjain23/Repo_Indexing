 PPU9UPC:    PGM        PARM(&OUTQ &HOLD &SAVE  &COPY +
                          &COMP &SLSRTE &RCDSTS &DAYS &UPDSTS)
/* PARMS                                                                      */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMP       *DEC    (3 0)         /* COMPANY                   */
        DCL    &SLSRTE     *CHAR   (3)           /* SALESPERSON CODE          */
        DCL    &RCDSTS     *CHAR   (1)           /* RECORD STS                */
        DCL    &DAYS       *DEC    (5 0)         /* DURATION                  */
        DCL    &UPDSTS     *CHAR   (1)           /* UPDATE RATE               */

/* INTERNAL FIELDS                                                            */
        DCL    &COMMAND    *CHAR   (512)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/


/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMWFPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*           OVRPRTF    FILE(PMWFPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE) */

             CALL       PGM(PMWFPFR) PARM(' ' &COMP &SLSRTE &RCDSTS +
                          &DAYS &UPDSTS)
 END:        ENDPGM
