PGM PARM(&COMP &ORDER &STATUS &INVTYP &JOB &OUTQ &HOLD &SAVE &DRAWER)

DCL &COMP *DEC (3 0)
DCL &ORDER *DEC (7 0)
DCL &STATUS *CHAR 1  /* Printed Status */
DCL &INVTYP *CHAR 2  /* Invoice Type */
DCL &JOB *CHAR 10  /* Job name */
DCL &OUTQ *CHAR 10
DCL &HOLD *CHAR 4
DCL &SAVE *CHAR 4
DCL &DRAWER *DEC (3 0)


OVRPRTF    FILE(PMIVPFR$) DRAWER(&DRAWER) OUTQ(&OUTQ) +
           HOLD(&HOLD) SAVE(&SAVE)

CALL PMIVPFR PARM(' ' &COMP &ORDER &STATUS &INVTYP &JOB)
DLTOVR PMIVPFR$
