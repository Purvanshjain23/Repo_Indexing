/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKVTUPC                                       */
/*  PROGRAM NAME:    CALCULATE PRODUCER HISTORY -MONTHLY           */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     07/18/00                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED FROM THE SCHEDULER       */
/*                   AT MONTH END FOR CALCULATING THE PRODUCER     */
/*                   MONTHLY HISTORY.                              */
/*                   THIS CL IS ALSO CALLED FROM A PROMPT.         */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC 8/15/02  USE SNDPGRMSG A LIEU SNDDST                       */
/*******************************************************************/
             PGM        PARM(&CO &FDTE &TDTE &PDEV &HOLD &SAVE +
                          &XCOPIES)

/*******************************************************************/
/*  VARIABLE DEFINITION                                            */
/*******************************************************************/

/* INPUT PARMS                                                    */
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDTE) TYPE(*DEC) LEN(7 0)

/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* SYSTEM NAME PARM                                               */

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/*******************************************************************/
/*  INITIALIZATION                                                 */
/*******************************************************************/
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/*******************************************************************/
/*  MAINLINE                                                       */
/*******************************************************************/
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKVRPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* CALCULATE PRODUCER HISTORY MONTHLY                          */
             CALL       PGM(PKVRPFR) PARM(' ' &CO &FDTE &TDTE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(PGMERR))

             GOTO       CMDLBL(COMPLETION)

/*******************************************************************/
/*  ERROR HANDLING                                                 */
/*******************************************************************/
PGMERR:
             CHGJOB     LOG(4 00 *SECLVL)
             SNDPGMMSG  MSG('Program Error')
/*           SNDDST     TYPE(*MSG) TOUSRID((PRKDEV BIGBYTE)) +
                          DSTD('Producer History') MSG('Calculate +
                   Monthly Producer History ended abnormally.') */
             SNDPGRMSG  TOPGR(PRKDEV) MSG('Calculate Monthly +
                          Producer History ended abnormally.') +
                          CONFMSG(*NO)
             GOTO       CMDLBL(ENDPGM)

/*******************************************************************/
/*  END PROGRAM                                                    */
/*******************************************************************/
COMPLETION:
 /*          SNDDST     TYPE(*MSG) TOUSRID((PRKDEV BIGBYTE)) +
                          DSTD('Producer History') MSG('Calculate +
                  Monthly Producer History ended normally.')      */
             SNDPGRMSG  TOPGR(PRKDEV) MSG('Calculate Monthly +
                          Producer History ended normally.') +
                          CONFMSG(*NO)

ENDPGM:
             DLTOVR *ALL

             ENDPGM
