/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PNHXUPC                                       */
/*  PROGRAM NAME:    Print SCHEDULE VARIANCE                     */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     04/27/09                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from Display SCHEDULE VARIANCE                */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES  +
            &APP  &PRDTYP &WHSE &XDATE &PRTSCH &PRIORAM &PRTVAR )

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGMPRTF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(6)
      DCL        VAR(&PRIORAM) TYPE(*CHAR) LEN(1)  /* PRT PRIOR DAY OR AM   */
             DCL        VAR(&PRTVAR) TYPE(*CHAR) LEN(1)  /* PRT VAR<>0 ONLY */
             DCL        VAR(&PRTSCH) TYPE(*CHAR) LEN(1)  /*PRINT ON SCHED */
             DCL        VAR(&WHSE) TYPE(*CHAR) LEN(3)

/* Function Specific                                               */
             DCL        VAR(&PRDTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&XDATE) TYPE(*CHAR) LEN(7)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

             DCL        VAR(&DATE) TYPE(*DEC) LEN(7 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&DATE) VALUE(&XDATE)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PNHWPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

  CALL PGM(PNHWPFR) PARM(' ' &PRDTYP &DATE &WHSE &APP &PRTSCH &PRIORAM &PRTVAR)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
