/*  PROGRAM NAME ..... CHECK FOR OBJECT EXISTANCE                  */
/*  PROGRAMMER ....... LEANNE FEDOR                                */
/*  DATE ............. 02/02/00                                    */
/*                                                                 */
/*  PROGRAM PURPOSE -> THIS CL CHECKS FOR THE EXISTANCE OF AN      */
/*                     OBJECT.                                     */
/*                                                                 */
/***************************************************************** */
/* MODIFICATIONS                                                   */
/***************************************************************** */

             PGM        PARM(&OBJ &TYPE &ERROR)

             DCL        VAR(&OBJ) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TYPE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&ERROR) TYPE(*CHAR) LEN(1)

             CHGVAR     VAR(&ERROR) VALUE(' ')

             CHKOBJ     OBJ(&OBJ) OBJTYPE(&TYPE)
             MONMSG     MSGID(CPF0000) EXEC(CHGVAR VAR(&ERROR)  +
                          VALUE('Y'))

             ENDPGM
