/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKX5UPR                                       */
/*  PROGRAM NAME:    PRINT CARCASS MERIT COMPETITOR                */
/*  PROGRAMMER:      SUSAN L. MASON                                */
/*  CREATE DATE:     12/11/95                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT PRINT CM COMPETITOR RPT .        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  ROSE CENTONZE   08/30/96    ADD CRTDUPOBJ FOR PKDFREL3         */
/*                                                                 */
/*  SUSAN MASON     10/04/96    CHG CRTDUPOBJL TO CRTDUPBOJL        */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &ACO &OMS)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&SRCT) TYPE(*CHAR) LEN(1) VALUE(' ')
             DCL        VAR(&SRCD) TYPE(*CHAR) LEN(3) VALUE('   ')
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6) VALUE('      ')
             DCL        VAR(&XFROM) TYPE(*CHAR) LEN(7) VALUE('0000000')
             DCL        VAR(&XTO) TYPE(*CHAR) LEN(7) VALUE('0000000')
             DCL        VAR(&ATO) TYPE(*CHAR) LEN(5) VALUE('00000')
             DCL        VAR(&ACO) TYPE(*CHAR) LEN(3)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FROM) TYPE(*DEC) LEN(7)
             DCL        VAR(&TO) TYPE(*DEC) LEN(7)
             DCL        VAR(&TTN) TYPE(*DEC) LEN(5)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FROM) VALUE(&XFROM)
             CHGVAR     VAR(&TO) VALUE(&XTO)
             CHGVAR     VAR(&TTN) VALUE(&ATO)
             CHGVAR     VAR(&CO) VALUE(&ACO)

/*        COMPETITOR TOTAL FILE                                 */

             CRTDUPOBJL  OBJ(PKDFREP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 /*          CRTDUPOBJL  OBJ(PKDFREL3) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)      */

/* CREATE PGM                                                  */


             CALL       PGM(PKXWXFR) PARM(' ')


/* PRINT PROGRAM    DETAIL                                     */

             CALL       PGM(PKX1PFR) PARM(' ' &CO &TTN &SRCT +
                          &SRCD &PROD &FROM &TO &LPP)

/* PRINT PROGRAM    TOTAL                                      */

             CALL       PGM(PKX4PFR) PARM(' ' &CO &TTN &SRCT +
                          &SRCD &PROD &FROM &TO &LPP)
/*******************************************************************/
             DLTOVR *ALL
             ENDPGM
