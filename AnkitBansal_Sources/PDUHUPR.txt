/*********************************************************************/
/*     Program: PDUHUPR                                              */
/*       Title: Generic program to execute an delete file command    */
/*        Date: 11/10/98                                             */
/* Description: - Execute a delete file command.                     */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &LIBRARY    Library to delete file from                  */
/*    I     &FILE       File to delete                               */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*                                                                   */
/*   Date     Pgmr     Description                                   */
/* -------- ---------- --------------------------------------------  */
/*                                                                   */
/*********************************************************************/
             PGM  PARM(&LIBRARY &FILE)

   /*-------------------*/
   /* Declare Variables */
   /*-------------------*/

             /* Parameter Fields */
             DCL  VAR(&LIBRARY) TYPE(*CHAR) LEN(10)
             DCL  VAR(&FILE)    TYPE(*CHAR) LEN(10)

   /*---------------------------------*/
   /* Execute the Delete File Command */
   /*---------------------------------*/
             DLTF FILE(&LIBRARY/&FILE)
             MONMSG MSGID(CPF0000)

 ENDPGM:
             ENDPGM
