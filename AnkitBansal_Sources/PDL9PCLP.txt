/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM WITH SELECTIONS FOR ISSUE ERRORS       */
/*             REPORT. RST ON CO, ORDER, TRX ID, TRX TYPE           */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT ISSUE ERRORS REPORT                           */
/*  DATE      -  6/25/96                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPNYN    - COMPANY NUMBER - BLANK FOR ALL            */
/*          &ORDRN      - ORDER NUMBER                              */
/*          &TRXID      - TRANSACTION ID                            */
/*          &TRXTYP     - TRANSACTION TYPE                          */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/*          &NIGHTQ     - NIGHT OUTPUT QUEUE                        */
/********************************************************************/

 PDL9PCLP:   PGM        PARM(&COMPNYN &ORDRN &TRXID &TRXTYP +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)

/* PARMS                                                                      */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &ORDRN      *DEC                  /* Order Number   (Passed)   */
        DCL    &TRXID      *CHAR   (1)           /* Transation ID             */
        DCL    &TRXTYP     *CHAR   (2)           /* Transaction Type          */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */
        DCL    &ORDER      *DEC    (7 0)         /* Order Number              */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&ORDER) VALUE(&ORDRN)

/*********************************************************************/
/*   CALL PRINT PROGRAM -
/*********************************************************************/

/* PRINT REPORT                                              */

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDL7PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PDL7PFR) PARM(' ' &COMPANY      +
                          &TRXID &TRXTYP &ORDER)
 END:        ENDPGM
