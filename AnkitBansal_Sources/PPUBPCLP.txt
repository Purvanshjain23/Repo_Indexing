/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  Print Internal Company Edit, Calculate Price, and   */
/*              then update the price in the Order Detail.          */
/*                                                                  */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT REPORTS                                       */
/*  DATE      - 10/31/05                                            */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*                         IF THIS IS BLANK AT START, PROMPT        */
/*                         PGM IS CALLED; OTHERWISE, THE PRINT CL   */
/*                         IS SUBMITTED WITHOUT PROMPTING           */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*     B    &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*     B    &FRMDATN    - FROM DATE      (PASSED)                   */
/*     B    &THRDATN    - THRU DATE      (PASSED)                   */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/********************************************************************/
/* 10/31/05 SLM  Created *new                                     */
/********************************************************************/

 BEGIN:      PGM        PARM(&RETURN &DESC &OUTQ &HOLD &SAVE &NIGHTQ +
                          &FDATEL &TDATEL &COPYN)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &CLP        *CHAR   (10)          /* CL to Submit              */
        DCL    &PGM        *CHAR   (10)          /* RPG TO SUBMIT             */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &BEGTO     *DEC    (7 0)          /*  BEGINNING TO   */
        DCL    &FDATE     *DEC    (7 0)          /*  FROM DATE      */
        DCL    &FDATEL    *DEC                   /*  FOR PASSING    */
        DCL    &TDATE     *DEC    (7 0)          /*  THROUGH DATE   */
        DCL    &TDATEL    *DEC                   /*  FOR PASSING    */
        DCL    &TYPRUN    *CHAR   (1)            /*  TYPE OF RUN           */
        DCL    &ERROR1    *CHAR   (1)            /*  ERROR ON REPORT 1     */
        DCL    &ERROR2    *CHAR   (1)            /*  ERROR PM REPORT 2     */
/*  HERE  */
        DCL    &PAPER      *CHAR    (1)
        DCL    &HOLDQ      *CHAR    (1)
        DCL    &JOBL       *CHAR    (1)
        DCL    &HOLDL      *CHAR    (1)
        DCL    &LOG        *CHAR    (1)
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)



/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR    VAR(&FDATE) VALUE(&FDATEL) /* Change to length +
                          for print program */
             CHGVAR    VAR(&TDATE) VALUE(&TDATEL) /* Change to length +
                          for print program */
             CHGVAR    VAR(&BEGTO) VALUE(&TDATEL) /* Change to length +
                          for print program */
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/*                        PPUBPFR$                                  */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPUBPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/********************************************************************/
/*                                                                  */
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/*                        PPUBPFR$                                  */
/*                        PPUCPFR$                                  */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPUBPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPUCPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL1:      CALL       PGM(PPUBPFR) PARM(' ' &BEGTO &FDATE &TDATE +
                          &ERROR1 &TYPRUN)

             DLTOVR     FILE(PPUBPFR$)

 CALL2:      CALL       PGM(PPUCPFR) PARM(' ' &BEGTO &FDATE &TDATE +
                          &ERROR2 &TYPRUN)

             DLTOVR     FILE(PPUCPFR$)

             IF COND(&ERROR2 *NE ' ') THEN(DO)
             GOTO CMDLBL(USERQUIT)
             ENDDO

             IF COND(&ERROR1 *NE ' ') THEN(DO)
             GOTO CMDLBL(USERQUIT)
             ENDDO

 CALL3:      CALL       PGM(PPUWXFR) PARM(' ' &OUTQ &HOLD &SAVE +
                          &NIGHTQ &COPYN)

/*------------------------------------------------------------------*/
 /* MAYBE THE PRINT OF INVOICE AND THE POST OF THE INVOICE        */
 /*LL4:      CALL       PGM(XXXXXXX) PARM(' ' &OUTQ &HOLD &SAVE + */
 /*                       &NIGHTQ &COPYN)                         */
/*------------------------------------------------------------------*/

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
