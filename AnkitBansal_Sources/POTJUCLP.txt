/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CLP Packaging Upload of MP2 to AS/400                */
/*                                                                  */
/*             This pgm was created to handle override              */
/*             to data base to PACKUPL to process the               */
/*             label name instead of the file name                  */
/*                                                                  */
/*            - Upload FILE into Packaging Daily Usage              */
/*            - POA0CPP  -       Packaging Daily Usage              */
/*                                                                  */
/*  System    - Item Management                                     */
/*  Date      - 04/03/07                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*      PARAMETERS USED IN POTJUPR                                  */
/*         &ENDDAT  70    -   CYYMMDD FORMAT                        */
/*         &SHIFT   1     -   SHIFT                                 */
/*         &P0MJDT  70    -   JOB DT                                */
/*         &P0CCVN 10     -   JOB USER                              */
/*         &COMP    30    -   JOB USER                              */
/*         &LABEL  10     -   LABEL                                 */
/*                                                                  */
/********************************************************************/
 PGMBEG:     PGM        PARM(&ENDDAT &SHIFT &JOBDT &JOBUS &COMPNYN +
                             &LABEL)

        DCL    &ENDDAT      *DEC   (7 0)    /* CYYMMDD FORMAT */
        DCL    &SHIFT       *CHAR  (01)     /* SHIFT          */
        DCL    &JOBDT       *DEC   (7 0)    /* JOB DATE       */
        DCL    &JOBUS       *CHAR  (10)     /* JOB DATE       */
        DCL    &COMPNYN     *DEC   (3 0)    /* JOB USER       */
        DCL    &LABEL       *CHAR  (10)     /* LABEL          */

 OVERFIDE:   OVRDBF     FILE(PACKUPL) TOFILE(&LABEL)

             CALL       PGM(POTJUPR) PARM(&ENDDAT &SHIFT &JOBDT +
                          &JOBUS &COMPNYN)

             DLTOVR     FILE(*ALL)

 ENDPGM:     ENDPGM
