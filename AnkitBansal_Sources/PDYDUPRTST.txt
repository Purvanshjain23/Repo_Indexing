/*   NIGHT DAY END PROCESSING FOR A/R FOR COMPANY 360   */
/*  THIS IS ONE OF 3 CLP PROGRAMS THAT HAVE COMPANY # HARD CODED */
/*  PDYKUPR  PDX8UPR  PDYDUPR   */
             PGM        (&OUTQ)
             DCL        VAR(&COMP) TYPE(*DEC)
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&COMPN) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&OUTQ) TYPE(*CHAR) LEN(10)
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)
             /* DCL VAR(&COMPN) TYPE(*DEC) LEN(3 0) VALUE(360) */
             CHGVAR     VAR(&COMPN) VALUE(360)


             /* PRINT NIGHTLY AGING BY INVOICE  */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDN4PFR$) +
                          OUTQ(&OUTQ) WIDTH(168) LANDSCAPE(*YES) +
                          COPIES(2) HOLD(*YES) SAVE(*NO)
             CALL       PGM(QCMDEXC) PARM(&command 512)
             CALL       PGM(PDN4PFR) PARM(&RETURN &COMPN)
             DLTOVR     FILE(PDN4PFR$)
             /* PRINT NIGHTLY AGING FOR CHARGEBACKS  */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDYJPFR$) +
                          OUTQ(&OUTQ) HOLD(*YES) SAVE(*NO)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)
             CALL       PGM(PDYJPFR) PARM(&RETURN &COMPN)
             DLTOVR     FILE(PDYJPFR$)

             ENDPGM
