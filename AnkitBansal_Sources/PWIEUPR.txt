/* PROGRAM:      PWIEUPR                                              */
/* PROGRAM NAME: Cpy Cold Scale UPL    CL                             */
/* PROGRAMMER:   Julius HIgh                                          */
/* CREATE DATE:  01/24/2019                                           */
/*                                                                    */
/* FUNCTION:     This CLP will copy the Cold Scale UPL file in        */
/*               PRKFxxx to PRKFSAVF library.                         */
/**********************************************************************/
/* Modification History                                               */
/* E14582 JJH 03/04/19 - Create PLARSVxxx files in PRKFGUP, PRPFSTP,  */
/*                       and PRKFTFP.  Add Input Parm &TOLIB          */
/**********************************************************************/
             PGM        PARM(&TOLIB &TOFILE &TOMBR)
/*--------------------------------------------------------------------*/
/*                      PARAMETER                                     */
/*--------------------------------------------------------------------*/
/* Function Specific                                                  */
             DCL        VAR(&TOLIB)  TYPE(*CHAR) LEN(10)
             DCL        VAR(&TOFILE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TOMBR)  TYPE(*CHAR) LEN(10)
/*--------------------------------------------------------------------*/
/* Add PRKFxxx to Library List                                        */
             ADDLIBLE   LIB(&TOLIB) POSITION(*LAST)
             MONMSG     MSGID(CPF2103) /* LIB ALREADY EXIST */
/*--------------------------------------------------------------------*/
             ADDPFM     FILE(&TOLIB/&TOFILE) MBR(&TOMBR) TEXT('Cold +
                          Scale UPL - PLARCPP')
             CPYF       FROMFILE(*LIBL/PLARCPP) +
                          TOFILE(&TOLIB/&TOFILE) TOMBR(&TOMBR) +
                          MBROPT(*ADD)
             MONMSG     MSGID(CPC2957) /* NO RECORDS IN FILE */
             MONMSG     MSGID(CPF2817) /* NO RECORDS IN FILE */
/*--------------------------------------------------------------------*/
/* PROGRAM END                                                        */
             ENDPGM
