             CRTDUPOBJL  OBJ(PDMHCPLC) FROMLIB(&DATA) OBJTYPE(*FILE) +

             DLTF       FILE(QTEMP/PDMHCPL0)

             DLTF       FILE(QTEMP/PDMHCPLB)
             DLTF       FILE(QTEMP/PDMHCPLC)

             DLTF       FILE(QTEMP/PDMHCPP)

             IF         COND(&REPROCS *EQ 'Y') THEN(GOTO REPEAT)

                          ENDPGM
