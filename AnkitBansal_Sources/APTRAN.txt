SELECT
  ALL       T01.LOC, T01.ACT, (T01.AMTDOL), T02.DOCSRC, T02.DTETRN, T02.BANKNO,
            T01.REFER, T02.VENDNO
  FROM      MGRALM0150/APTRAN T01 LEFT OUTER JOIN
            MGRALM0150/APOPEN T02
  ON        T01.REFER = T02.REFER

    AND     T02.DTETRN <= 19920125
