/*  PROGRAM NAME:    PRINT MOVEMENT                                */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     02/19/03                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED FROM THE WORK WITH       */
/*                   DELIVERY SCHEDULE DETAIL.                     */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&CO &KDTE &DELDT &DELTM &DELSQ &TRANS +
                          &TRKR &DRVR &PDEV &HOLD &SAVE &XCOPIES)

/*******************************************************************/
/*  VARIABLE DEFINITION                                            */
/*******************************************************************/

/* INPUT PARMS                                                    */
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&DELDT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&DELTM) TYPE(*DEC) LEN(4 0)
             DCL        VAR(&DELSQ) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&TRANS) TYPE(*CHAR) LEN(5)
             DCL        VAR(&TRKR) TYPE(*CHAR) LEN(5)
             DCL        VAR(&DRVR) TYPE(*CHAR) LEN(5)

/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* SYSTEM NAME PARM                                               */

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/*******************************************************************/
/*  INITIALIZATION                                                 */
/*******************************************************************/
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/*******************************************************************/
/*  MAINLINE                                                       */
/*******************************************************************/
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLCGPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

             OVRPRTF   FILE(PLCGPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
/* PRINT MOVEMENT                                              */
             CALL       PGM(PLCGPFR) PARM(' ' &CO &KDTE &DELDT +
                          &DELTM &DELSQ &TRANS &TRKR &DRVR)
             GOTO       CMDLBL(COMPLETION)

/*******************************************************************/
/*  ERROR HANDLING                                                 */
/*******************************************************************/

/*******************************************************************/
/*  END PROGRAM                                                    */
/*******************************************************************/
COMPLETION:

ENDPGM:
             DLTOVR *ALL

             ENDPGM
