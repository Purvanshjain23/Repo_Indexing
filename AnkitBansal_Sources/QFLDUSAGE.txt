SELECT
  ALL       T01.@@DOM, T02.OBJATR, T02.FLD, T01.INTLEN, T01.INTINT, T01.INTDEC
  FROM      HPEDEVMDL/YFLDDOMRFP T01 LEFT OUTER JOIN
            HPEDEVMDL/YFLDDTARFP T02
  ON        T01.@@DOM = T02.@@DOM
  WHERE     T02.FLD NOT LIKE '*%'
    AND     T02.OBJATR <> 'DTE'
    AND     T02.OBJATR <> 'THM'
    AND     T02.OBJATR <> 'TME'
  ORDER BY  T01.@@DOM ASC, T02.FLD ASC
