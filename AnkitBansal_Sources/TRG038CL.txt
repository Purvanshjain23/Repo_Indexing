/**************************************************************************/
/*  PROGRAM:       TRG038CL                                               */
/*  PROGRAM DESC:  APPLY TRIGGER TO HPS FEED TICKET DTL (HSP038) FILE     */
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   01/20/2022                                             */
/*  PROJECT:       W88663                                                 */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY PHYSICAL FILE TRIGGER TO THE HPS FEED  */
/*                 TICKET DTL (HSP038) FILE FOR ALL TRIGGER DATABASE      */
/*                 EVENTS:                                                */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 04/05/22  BLG WHD88663 Changed to execute trigger wrapper program      */
/*                        TRG038W instead of TRG001W to follow out HPS    */
/*                        standards. Also renamed prg from TRG001CL to    */
/*                        TRG038CL.                                       */
/* 09/08/22  E.L SDN736   RECOMPILE                                       */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/HSP038) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(TRG038W) +
                          RPLTRG(*YES) TRG(SEB_HSP038_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/HSP038) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(TRG038W) +
                          RPLTRG(*YES) TRG(SEB_HSP038_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/HSP038) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(TRG038W) +
                          RPLTRG(*YES) TRG(SEB_HSP038_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/HSP038) TRGEVENT(*ALL)
             ENDDO

 END:        ENDPGM
