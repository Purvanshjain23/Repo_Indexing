SELECT
  ALL       T05.REQSYN, T05.DSPSEQ, T01.APPCDE, T03.FIL, T05.MSG
  FROM      HPEMDL/YMSGDTARFP T05 INNER JOIN
            HPEMDL/YFILDTARFP T03
  ON        T03.@@FIL = T05.@@FIL INNER JOIN
            HPEMDL/YAPPFILRFP T02
  ON        T02.@@FIL = T03.@@FIL INNER JOIN
            HPEMDL/YAPPDTARFP T01
  ON        T01.@@APP = T02.@@APP
  WHERE     T05.OBJATR = 'RPG'
    AND     T03.FIL NOT LIKE '*%'
  ORDER BY  T03.FIL ASC, T01.APPCDE ASC
