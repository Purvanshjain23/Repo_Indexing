SELECT
  ALL       GLDCT, GLDOC,
            GLMCU CONCAT '.' CONCAT SUBSTR(GLOBJ,1,4) CONCAT '.' CONCAT GLSUB
            AS QACCT,
            DIGITS(GLDGM) CONCAT '/' CONCAT DIGITS(GLDGD) CONCAT '/' CONCAT
            DIGITS(GLDGY) AS QGLDATE, GLEXA, GLEXR, GLAA
  FROM      JTRANPROD/F0911 T01

    AND     GLPN = 3
