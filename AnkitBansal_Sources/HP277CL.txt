/*  PROGRAM NAME:    SYNON/WEB: EDIT/POST SALES MOVEMENTS INTO HPS */
/*  PROGRAMMER:      LEANNE FEDOR                                  */
/*  CREATE DATE:     01/23/03                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED FROM SYNON/WEB           */
/*                                                                 */
/*  MODIFIED BY:     BARB GUTIERREZ   12/14/21                     */
/*                   CHANGED INCOMING PARM TO CHARACTER TO FIX     */
/*                   ISSUE COMING FROM A LINKED SERVER IN THE      */
/*                   INTERFACE.  THEN CHANGE IT TO A DECIMAL FOR   */
/*                   FURTHER PROCESSING IN HP277.                  */
/*                   PROJECT SDN616.                               */
/*******************************************************************/

             PGM        PARM(&CYYMMDDC)

             DCL        VAR(&CYYMMDDC) TYPE(*CHAR) LEN(7)
             DCL        VAR(&CYYMMDD) TYPE(*DEC) LEN(7)
             DCL        VAR(&RETURN)  TYPE(*CHAR) LEN(7)
             DCL        VAR(&COMPANY) TYPE(*DEC) LEN(3)

             CHGVAR     VAR(&COMPANY) VALUE(360)

             CHGVAR     VAR(&CYYMMDD) VALUE(&CYYMMDDC)

/*--------------------------------------------------------------------*/
/* CALL PROGRAM TO EDIT SYNON DATA AND POST ERROR-FREE RECORDS TO HPS */
/*--------------------------------------------------------------------*/

             CALL       PGM(HP277) PARM(&CYYMMDD)

/*--------------------------------------------------------------------*/
/* CALL A SYNON PROGRAM TO UPDATE SYNON FILES                         */
/*--------------------------------------------------------------------*/

             CALL       PGM(PLBUXFR) PARM(&RETURN &COMPANY &CYYMMDD)

             ENDPGM
