/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKBXUPR                                       */
/*  PROGRAM NAME:    PRINT FREIGHT ZONE PERFORMANCE                */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     08/30/96                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT PRINT FRT ZONE PERF RPT .        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                   TATTOO IS PASSED ON AS 0 AS IT IS NOT A       */
/*                   SELECTOR FOR THIS REPORT.                     */
/*******************************************************************/
/* RMC 11/23/07 PRICE TYPE WAS MISSING FROM THE PROMPT. SO I ADDED IT +
/*              TO PASS TO HERE TO PKX3XFR, WHICH ALREADY HAD IT  */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  Modification:    10/04/96                                      */
/*                                                                 */
/*  Change CRTDUP TO JOHNE. CRTDUPL                                */
/* RMC 11/23/07 PRICE TYPE WAS MISSING FROM THE PROMPT. SO I ADDED IT +
/*              TO PASS TO HERE TO PKX3XFR, WHICH ALREADY HAD IT  */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &ACO &SRCT &SRCD &PROD &FZ &XFROM +
                          &XTO &PTYP  &OMS &PDEV &HOLD &SAVE &XCOPIES)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&SRCT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SRCD) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6)
             DCL        VAR(&XFROM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTO) TYPE(*CHAR) LEN(7)
             DCL        VAR(&ACO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PTYP) TYPE(*CHAR) LEN(1)
FRTZN:       DCL        VAR(&FZ) TYPE(*CHAR) LEN(6)
 TATTOO:     DCL        VAR(&TTN) TYPE(*DEC) LEN(5) VALUE(00000)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FROM) TYPE(*DEC) LEN(7)
             DCL        VAR(&TO) TYPE(*DEC) LEN(7)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FROM) VALUE(&XFROM)
             CHGVAR     VAR(&TO) VALUE(&XTO)
             CHGVAR     VAR(&CO) VALUE(&ACO)

/*        COMPETITOR TOTAL FILE                                 */

             CRTDUPOBJL OBJ(PKDFREP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDFREL3) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

/* CREATE PGM                                                  */


             CALL       PGM(PKX3XFR) PARM(' ' &CO &TTN &FROM &SRCT +
                          &SRCD &PROD &FZ &TO &PTYP)


/*           CPYF       FROMFILE(QTEMP/PKDFREP) +        */
/*                        TOFILE(XXXXXX/PKDFREPESLM) +    */
/*                        MBROPT(*REPLACE) FROMRCD(1)     */

             MONMSG     MSGID(CPF0000)

/* CREATE PRINTER FILE OVERRIDE CODE                            */

             SBDPRTOVR  CMDVAR(&CMD) FILE(PKBTPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)


/* PRINT PROGRAM                                               */

             CALL       PGM(PKBTPFR) PARM(' ' &CO  &SRCT +
                          &SRCD &PROD &FZ &FROM &TO &LPP)

/*******************************************************************/
             DLTOVR *ALL
             ENDPGM
