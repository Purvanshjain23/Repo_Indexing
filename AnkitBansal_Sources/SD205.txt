     h option(*SRCSTMT:*NODEBUGIO)
      *
      * ENVIRONMENT: Pork Division
      * SYSTEM:      Margin Management
      * PROGRAM:     SD205 - Signal Demand: Item Structure
      * PROGRAMMER:  LeAnne Ramsey
      * CREATED:     03/05/08
      *
      * FUNCTION: This program builds an Item Structure file.
      *
      *           In OMS, we have 3 master files for:
      *               Item structure type
      *               Item structure group
      *               Item structure class
      *
      *           Here, we will dump all 3 files into one "item structure"
      *           file.  We will identify each record (link it to its source
      *           file) by a field:
      *                 C = Item structure class
      *                 G = Item structure group
      *                 T = Item structure type
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * Date      Programmer
      *
      /eject
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
     Fome8rel0  if   e           k disk
      *   OM Product type
      *
      *
     Fome9rel0  if   e           k disk
      *   OM Product group
      *
      *
     Fomfarel0  if   e           k disk
      *   OM Product class
      *
      *
     Fsdp205    o    e           k disk
      *    Signal Demand: Item structure
      *
      /EJECT
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *---------------------------------------------------------------
      *  Named Constants
      *---------------------------------------------------------------
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      ****************************************************************
      * Data Structures
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ***************************************************************************************
      * Mainline
      ***************************************************************************************
      *
      * Item Structure "Type"
      *
     C     *loval        setll     ome8rel0
     C                   dou       *in90 = *on                                  Do type
     C                   read      ome8rel0                               90
     C                   if        *in90 = *off                                 If not EOF
     C                   move      'T'           islink
     C                   z-add     e8rgnb        isiscd
     C                   move      e8dctx        isisds
     C                   move      e8vsst        isaist
     C                   write     isrec
     C                   endif                                                  If not EOF
     C                   enddo                                                  Do type
      *
      * Item Structure "Group"
      *
     C     *loval        setll     ome9rel0
     C                   dou       *in90 = *on                                  Do group
     C                   read      ome9rel0                               90
     C                   if        *in90 = *off                                 If not EOF
     C                   move      'G'           islink
     C                   z-add     e9rhnb        isiscd
     C                   move      e9ddtx        isisds
     C                   move      e9vsst        isaist
     C                   write     isrec
     C                   endif                                                  If not EOF
     C                   enddo                                                  Do group
      *
      *
      * Item Structure "Class"
      *
     C     *loval        setll     omfarel0
     C                   dou       *in90 = *on                                  Do class
     C                   read      omfarel0                               90
     C                   if        *in90 = *off                                 If not EOF
     C                   move      'C'           islink
     C                   z-add     farinb        isiscd
     C                   move      fadetx        isisds
     C                   move      favsst        isaist
     C                   write     isrec
     C                   endif                                                  If not EOF
     C                   enddo                                                  Do class
      *
      *
     C                   seton                                        lr
      /EJECT
      *---------------------------------------------------------------
      * Initialization subroutine
      *---------------------------------------------------------------
      *
     C     *inzsr        begsr
      *
     C                   endsr
      /EJECT
