/*  SYNOPSIS:                                                      */
/*              CL program that automatically processes the         */
/*              Copy file of PDKWCPP to ASKWCPP                     */
/*                                                                  */
/*  SYSTEM:                                                         */
/*              Order Management System                             */
/*  DATE:                                                           */
/*              5/08/96                                              */
/*  PARAMETERS:                                                     */
/********************************************************************/
             PGM
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(10)
             DCL        VAR(&FLIB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HLIB) TYPE(*CHAR) LEN(10)

 /*         RETREIVE FILE LIBRARY INTO FLIB                          */
             CALL       PGM(CAQXXFR) PARM(&RETURN &FLIB &HLIB)


             IF         COND(&FLIB *NE '          ') THEN(DO)

                    CPYF       FROMFILE(&FLIB/PDKWCPP) +
                               TOFILE(&FLIB/ASKWCPP) MBROPT(*ADD) +
                               CRTFILE(*YES)

             ENDDO

 END: ENDPGM
