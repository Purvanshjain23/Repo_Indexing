/***************************************************************** */
/*                                                                 */
/*  Program Number  -> GETMASTER                                   */
/*                                                                 */
/*  Program Name    -> GET MASTER FILES FROM E1 - NIGHLTY FROM JS  */
/*                                                                 */
/*  Program Purpose -> This CLP is used to call a program to push E1     */
/*                     master tables to an IBMi EZnnnn version of the same */
/*                     table.  After this job completes the CLP          */
/*                     runs that pushes the data from the EZnnnn table to  */
/*                     the E1nnnn version of the table.                   */
/*             *Note:  &getactbal controls the E10902 account balance push:*/
/*                     IF parm is "N" then do not push the E10902       */
/*                     If parm is "O" then only push the E10902         */
/*                     If parm is "Y" then do push the E10902, et al   */
/*                                                                 */
/*  Programmer      -> Rose Centonze            09/22/2020         */
/*******************************************************************/
 GETMASTER:  PGM        PARM(&GETACTBAL)

             DCL        VAR(&GETACTBAL) TYPE(*CHAR) LEN(1)

ONLYBAL:     IF         COND(&GETACTBAL = 'O') THEN(GOTO +
                          CMDLBL(ACTBAL))

 /* BUSINESS UNIT MASTER  - GET E10006             */

            CALL       PGM(GETE10006)

 /* COMPANY CONSTANTS  - GET E10010               */

            CALL       PGM(GETE10010)

 /* PAYMENT TERMS  - GET E10014                   */

            CALL       PGM(GETE10014)

 /* ADDRESS BOOK --  Get F0101/F0116/F0111/f0115/f01151  */

            CALL       PGM(GETE10101)

 /* ACCOUNT MASTER  - GET E10901           */

            CALL       PGM(GETE10901)

 /* ACCOUNT BALANCES - GET E10902 - DO PER PARM       */

ACTBAL:     IF         COND(&GETACTBAL = 'O' *OR &GETACTBAL = 'Y') THEN(DO)

            CALL       PGM(GETE10902)
            ENDDO

ENDPGM:
            ENDPGM
