/*******************************************************************/
/*  PROGRAM NUMBER:  PLRAUPC                                       */
/*  PROGRAM DESC:    ESEND EMAIL WITH A PRINTER FILE AND EMAIL     */
/*                   LINE OF 50                                    */
/*                   DO NOT REMOVE SPOOL FILES IN THIS CL          */
/*                   THIS CL COULD BE CALL MULTIPLE TIMES          */
/*                   USING THE SAME SPOOL - THIS IS TO ALL         */
/*                   ULTIPLE USERS TO GET THE SAME SPOOL REPORT    */
/*  PROGRAMMER:      SUSAN MASON                                   */
/*  CREATE DATE:     09/15/11 CLONE FROM THE EMAIL ROUTINE         */
/*                   CLONE FROM PLK8UPC                            */
/*  CHANGE MODIFICATION:                                           */
/*  SUSAN MASON      WITH A ATTACHMENT OR WITHOUT BUT REPORT       */
/*  9/23/2011                                                      */
/*                                                                 */
/*******************************************************************/
/*******************************************************************/
 PLK8UPC:    PGM        PARM(&PRTF &EMAIL &SUBJ &ATTACH)

             DCL        VAR(&PRTF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)
             DCL        VAR(&SUBJ) TYPE(*CHAR) LEN(60)
             DCL        VAR(&ATTACH) TYPE(*CHAR) LEN(10)

             DCL        VAR(&USRCLS) TYPE(*CHAR) LEN(10)

/*----------------------------------------------------------------- */
/* CHANGE VARIABLES                                                 */
/*----------------------------------------------------------------- */

             RTVUSRPRF  USRCLS(&USRCLS)


             IF         COND(&ATTACH *NE 'R         ') THEN(DO)

   /*********** NO ATTACHMENT  - SUBJECT LINE **********/
             ESEND/ESNDMAIL RECIPIENT(&EMAIL) SUBJECT(&SUBJ)

             ENDDO

             ELSE DO

   /*********** SEND REPORT        **********/
             ESEND/ESNDMAIL RECIPIENT(&EMAIL) SUBJECT(&SUBJ) +
                          ATTLIST((* *PDF *N &PRTF))

             ENDDO

 END:

             DMPCLPGM

            ENDPGM
