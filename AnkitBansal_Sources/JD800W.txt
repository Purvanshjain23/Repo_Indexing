      * ENVIRONMENT:  J.D. Edwards
      * PROGRAM:      JD800
      * TITLE:        Retrieve Company Name
      * AUTHOR:       LeAnne Fedor
      * CREATED:      04/01/98
      *
      * FUNCTION: This is a generic program that retrieves the name of a
      *           company in the Company Constants File.
      *
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      * 99/99/99
      *
      /EJECT
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
     Ff0010     if   e           k disk
      *    company constants
      *
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      ****************************************************************
      * ARRAYS AND TABLES
      ****************************************************************
      *
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      * Standard fields
      *
     D pgm             s             10
     D routne          s              6
     D rtncd           s              2
      *
      * Parm fields
      *
     D xxco            s                   like(ccco)
     D xxname          s                   like(ccname)
      *
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      *---------------------------------------------------------------
      * Standard database file information data structure
      *---------------------------------------------------------------
      *    externally defined as UTDBGR (record format: FDBCKD)
     D dbfeed        e ds                  extname(utdbfr)
      *
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * MAINLINE
      ****************************************************************
      *
     C     xxco          chain     f0010                              92
     C                   select
     C                   when      *in92 = *off
     C                   eval      xxname = ccname
     C                   eval      rtncd = *blank
     C                   other
     C                   eval      rtncd = '99'
     C                   endsl
      *
     C                   seton                                        LR
      /EJECT
      *---------------------------------------------------------------
      * *inzsr - initialization subroutine
      *---------------------------------------------------------------
      *
     C     *inzsr        begsr
      *
      * Key lists
      *   None
      *
      * Parm lists
      *
     C     *entry        plist
     C                   parm                    xxco
     C                   parm                    xxname
     C                   parm                    rtncd
      *
     C                   endsr
      /EJECT
