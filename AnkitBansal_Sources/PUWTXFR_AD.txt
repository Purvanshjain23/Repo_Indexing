EXECUTE FUNCTION(Rtv Count No Dtls Cmb3RT) TYPE(RTVOBJ) FILE(PDMSREP)           AC2206687;
PARAMETER(PAR.Company_Number);
PARAMETER(PAR.Order_Number);
PARAMETER(PAR.ED_Export_Document_Type);
PARAMETER(PAR.Country_Code);
PARAMETER(LCL.ED_Xlate_Desc_to_Country);
PARAMETER(PAR.Page_Overflow_USR);
{
 //?USER: Initialize routine

 // PAR.Page Overflow USR = CND.No
 PAR.Page_Overflow_USR = 'N';

 // LCL.# of Lines Reqd to Print = CON.*ZERO
 LCL.#_of_Lines_Reqd_to_Print = *ZERO;

 //?USER: Process Data record

 CASE;

 // IF DB1.ED Product Net Weight lbs GT WRK.ZERO
 IF DB1.ED_Product_Net_Weight_lbs > WRK.ZERO;

 CASE;

 // IF DB1.ED Item Code NE LCL.ED Item Code
 IF DB1.ED_Item_Code <> LCL.ED_Item_Code;

 // LCL.ED Item Code = DB1.ED Item Code
 LCL.ED_Item_Code = DB1.ED_Item_Code;

 // LCL.ED Item Label Line 2 = DB1.ED Item Label Line 2
 LCL.ED_Item_Label_Line_2 = DB1.ED_Item_Label_Line_2;

 // LCL.ED Language Xlate Desc 2 = DB1.ED Language Xlate Desc 2
 LCL.ED_Language_Xlate_Desc_2 = DB1.ED_Language_Xlate_Desc_2;

 //?Determine '# of Lines Reqd to Print'.
 CASE;

 // IF *OTHERWISE
 IF *OTHERWISE;

 // LCL.# of Lines Reqd to Print = LCL.# of Lines Reqd to Print + CON.2
 LCL.#_of_Lines_Reqd_to_Print = LCL.#_of_Lines_Reqd_to_Print + 2;

 CASE;

 // IF DB1.ED Language Xlate Desc is Entered
 IF DB1.ED_Language_Xlate_Desc <> *BLANK;

 // AND NOT PAR.ED Xlate Desc to Country is Not Entered
 AND ! PAR.ED_Xlate_Desc_to_Country = *BLANK;

 // LCL.# of Lines Reqd to Print = LCL.# of Lines Reqd to Print + CON.2
 LCL.#_of_Lines_Reqd_to_Print = LCL.#_of_Lines_Reqd_to_Print + 2;

 ENDIF;

 //?Set Overflow On if More Than 9 Detail Lines.
 CASE;

 // IF LCL.# of Lines Reqd to Print is GT 9
 IF LCL.#_of_Lines_Reqd_to_Print > 9;

 // PAR.Page Overflow USR = CND.Yes
 PAR.Page_Overflow_USR = 'Y';

 //?Quit If >= 12.
 CASE;

 // IF LCL.# of Lines Reqd to Print is GE 12
 IF LCL.#_of_Lines_Reqd_to_Print >= 12;

 QUIT;

 ENDIF;

 ENDIF;

 ENDIF;

 ENDIF;

 ENDIF;

 //?USER: Exit processing

 //?1st Page Exception to Print 9th Detail Line if No Label Ln2 or Xlate Ln2.
 CASE;

 // IF LCL.# of Lines Reqd to Print is 10
 IF LCL.#_of_Lines_Reqd_to_Print = 10;

 CASE;

 // IF LCL.ED Item Label Line 2 is Not Entered
 IF LCL.ED_Item_Label_Line_2 = *BLANK;

 // OR LCL.ED Language Xlate Desc 2 is Not Entered
 OR ( LCL.ED_Language_Xlate_Desc_2 = *BLANK;

 // AND NOT PAR.ED Xlate Desc to Country is Not Entered
 AND ! PAR.ED_Xlate_Desc_to_Country = *BLANK );

 // PAR.Page Overflow USR = CND.No
 PAR.Page_Overflow_USR = 'N';

 ENDIF;

 ENDIF;

}
