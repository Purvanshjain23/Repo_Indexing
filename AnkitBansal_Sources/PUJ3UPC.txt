/*********************************************************************/
/* PROGRAM    : PUJ3UPC                                              */
/* TITLE      : OVRPRTF COMMAND                                      */
/* DATE       : 10/25/2018                                           */
/* Description: OVERRIDE PRINTER FILE                                */
/* NAME       : ROSE CENTONZE                              */
/*                                                                   */
/* Parameters:                                                       */
/*  &FILE       PRINTER FILE NAME                                    */
/*  &DESC       PRT DESCRIPTION                                      */
/*  &OUTQ       OUTPUT QUEUE                                         */
/*  &HOLD       HOLD ON QUEUE - YES OR NO                            */
/*  &SAVE       SAVE - YES OR NO                                     */
/*  &COPIES     NUMBER OF COPIES, DEFAULT IS 1                       */
/*********************************************************************/
/* Modifications:                                                    */
/*********************************************************************/
             PGM        PARM(&FILE &DESCR &OUTQ &HOLD &SAVE &COPIES &OVL +
                    &CO &ULSD &BATCH )
/* Declare Variables                                                 */
             DCL  VAR(&FILE)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&DESCR)   TYPE(*CHAR) LEN(25)
             DCL  VAR(&OUTQ)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&OVL)    TYPE(*CHAR) LEN(6)
             DCL  VAR(&COPIES)  TYPE(*DEC)
             DCL  VAR(&COPYN)   TYPE(*DEC)  LEN(2 0)
             DCL  VAR(&CO)    TYPE(*CHAR) LEN(3)
             DCL  VAR(&ULSD)   TYPE(*CHAR) LEN(4)
             DCL  VAR(&BATCH)     TYPE(*CHAR) LEN(5)

/* OVRPRTF                                                           */

             IF         COND(&COPIES *EQ 0) THEN(CHGVAR VAR(&COPIES) +
                          VALUE(1))
             CHGVAR     VAR(&COPYN) VALUE(&COPIES)

             OVRPRTF    FILE(&FILE) DEV(&OUTQ) DEVTYPE(*AFPDS) +
                          PAGESIZE(85 136) LPI(8) CPI(15) +
                          OVRFLW(82) FONT(254 7) FRONTOVL(&OVL) +
                          OUTQ(&OUTQ) COPIES(&COPYN) HOLD(&HOLD) +
                          SAVE(&SAVE)

             CALL       PGM(PUJ7PFR) PARM('' &CO &ULSD &BATCH)

/* End of Program */
 ENDPGM:     ENDPGM
