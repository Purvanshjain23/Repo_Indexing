SELECT
  ALL       T01.CONO, T01.CUSTNO, T01.SHIPTO, T01.CNAME, T01.CADDR, T01.CADDR1,
            T01.CCITY, T01.CSTATE, T01.CZIP, T02.EXCSTP, 'ATH' AS LOCID
  FROM      ATHFLIB/ARSP001 T01 INNER JOIN
            ATHFLIB/ARSP011 T02
  ON        T01.CONO = T02.EXCONO
    AND     T01.CUSTNO = T02.EXCSTN
    AND     T01.SHIPTO = T02.EXSHPT
  WHERE     T01.CONO = 02
    AND     T02.EXCSTP <> 'D'
  ORDER BY  T01.CONO ASC, T01.CUSTNO ASC, T01.SHIPTO ASC
