/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING STATE CODE LISTING           */
/*                                                                  */
/*  SYSTEM    - OMS                                                 */
/*  DATE      -  2/28/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &STATUS        -  RECORD STATUS (PASSED)                 */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 PDFUCLP:    PGM        PARM(&STATE &TERR &OUTQ &HOLD &SAVE &COPYN +
                          &NIGHTQ)
        DCL    &STATE       *CHAR  (02)
        DCL    &TERR        *CHAR  (6)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDFUPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)


 CALL:       CALL       PGM(PDFUPFR) PARM(' ' &STATE &TERR)

 ENDPGM:     ENDPGM
