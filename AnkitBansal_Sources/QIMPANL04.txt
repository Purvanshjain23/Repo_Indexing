SELECT
  ALL       T03.FIL, T01.ELMTTL, T02.MSG
  FROM      HPEMDL/YMSGACTRFP T01 INNER JOIN
            HPEMDL/YMSGDTARFP T02
  ON        T01.@@MSG = T02.@@MSG INNER JOIN
            HPEMDL/YFILDTARFP T03
  ON        T02.@@FIL = T03.@@FIL
  WHERE     T01.ELMTTL LIKE 'Sel%'
  ORDER BY  T03.FIL ASC, T01.ELMTTL ASC
