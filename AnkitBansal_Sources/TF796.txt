      *----------------------------------------------------------------
      * Set environment for Screen 1

     C     $set1         begsr
      *
      * Retrieve the Base Marketing Fees record
      *
     C     xxyrno        chain     tfp003                             92
     C                   if        *in92 = *on                                  If not found
     C                   add       1             error
     C                   movel     'HS09008'     mgid(e)
     C                   else
      *
     C                   z-add     bmyrno        d1yrno
     C                   z-add     bmbmcfee      d1bmcfee
     C                   z-add     bmbmdfee      d1bmdfee
     C                   z-add     bmbmtfee      d1bmtfee
      *
      * Year-beginning date
      *
     C     *iso          test(d)                 bmybdt                 92
     C                   if        *in92 = *off                                 If OK begin
     C                   move      bmybdt        wkisodate
     C     *mdy          move      wkisodate     d1ybmdy
     C                   endif                                                  If OK begin
      *
      *
      * Year-ending date
      *
     C     *iso          test(d)                 bmyedt                 92
     C                   if        *in92 = *off                                 If OK end
     C                   move      bmyedt        wkisodate
     C     *mdy          move      wkisodate     d1yemdy
     C                   endif                                                  If OK end

     C                   endif                                                  If not found

      *
     C                   endsr

      * Write the error messages from the error arrays to the
      * error message subfile

     C                   exsr      $wrmsg
      *
      * Write screen 1 to CRT
      *
     C                   write     tf796k1
     C                   write     tf796ec
     C                   exfmt     tf796r1
      *
      * Clear all messages
      *
     C                   exsr      $clmsg
      *
      *
      * Process user's entry and set routine
      *
     C                   select
     C                   when      *in03 = *on                                  F3-exit
     C                   move      '03'          rtncd
     C                   eval      routne = exit
      *
     C                   when      *in05 = *on                                  F5-refresh
     C                   eval      routne = set1
     C                   other
      *
     C                   eval      routne = exit
     C                   endsl
      *
     C                   endsr
      /eject

     C                   call      'UT80045J'                           43

     C                   parm                    mgdt
     C                   parm                    error
     C                   parm      sdpgm         pgm
     C                   parm      msgfil        msgfl

     C                   endif
      *
      *
     C                   endsr

     C                   parm      msgfil        msgfl

      *
