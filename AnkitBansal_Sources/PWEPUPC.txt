/*  PROGRAM NUMBER:  PWEPUPC                                       */
/*  PROGRAM NAME:    PAYMENT PRODUCER POST CLP                     */
/*  PROGRAMMER:      JULIUS HIGH                                   */
/*  CREATE DATE:     058/15/17                                     */
/*                                                                 */
/*******************************************************************/
/*  FUNCTION:        THIS CL IS SUBMITTED SO THAT IT CAN ADD THE   */
/*                   HALPROD LIBRARY WHICH IS USED BY HALCYON TO   */
/*                   SEND MESSAGES.  IT THEN WILL PERFORM THE      */
/*                   FTP/SFTP/EMAIL PROCESS FOR EDI CARCASS        */
/*                   SUMMARY AND DETAIL.                           */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&COMPANYA &PAYDATEA)

/* INPUT  PARMS                                                   */
             DCL        VAR(&COMPANYA) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PAYDATEA) TYPE(*CHAR) LEN(7)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&COMPANYN) TYPE(*DEC) LEN(3)
             DCL        VAR(&PAYDATEN) TYPE(*DEC) LEN(7)
/**********************************************************************/

             ADDLIBLE   LIB(HALPROD) POSITION(*LAST)

/**********************************************************************/

/* PROCESS PAYMENT PRODUCER POST                               */

             CHGVAR     VAR(&COMPANYN) VALUE(&COMPANYA)
             CHGVAR     VAR(&PAYDATEN) VALUE(&PAYDATEA)

             CALL       PGM(PWEHXFR) PARM(' ' &COMPANYN &PAYDATEN)

             ENDPGM
