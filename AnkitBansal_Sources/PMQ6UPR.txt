/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PMQ6UPR                                       */
/*  PROGRAM NAME:    Print ACCRUAL HISTORY                         */
/*  PROGRAMMER:      SHARON ZURN                                   */
/*  CREATE DATE:     12/19/07                                      */
/*                                                                 */
/*  FUNCTION:        THIS CLP IS SUBMITTED FROM DSP ACCRU HISTORY  */
/*                                                      (PMQVDFR)  */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  SHARON ZURN 10/06/09 ADDED MKTCD & SLSPR AS PARMS              */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&CUST &DATE &CODE &PDEV &HOLD &SAVE +
                          &COPIES &MKTCD &SLSPR)
                                                                       /**/
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
             DCL        VAR(&XCOPIES) TYPE(*DEC) LEN(2 0)
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* Function Specific                                               */
               DCL        VAR(&CUST) TYPE(*DEC) LEN(8 0)
               DCL        VAR(&DATE) TYPE(*DEC) LEN(8 0)
               DCL        VAR(&CODE) TYPE(*DEC) LEN(5 0)
               DCL        VAR(&MKTCD) TYPE(*CHAR) LEN(3)
               DCL        VAR(&SLSPR) TYPE(*CHAR) LEN(3)


/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             OVRPRTF    FILE(PMQ5PFR$) OUTQ(&PDEV) COPIES(&COPIES) +
                          HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(PMQ5PFR) PARM(' ' &CUST &DATE &CODE +
                          &MKTCD &SLSPR)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
