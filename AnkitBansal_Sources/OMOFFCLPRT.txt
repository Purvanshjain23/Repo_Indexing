/*  SYNOPSIS -  PRINT ITEM SPECIFICATIONS BY ITEM NUMBER            */
/*              *** THIS IS FOOD SERVICE GROUP                      */
/*  COMPANY   - SEABOARD CORPORATION                                */
/*  SYSTEM    - IDEAS/400 CROSS APPLICATION                         */
/*  PROGRAMMER- SHEILA SCHERTZ                                      */
/*  PROGRAM   - SYSTEM CONTROL SELECTIONS                           */
/*  DATE      - 11/29/93                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &CO         - COMPANY NUMBER ALPHA                      */
/*          &ITEM       - ITEM CODE ALPHA                           */
/*          &CODE       - SPECIFICATION TYPE = S                    */
/*                                                                  */
/********************************************************************/

             PGM        PARM(&CO &ITEM &CODE)
             DCL        VAR(&CO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&ITEM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&CODE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&FOLDER) TYPE(*CHAR) LEN(8)
             DCL        VAR(&DOC) TYPE(*CHAR) LEN(8)
             MONMSG     MSGID(CPF0000)

      /*  BUILD FOLDER NAME (&FOLDER)                                     */
      CHGVAR VAR(&FOLDER) VALUE('FS' *CAT &CO)

      /*  BUILD DOCUMENE NAME (&DOC)                                      */
      CHGVAR VAR(&DOC   ) VALUE(&CODE *CAT &ITEM)

             PRTDOC     DOC(&DOC) FLR(&FOLDER)

             ENDPGM
