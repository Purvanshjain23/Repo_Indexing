/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  PRINT TRANSFER TEMPLATE ITEMS LIST                  */
/*                                                                  */
/*  SYSTEM    - INVENTORY SYSTEM                                    */
/*  DATE      - 12/21/2006                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     B    &COPYN      -  NO. OF COPIES -PASSED AS 15.5            */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*                         IF THIS IS BLANK AT START, PROMPT        */
/*                         PGM IS CALLED; OTHERWISE, THE PRINT CL   */
/*                         IS SUBMITTED WITHOUT PROMPTING           */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &COMPYN     -  COMPANY NUMBER - PASSED AS 15.5          */
/*     B    &CUSTN      -  CUSTOMER NUMBER- PASSED AS 15.5          */
/*     B    &WHSTYP     -  WAREHOUSE TYPE                           */
/*     B    &RECSTS     -  RECORD STATUS                            */
/********************************************************************/
 PMC4UPC:    PGM        PARM(&RETURN &OUTQ &HOLD &SAVE &COMPYN +
                          &CUSTN &COPYN &WHSTYP &RECSTS)

/* PARMS                                                            */
        DCL    &RETURN     *CHAR   (7)  /* RETURN CODE              */
        DCL    &COPYN      *DEC         /* NO. OF COPIES            */
        DCL    &OUTQ       *CHAR   (10) /* OUTPUT QUEUE             */
        DCL    &HOLD       *CHAR   (4)  /* HOLD STATUS              */
        DCL    &SAVE       *CHAR   (4)  /* SAVE STATUS              */
        DCL    &COMPYN     *DEC         /* CO NUMBER - PASSED       */
        DCL    &CUSTN      *DEC         /* CUSTOMER# - PASSED       */
        DCL    &WHSTYP     *CHAR   (2)  /* WAREHOUSE TYPE           */
        DCL    &RECSTS     *CHAR   (1)  /* RECORD STATUS            */

/* INTERNAL FIELDS                                                  */
        DCL    &COPY       *DEC    (2 0) /* NUMBER OF COPIES        */
        DCL    &COMP       *DEC    (3 0) /* COMPANY NUMBER          */
        DCL    &CUST       *DEC    (7 0) /* CUSTOMER #              */

/* ERROR MESSAGE FIELDS                                             */
        DCL    &MSG        *CHAR   (512) /* MESSAGE                 */
        DCL    &MSGID      *CHAR   (7)   /* MESSAGE ID              */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/
             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&COMP) VALUE(&COMPYN)
             CHGVAR     VAR(&CUST) VALUE(&CUSTN)

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PRINT                                                   */
/********************************************************************/

             OVRPRTF    FILE(PMC3PFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(PMC3PFR) PARM(' ' &COMP &WHSTYP &CUST +
                          &RECSTS)
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
