/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PBL1UPR                                       */
/*  PROGRAM NAME:    Print Deadline Departure Report By specify    */
/*                   selection.                                    */
/*                                                                 */
/*  PROGRAMMER:      Susan Mason                                   */
/*  CREATE DATE:     03/05/2014                                    */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with fronm the Job Sch  */
/*                   Was Design to select on specific information  */
/*                   by Jeff Ruby                                  */
/*                                                                 */
/*                   Send in company 360 - date, wharehouse GP1    */
/*                   deadline departure date/time                  */
/*                   always send report by email                   */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  SLM 3/05/14 CREATED and place on the job schedule to start     */
/*              at a specific time                                 */
/*   User request to run Monday - Friday at 15:30PM using          */
/*   current date                                                  */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &COMP &XWHSE)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)

             DCL        VAR(&DISTDLN) TYPE(*CHAR) LEN(40)

/* Function Specific                                               */
             DCL        VAR(&COMP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XWHSE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&RTN1) TYPE(*CHAR) LEN(7)
             DCL        VAR(&RTN2) TYPE(*CHAR) LEN(7)
             DCL        VAR(&JOBDATE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&JOBDTE) TYPE(*CHAR) LEN(7)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0) VALUE(01)
             DCL        VAR(&COMPN) TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&FDATE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDATE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)
/*-----------------------------------------------------------------*/
/*    GET JOB DATE                                                 */
/*-----------------------------------------------------------------*/
             RTVJOBA    DATE(&JOBDATE)
             CVTDAT     DATE(&JOBDATE) TOVAR(&JOBDTE) FROMFMT(*MDY) +
                          TOFMT(*CYMD) TOSEP(*NONE)
             CHGVAR     VAR(&FDATE) VALUE(&JOBDTE)
             CHGVAR     VAR(&TDATE) VALUE(&JOBDTE)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COMPN) VALUE(&COMP)


/* GET EMAIL DIST LIST FOR SCHED REPORTS            */
 GETEMLSC:  CALL       PGM(POMTXFR) PARM(' ' &COMPN 'OMSDEADLNE' &DISTDLN)
/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
/****    build data in qtemp                                    ****/
             CALL       PGM(PMD5UCLP) PARM(&RTN1)

             IF         COND(&RTN1 *EQ '       ') THEN(DO)
/*-----------------------------------------------------------------*/
/******  IF no error then                                       ****/
/******  call program to print report and send email            ****/
/******  this request is to print and send email all the time   ****/
/******  email is not optional                                  ****/
/*-----------------------------------------------------------------*/

/******  override printer file   TO QPRINT2                     ****/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDUDPFR$) +
                          OUTQ(QPRINT2) COPIES(1) HOLD(*YES) SAVE(*NO)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PBL3XFR) PARM(&RTN2 &COMPN &FDATE +
                  &TDATE &XWHSE)

             ESEND/ESNDMAIL RECIPIENT(&DISTDLN) SUBJECT('Deadline +
                          Departure Report') ATTLIST((* *PDF *N +
                          PDUDPFR$))

             ENDDO
/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
