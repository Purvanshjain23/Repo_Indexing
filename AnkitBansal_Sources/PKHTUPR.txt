/*                                                                 */
/*  PROGRAM NUMBER:  PKHTUPR                                       */
/*  PROGRAM NAME:    PRINT CARCASS SUMMARY / DETAIL                */
/*  PROGRAMMER:      MARK ARENSBERG                                */
/*  CREATE DATE:     02/24/97                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS CALLED BY THE PAYMENT POST PROGRAM */
/*                   (PKTFXFR) AND THE PRINT CARC DTL & SUM CLP    */
/*                   (PKXCUPR).  THE PURPOSE OF THE CL IS TO       */
/*                   CHANGE THE OVERRIDE ON THE PRINT FILE FOR     */
/*                   FAXING OR FOR REPORTING.                      */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  SLM     03/21/97 DO NOT ALLOW PROGRAMERS TO FAST FAX ON        */
/*          EITHER LITTBIT OR BIGBYTE                              */
/*                                                                 */
/*  RMC     10/10/97  CHANGED TO JUST PRINT THE CARC DTL/SUMM -    */
/*                    A DIFFERENT CL IS USED TO FAX THE RPTS       */
/*  RMC     05/29/03  ADDED PARM FOR SORT LOSS PRINT = "N"         */
/*  SLM     04/16/04  TFM - REPLACE PKMLPFR WITH PLEMPFR           */
/*  JJH     05/09/18  ADD &NODATA PARM TO PKMOXFR, NO ADDITIONAL   */
/*                    LOGIC REQUIRED.                              */
/*******************************************************************/
/*           PGM        PARM(&RTN &XCOMP &XBYORD &XLOAD &XTATTOO +
                             &ANAL &PDEV1 &HOLD &SAVE &XCOPIES +
                             &XPRDVN# &XPRDLCVN# &FXPROD &FXPRDLC +
                             &IGNFAX)                            */
         PGM        PARM(&RTN &OMS &PDEV1 &HOLD &SAVE &XCOPIES)
/* INPUT PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(40)
             DCL        VAR(&PDEV1) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SORT) TYPE(*CHAR) LEN(1) VALUE(N)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&CMD)     TYPE(*CHAR) LEN(512)
             DCL        VAR(&NODATA) TYPE(*CHAR) LEN(1)

/*  SET POST_OR_SELECT_STATUS TO = POST, ANAL = ' ' */
             DCL        VAR(&POST) TYPE(*CHAR) LEN(1) VALUE('P')
             DCL        VAR(&ANAL) TYPE(*CHAR) LEN(1) VALUE(' ')

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)


/* OVERRIDE  FOR PRINTING  */

/*           OVRPRTF    FILE(PKTBPFR$) OUTQ(&PDEV1) FORMTYPE(*STD) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)  +
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKTCPFR$) OUTQ(&PDEV1) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)                          +
             CALL       PGM(QCMDEXC) PARM(&CMD 512)          */

/*           SBDPRTOVR  CMDVAR(&CMD) FILE(PKMLPFR$) OUTQ(&PDEV1) +  */
/*                        COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) + */
/*                        LETTER(*YES)                              */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLEMPFR$) OUTQ(&PDEV1) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT CARCASS SUMMARY AND DETAIL                      */

 /*          CALL       PGM(PKTBPFR) PARM(&RTN &COMP &BYORD &LOAD +
                                          &TATTOO &ANAL)        */

             CALL       PGM(PKMOXFR) PARM(&RTN &POST &ANAL &SORT +
                          &OMS &NODATA)


             DLTOVR *ALL

             ENDPGM
