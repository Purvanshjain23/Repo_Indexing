/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  THIS CL IS CALLED FROM THE SCHEDULER TO PROCESS     */
/*              THE CARRIER INTERNET LOAD STOP DATA UPL FILE.       */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PROCESS CARRIER INTERNET LOAD STOP DATA FILE       */
/*  DATE      - 02/09/98                                            */
/*                                                                  */
/*  DESCRIPTION -  PROCESS THE LOAD STOP DATA FILE UPLOADED FROM    */
/*                 THE INTERNET.                                    */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*                                                                  */
/********************************************************************/
/*  REVISIONS                                                       */
/*                                                                  */
/*                                                                  */
/********************************************************************/

 PDVMUPC:    PGM        PARM(&RETURN)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &ARRN       *CHAR   (5)           /* LAST RRN                  */
        DCL    &LRRN       *DEC    (5 0)         /* LAST RRN                  */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   RETRIEVE LASTRRN DATA AREA                               */
/********************************************************************/

             RTVDTAARA  DTAARA(LASTRRN    (1 5)) RTNVAR(&ARRN)
             CHGVAR     VAR(&LRRN) VALUE(&ARRN)

             CALL       PGM(PDJYUPR) PARM(&RETURN &LRRN)
             IF COND(&RETURN *NE '      ') THEN(GOTO USERQUIT)

             CHGVAR     VAR(&ARRN) VALUE(&LRRN)
             CHGDTAARA  DTAARA(LASTRRN (1 5)) VALUE(&ARRN)

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
