/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING PREDICTED MKT PRICES         */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  1/02/96                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPN         -  COMPANY DFLT  (PASSED)                 */
/*         &MRKTCD        -  MARKET CODE   (PASSED)                 */
/*         &COMGRP        -  COMMODITY GROUP (PASSED)               */
/*         &DATEN         -  DATE    (PASSED)                       */
/*         &RPT1          -  REPORT SEL 1  (PASSED)                 */
/*         &RPT2          -  REPORT SEL 2  (PASSED)                 */
/*         &RPT3          -  REPORT SEL 3  (PASSED)                 */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 PDF4CLP:    PGM        PARM(&COMPN &MRKTCD &COMGRP &DATEN &RPT1 +
                          &RPT2 &RPT3 &APPLIC &OUTQ &HOLD &SAVE +
                          &COPYN &NIGHTQ)
        DCL    &MRKTCD      *CHAR  (03)
        DCL    &COMGRP      *CHAR  (07)
        DCL    &DATEN       *DEC
        DCL    &COMPN       *DEC
        DCL    &RPT1        *CHAR  (01)
        DCL    &RPT2        *CHAR  (01)
        DCL    &RPT3        *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &DATE        *DEC   (7 0)
        DCL    &COMPANY     *DEC   (3 0)
             DCL        VAR(&COPY) TYPE(*DEC) LEN(2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)
             CHGVAR     VAR(&DATE)  VALUE(&DATEN)
             CHGVAR     VAR(&COMPANY)  VALUE(&COMPN)

/*********************************************************************/
/*   CALL LISTING PROGRAM                                            */
/*********************************************************************/
  /* COMMODITY MARKET PRICE LIST                   */

     IF COND(&RPT1 *EQ 'Y') THEN(DO)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDF0PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PDF0PFR) PARM(' ' &MRKTCD +
                          &COMGRP &DATE)
     ENDDO

     IF COND(&RPT2 *EQ 'Y') THEN(DO)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDF5PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PDF5PFR) PARM(' ' &COMPANY &MRKTCD +
                          &COMGRP &DATE)
     ENDDO


 ENDPGM:     ENDPGM
