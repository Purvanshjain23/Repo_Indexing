/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKOEUPR                                       */
/*  PROGRAM NAME:    PRINT RESERVE ACCOUNT                         */
/*  PROGRAMMER:      PEG LYONS                                     */
/*  CREATE DATE:     05/11/98                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH SYNON 6.0 SBMJOB    */
/*                   FUNCTION.                                     */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&PDEV &HOLD &SAVE  +
                          &CONTR &XCOPIES &SYSNAM)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV)    TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)
             DCL        VAR(&SYSNAM)  TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&CONTR)   TYPE(*CHAR) LEN(5)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES)  TYPE(*DEC) LEN(2 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD)     TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKOBPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PROGRAM                                                  */
             CALL       PGM(PKOBPFR) PARM(' ' &CONTR &SYSNAM)

             DLTOVR *ALL
             ENDPGM
