/*  PROGRAM NAME:    PRINT DELIVERY SCHEDULE                       */
/*  PROGRAMMER:      SUSAN L. MASON                                */
/*  CREATE DATE:     03/09/95                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PMT OF PRT DELIVERY SCHEDULE             */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  04/20/95                                                       */
/*  Rose M. CENTONZE Added SBDPRTOVR code and command.             */
/*                                                                 */
/*  08/29/95                                                       */
/*  Rose M. CENTONZE Changed Delv Date input parm to kill date     */
/*  12/17/98                                                       */
/*  Rose M. CENTONZE Changed Date input parm to ALPHA FOR Y2K      */
/*  02/28/03                                                       */
/*  Rose M. CENTONZE ADDED TO DATE PARM                            */
/*  08/08/18 DN  T13559 - Added 'PL Location Code' as input parm   */
/*     to allow report PKARPFR to print down to the Producer       */
/*     Location level.                                             */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &XCOMP +
                  &XKILL &XPP &SRCTYP &SRCCDE &PRDCDE &XTODAT &PRLOCD)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XKILL) TYPE(*CHAR) LEN(7)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(40)
             DCL        VAR(&XPP) TYPE(*CHAR) LEN(33)
             DCL        VAR(&NME) TYPE(*CHAR) LEN(40) +
                          VALUE('Seaboard Farms, Inc.')
             DCL        VAR(&SRCTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SRCCDE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PRDCDE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&XTODAT) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PRLOCD) TYPE(*CHAR) LEN(6)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KILL) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&FRDAT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TODAT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&Y) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&Z) TYPE(*DEC) LEN(2 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
 LOOP:          CHGVAR &Y VALUE(&Y + 1)
             IF         COND(&Y = 40) THEN(GOTO CMDLBL(GOON))
             IF         COND((%SST(&NME &Y 2)) = '  ') THEN(DO)
             CHGVAR &Z VALUE(40 - &Y)
             CHGVAR     VAR(%SST(&NME &Y &Z)) VALUE(' ')
             GOTO       CMDLBL(GOON)
             ENDDO
             GOTO       CMDLBL(LOOP)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
  GOON:      CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&APP) VALUE(&XPP)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&KILL) VALUE(&XKILL)
             CHGVAR     VAR(&FRDAT) VALUE(&XKILL)
             CHGVAR     VAR(&TODAT) VALUE(&XTODAT)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKARPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKARPFR) PARM(' ' &COMP &KILL &NME &APP +
                          &SRCTYP &SRCCDE &PRDCDE &FRDAT &TODAT &PRLOCD)

             DLTOVR *ALL
             ENDPGM
