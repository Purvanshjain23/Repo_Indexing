/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKB6UPR                                       */
/*  PROGRAM NAME:    PRINT RECEIVING REPORT                        */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     09/11/96                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT FOR PRINT.                        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  R CENTONZE      10/23/96    ADD PARM TO PRINT DETAIL,SUMMARY,  */
/*                              OR LOG RECEIVING REPORT.           */
/*  S MASON         01/29/97    POSITION ON FROM DATE              */
/*  R CENTONZE      04/18/97    ADD PARM TO SELECT KILL SHIFT      */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &ACO +
                          &ADFROM &ADTO &CLTYP &SRTYP &DTLSUM +
                          &KSHIFT &AFSITE &PROD &OMS)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN)     TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV)    TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS)     TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP)     TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&ADTO)    TYPE(*CHAR) LEN(7)
             DCL        VAR(&ADFROM)  TYPE(*CHAR) LEN(7)
             DCL        VAR(&ACO)     TYPE(*CHAR) LEN(3)
             DCL        VAR(&CLTYP)   TYPE(*CHAR) LEN(1)
             DCL        VAR(&SRTYP)   TYPE(*CHAR) LEN(1)
             DCL        VAR(&DTLSUM)  TYPE(*CHAR) LEN(1)
             DCL        VAR(&KSHIFT)  TYPE(*CHAR) LEN(1)
             DCL        VAR(&AFSITE)  TYPE(*CHAR) LEN(5)
             DCL        VAR(&PROD)  TYPE(*CHAR) LEN(6)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES)  TYPE(*DEC)  LEN(2 0)
             DCL        VAR(&CO)      TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&DFROM)   TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&DTO)     TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&FSITE)   TYPE(*DEC)  LEN(5 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD)     TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&CO) VALUE(&ACO)
             CHGVAR     VAR(&DFROM) VALUE(&ADFROM)
             CHGVAR     VAR(&DTO) VALUE(&ADTO)
             CHGVAR     VAR(&FSITE) VALUE(&AFSITE)

/****************************************************************/
/*  PRINT RECEIVING BY DAY                                      */
/****************************************************************/

             IF         COND(&DTLSUM = 'D') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKBOPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKBOPFR) PARM(' ' &CO &DFROM &SRTYP +
                    &CLTYP &KSHIFT &FSITE &PROD &OMS &DFROM &DTO)

             DLTOVR *ALL
             ENDDO

/****************************************************************/
/*  PRINT RECEIVING SUMMARY                                     */
/****************************************************************/

             IF         COND(&DTLSUM = 'S') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKDWPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKDWPFR) PARM(' ' &CO &SRTYP &CLTYP &OMS +
                          &DFROM &DTO)

             DLTOVR *ALL
             ENDDO

/****************************************************************/
/*  PRINT RECEIVING LOG                                         */
/****************************************************************/

             IF         COND(&DTLSUM = 'L') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKD8PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKD8PFR) PARM(' ' &CO &DFROM &SRTYP +
                          &CLTYP &OMS &DFROM &DTO)

             DLTOVR *ALL
             ENDDO

/****************************************************************/
/*  PRINT FARM SITE DETAILS                                     */
/****************************************************************/

             IF         COND(&DTLSUM = 'F') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKO0PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKO0PFR) PARM(' ' &CO &DFROM &SRTYP +
                          &CLTYP &KSHIFT &FSITE &OMS &DFROM &DTO)

             DLTOVR *ALL
             ENDDO

             ENDPGM
