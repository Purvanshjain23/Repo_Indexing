 PPWQUPC:    PGM        PARM(&RETURN &FDATEN  &TDATEN &UPDYN &COMPN +
                        &OUTQ &HOLD &SAVE &COPYN)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &FDATEN     *DEC
        DCL    &TDATEN     *DEC
        DCL    &COMPN      *DEC
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COPYN      *DEC
        DCL    &UPDYN      *CHAR  LEN(1)

/* INTERNAL FIELDS                                                            */
        DCL    &FDATE      *DEC  LEN(7 0)
        DCL    &TDATE      *DEC  LEN(7 0)
        DCL    &COMP       *DEC  LEN(3 0)
        DCL    &COPY       *DEC  LEN(2 0)
        DCL    &COMMAND    *CHAR   (512)         /* message                   */
/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&FDATE) VALUE(&FDATEN)
             CHGVAR     VAR(&TDATE) VALUE(&TDATEN)
             CHGVAR     VAR(&COMP) VALUE(&COMPN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPWMPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PPWMPFR) PARM(' '  &FDATE +
               &TDATE &UPDYN  &COMP)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
 END:        ENDPGM
