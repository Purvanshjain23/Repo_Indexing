/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  CMPSNDEML                                     */
/*  PROGRAM NAME:    Send the CMP Program and its records          */
/*                   in a xls spread sheet to the user             */
/*  PROGRAMMER:      Susan Mason                                   */
/*  CREATE DATE:     08/22/08                                      */
/*                                                                 */
/*  FUNCTION:        this cl was developed to send the CMP         */
/*                   program and its lower level files             */
/*                   due to the fact the user selected             */
/*                   the mass delete.                              */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&CMPCDE &EMAIL)

             DCL        VAR(&CMPCDE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)
             DCL        VAR(&TEXT) TYPE(*CHAR) LEN(50) +
                          VALUE('Carcass Merit Program was selected +
                          for delete')

/*****************************************************************/
             EXECUTE    VIEW(*LIBL/CMPPRTQ01) PCFMT(*XLS) +
                          REPLACE(*YES) RECIPIENT(&EMAIL) +
                          EMLMSG(&TEXT) SETVAR((&CMPCDE &CMPCDE))


             EXECUTE    VIEW(*LIBL/CMPPRTQ02) PCFMT(*XLS) +
                          REPLACE(*YES) RECIPIENT(&EMAIL) +
                          EMLMSG(&TEXT) SETVAR((&CMPCDE &CMPCDE))


             EXECUTE    VIEW(*LIBL/CMPPRTQ03) PCFMT(*XLS) +
                          REPLACE(*YES) RECIPIENT(&EMAIL) +
                          EMLMSG(&TEXT) SETVAR((&CMPCDE &CMPCDE))


             EXECUTE    VIEW(*LIBL/CMPPRTQ04) PCFMT(*XLS) +
                          REPLACE(*YES) RECIPIENT(&EMAIL) +
                          EMLMSG(&TEXT) SETVAR((&CMPCDE &CMPCDE))

ENDPGM
