/*                                                                   */
/*  PROGRAM NUMBER:  HP471CL                                         */
/*  PROGRAM NAME:    PRELIMINARY EOP EDITS                           */
/*  PROGRAMMER:      LEANNE FEDOR                                    */
/*  CREATE DATE:     05/13/04                                        */
/*                                                                   */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH QCMDEXC FROM          */
/*                   HP471-SPECIFY PRELIMINARY EOP EDITS.            */
/*                                                                   */
/*                   OPTIONS ARE PASSED THROUGH THE LDA.             */
/*                                                                   */
/*  MODIFIED BY:     BARB GUTIERREZ   3/21/2016                      */
/*                   ADDED COMPANY NUMBER TO SELECTION SCREEN TO     */
/*                   PASS INTO THE REPORTS.   T005449.               */
/*                                                                   */
/*******************************************************************/

             PGM

             DCL        VAR(&ERROR) TYPE(*CHAR) LEN(1)

             DCL        VAR(&LDCOCD) TYPE(*CHAR) LEN(3)
             DCL        VAR(&LDCLS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&LDGRP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&LDNEG) TYPE(*CHAR) LEN(1)
             DCL        VAR(&LDLOW) TYPE(*CHAR) LEN(1)
             DCL        VAR(&LDEXP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&LDFED) TYPE(*CHAR) LEN(1)

             CHGVAR     VAR(&ERROR) VALUE('N')
             CHGVAR     VAR(&LDCOCD) VALUE(%SST(*LDA  200 3))
             CHGVAR     VAR(&LDCLS) VALUE(%SST(*LDA 300 1))
             CHGVAR     VAR(&LDGRP) VALUE(%SST(*LDA 301 1))
             CHGVAR     VAR(&LDNEG) VALUE(%SST(*LDA 302 1))
             CHGVAR     VAR(&LDEXP) VALUE(%SST(*LDA 303 1))
             CHGVAR     VAR(&LDLOW) VALUE(%SST(*LDA 304 1))
             CHGVAR     VAR(&LDFED) VALUE(%SST(*LDA 305 1))


/*-------------------------------------------------------------------*/
/*  PRELIMINARY EDITS AND EXCEPTION REPORTS                          */
/*-------------------------------------------------------------------*/

/*  PERIOD CLOSE EDITS                                               */

             IF         COND(&LDCLS *EQ 'X') THEN(DO)
             CRTDUPOBJL OBJ(HSP320) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CALL       PGM(HP373CL) PARM(&ERROR &LDCOCD)
             ENDDO

/*  DISPOSED GROUP EDITS                                             */

             IF         COND(&LDGRP *EQ 'X') THEN(DO)
             CALL       PGM(HP372CL) PARM(&ERROR &LDCOCD)
             ENDDO

/*  NEGATIVE INVENTORY EXCEPTION REPORT                              */

             IF         COND(&LDNEG *EQ 'X') THEN(DO)
             CALL       PGM(HP3081CL) PARM(&LDCOCD)
             ENDDO

/* LOW INVENTORY REPORT                                              */

             IF         COND(&LDLOW *EQ 'X') THEN(DO)
             CALL       PGM(HP3084CL) PARM(&LDCOCD)
             ENDDO

/*  G/L ACCOUNT/HPS EXPENSE CODE EXCEPTION REPORT                    */

             IF         COND(&LDEXP *EQ 'X') THEN(DO)
             CALL       PGM(HP3082CL) PARM(&LDCOCD)
             ENDDO


/* FEED REPORT FOR CREATED GROUPS                                    */

             IF         COND(&LDFED *EQ 'X') THEN(DO)
             CALL       PGM(HP3085CL) PARM(&LDCOCD)
             ENDDO

             ENDPGM
