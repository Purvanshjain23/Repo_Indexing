/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKUYUPR                                       */
/*  PROGRAM NAME:    PRINT HOG POPULATION REPORT FOR PRIMAL OPT    */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     03/28/01                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT PRINT HOG POPULATION RPT .        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  ROSE CENTONZE - 08/16/96      ADD "TYPE OF POPULATION" PARM.    */
/*                  THE PARM IS &TYPOP.                            */
/*  ROSE CENTONZE - 01/07/97      ADD FARM SITE PARM (&XFSCD)      */
/*  PURVA DROGE   - 02/28/99      ADD KILL SHIFT     (&SHIFT)      */
/*  ROSE CENTONZE - 09/13/99      ADD PRODUCER LOCATION (&PLOC)  */
/*  ROSE CENTONZE - 03/23/01      CALL PGM TO BLD OUTF           */
/*                                DTL FILE IS FORMAT *XLS        */
/*                                SUMMARY FILE IS FORMAT *DELIMITED  */
/*                                CALL PKW6XFR INSTEAD OF PKUXXFR   */
/*                                BECAUSE THE KILL DT MUST BE IN THE */
/*                                REPORT BREAK FIELD                */
/*  ROSE CENTONZE - 04/26/01      ADD OUTPUT FIELD AVG HOT WGT     */
/*******************************************************************/
             PGM        PARM(&RTN &RPT &TYD &ACO &SRCT &PAYT &SRCD +
                          &PROD &PLOC &ATO &XFROM &XTO &XFSCD +
                          &SHIFT &OUTF &TYPOP &OMS &CUTMTX &XMODY +
                  &DOCUM  &PDEV &HOLD &SAVE &XCOPIES)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&RPT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&TYD) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SRCT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PAYT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6)
             DCL        VAR(&PLOC) TYPE(*CHAR) LEN(6)
             DCL        VAR(&SRCD) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XFROM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTO) TYPE(*CHAR) LEN(7)
             DCL        VAR(&ATO) TYPE(*CHAR) LEN(5)
             DCL        VAR(&ACO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&CUTMTX) TYPE(*CHAR) LEN(6)
             DCL        VAR(&XMODY) TYPE(*CHAR) LEN(15)
             DCL        VAR(&TYPOP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&XFSCD) TYPE(*CHAR) LEN(5)
             DCL        VAR(&SHIFT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OUTF) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&MODY) TYPE(*DEC) LEN(4 0)
             DCL        VAR(&FROM) TYPE(*DEC) LEN(7)
             DCL        VAR(&TO) TYPE(*DEC) LEN(7)
             DCL        VAR(&TTN) TYPE(*DEC) LEN(5)
             DCL        VAR(&FSCD) TYPE(*DEC) LEN(5)
             DCL        VAR(&FIRSTD) TYPE(*DEC) LEN(7)
             DCL        VAR(&LASTD) TYPE(*DEC) LEN(7)
             DCL        VAR(&FROMA) TYPE(*CHAR) LEN(7)
             DCL        VAR(&NXTDY) TYPE(*DEC) LEN(7)
             DCL        VAR(&NXTDYA) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DAY) TYPE(*DEC) LEN(8) VALUE(1)

        DCL    &DOCUMENT  *CHAR   (10)
        DCL    &DOCUM     *CHAR    (06)           /*  FOR PASSING    */
        DCL    &PCFILE    *CHAR   (32)
        DCL    &SEL       *CHAR    (1500)         /*  FOR PASSING    */
        DCL    &DOCUMSM   *CHAR    (08)           /*  SUMMARY FILE   */
        DCL    &DOCUMDT   *CHAR    (08)           /*  DETAIL FILE    */
/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FROM) VALUE(&XFROM)
             CHGVAR     VAR(&TO) VALUE(&XTO)
             CHGVAR     VAR(&TTN) VALUE(&ATO)
             CHGVAR     VAR(&CO) VALUE(&ACO)
             CHGVAR     VAR(&MODY) VALUE(&XMODY)
             CHGVAR     VAR(&FSCD) VALUE(&XFSCD)

/* CREATE HOG POPULATION REPORT FILES IN QTEMP                  */
/*           CLRPFM     FILE(PKC9CPP)                           */
/*           CLRPFM     FILE(PKDACPP)                           */
/*  GOTO       CMDLBL(FTPFILES)                                 */
/*        HOG POPULATION REPORT DETAIL FILE */
             CRTDUPOBJL  OBJ(PKC9CPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKC9CPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKC9CPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKC9CPL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

/*        HOG POPULATION REPORT HEADER FILE                     */

             CRTDUPOBJL  OBJ(PKDACPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDACPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PKDACPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

 FTPFILES:   CLRPFM     FILE(PKFBCPP)
             CLRPFM     FILE(PKFCCPP)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKUZPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/*                                                             */
 /* LOOP UNTIL THE BUILD HAS RAN FOR THE EACH DATE IN THE RANGE */
/* BUILD OUTFILE WITH ONE RECORD PER EACH COLUMN ON THE REPORT */
/*   OUTFILES:  PKFBCPP,PKFCCPP                                 */
/*                                                             */
 SAVEDT:     CHGVAR     VAR(&FIRSTD) VALUE(&FROM)
             CHGVAR     VAR(&LASTD) VALUE(&TO)

 LOOP:       CHGVAR     VAR(&TO) VALUE(&FROM)

/* ORIGINAL:CALL       PGM(PKUXXFR) PARM(' ' &RPT &TYD &CO &SRCT +
                          &PAYT &TTN &FROM &TO &FSCD &SHIFT &SRCD +
                          &PROD &PLOC &CUTMTX &MODY)           */

            CALL       PGM(PKW6XFR) PARM(' ' &RPT &TYD &CO &SRCT +
                          &PAYT &TTN &FROM &TO &FSCD &SHIFT &SRCD +
                          &PROD &PLOC &CUTMTX &MODY &FIRSTD &LASTD)

/*  PRINT WONT WORK RIGHT FOR THIS PURPOSE                 */
/*    CALL HOG POPULATION REPORT                                 */


  /*         CALL       PGM(PKUZPFR) PARM(' ' &LPP &RPT &TYD &CO +
                          &SRCT &PAYT &TTN &FROM &TO &FSCD &SHIFT +
                     &SRCD &PROD &PLOC &CUTMTX &MODY &TYPOP &OUTF) */
     /* CREATE NULL RECORDS IN THE WORKFILES  */
       CALL PKW7XFR (' ' &FROM &FIRSTD &LASTD)
/*                                                             */
/*   GET THE NEXT DAY, DO LOOP AGAIN IF END DATE NOT EXCEEDED  */
/*                                                             */
             CALL       PGM(PDX8XFR) PARM('' &FROM &NXTDY &DAY)
             CHGVAR     VAR(&FROM) VALUE(&NXTDY)
             IF         COND(&FROM > &LASTD) THEN(GOTO CMDLBL(FTP))
             GOTO       CMDLBL(LOOP)
FTP:
 /*------------------------------------------------*/
   /* RUN SQL TO PUSH document to /DOWNLOAD.PORK    */
 /*------------------------------------------------*/
             IF         COND(&OUTF *EQ 'Y') THEN(DO)

             ADDLIBLE LIB(SEQUEL) POSITION(*LAST)
             MONMSG MSGID(CPF2103)
   /*------------------------------------------------*/
   /* EXECUTE SEQUEL TO SELECT THE WORKF FILE AND FTP IT -*/
   /*-----------------------------------------------*/
 SUMMARY:    CHGVAR     VAR(&DOCUMSM) VALUE(&DOCUM *TCAT 'SM')
             CHGVAR     VAR(&SEL) VALUE('SELECT FBK1ST +
                          COLHDG("DIST" "TYPE"), FBFJA1 COLHDG("MIN +
                          CARC" "WGT"), FBFKA1 COLHDG("MAX CARC" +
                          "WGT"), FBFIA1 COLHDG("TYPE"), FBCMDT +
                          COLHDG("FROM KILL" "DATE"), FBCODT +
                          COLHDG("TO KILL" "DATE"), FBFLA1 +
                          COLHDG("MIN LIVE" "WGT"), FBFMA1 +
                          COLHDG("MAX LIVE" "WGT"), FBFNA1 +
        COLHDG("COUNT"), FBFPA1 COLHDG("AVG HOT" "WGT") +
        FROM  PKFBCPP ORDER BY FBK1ST, FBFJA1')
             EXECUTE    SQL(&SEL) MBROPT(*REPLACE) NBRRCDS(*ALL) +
                          KEYFLDCNT(*ALL) ALWNULL(*NO) +
                          TOFLR('GUY.IS') TODOC(&DOCUMSM) +
                          PCFMT(*DELIMITED) REPLACE(*YES) TEXT('Hog +
                          Population') OPTIMIZE(*TOTAL) +
                          ALWCPY(*YES) MSG(*YES) UNIQUEKEY(*NONE) +
                          IGNDECERR(*NO) DTSTYLE(*FIELD '/' *FIELD ':')

   /*------------------------------------*/
   /* PUSH documents to /DOWNLOAD.PORK    */
   /*------------------------------------*/
             CHGVAR VAR(&DOCUMENT) VALUE(&DOCUMSM)
             CHGVAR VAR(&PCFILE)   VALUE(&DOCUMSM *TCAT '.CSV')
             CALL PGM(FTPTEST) PARM(&DOCUMENT &PCFILE)

DETAIL:      CHGVAR     VAR(&DOCUMDT) VALUE(&DOCUM *TCAT 'DT')
             CHGVAR     VAR(&SEL) VALUE('SELECT FCK1ST COLHDG("DIST" +
                          "TYPE"), FCB0DT COLHDG("KILL" "DATE"), +
                          FCQUNB COLHDG("MIN CARC" "WGT"), FCQVNB +
                          COLHDG("MAX CARC" "WGT"), FCFIA1 +
                          COLHDG("TYPE"), FCQWNB COLHDG("MIN LIVE" +
    "WGT"), FCQXNB COLHDG("MAX LIVE" "WGT"), +
 FCFHA1 COLHDG("COUNT") FROM  PKFCCPP ORDER BY FCK1ST,FCB0DT,FCQUNB')
            EXECUTE    SQL(&SEL) MBROPT(*REPLACE) NBRRCDS(*ALL) +
                          KEYFLDCNT(*ALL) ALWNULL(*NO) +
                          TOFLR('GUY.IS') TODOC(&DOCUMDT) +
                          PCFMT(*DELIMITED) REPLACE(*YES) TEXT('Hog +
                          Population Detail') OPTIMIZE(*TOTAL) +
                          ALWCPY(*YES) MSG(*YES) UNIQUEKEY(*NONE) +
                          IGNDECERR(*NO) DTSTYLE(*FIELD '/' *FIELD ':')

   /*------------------------------------*/
   /* PUSH documents to /DOWNLOAD.PORK    */
   /*------------------------------------*/
             CHGVAR VAR(&DOCUMENT) VALUE(&DOCUMDT)
             CHGVAR VAR(&PCFILE)   VALUE(&DOCUMDT *TCAT '.CSV')
             CALL PGM(FTPTEST) PARM(&DOCUMENT &PCFILE)
             ENDDO
             DLTOVR *ALL
             ENDPGM
