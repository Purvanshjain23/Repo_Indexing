/*********************************************************************/
/*     Program: PMEQUPC                                              */
/*       Title: CALL THE CUSTOMER A/R ACTIVITY REPORT PROGRAM        */
/*        Date: 03/09/2007                                           */
/* Description: - THIS PROGRAM IS SUBMITTED FROM PMETUPC             */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/* &RETURN     - RETURN CODE                                         */
/* &APPLIC     - APPLICATION CODE                                    */
/* &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT               */
/*                SCREEN)                                            */
/* &OUTQ       -  OUTPUT QUEUE                                       */
/* &HOLD       -  HOLD STATUS                                        */
/* &SAVE       -  SAVE STATUS                                        */
/* &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q              */
/* &NIGHTQ     -  NIGHT JOB QUEUE                                    */
/* &CUSTN      -  CUSTOMER#     (PASSED)                             */
/* &COPIESN    -  NO. OF COPIES (PASSED)                             */
/* &FDATEN     -  FROM DATE (PASSED AS 15.5)                         */
/* &TDATEN     -  TO DATE   (PASSED AS 15.5)                         */
/* &FDINV      -  FROM INVOICE # (PASSED AS 15.5)                    */
/* &TOINV      -  TO INVOICE # (PASSED AS 15.5)                      */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*                                                                   */
/*   Date     Pgmr     Description                                   */
/* -------- ---------- --------------------------------------------  */
/*********************************************************************/
 PMEQUPC:    PGM        PARM(&RETURN &APPLIC &DESC &OUTQ &HOLD &SAVE +
                          &NQSTAT &NIGHTQ &CUSTN &COPIESN &FDATEN +
                          &TDATEN &FRINV &TOINV)

/*-------------------------------------------------------------------*/
/* Declare Variables                                                 */
/*-------------------------------------------------------------------*/
/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)  /* RETURN CODE               */
        DCL    &APPLIC     *CHAR   (06) /* APPLICATION CODE          */
        DCL    &DESC       *CHAR   (25) /* PRT PGM DESCRIPTION       */
        DCL    &COPYN      *DEC         /* NO. OF COPIES             */
        DCL    &OUTQ       *CHAR   (10) /* OUTPUT QUEUE              */
        DCL    &HOLD       *CHAR   (4)  /* HOLD STATUS               */
        DCL    &SAVE       *CHAR   (4)  /* SAVE STATUS               */
        DCL    &FDATEN     *DEC         /* FROM DATE 15.5            */
        DCL    &TDATEN     *DEC         /* THRU DATE 15.5            */
        DCL    &FRDATE     *DEC    (7 0)/* FROM DATE 7.0             */
        DCL    &TODATE     *DEC    (7 0)/* THRU DATE 7.0             */
        DCL    &CUST       *DEC    (7 0)/* CUSTOMER NUMBER           */
        DCL    &NQSTAT     *CHAR   (01) /* IF MUST RUN IN NIGHT Q    */
        DCL    &NIGHTQ     *CHAR   (10) /* NIGHT JOB QUEUE           */
        DCL    &CUSTN      *DEC         /* CUSTOMER NUMBER           */
        DCL    &COPIESN    *DEC         /* NO. OF COPIES (PASSED)    */
        DCL    &FRINV      *DEC         /* FROM INV# 15.5            */
        DCL    &TOINV      *DEC         /* THRU INV# 15.5            */
        DCL    &FRINV#     *DEC    (7 0)/* FROM INV# 7.0             */
        DCL    &TOINV#     *DEC    (7 0)/* THRU INV# 7.0             */

/* Internal Work Fields                                              */
             DCL        VAR(&COPY) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&MSG) TYPE(*CHAR) LEN(512)
             DCL        VAR(&MSGID) TYPE(*CHAR) LEN(7)
             DCL        VAR(&MSGL) TYPE(*DEC) LEN(5 0)

/*-------------------------------------------------------------------*/
/* UNCOMMENT FOR TESTING                                             */
/********    CHGJOB     LOG(4 99 *SECLVL)                   **********/
/*-------------------------------------------------------------------*/
/* Monitor for all message at the program level                      */
/*-------------------------------------------------------------------*/
             MONMSG MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/*-------------------------------------------------------------------*/
/* CHANGE INPUT PARMS TO CORRECT LENGTH AND TYPE                     */
/*-------------------------------------------------------------------*/
             CHGVAR     VAR(&CUST) VALUE(&CUSTN)
             CHGVAR     VAR(&FRDATE) VALUE(&FDATEN)
             CHGVAR     VAR(&TODATE) VALUE(&TDATEN)
             CHGVAR     VAR(&COPY) VALUE(&COPIESN)
             CHGVAR     VAR(&FRINV#) VALUE(&FRINV)
             CHGVAR     VAR(&TOINV#) VALUE(&TOINV)

/*-------------------------------------------------------------------*/
/* CALL PRINT PROGRAM BY INVOICE NUMBER - COMMENTED OUT FOR NOW      */
/*-------------------------------------------------------------------*/
/**********  OVRPRTF    FILE(PME2PFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)                  +
                                                                   +
             CALL       PGM(PME2PFR) PARM('' &CUST &FRINV# &TOINV# +
                          &FRDATE &TODATE)          ***************/

/*-------------------------------------------------------------------*/
/* CALL PRINT PROGRAM BY INVOICE DATE                                */
/*-------------------------------------------------------------------*/
             OVRPRTF    FILE(PMEOPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(PMEOPFR) PARM('' &CUST &FRDATE &TODATE)
             GOTO       CMDLBL(END)

/*-------------------------------------------------------------------*/
/* Errors                                                            */
/*-------------------------------------------------------------------*/
 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))
             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000)

/*-------------------------------------------------------------------*/
/* End of Program                                                    */
/*-------------------------------------------------------------------*/
 END:        ENDPGM
