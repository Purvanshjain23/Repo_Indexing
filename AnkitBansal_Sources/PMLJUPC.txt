PGM PARM(&CRCDE)

DCL &CRCDE *CHAR 3  /* Accrual Code */
DCL &LIBRARY *CHAR 10  /* Override Library  */

/* Override to DFDATA for first run */
CHGVAR &LIBRARY 'DFDATA'
OVRDBF     FILE(REBATBGX) TOFILE(&LIBRARY/REBATBGX)

/* Call EUP Bld Cust Rebate    */
CALL PMLIUPR PARM(&CRCDE &LIBRARY)

/* Delete overrides */
DLTOVR  REBATBGX

/* Override to JRDATA for second run */
CHGVAR &LIBRARY 'JRDATA'
OVRDBF     FILE(REBATBGX) TOFILE(&LIBRARY/REBATBGX)

/* Call EUP Bld Cust Rebate    */
CALL PMLIUPR PARM(&CRCDE &LIBRARY)

/* Delete overrides */
DLTOVR  REBATBGX
ENDPGM
