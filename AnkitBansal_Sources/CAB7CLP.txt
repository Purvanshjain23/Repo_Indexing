/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING UNIT OF MEASURE LISTING      */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  2/28/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &RCDSTS        -  RECORD STATUS                          */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 CAB7CLP:    PGM        PARM(&RCDSTS &APPLIC &OUTQ &HOLD &SAVE &COPYN +
                             &NIGHTQ)

        DCL    &RCDSTS      *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND    *CHAR   (512)

        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(CAB7PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(CAB7PFR) PARM(' ' &RCDSTS)

 ENDPGM:     ENDPGM
