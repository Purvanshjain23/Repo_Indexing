//?Execute user function

// PAR.Extended Amount = CON.*ZERO
PAR.Extended_Amount = *ZERO;

//?UOM Type
CASE;

// IF PAR.U/M - Base Price is Carcass Hundred Weight
IF PAR.U_M_Base_Price = 'CW';

// OR PAR.U/M - Base Price is Pounds
OR PAR.U_M_Base_Price = 'LB';

// OR PAR.U/M - Base Price is Tons
OR PAR.U_M_Base_Price = 'TN';

// OR PAR.U/M - Base Price is Thousand
OR PAR.U_M_Base_Price = 'M';

// LCL.UOM Type USR = CND.Weight
LCL.UOM_Type_USR = 'W';

//?UOM Type
// IF *OTHERWISE
IF *OTHERWISE;

// LCL.UOM Type USR = CND.Quantity
LCL.UOM_Type_USR = 'Q';

ENDIF;

//?UOM Type = Weight
CASE;

// IF LCL.UOM Type USR is Weight
IF LCL.UOM_Type_USR = 'W';

//?Hundred Weight
CASE;

// IF PAR.U/M - Base Price is Carcass Hundred Weight
IF PAR.U_M_Base_Price = 'CW';

// PAR.Extended Amount =       *
PAR.Extended_Amount = PAR.Unit_Price * ( PAR.Weight_Shipped_Total / 100 'H' ) 'H';

ENDIF;

//?Thousand
CASE;

// IF PAR.U/M - Base Price is Thousand
IF PAR.U_M_Base_Price = 'M';

// PAR.Extended Amount =       *
PAR.Extended_Amount = PAR.Unit_Price * ( PAR.Weight_Shipped_Total / 1000 'H' ) 'H';

ENDIF;

//?Tons
CASE;

// IF PAR.U/M - Base Price is Tons
IF PAR.U_M_Base_Price = 'TN';

// PAR.Extended Amount =       *
PAR.Extended_Amount = PAR.Unit_Price * ( PAR.Weight_Shipped_Total / 2000 'H' ) 'H';

ENDIF;

//?Pounds
CASE;

// IF PAR.U/M - Base Price is Pounds
IF PAR.U_M_Base_Price = 'LB';

// PAR.Extended Amount =       *
PAR.Extended_Amount = PAR.Unit_Price * PAR.Weight_Shipped_Total 'H';

ENDIF;

ENDIF;

//?UOM Type = Quantity
CASE;

// IF LCL.UOM Type USR is Quantity
IF LCL.UOM_Type_USR = 'Q';

// PAR.Extended Amount =       *
PAR.Extended_Amount = PAR.Unit_Price * PAR.Quantity_Shipped_Total 'H';

ENDIF;
