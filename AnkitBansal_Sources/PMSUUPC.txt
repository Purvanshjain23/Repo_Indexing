/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING PCKG VARIANCE SUMMARY       */
/*                  TRANSACITONS THAT CAME UP AFTER PROD DATE END */
/*  SYSTEM    - INVENTORY MANAGEMENT                                */
/*  DATE      - 03/12/08                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &RCDSTS        -  PASSED RECORD STATUS                   */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
/*  RMC 06/18/03  ADDED EMAIL ADDRESS PARM                         */
/* RMC   2/03/05  MADE EMAIL ADDR 50 LONG, FROM 2 FIELDS BECAUSE OF */
/*                CL LIMITS                                    */
/* RMC   3/10/08  BUILD WORKF FIRST WITH FG ITEMS RR/AP/A TOTALS THEN */
/*                READ THAT IN THE PRTF, TO SPEED UP PROCESSING       */
/* RMC   3/12/08  ADDED &AFTPRD PARM: WILL BE "Y" HERE MEANING TO ACCUM */
/*                ALL TRANS FOUND FOR THE PRODUCTION DATE RANGE THAT */
/*                HAVE A TRANS DATE AFTER THE TO PROD DATE           */
/********************************************************************/
 PMSUUPC:    PGM        PARM(&RETURN &COMPNYN &XFDATE &XTDATE &OUTQ +
                          &HOLD &SAVE &COPYN &EMAIL1 &EMAIL2)
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &COMPNYN     *CHAR (3)
        DCL    &XFDATE       *CHAR  (07)
        DCL    &XTDATE       *CHAR  (07)
        DCL    &AFTPRD      *CHAR  (1)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *CHAR  (15)
        DCL    &NIGHTQ      *CHAR  (10)
             DCL        VAR(&EMAIL1) TYPE(*CHAR) LEN(25)
             DCL        VAR(&EMAIL2) TYPE(*CHAR) LEN(25)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)
        DCL    &COMMAND     *CHAR  (512)
        DCL     &DATA     *CHAR   10              /* DATA FILES      */
        DCL     &DATAH    *CHAR   10              /* HIST FILES      */

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &FDATE       *DEC   (7 0)
        DCL    &TDATE       *DEC   (7 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)
             CHGVAR     VAR(&FDATE)    VALUE(&XFDATE)
             CHGVAR     VAR(&TDATE)    VALUE(&XTDATE)
             ADDLIBLE   LIB(ESEND)
             MONMSG CPF0000
/* -----------------------------------------------------------------------*/
/*   BUILD WORKFILE WITH FG TRANSACTION TOTALS FOR : RR/AP/A     */
/* -----------------------------------------------------------------------*/
             CALL      PGM(CAQXXFR) PARM(' ' &DATA &DATAH)
             CRTDUPOBJ  OBJ(PMALCPP) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJ  OBJ(PMALCPL0) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJ  OBJ(PMALCPL1) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJ  OBJ(PMALCPL2) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CHGVAR     VAR(&AFTPRD) VALUE('Y') /* ONLY accum +
                          TRANSactions THAT HAVE A PROD DATE WITHIN +
                          THE PROD DATE RANGE BUT CAME UP AFTER THE +
                          TO PROD DATE  */

             CALL PMSTXFR ('' &COMPANY &FDATE &TDATE &AFTPRD)

/* -----------------------------------------------------------------------*/
/*   CALL PRINT FILE PER THE INPUT PARMS                        */
/* -----------------------------------------------------------------------*/
/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PMSVPFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)


 CALL:       CALL       PGM(PMSVPFR) PARM(' ' &COMPANY &FDATE &TDATE)
             IF         COND(&EMAIL1 *NE ' ') THEN(DO)
             CHGVAR &EMAIL VALUE(&EMAIL1 *CAT &EMAIL2)
             ESNDMAIL   RECIPIENT(&EMAIL) +
                          SUBJECT(VARIANCE) ATTLIST((* *PDF *N +
                       PMSVPFR$))
             ENDDO
ENDPGM:     ENDPGM
