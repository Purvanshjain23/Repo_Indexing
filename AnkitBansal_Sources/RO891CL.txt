             PGM

             DCL        VAR(&OUTQ) TYPE(*CHAR) LEN(10)
             DCL        VAR(&ERROR) TYPE(*CHAR) LEN(1)

             CHGVAR     VAR(&ERROR) VALUE(' ')
             CHGVAR     VAR(&OUTQ) VALUE(%SST(*LDA 401 10))

             CHKOBJ     OBJ(&OUTQ) OBJTYPE(*OUTQ)
             MONMSG     MSGID(CPF0000) EXEC(CHGVAR VAR(&ERROR)  +
                          VALUE('Y'))

             CALL       PGM(RO891) PARM(&ERROR)

             ENDPGM
