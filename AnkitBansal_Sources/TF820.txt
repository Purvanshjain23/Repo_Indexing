      * TITLE:       Retrieve the Date 7-Days-Prior (Synon formats)

      *            to that date. Both dates are in Synon format.
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      *
      /EJECT

      *
      ****************************************************************

      ****************************************************************

     D no              c                   'N'
      *
      ****************************************************************
      * ARRAYS AND TABLES
      ****************************************************************
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      * Workfields for date manipulation
      *
     D wkisodate       s               d   datfmt(*iso)
      *
      *
      * Parms
      *
     D xxindt7         s              7  0
     D xxoutdt7        s              7  0
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *

      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************

     C     *cymd         test(d)                 xxindt7                92
     C                   if        *in92 = *off                                 If OK date
     C     *cymd         move      xxindt7       wkisodate
     C                   subdur    7:*d          wkisodate
     C     *cymd         move      wkisodate     xxoutdt7
     C                   endif                                                  If OK date
      *
     C                   seton                                        lr
      /EJECT
      *---------------------------------------------------------------
