/*********************************************************************/
/*     Program: PDPOUPR                                              */
/*       Title: Execute the command SBDOVR                           */
/*        Date: 06/12/98                                             */
/* Description: - Construct the command string.                      */
/*              - Execute the command.                               */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &PRTFILE    Printer File Name                            */
/*    I     &OUTQ       Output Queue                                 */
/*    I     &HOLD       Hold Flag                                    */
/*    I     &SAVE       Save Flag                                    */
/*    I     &COPY       Number of Copies                             */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*                                                                   */
/*   Date     Pgmr     Description                                   */
/* -------- ---------- --------------------------------------------  */
/*                                                                   */
/*********************************************************************/
             PGM  PARM(&PRTFILE &OUTQ &HOLD &SAVE &COPY)

   /*-------------------*/
   /* Declare Variables */
   /*-------------------*/

             /* Parameter Fields */
             DCL  VAR(&PRTFILE) TYPE(*CHAR) LEN(10)
             DCL  VAR(&OUTQ)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&COPY)    TYPE(*CHAR) LEN(2)

             /* Internal Work Fields */
             DCL  VAR(&COPYN)   TYPE(*DEC)  LEN(2 0)
             DCL  VAR(&COMMAND) TYPE(*CHAR) LEN(512)

   /*-----------------------------*/
   /* Initialize & Load Variables */
   /*-----------------------------*/
             CHGVAR VAR(&COPYN)   VALUE(&COPY)

   /*---------------------------*/
   /* Create Work File in QTEMP */
   /*---------------------------*/
             SBDPRTOVR CMDVAR(&COMMAND) +
                       FILE(&PRTFILE) +
                       OUTQ(&OUTQ) +
                       COPIES(&COPYN) +
                       HOLD(&HOLD) +
                       SAVE(&SAVE)

             CHGVAR VAR(&COMMAND) +
                    VALUE(&COMMAND *BCAT 'OVRSCOPE(*JOB)')

             CALL PGM(QCMDEXC) PARM(&COMMAND 512)

   /*----------------*/
   /* End of Program */
   /*----------------*/
ENDPGM:
             ENDPGM
