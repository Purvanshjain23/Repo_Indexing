SELECT
  ALL       T01.FIL, T03.FLD, T04.L01, T04.L02, T04.L03, T04.L04, T04.L05,
            T04.L06
  FROM      OMSDEVMDL/YFILDTARFP T01 LEFT OUTER JOIN
            OMSDEVMDL/YRELDTARFP T02
  ON        T01.@@FIL = T02.@@OBJ LEFT OUTER JOIN
            OMSDEVMDL/YFLDDTARFP T03
  ON        T02.@@REF = T03.@@FLD LEFT OUTER JOIN
            OMSDEVMDL/YTXTDTARFP T04
  ON        T03.@@FLD = T04.@@OBJ
  WHERE     T01.FIL LIKE 'PS %'
    AND     T03.FLD <> '     '
