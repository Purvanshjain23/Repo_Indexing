             PGM        PARM(&FILE &DESCRIPT &RTNCDE)

             DCL        VAR(&FILE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&DESCRIPT) TYPE(*CHAR) LEN(50)
             DCL        VAR(&RTNCDE) TYPE(*CHAR) LEN(7)

             RTVMBRD    FILE(&FILE) TEXT(&DESCRIPT)
             MONMSG     MSGID(CPF3812 CPF9812) EXEC(CHGVAR +
                          VAR(&RTNCDE) VALUE(Y2U0005))

             RETURN
             ENDPGM
