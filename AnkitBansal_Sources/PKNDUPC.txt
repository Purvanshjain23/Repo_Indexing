/*  FUNCTION:        This CLP is submitted with an Execute Message */

/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */

/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &XCOMP &XDTE1 &XDTE2)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)

/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XDTE1) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XDTE2) TYPE(*CHAR) LEN(7)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&DTE1) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&DTE2) TYPE(*DEC) LEN(7 0)


/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&DTE1) VALUE(&XDTE1)
             CHGVAR     VAR(&DTE2) VALUE(&XDTE2)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             CALL       PGM(PKNBXFR) PARM(' ' &COMP &DTE1 &DTE2)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
