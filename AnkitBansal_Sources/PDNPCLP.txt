/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING ITEM DEFAULT LISTING         */
/*                                                                  */
/*  SYSTEM    - INVENTORY CONTROL                                   */
/*  DATE      -  5/08/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &RCDSTS        -  RECORD STATUS                          */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 PDNPCLP:    PGM        PARM(&RCDSTS &APPLIC &OUTQ &HOLD &SAVE &COPYN +
                             &NIGHTQ)

        DCL    &RCDSTS      *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND     *CHAR  (512)

        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDNPPFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDNPPFR) PARM(' ' &RCDSTS)

 ENDPGM:     ENDPGM
