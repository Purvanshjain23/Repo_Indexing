/* Some Dailys conversion programs requiring writing to or updating */
/* production files in DFDATA or JRDATA. In every case the process  */
/* does an OVRDBF explictly to the appropriate library.             */
/*                                                                  */
/* Copies of the files were made in DFDATATST and JRDATATST for the */
/* purpose of testing. Instead of updating the programs with the    */
/* correct library name, update programs will get the library name  */
/* from data area ISTJOHNLIB/DAILYLIB. That way, when testing is    */
/* complete the library names in the data area can be changed       */
/* instead of the programs.                                         */
/********************************************************************/
PGM PARM(&REQLIB &LIBRARY)

DCL &REQLIB *CHAR 2                        /* Library requested */
DCL &LIBRARY *CHAR 10                      /* Library name */

IF COND(&REQLIB *EQ 'DF') THEN(DO)
     RTVDTAARA  DTAARA(ISTJOHNLIB/DAILYLIB (1 10)) RTNVAR(&LIBRARY)
ENDDO

IF COND(&REQLIB *EQ 'JR') THEN(DO)
     RTVDTAARA  DTAARA(ISTJOHNLIB/DAILYLIB (11 10)) RTNVAR(&LIBRARY)
ENDDO

ENDPGM
