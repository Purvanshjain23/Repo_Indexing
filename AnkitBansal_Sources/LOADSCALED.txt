SELECT
  ALL       B3ANTM/10000 AS "HOUR", B3B6DT, (B3FMNB)
  FROM      PRKFLIB/PKB3CPP T01
  WHERE     B3B6DT >= 990104
    AND     B3ANTM <= 990123
    AND     B3ANTM >= 230000
    AND     B3ANTM <= 240000
    OR      B3B6DT >= 990105
    AND     B3ANTM <= 990123
    AND     B3ANTM >= 000000
    AND     B3ANTM <= 053000
  ORDER BY  002 ASC, 001 ASC
