/***************************************************************** */
/*  PROGRAM NUMBER --> POSTUPR                                     */
/*  PROGRAM NAME ----> CALCULATE BONING & EXPORT RATE              */
/*                                                                 */
/*  PROGRAMMER ------> PURVA DROGE                                 */
/*  DATE ------------> 12/15/02                                    */
/*                                                                 */
/*                                                                 */
/*  PROGRAM PURPOSE -> THIS CL CALLS A PROGRAM TO CACULATE THE     */
/*                     BONING RATE AND THE EXPORT RATE FOR USE     */
/*                     IN PRODUCTION SCHEDULING.                   */
/*                     THIS PROGRAM WILL RUN EVERY SUNDAY FROM THE */
/*                     JOB SCHEDULER.                              */
/***************************************************************** */
/* MODIFIED:                                                       */
/* L. BUSER   03/02/2006 T00006 IF THE OUTQ IS PRKPRT05 THEN       */
/*            CHANGE THE PRINTER FILE TO HOLD(*YES)                */
/***************************************************************** */
             PGM        PARM(&CO &OUTQ)

             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&OUTQ) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

             CHGVAR     VAR(&HOLD) VALUE(*NO)
/***************************************************************** */
/*     INITIALIZE                                                  */
/***************************************************************** */

/***************************************************************** */
/*     MAINLINE                                                    */
/***************************************************************** */
             IF         COND(&OUTQ *EQ 'PRKPRT05') THEN(CHGVAR +
                          VAR(&HOLD) VALUE(*YES))
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(POSQPFR$) OUTQ(&OUTQ) +
                          COPIES(1) HOLD(&HOLD) SAVE(*YES)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(POSQPFR) PARM(' ' &CO)
             MONMSG     MSGID(CPF0000)

/***************************************************************** */
/*     ENDPGM                                                      */
/***************************************************************** */
ENDPGM:
             ENDPGM
