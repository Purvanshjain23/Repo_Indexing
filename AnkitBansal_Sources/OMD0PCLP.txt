/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING MANIFEST                     */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT                                    */
/*  DATE      -  5/09/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &ORDERN        -  PASSED ORDER NUMBER                    */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 OMD0PCLP:   PGM        PARM(&COMPNYN &ORDERN &MANTYPE &APPLIC &OUTQ +
                          &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &COMPNYN     *DEC
        DCL    &ORDERN      *DEC
        DCL    &MANTYPE     *CHAR  (02)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &ORDER       *DEC   (7 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&ORDER)  VALUE(&ORDERN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/
             OVRPRTF    FILE(OMD0PFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

             OVRPRTF    FILE(OMDZPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

/* CUSTOMER SUMMARY */
             IF         COND(&MANTYPE *EQ 'C' *OR &MANTYPE *EQ +
                          'CS') THEN(DO)
             CALL       PGM(OMDZPFR) PARM(' ' &COMPANY &ORDER 'CS')
             ENDDO
/* CUSTOMER DETAIL */
             IF         COND(&MANTYPE *EQ 'C' *OR &MANTYPE *EQ +
                          'CD') THEN(DO)
             CALL       PGM(OMDZPFR) PARM(' ' &COMPANY &ORDER 'CD')
             ENDDO
/* SALES SUMMARY   */
             IF         COND(&MANTYPE *EQ 'S' *OR &MANTYPE *EQ +
                          'SS') THEN(DO)
             CALL       PGM(OMD0PFR) PARM(' ' &COMPANY &ORDER 'SS')
             ENDDO
/* SALES DETAIL    */
             IF         COND(&MANTYPE *EQ 'S' *OR &MANTYPE *EQ +
                          'SD') THEN(DO)
             CALL       PGM(OMD0PFR) PARM(' ' &COMPANY &ORDER 'SD')
             ENDDO

 ENDPGM:     ENDPGM
