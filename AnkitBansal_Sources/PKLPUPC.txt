/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKLPUPC                                       */
/*  PROGRAM NAME:    PRINT STRESSED HOGS REPORT                    */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     08/28/97                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PMT PRT STRESSED HOGS                    */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &APP +
                          &XCOMP &XFDATE &XTDATE &PEN &SRTYPE &PRD)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)
             DCL        VAR(&XFDATE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTDATE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PEN) TYPE(*CHAR) LEN(3)
             DCL        VAR(&SRTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PRD) TYPE(*CHAR) LEN(6)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&FDATE) TYPE(*DEC) LEN(7)
             DCL        VAR(&TDATE) TYPE(*DEC) LEN(7)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&LPP) VALUE(&APP)
             CHGVAR     VAR(&FDATE) VALUE(&XFDATE)
             CHGVAR     VAR(&TDATE) VALUE(&XTDATE)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKLQPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PKLQPFR) PARM(' ' &COMP &FDATE &PEN &LPP +
                 &FDATE &TDATE &SRTYPE &PRD)
             DLTOVR *ALL
             ENDPGM
