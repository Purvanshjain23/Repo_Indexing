SELECT
  ALL       T01.GRPID#, T01."USERID", T02.UNAME
  FROM      WM400/WMGROUP T01 INNER JOIN
            WM400/WMUSER T02
  ON        T01."USERID" = T02."USERID"
  WHERE     T01.COID = 'GUY'
    AND     T01.GRPID# BETWEEN 'GPPPS    ' AND 'GPPPSZZZZ'
  ORDER BY  T01.GRPID# ASC, T01."USERID" ASC
