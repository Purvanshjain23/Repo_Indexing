/*********************************************************************/
/*     Program: PBS3UPR                                              */
/*       Title: Copy PFS Prod Sched Batcth Dtl to PPB1CPPxxx         */
/*        Date: 09/23/2015                                           */
/* Description: Copy PFS Prod Sched Batch Dtl (PBB1CPP) to           */
/*              PBB1CPPGUY or PPB1CPPSTJ                             */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &PFSCOMNBA  PFS Company Number Alpha                     */
/*    I     &PFSBTCDTA  PFS Batch Date Alpha                         */
/*    I     &PFSBTCTMA  PFS Batch Time Alpha                         */
/*    I     &PFSUSERID  PFS User ID                                  */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*                                                                   */
/*   Date   Pgmr/Task  Description                                   */
/* -------- ---------- --------------------------------------------  */
/*  2/16/17 Brad Baden Add copy of file for company 440.  New file   */
/*          E9064      name is PBB1CPPSTF.                           */
/*                                                                   */
/*********************************************************************/
             PGM        PARM(&PFSCOMNBA &PFSBTCDTA &PFSBTCTMA +
                          &PFSUSERID)

   /*-------------------*/
   /* Declare Variables */
   /*-------------------*/

             /* Parameter Fields */
             DCL        VAR(&PFSCOMNBA) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PFSBTCDTA) TYPE(*CHAR) LEN(8)
             DCL        VAR(&PFSBTCTMA) TYPE(*CHAR) LEN(6)
             DCL        VAR(&PFSUSERID) TYPE(*CHAR) LEN(10)
             /* Other Variables  */
             DCL        VAR(&TOFILE) TYPE(*CHAR) LEN(10)

   /*-------------------------------*/
   /* Set &TOFILE based on Company  */
   /*-------------------------------*/
             IF         COND(&PFSCOMNBA *EQ '360') THEN(DO)
             CHGVAR     VAR(&TOFILE) VALUE(PBB1CPPGUY)
             ENDDO

             IF         COND(&PFSCOMNBA *EQ '440') THEN(DO)
             CHGVAR     VAR(&TOFILE) VALUE(PBB1CPPSTF)
             ENDDO

             IF         COND(&PFSCOMNBA *EQ '960') THEN(DO)
             CHGVAR     VAR(&TOFILE) VALUE(PBB1CPPSTJ)
             ENDDO
   /*-------------------------------*/
   /* Execute the Copy File Command */
   /*-------------------------------*/
             CPYF       FROMFILE(*LIBL/PBB1CPP) +
                          TOFILE(*LIBL/&TOFILE) MBROPT(*REPLACE) +
                          INCREL((*IF  B1Y1NX *EQ &PFSCOMNBA)    +
                                 (*AND B1ASD8 *EQ &PFSBTCDTA)    +
                                 (*AND B1AOTS *EQ &PFSBTCTMA)    +
                                 (*AND B1TVAA *EQ &PFSUSERID))

             MONMSG MSGID(CPF0000)

 ENDPGM:
             ENDPGM
