      * TITLE:       Retrieve Beginning/Ending Dates for a Calendar Year/Week
      * PROGRAMMER:  LeAnne Fedor
      * CREATED:     05/04/04
      *
      * FUNCTION:  Given an incoming calendar year/week, this program:
      *                   1) calls a SYNON program to retrieve the beginning/ending week dates
      *                   2) calls an HPS program to format the returned SYNON dates into
      *                      CCYYMMDD format
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      * 99/99/99
      *
      /EJECT
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * CONSTANTS
      ****************************************************************
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      *
      ****************************************************************
      * ARRAYS AND TABLES
      ****************************************************************
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      * Parms
      *
     D xxreturn        s              7a
     D xxcocd          s              3  0
      *
     D xxcdyr          s              4  0
     D xxcdwk          s              2  0
      *
     D xxsynbegdt      s              7  0
     D xxsynenddt      s              7  0
      *
     D xxwbdt          s              8  0
     D xxwedt          s              8  0
      *
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)

      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************
      *
      * Retrieve the beginning/ending dates for the calendar year/week.
      *
     C                   call      'PPENXFR'
     C                   parm      *blank        xxreturn
     C                   parm      350           xxcocd
     C                   parm                    xxcdyr
     C                   parm                    xxcdwk
     C                   parm      0             xxsynbegdt
     C                   parm      0             xxsynenddt
      *
     C                   if        xxsynenddt <> 0 and xxsynbegdt <> 0          If got OMS hit
      *
      * Get Synon beginning date into CCYYMMDD format
      *
     C                   call      'HP8011'
     C                   parm                    xxsynbegdt
     C                   parm      0             xxwbdt
      *
      *
      * Get Synon ending date into CCYYMMDD format
      *
     C                   call      'HP8011'
     C                   parm                    xxsynenddt
     C                   parm      0             xxwedt
     C                   endif                                                  If got OMS hit
      *
     C                   seton                                        lr
      /EJECT
      *---------------------------------------------------------------
      * Initialization subroutine
      *---------------------------------------------------------------
      *
     C     *inzsr        begsr
      *
      * Parm lists
      *
     C     *entry        plist
     C                   parm                    xxcdyr
     C                   parm                    xxcdwk
     C                   parm                    xxwbdt
     C                   parm                    xxwedt
      *
     C                   endsr
      /EJECT
