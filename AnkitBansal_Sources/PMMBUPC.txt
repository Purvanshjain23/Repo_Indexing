PGM

DCL &LIBRARY *CHAR 10  /* Override Library  */

/* Override to DFDATA for first run */
CHGVAR &LIBRARY 'DFDATA'
OVRDBF     FILE(ACCRHIST1) TOFILE(&LIBRARY/ACCRHIST1) SHARE(*YES)
OVRDBF     FILE(SHP204D) SHARE(*YES)
OPNQRYF    FILE((SHP204D)) QRYSLT('DSCONO *EQ 364 *AND DSITEMNO *NE 72591 *AND +
                                   DSITEMNO *NE 72592') KEYFLD(*FILE)

/* Call EUP Convert SHP204D  */
CALL PMK4UPR

/* Delete override */
CLOF SHP204D
DLTOVR  SHP204D
DLTOVR  ACCRHIST1

/* Override to JRDATA for second run */
CHGVAR &LIBRARY 'JRDATA'
OVRDBF     FILE(ACCRHIST1) TOFILE(&LIBRARY/ACCRHIST1) SHARE(*YES)
OVRDBF     FILE(SHP204D) SHARE(*YES)
OPNQRYF    FILE((SHP204D)) QRYSLT('DSCONO *EQ 363 *AND DSITEMNO *NE 72591 *AND +
                                   DSITEMNO *NE 72592') KEYFLD(*FILE)

/* Call EUP Convert SHP204D  */
CALL PMK4UPR

/* Delete override */
CLOF SHP204D
DLTOVR  SHP204D
DLTOVR  ACCRHIST1

ENDPGM
