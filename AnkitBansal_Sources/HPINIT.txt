/*  PROGRAM NUMBER --> HPINIT                                       */
/*                                                                  */
/*  PROGRAM NAME ----> HOG PRODUCTION INITIALIZE PGM                */
/*                                                                  */
/*  PROGRAM PURPOSE -> THIS CL RETRIEVES THE USERS OUTPUT QUEUE AND */
/*                     LOADS IT INTO THEIR *LDA TO BE USED AS A    */
/*                     DEFAULT WHEN DIRECTING PRINTED OUTPUT.      */
/*                                                                 */
/*  PROGRAMMER ------> ROSE CENTONZE             07/25/94          */
/*                                                                 */
/***************************************************************** */
/* MODIFICATIONS                                                   */
/* 05/14/97  LEANNE FEDOR                                          */
/*           THE USER'S DEFAULT OUTPUT QUEUE WILL NOW COME FROM    */
/*           THE USER'S RECORD IN THE USERS FILE---NOT THE USER'S */
/*           PROFILE.                                              */
/*                                                                 */
/***************************************************************** */

             PGM

             DCL        VAR(&USRPRF) TYPE(*CHAR) LEN(10)

             RTVUSRPRF  RTNUSRPRF(&USRPRF)
             CALL       PGM(HPOUTQ) PARM(&USRPRF)

             ENDPGM
