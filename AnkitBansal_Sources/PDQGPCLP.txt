 PDRVPCLP:   PGM        PARM(&COMPL)

        DCL     &COMPL       *DEC
        DCL     &COMP        *DEC     (3 0)      /*  COMPANY #  */
        DCL     &RETURN      *CHAR    7

/********************************************************************/
/* load Company                                                     */
/********************************************************************/
             CHGVAR     VAR(&COMP) VALUE(&COMPL)

/*********************************************************************/
/*   Call DOWNLOAD ITEM AND BOX ID FILES                             */
/*********************************************************************/

             CALL       PGM(PDQAXFR) PARM(&RETURN &COMP)

 END:        ENDPGM
