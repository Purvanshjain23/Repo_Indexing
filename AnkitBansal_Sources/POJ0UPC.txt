/*********************************************************************/
/*     THIS CL CREATES THE LOAD SLOT ACTIVITY FILE IN QTEMP          */
/*                                                                   */
/*     ROSE CENTONZE   3/05/01                                       */
/*                                                                   */
/*********************************************************************/

POJ0UPC:  PGM
             MONMSG     MSGID(CPF0000)

             CRTDUPOBJL OBJ(POA4CPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
EXISTS:      MONMSG     MSGID(CPF2130) EXEC(DO)
             CLRPFM FILE(QTEMP/POA4CPP)
             GOTO EXIT
             ENDDO
             CRTDUPOBJL OBJ(POA4CPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL OBJ(POA4CPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL OBJ(POA4CPL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL OBJ(POA4CPL3) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

EXIT:  ENDPGM
