/*  PROGRAM NUMBER:  PKLKUPR                                       */
/*  PROGRAM NAME:    CREATE PRODUCER EDIT INFORMATION              */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     08/04/97                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from A PROMPT                                 */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  ROSE CENTONZE 12/17/98  SEND IN ALPHA DATE FOR Y2K            */
/*  SUSAN MASON   11/14/05  S00555 PAS IN PARM TO STATE TO        */
/*                CREATE THE DATA AND SEND OUT TO THE WEB         */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &XCOMP &PRD &XKILLDTE)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)

/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PRD)   TYPE(*CHAR) LEN(6)
             DCL        VAR(&XKILLDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&SNDOUT) TYPE(*CHAR) LEN(1) VALUE(S)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KILLDTE) TYPE(*DEC) LEN(7 0)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&KILLDTE) VALUE(&XKILLDTE)

/* ADD LIBRARIES FOR EDI                                        */
             ADDLIBLE   LIB(EDIFLIB) POSITION(*LAST)
             MONMSG     MSGID(CPF0000)
             ADDLIBLE   LIB(EDIPLIB) POSITION(*LAST)
             MONMSG     MSGID(CPF0000)
             ADDLIBLE   LIB(EDIV21F) POSITION(*LAST)
             MONMSG     MSGID(CPF0000)
             ADDLIBLE   LIB(EDIV21P) POSITION(*LAST)
             MONMSG     MSGID(CPF0000)
             ADDLIBLE   LIB(EDIV21O) POSITION(*LAST)
             MONMSG     MSGID(CPF0000)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

/* BUILD HPE EDI FILE */
             CALL       PGM(PKLLXFR) PARM(' ' &COMP &KILLDTE &PRD +
                          &SNDOUT)

/* CREATE EDI DOWNLOAD FILE */
/*           CALL       PGM(CARCASSCB) PARM(' ' 0 ' ') */

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
