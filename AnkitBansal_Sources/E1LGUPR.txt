      * ‚Entry parameters ....
     d MainRoutine     PR                  extpgm('E1LGUPR')
     d $s2ev01                        1a                                        everest point
     d $s2user                       10a                                        user
     d $s2urrf                       15a                                        version

     d MainRoutine     PI
     d $s2ev01                        1a                                        everest point
     d $s2user                       10a                                        user
     d $s2urrf                       15a                                        version
      /FREE

   ‚   // Read F55CONFIG3 for letter corresponding to user
       setll *start F55CONFIG3;
       READ F55CONFIG3;
       Dow Not %EOF(F55CONFIG3);
       if $s2user=s2user;
       $s2ev01= s2ev01;
       $s2urrf= s2urrf;
       endif;
       READ F55CONFIG3;
       EndDo;
