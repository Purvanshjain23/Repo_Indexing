/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  BUILD THE PRODUCT COST RECORDS IN THE               */
/*              ITEM STANDARD COST FILE FOR COGNOS                  */
/*                                                                  */
/*  SYSTEM    - DAILY FOODS                                         */
/*  PROGRAM   -                                                     */
/*                                                                  */
/*  DATE      - 02/20/2009                                          */
/*  PROGRAMMER- SHARON ZURN                                         */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &EFTDTE     - EFFECTIVE DATE                            */
/********************************************************************/
 PMEYUPC:    PGM        PARM(&EFTDTE)

/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)  /* RETURN CODE               */
        DCL    &EFTDTE     *DEC    (7 0)/* EFFECTIVE DATE            */
        DCL    &CO         *DEC    (3 0)/* COMPANY                   */

/********************************************************************/
             CHGVAR     VAR(&CO) VALUE(363)
             SETLIBL    JOBD(DMMJOBD)
             CALL       PGM(PME2XFR) PARM( &RETURN &CO &EFTDTE)


             CHGVAR     VAR(&CO) VALUE(364)
             SETLIBL    JOBD(DSLJOBD)
             CALL       PGM(PME2XFR) PARM( &RETURN &CO &EFTDTE)

 END:        ENDPGM
