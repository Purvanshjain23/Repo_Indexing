/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PLH7PCLP                                      */
/*  PROGRAM NAME:    PRT Tattoo/Unloader Report                    */
/*  PROGRAMMER:      SUSAN L. MASON                                */
/*  CREATE DATE:     07/21/2006                                    */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT Prt Tattoo/Unload Report .        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*******************************************************************/
             PGM        PARM(&RTN &ACO &XFROM &SHIFT &XTO &OMS +
                          &PDEV &HOLD &SAVE &XCOPIES)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&ACO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XFROM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTO) TYPE(*CHAR) LEN(7)
             DCL        VAR(&SHIFT) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FROM) TYPE(*DEC) LEN(7)
             DCL        VAR(&TO) TYPE(*DEC) LEN(7)

             /* */
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FROM) VALUE(&XFROM)
             CHGVAR     VAR(&TO) VALUE(&XTO)
             CHGVAR     VAR(&CO) VALUE(&ACO)

/* CREATE HOG POPULATION REPORT FILES IN QTEMP                  */

/*     Tattoo UnLoader Cnt Wrk                                  */
             CRTDUPOBJL  OBJ(PLAXCPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PLAXCPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PLAXCPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJL  OBJ(PLAXCPL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

/* CREATE WORK FILES                                           */

             CALL       PGM(PLH5XFR) PARM(' ' &CO &FROM &SHIFT +
                          &TO)

/*******************************************************************/
/*    CALL HOG POPULATION REPORT                                 */
/*                                                               */
/* CREATE PRINTER FILE OVERRIDE CODE                            */
/*******************************************************************/

             OVRPRTF    FILE(PLH7PFR$) OUTQ(&PDEV) COPIES(&COPIES) +
                          HOLD(&HOLD) SAVE(&SAVE)

/* PRINT PROGRAM                                               */

             CALL       PGM(PLH7PFR) PARM(' ' &FROM &SHIFT &LPP +
                          &TO &CO)


             DLTOVR *ALL
             ENDPGM
