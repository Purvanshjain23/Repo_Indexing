/*                   FROM THE JOB SCHEDULER.                       */
/*  FUNCTION:                                                      */
/*    PASS IN THE DATES AS MMDDYY TO WORK IN THE EXECUTE.          */
/*    -----------------------------------------                    */
/*  MODIFIED BY:                                                   */
/*******************************************************************/
             PGM        PARM(&FDTE &TDTE)

             DCL        VAR(&FDTE) TYPE(*CHAR) LEN(6) /* MMDDYY */
             DCL        VAR(&TDTE) TYPE(*CHAR) LEN(6) /* MMDDYY */
             DCL        VAR(&COALPHA) TYPE(*CHAR) LEN(3) VALUE(SLC) +
                          /* COMPANY*/
             DCL        VAR(&TITLE) TYPE(*CHAR) LEN(49)
/*------------------------------------------------------------------*/
             CHGVAR     VAR(&TITLE) VALUE('      ')
             CHGVAR     VAR(%SST(&TITLE 1 26)) VALUE('Tonnage Rpt by +
                          Group From')

             CHGVAR     VAR(%SST(&TITLE 27 6)) VALUE(&FDTE)
             CHGVAR     VAR(%SST(&TITLE 34 2)) VALUE('To')
             CHGVAR     VAR(%SST(&TITLE 37 6)) VALUE(&TDTE)
             CHGVAR     VAR(%SST(&TITLE 44 3)) VALUE(&COALPHA)

             EXECUTE    VIEW(TNNG364.1) OUTFILE(QTEMP/TNNG364) +
                          SETVAR((&FDATE &FDTE) (&TDATE &TDTE))

             REPORT     REPORT(*LIBL/TNNG364.R) TITLE(&TITLE) +
                          HOLD(*YES) PCFMT(*SPL) OUTQ(SLCPRT07)

             ENDPGM
