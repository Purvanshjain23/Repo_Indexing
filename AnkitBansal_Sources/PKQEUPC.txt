/*  PROGRAM NUMBER:  PKQEUPC                                       */
/*  PROGRAM NAME:    DSP WINDOW THAT  'WEIGHT HAS BEEN ACCEPTED'   */
/*  PROGRAMMER:      ROSE CENTONZE                                 */
/*  CREATE DATE:     01/15/99                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS CALLED FROM THE CLP THAT ACCEPTS   */
/*                   THE WEIGHT FROM THE WEDGE.  SET THE WAITRCD   */
/*                   TIME ON THE DSPF TO 5 OR 10 SECONDS SO THE    */
/*                   SCREEN WILL ONLY DISPLAY FOR THAT LONG.       */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&WEIGHT &ORDER &LOAD) /* Display Window +
                          that Weight has been accepted */

             DCLF       FILE(PKQEUPC#)


             SNDRCVF    RCDFMT(WINDOW) WAIT(*NO)
             WAIT
             MONMSG     MSGID(CPF0889)
 /*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
 ENDPGM:     ENDPGM
