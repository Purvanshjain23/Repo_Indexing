/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKMEUPR                                       */
/*  PROGRAM NAME:    PRINT PRODUCER COMPARISON REPORT  -DTL,SUM    */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     09/23/97                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT PRINT PRODUCER COMPARISON REPORT  */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*    ROSE CENTONZE  12/17/98 SEND IN ALPHA DATES FOR Y2K          */

/*******************************************************************/
             PGM        PARM(&RTN &XCO &SELCDE &SRCE &PAYM &PROD +
                          &SCODE &BUYR &DTLSUM &SELDSC &XFROM &XTO +
                          &SEQ &SELTTO &PDEV &HOLD &SAVE &XCOPIES &OMS)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&SELCDE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&SELDSC) TYPE(*CHAR) LEN(30)
             DCL        VAR(&SRCE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PAYM) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SCODE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6)
             DCL        VAR(&BUYR) TYPE(*CHAR) LEN(3)
             DCL        VAR(&XFROM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTO) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DTLSUM) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SEQ) TYPE(*CHAR) LEN(1)
             DCL        VAR(&SELTTO) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGMPRTF) TYPE(*CHAR) LEN(10)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FROM) TYPE(*DEC) LEN(7)
             DCL        VAR(&TO) TYPE(*DEC) LEN(7)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

             /* */
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&CO) VALUE(&XCO)
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&FROM) VALUE(&XFROM)
             CHGVAR     VAR(&TO) VALUE(&XTO)
/*******************************************************************/
/*     PRINT REPORTS                                              */
/*******************************************************************/


             IF         COND(&SEQ *EQ '3') THEN(DO)
             CHGVAR     VAR(&PGMPRTF) VALUE('PKRIPFR$')
             CHGVAR     VAR(&PGM) VALUE('PKRIPFR')
             ENDDO
             ELSE (DO)
             CHGVAR     VAR(&PGMPRTF) VALUE('PKQSPFR$')
             CHGVAR     VAR(&PGM) VALUE('PKQSPFR')
             ENDDO

             IF         COND(&SEQ *EQ '4') THEN(DO)
             CHGVAR     VAR(&PGMPRTF) VALUE('PKR9PFR$')
             CHGVAR     VAR(&PGM) VALUE('PKR9PFR')
             ENDDO

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(&PGMPRTF) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

BOTH:        IF         COND(&DTLSUM *EQ 'B') THEN(DO)

/* PRINT SUMMARY AND DETAIL                                    */
             CALL       PGM(&PGM) PARM(' ' &CO &SELCDE &SRCE +
                          &PAYM &SCODE &PROD &BUYR 'D' &LPP &SELDSC +
                          &FROM &TO &SELTTO)
             CALL       PGM(&PGM) PARM(' ' &CO &SELCDE &SRCE +
                          &PAYM &SCODE &PROD &BUYR 'S' &LPP &SELDSC +
                          &FROM &TO &SELTTO)
             ENDDO
             ELSE (DO)
/* PRINT SUMMARY OR DETAIL                                     */
             CALL       PGM(&PGM) PARM(' ' &CO &SELCDE &SRCE +
                          &PAYM &SCODE &PROD &BUYR &DTLSUM &LPP +
                          &SELDSC &FROM &TO &SEQ &SELTTO)
             ENDDO
/*******************************************************************/
             DLTOVR *ALL
             ENDPGM
