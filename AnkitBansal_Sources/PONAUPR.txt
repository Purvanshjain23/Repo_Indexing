/*  Modification History                                           */

/*******************************************************************/
             PGM        PARM(&COMP &ITEM &EFDATE &CURENT &MTOCDE +
                          &MTOTYP &PDEV &HOLD &SAVE &COPIES)
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* INPUT PARAMETERS                                                */
             DCL        VAR(&COMP)  TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&ITEM) TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&EFDATE) TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&CURENT) TYPE(*CHAR) LEN(1)
             DCL        VAR(&MTOCDE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&MTOTYP) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* PROGRAM VARIABLES                                               */
             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/* PRINTER OVERRIDE                                                */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(POM9PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*-----------------------------------------------------------------*/
/* PRINT REPORT                                                    */
/*-----------------------------------------------------------------*/

             CALL       PGM(POM9PFR) PARM(' ' &COMP &ITEM &EFDATE +
                          &CURENT &MTOCDE &MTOTYP)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:

             ENDPGM
