/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKAOUPR                                       */
/*  PROGRAM NAME:    PRINT TATTOO/CARCASS DETAIL                   */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     06/25/96                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from MEL CMP/FRT/DA'S/NET PROMPT.             */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &XCOMP &PRD &XKILLRGF &XKILLRGT +
                          &RPTTYP &PDEV &HOLD &SAVE &XCOPIES)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* Function Specific                                               */
             DCL        VAR(&PRD)   TYPE(*CHAR) LEN(6)
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XKILLRGF) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XKILLRGT) TYPE(*CHAR) LEN(15)
             DCL        VAR(&RPTTYP) TYPE(*CHAR) LEN(1)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/

             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KILLRGF) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&KILLRGT) TYPE(*DEC) LEN(7 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&KILLRGF) VALUE(&XKILLRGF)
             CHGVAR     VAR(&KILLRGT) VALUE(&XKILLRGT)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PKANPFR$) OUTQ(&PDEV) +
                          DRAWER(2) COPIES(&COPIES) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             R          LIBLST(PRKPROD)
             CHGCURLIB  CURLIB(*CRTDFT)
             CALL       PGM(PKANPFR) PARM(' ' &COMP &KILLRGF &PRD +
                          &KILLRGF &KILLRGT &RPTTYP)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             ENDPGM
