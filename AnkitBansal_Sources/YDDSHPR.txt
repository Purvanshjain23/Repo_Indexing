/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER  -> YDDSHPR                                     */
/*  PROGRAM NAME    -> PROCESS HELP TEXT                           */
/***************************************************************** */
             PGM        PARM(&W0HPMB &YYHPFL &YYHPLB &W0RTN &YYHLVN +
                          &YYUSOP &YYRW &YYCL &YYLGCT &YYLGVN)

             DCL        VAR(&W0HPMB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&YYHPFL) TYPE(*CHAR) LEN(10)
             DCL        VAR(&YYHPLB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&W0RTN)  TYPE(*CHAR) LEN(7)
             DCL        VAR(&YYHLVN) TYPE(*CHAR) LEN(10)
             DCL        VAR(&YYUSOP) TYPE(*CHAR) LEN(10)
             DCL        VAR(&YYRW)   TYPE(*DEC) LEN(5 0)
             DCL        VAR(&YYCL)   TYPE(*DEC) LEN(5 0)
             DCL        VAR(&YYLGCT) TYPE(*DEC) LEN(5 0)
             DCL        VAR(&YYLGVN) TYPE(*CHAR) LEN(10)

             DCL        VAR(&DOC)    TYPE(*CHAR) LEN(8)
             DCL        VAR(&FLDR)   TYPE(*CHAR) LEN(17)
             DCL        VAR(&LBL)    TYPE(*CHAR) LEN(10)


             CALL       PGM(PKJGXFR) PARM('       ' &W0HPMB &DOC +
                          &FLDR &LBL)

             IF         COND(&DOC *NE ' ' & &FLDR *NE ' ' & &LBL *NE +
                          ' ') THEN(DSPHLPDOC DOC(&DOC) FLR(&FLDR) +
                          HLPLBL(&LBL))
             ELSE       CMD(SNDPGMMSG MSG('Help does not exist'))


ENDPGM
