/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT CUSTOMER CLAIM REPORT                          */
/*                                                                  */
/*  COMPANY   - SEABOARD COPRORATION                                */
/*  SYSTEM    - ORDER MANAGEMENT                                    */
/*  USER NAME - SUSAN MASON                                         */
/*  DATE      - 06/29/06                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/********************************************************************/
 PPY5PFR:    PGM        PARM(&RETURN &APPLIC &DESC &OUTQ +
                          &HOLD &SAVE &COPYN &NQSTAT &NIGHTQ &LOAD +
                          &CLM &CLMT &CLMST &CLMDT &CLMNM &CLMSP +
                          &CLMCO &ORDNO &STPCL &STPCM +
                          &STPRS &STPEX)

        DCL            &LOAD   *DEC         /*  LOAD ID                  */
        DCL            &CLM    *DEC         /*  CLAIM NUMBER             */
        DCL            &CLMT   *CHAR (02)   /*  CLAIM TYPE               */
        DCL            &CLMST  *CHAR (01)   /*  CLAIM STATUS             */
        DCL            &CLMDT  *DEC         /*  CLAIM ENTRY DAT          */
        DCL            &CLMNM  *CHAR (30)   /*  CLAIM ENTRY DAT          */
        DCL            &CLMSP  *DEC         /*  CLAIM SHIP TO C          */
        DCL            &CLMCO  *DEC         /*  CLAIM COMPANY N          */
        DCL            &ORDNO  *DEC         /*  CLAIM ORDER NUM          */
        DCL            &STPCL  *CHAR (01)   /*  CLM DTA STEPS-CLM        */
        DCL            &STPCM  *CHAR (01)   /*  CLM DTA STEPS-CMT        */
        DCL            &STPRS  *CHAR (01)   /*  CLM DTA STEPS-RSL        */
        DCL            &STPEX  *CHAR (01)   /*  CLM DTA STEPS-EXP        */

/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &DESC       *CHAR   (25)          /* PRT Pgm Description       */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &COPIES     *DEC  LEN(2 0) /* No. of +
                          Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMMAND    *CHAR   (512)         /* Printer override field    */

/********************************************************************/
/* Override printer file with user specified values                 */
/********************************************************************/

             CHGVAR     VAR(&COPIES) VALUE(&COPYN)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPY5PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)

             CALL PGM(QCMDEXC) PARM(&COMMAND 512)

/*********************************************************************/
/*   CALL LISTING PROGRAM                                            */
/*********************************************************************/

             CALL       PGM(PPY5PFR) PARM('' &LOAD &CLM &CLMT &CLMST +
                          &CLMDT &CLMNM &CLMSP &CLMCO &ORDNO &STPCL +
                          &STPCM &STPRS &STPEX)

 END:        ENDPGM
