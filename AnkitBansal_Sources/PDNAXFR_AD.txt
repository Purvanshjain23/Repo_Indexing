//?Calculating MTD Avg Days to Pay
CASE;

// IF PAR.MTD Invoices Paid is Not zero
IF PAR.MTD_Invoices_Paid <> *ZERO;

// PAR.MTD Avg Days to Pay Calc = PAR.MTD Days Paid / PAR.MTD Invoices Paid *Rounde
PAR.MTD_Avg_Days_to_Pay_Calc = PAR.MTD_Days_Paid / PAR.MTD_Invoices_Paid 'H';

// PAR.MTD Avg Days to Pay = PAR.MTD Avg Days to Pay Calc * CON.1 *Rounded
PAR.MTD_Avg_Days_to_Pay = PAR.MTD_Avg_Days_to_Pay_Calc * 1 'H';

ENDIF;

//?Calculating YTD Avg Days to Pay
CASE;

// IF PAR.YTD Invoices Paid is Not Equal to Zero
