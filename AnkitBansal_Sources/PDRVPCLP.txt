/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - SUBMIT REVALUE INVENTORY WITH MARKET PRICE           */
/*                                                                  */
/*  COMPANY   - SEABOARD CORP.                                      */
/*  SYSTEM    - IDeAS/400 INVENTORY MANAGEMENT                      */
/*  USER NAME - MARK ARENSBERG                                      */
/*  DATE      - 05/24/95                                            */
/*                                                                  */
/********************************************************************/

 PDRVPCLP:   PGM        PARM(&COMPL &WHSE &DATEL &TYPE)

        DCL     &COMPL       *DEC
        DCL     &COMP        *DEC     (3 0)      /*  COMPANY #  */
        DCL     &DATEL       *DEC
        DCL     &DATE        *DEC     (7 0)      /*  DATE       */
        DCL     &WHSE        *CHAR    3          /*  WAREHOUSE  */
        DCL     &TYPE        *CHAR    1          /*  REVALUE TYPE */
        DCL     &RETURN      *CHAR    7

/********************************************************************/
/* load Company                                                     */
/********************************************************************/
             CHGVAR     VAR(&COMP) VALUE(&COMPL)
             CHGVAR     VAR(&DATE) VALUE(&DATEL)

/*********************************************************************/
/*   Call REVALUE INVENTORY WITH MARKET PRICE                        */
/*********************************************************************/

             IF         COND(&TYPE *EQ 'M') THEN(DO)
             CALL       PGM(PDPPXFR) PARM(&RETURN &COMP &WHSE &DATE)
             ENDDO
             ELSE (DO)
             CALL       PGM(PDFJXFR) PARM(&RETURN &COMP &WHSE &DATE)
             ENDDO

 END:        ENDPGM
