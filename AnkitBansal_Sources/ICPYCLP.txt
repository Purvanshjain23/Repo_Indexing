/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING PHYSICAL TO BOOK COMPARE     */
/*                                                                  */
/*  SYSTEM    - INVENTORY MANAGEMENT                                */
/*  DATE      -  8/18/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  COMPANY                                */
/*         &WHSE          -  WAREHOUSE                              */
/*         &CYCLE         -  CYCLE                                  */
/*         &PRTZR         -  PRINT NON-ZERO QTYS ONLY               */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/*------------------------------------------------------------------*/
/* MAINTENANCE LOG                                                  */
/*------------------------------------------------------------------*/
/* 05/02/00 PKD ADDED CODE TO SAVE THE REPORT AFTER IT IS PRINTED.  */
/* 07/22/02 RMC PRINT ON PRKPRT04 ALSO                              */
/* 01/29/04 RMC PRINT ON PRKPRT02 INSTEAD OF PRKPRT04               */
/*                                                                  */
/********************************************************************/
 ICPYCLP:    PGM        PARM(&COMPNYN &WHSE &CYCLE &PRTZR &APPLIC +
                             &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)

        DCL    &COMPNYN     *DEC
        DCL    &WHSE        *CHAR  (03)
        DCL    &CYCLE       *CHAR  (03)
        DCL    &PRTZR       *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND     *CHAR  (512)
        DCL    &RETURN      *CHAR  (7)

        DCL    &COMP        *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)  VALUE(&COPYN)
             CHGVAR     VAR(&SAVE) VALUE(*YES)      /* 5/02/00 PKD */

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(ICPYPFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(ICPYPFR) PARM(&RETURN &COMP &WHSE &CYCLE +
                                          &PRTZR)
/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(ICPYPFR$) +
                          OUTQ(PRKPRT02) COPIES(&COPY) HOLD(*YES)  +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL2:      CALL       PGM(ICPYPFR) PARM(&RETURN &COMP &WHSE &CYCLE +
                                          &PRTZR)

 ENDPGM:     ENDPGM
