/*&                                                                  */
/*&    DESCRIPTION   -  PASS PARMS TO A BREAK MESSAGE.               */
/*&                                                                  */
/*&    CREATED       -  01/02/08  SHARON ZURN                        */
/*&    MODIFIED      -    /  /                                       */
/* ------------------------------------------------------------------*/
             PGM        PARM(&FSTS &EMAIL &TSCDE &TEXT)

 /*  DECLARE VARIABLES                                               */

             DCL        VAR(&FSTS) TYPE(*CHAR) LEN(1)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(60)
             DCL        VAR(&TSCDE) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TEXT) TYPE(*CHAR) LEN(30)

                                                                       /**/
             SNDMSG     MSG(&FSTS *BCAT '*' *BCAT &EMAIL *BCAT '*' +
                          *BCAT &TSCDE *BCAT '*' *BCAT &TEXT *BCAT +
                          ' ') TOUSR(ISSZURN)
             ENDPGM
ENDPGM
