/********************************************************************/
/*    GET THE SYSTEM DATE FORMATTED YYYYMMDD                       */
/*                                                                 */
/********************************************************************/
PBRRUPC:    PGM  PARM(&DATEF &DATET)

/* PARMS */
            DCL        VAR(&DATEF) TYPE(*DEC) LEN(7 0)
            DCL        VAR(&DATEA) TYPE(*CHAR) LEN(7)
            DCL        VAR(&DATET) TYPE(*CHAR) LEN(8)
         CHGVAR &DATEA VALUE(&DATEF)
             CVTDAT     DATE(&DATEA) TOVAR(&DATET) FROMFMT(*CYMD) +
                          TOFMT(*YYMD) TOSEP(*NONE)

ENDPGM
