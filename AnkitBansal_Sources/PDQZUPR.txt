/*  SYNOPSIS:                                                      */
/*              CL program that automatically polls the pick file   */
/*              from Acu-Sort to determine orders that have been    */
/*              shipped.                                            */
/*                                                                  */
/*  SYSTEM:                                                         */
/*              Order Management System                             */
/*  DATE:                                                           */
/*              June 9, 1995                                        */
/*  PARAMETERS:                                                     */
/********************************************************************/
             PGM
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(10)
             DCL        VAR(&FLIB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HLIB) TYPE(*CHAR) LEN(10)

 /*         RETREIVE FILE LIBRARY INTO FLIB                          */
             CALL       PGM(CAQXXFR) PARM(&RETURN &FLIB &HLIB)


 PROGRAMS:
 /*         * PROCESS AS PICK ACKNOWLEDGMENT                         */

             CALL       PGM(PDPCKXFR) PARM(' ')


 /*         * PROCESS AS SHIP ORDER DETAIL                           */

             CALL       PGM(PDRXXFR) PARM(' ')


 /*         * PROCESS AS SHIP ORDER HEADER (LOAD)                    */

             CALL       PGM(PDZ5XFR) PARM(' ')


             IF         COND(&FLIB *NE '          ') THEN(DO)


                    CPYF       FROMFILE(&FLIB/PDJSCPP) +
                               TOFILE(&FLIB/ASJSCPP) MBROPT(*ADD) +
                               CRTFILE(*YES)
                    CPYF       FROMFILE(&FLIB/PDJUCPP) +
                               TOFILE(&FLIB/ASJUCPP) MBROPT(*ADD) +
                               CRTFILE(*YES)
                    CPYF       FROMFILE(&FLIB/PDJ9CPP) +
                               TOFILE(&FLIB/ASJ9CPP) MBROPT(*ADD) +
                               CRTFILE(*YES)

             ENDDO

 END: ENDPGM
