/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - FAX INVOICE FROM REPRINT SCREEN                      */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - FAX INVOICE                                         */
/*  DATE      -  1/13/97                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &RETURN     - RETURN CODE                               */
/*          &COMPNYN    - COMPANY NUMBER                            */
/*          &ORDRN      - ORDE NUMBER                               */
/*          &CUSTNON    - CUSTOMER NUMBER                           */
/*          &REPRINT    - REPRINT FLAG                              */
/*          &JOBCDE     - JOB NAME CODE                             */
/*          &CVRNME     - COVER SHEET NAME                          */
/********************************************************************/
/* MODIFIED:  RMC 5/02/02  CHANGE TLX400J TO TLX400P               */
/********************************************************************/

 PDOTPCLP:   PGM        PARM(&RETURN &COMPNYN &ORDRN &CUSTNON +
                          &REPRINT &JOBCDE &CVRNME &CRDFLAG &FAX# +
                          &CONTACT)


/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* RETURN CODE               */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &ORDRN      *DEC                  /* Order Number   (Passed)   */
        DCL    &CUSTNON    *DEC                  /* CUSTOMER       (Passed)   */
        DCL    &JOBCDE     *CHAR   (10)          /* JOB NAME CODE             */
        DCL    &CVRNME     *CHAR   (10)          /* COVER SHEET NAME          */
        DCL    &CRDFLAG    *CHAR   (1)           /* USE CRD CONTACT INFO      */
        DCL    &FAX#       *CHAR   (15)
        DCL    &CONTACT    *CHAR   (25)

/* INTERNAL FIELDS                                                            */
        DCL    &REPRINT    *CHAR   (1)           /* Y = REPRINT               */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */
        DCL    &ORDR       *DEC    (7 0)         /* Order Number              */
        DCL    &CUSTNO     *DEC    (7 0)         /* CUSTOMER                  */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&ORDR) VALUE(&ORDRN)
             CHGVAR     VAR(&CUSTNO) VALUE(&CUSTNON)

/********************************************************************/
/*   SEND FAX IF USER SELECTED YES ON THE PROMPT                    */
/********************************************************************/

             OVRPRTF    FILE(PDVGPFR$) OUTQ(PRKFLIB/FAXQ)

             CALL       PGM(PODUXFR) PARM(&RETURN &COMPANY &ORDR +
                          &CUSTNO &REPRINT &JOBCDE &CVRNME &CRDFLAG)

             TLX400P/SNDFSPL SPF(PDVGPFR$) DESC('Invoice') DIAL(*N +
                          &FAX# &CONTACT) CVRID(&CVRNME)

             DLTSPLF    FILE(PDVGPFR$) SELECT(*ALL)

 END:        ENDPGM
