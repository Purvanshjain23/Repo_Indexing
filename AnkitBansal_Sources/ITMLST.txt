SELECT
  ALL       T01.BYHGCD, T01.BYAVNA, T01.BYQZCD, T02.BYQZCD, T01.BYVMST,
            T02.BYVMST, T01.BYX5ST, T02.BYX5ST
  FROM      PRKTESTF/OMBYREP T01 INNER JOIN
            PRKTESTX/OMBYREP T02
  ON        T01.BYHGCD = T02.BYHGCD
  WHERE     T01.BYQZCD <> T02.BYQZCD
    OR      T01.BYVMST <> T02.BYVMST
    OR      T01.BYX5ST <> T02.BYX5ST
  ORDER BY  T01.BYHGCD ASC
