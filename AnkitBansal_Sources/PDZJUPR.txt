/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR CREATING THE DOWNLOAD INV/SALES TO PC */
/*             FILE.                                                */
/*                                                                  */
/*  SYSTEM    - ITEM MANAGEMENT                                     */
/*  DATE      - 05/29/97                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &ACOMP         -  PASSED COMPANY NUMBER                  */
/*         &AFSHIP        -  PASSED FROM SHIP DATE ALPHA           */
/*         &ATSHIP        -  PASSED TO SHIP DATE ALPHA             */
/*                                                                  */
/********************************************************************/
 PDZJUPR:    PGM        PARM(&RTN &ACOMP &AFSHIP &ATSHIP)

        DCL    &RTN         *CHAR  (7)
        DCL    &ACOMP       *CHAR  (3)
        DCL    &AFSHIP      *CHAR  (7)
        DCL    &ATSHIP      *CHAR  (7)
        DCL    &COMP        *DEC   (3 0)
        DCL    &FSHIP       *DEC   (7 0)
        DCL    &TSHIP       *DEC   (7 0)
        DCL    &NBRREC      *DEC   (10 0)
        DCL    &RECS        *CHAR  (10)
        DCL    &USER        *CHAR  (10)

/* CONVERT PARAMETERS TO REQUIRED SIZE                                     */

             CHGVAR     VAR(&COMP)  VALUE(&ACOMP)
             CHGVAR     VAR(&FSHIP)  VALUE(&AFSHIP)
             CHGVAR     VAR(&TSHIP)  VALUE(&ATSHIP)

/* CLEAR THE DOWNLOAD FILE                                                 */

 CLR:        CLRPFM     FILE(PDLKCPP)

/* CALL XF TO CREATE THE FILE                                              */

 CALL:       CALL       PGM(PDZLXFR) PARM(' ' &COMP &FSHIP &TSHIP)

             RTVMBRD    FILE(PDLKCPP) NBRCURRCD(&NBRREC)
             RTVJOBA    USER(&USER)
             CHGVAR     VAR(&RECS)  VALUE(&NBRREC)
             SNDPGMMSG  MSGID(CPF9898) MSGF(QCPFMSG) MSGDTA('Created +
                          ' *CAT &RECS *CAT ' Records for +
                          Download') TOUSR(&USER) MSGTYPE(*INFO)

 ENDPGM:     ENDPGM
