/*                                                                  */

 CAZLLCLP:   PGM

        DCL    &CONTACT    *DEC      (7 0)       /*  CONTACT NUMBER */

             CALL        PGM(CAZLDFR) PARM('' &CONTACT)

 END:           ENDPGM
