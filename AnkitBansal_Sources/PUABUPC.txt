/**************************************************************************/
/*  PROGRAM:       PUABUPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO ORDER HEADER TRG (OPBFCPP) FILE.      */
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   01/10/2017                                             */
/*  PROJECT:       E8248                                                  */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY PHYSICAL FILE TRIGGER TO THE           */
/*                 ORDER HEADER TRG (OPBFCPP) FILE FOR ALL TRIGGER        */
/*                 DATABASE EVENTS:                                       */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 03/03/23 DN   W122889 ADDED MONITOR MESSAGE FOR 'RMVPFTRG' TO PREVENT  */
/*                       ERROR.                                           */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/OPBFCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PTPTUPR) +
                          RPLTRG(*YES) TRG(SEB_OPBFCPP_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/OPBFCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PTPTUPR) +
                          RPLTRG(*YES) TRG(SEB_OPBFCPP_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/OPBFCPP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PTPTUPR) +
                          RPLTRG(*YES) TRG(SEB_OPBFCPP_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/OPBFCPP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
