/*                                                                 */
/*  PROGRAM NUMBER:  PLD9UPC                                       */
/*  PROGRAM NAME:    PRINT TRACKING TIMES  & TRACKING SUMMARY      */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     01/14/04                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT FOR PRINT.                        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  RMC  02/03/04    PASS IN REPORT TYPE, AND CALL TRACKING TIMES  */
/*                   OR SUMMARY BASED ON REPORT TYPE               */
/*  ISE  11/05/20    RECOMPILED DUE TO CHANGE IN BUY ORDER NUMBER  */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &CO +
                          &TRKR &DRVR &TRNOFC &SRTYP &PROD &RPTTYP +
               &EXCEP     &DFROM &DTO &OMS)


/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN)     TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV)    TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)

/* SYSTEM NAME PARM                                              */
             DCL        VAR(&OMS)     TYPE(*CHAR) LEN(33)
             DCL        VAR(&LPP)     TYPE(*CHAR) LEN(40)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&ADTO)    TYPE(*CHAR) LEN(7)
             DCL        VAR(&ADFROM)  TYPE(*CHAR) LEN(7)
             DCL        VAR(&ACO)     TYPE(*CHAR) LEN(3)
             DCL        VAR(&TRNOFC) TYPE(*CHAR) LEN(6)
             DCL        VAR(&TRKR)   TYPE(*CHAR) LEN(6)
             DCL        VAR(&DRVR)   TYPE(*CHAR) LEN(6)
             DCL        VAR(&SRTYP)   TYPE(*CHAR) LEN(1)
             DCL        VAR(&PROD)  TYPE(*CHAR) LEN(6)
             DCL        VAR(&RPTTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&EXCEP) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES)  TYPE(*DEC)  LEN(2 0)
             DCL        VAR(&CO)      TYPE(*DEC)  LEN(3 0)
             DCL        VAR(&DFROM)   TYPE(*DEC)  LEN(7 0)
             DCL        VAR(&DTO)     TYPE(*DEC)  LEN(7 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD)     TYPE(*CHAR) LEN(512)
/* OPNQRY FILE VARIABLES                                       */
        DCL     &A        *CHAR   45              /* QRY KILL DATES  */
        DCL    &FRMDATA    *CHAR    7            /* FROM KILL DATE (Passed)   */
        DCL    &THRDATA    *CHAR    7            /* THRU KILL DATE (Passed)   */
             CHGVAR     VAR(&FRMDATA) VALUE(&DFROM)
             CHGVAR     VAR(&THRDATA) VALUE(&DTO)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/****************************************************************/
/*  PRINT TRACKING TIMES                                        */
/****************************************************************/
             IF         COND(&RPTTYP = 'T') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLD8PFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT PGM                                                   */
             CALL       PGM(PLD8PFR) PARM(' ' &CO &DFROM &TRKR &DRVR +
                          &TRNOFC &SRTYP &PROD &EXCEP &DFROM &DTO &LPP)
             ENDDO
/****************************************************************/
/*  PRINT TRACKING SUMMARY                                      */
/****************************************************************/
             IF         COND(&RPTTYP = 'S') THEN(DO)
/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLEAPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* OPEN QUERY FILE TO GET THE DATE RANGE                    */
             CHGVAR     VAR(&A) VALUE('AFE0DT *GE ' *CAT +
                         &FRMDATA *CAT ' *AND AFE0DT *LE ' +
                          *CAT &THRDATA)
             OVRDBF     FILE(PLAFCPL6) SHARE(*YES)
             OPNQRYF    FILE((PLAFCPL6))  +
                        QRYSLT(&A)   KEYFLD(*FILE)
/* PRINT PGM                                                   */
             CALL       PGM(PLEAPFR) PARM(' ' &CO &TRKR &DRVR +
                          &TRNOFC &SRTYP &PROD &EXCEP &DFROM &DTO &LPP)
             ENDDO
             DLTOVR *ALL

             ENDPGM
