/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  SUBMIT PROCESS PROGRAM                              */
/*                                                                  */
/*  SYSTEM    - INVENTORY MANAGEMENT SYSTEM                         */
/*  PROGRAM   - SUBMITS PROGRAM TO CALL THE PROGRAM THAT WILL       */
/*              CREATE TRANSFER TEMPLATE ORDERS                     */
/*  DATE      - 12/27/2006                                          */
/*  PROGRAMMER- LARA BUSER                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &COMPYN     - COMPANY NUMBER - PASSED AS 15.5           */
/********************************************************************/
PMC8UPC:     PGM        PARM(&RETURN &COMPYN)

             DCL        VAR(&RETURN)  TYPE(*CHAR) LEN(7)
             DCL        VAR(&COMPYN) TYPE(*DEC)

/*------------------- CALL TRANSFER TEMPLATE ORDER ------------------*/
             SBMJOB     CMD(CALL PGM(PMC8UPC) PARM(' ' &COMPYN)) +
                          JOB(SBMXFRTMPL)

 END:        ENDPGM
