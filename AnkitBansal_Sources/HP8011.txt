      *****************  RPG PROGRAM HEADING  ************************

      * SYSTEM:      HOG PRODUCTION
      * PROGRAM:     HP8011
      * TITLE:       Format SYNON Date into CCYYMMDD Format
      * PROGRAMMER:  LeAnne Fedor
      * CREATED:     01/16/03
      *
      * FUNCTION:  Given an incoming date in SYNON format, this program formats it
      *            into the CCYYMMDD format.
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      *
      /EJECT
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * CONSTANTS
      ****************************************************************
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      *
      ****************************************************************
      * ARRAYS AND TABLES
      ****************************************************************
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      * Parms
      *
     D xxccyymmdd      s              8  0
     D xxsynondt       s              7  0
      *
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      *---------------------------------------------------------------
      * Data structures for dates
      *---------------------------------------------------------------
      *
      * Dates in the Hog Production System are ccyymmdd.
      *
     d                 ds
     d  wkcymd                 1      8  0
     d  wkcen                  1      2  0
     d  wkyer                  3      4  0
     d  wkmon                  5      6  0
     d  wkday                  7      8  0
      *
      *
      * Synon dates are cyymmdd.
      *
     D                 ds
     D  wk0ymd                 1      7  0
     D  wkcc                   1      1  0
     D  wkyy                   2      3  0
     D  wkmm                   4      5  0
     D  wkdd                   6      7  0
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************
      *
      * Synon dates are 7,0 numeric fields where the first digit is:
      *          0=20th century
      *          1=21st century
      * Example: 0951225 for Christmas in 1995
      *          1951225 for Christmas in 2095
      *
      *
      * Format the SYNON date into an 8,0 field
      *
     C                   z-add     xxsynondt     wk0ymd
     C                   z-add     wkyy          wkyer
     C                   z-add     wkmm          wkmon
     C                   z-add     wkdd          wkday
     C                   if        wkcc = 0
     C                   z-add     19            wkcen
     C                   else
     C                   z-add     20            wkcen
     C                   endif
      *
     C                   z-add     wkcymd        xxccyymmdd
      *
     C                   seton                                        lr
      /EJECT
      *---------------------------------------------------------------
      * Initialization subroutine
      *---------------------------------------------------------------
      *
     C     *inzsr        begsr
      *
      * Parm lists
      *
     C     *entry        plist
     C                   parm                    xxsynondt
     C                   parm                    xxccyymmdd
      *
     C                   endsr
      /EJECT
