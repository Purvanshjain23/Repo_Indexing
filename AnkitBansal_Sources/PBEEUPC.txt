/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  COPIED FROM PDUCUPC TO RUN THE LATE DELIVERY     */
/*              REPORT AND EMAIL TO RECIPIENTS F/INVNIGHTRN          */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT REPORTS                                       */
/*  DATE      - 07/19/2012                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &PGM        - CL PROGRAM TO SUBMIT                      */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*                         IF THIS IS BLANK AT START, PROMPT        */
/*                         PGM IS CALLED; OTHERWISE, THE PRINT CL   */
/*                         IS SUBMITTED WITHOUT PROMPTING           */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*     B    &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*     B    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     B    &FRMDATN    - FROM DATE      (PASSED)                   */
/*     B    &THRDATN    - THRU DATE      (PASSED)                   */
/*     B    &LOADN      - LOAD NUMBER    (PASSED)                   */
/*     B    &CARRIER    - CARRIER CODE   (PASSED)                   */
/*     B    &LOADT      - LOAD TYPE      (PASSED)                   */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     B    &RPTSLT     - SELECTED REPORT TYPE                      */
/*     B    &UOM        - UNIT OF MEASURE                           */
/*     B    &EXPORTYN   - EXCLUDE EXPORTS Y OR N                    */
/*     B    &OTRGROUP   - ON TIME REASON GROUP CODE                 */
/********************************************************************/
/*  RMC 1/02/2001 ADD SALESPERSON AS INPUT PARM AND PASS TO CALLED  */
/*                PROGRAMS                                          */
/*  RMC 6/27/2003 ADDED REASON PARM                                 */
/*  LJB 6/14/2006 ADDED EXCLUDE EXPORT PARM, REMOVED CUSTOMER#      */
/*                IT IS NOT USED AND SYNON PARMS ARE MAX'D OUT      */
/*  LJB 6/21/2006 ADDITIONAL PARMS WERE ADDED. CHANGED INPUT PARMS  */
/*                TO MATCH SYNON PARM LIST                          */
/*  LJB 6/30/2006 REPORTS WERE CHANGED TO REPLICATE THE CARRIER     */
/*                PERFORMANCE REPORTS. CARRIER PERFORMANCE AND      */
/*                LATE DELIVERY REPORTS ALL CALL THIS CL PROGRAM.   */
/*  RMC 7/19/2006 ADDED SHIP TO CUST SELECTION                      */
/*  LJB 8/25/2006 ADDED ON TIME REASON AND ON TIME DEPARTMENT       */
/*                CODES TO PARMS AND PASS TO CALLED PROGRAMS        */
/*  LJB 5/08/2015 E004076                                           */
/*                ADDED EMAIL ADDR IN 2 PARTS TO PASS TO PPY4UPC    */
/*  RMC 3/01/2020 ADDED EMXLS PARM = N                              */
/********************************************************************/
 PBEEUPC:    PGM        PARM(&RETURN &PGM &APPLIC &CARRIER &DESC +
                          &OUTQ &HOLD &SAVE &NQSTAT &NIGHTQ +
                          &COMPNYN &YEARN &FRMDATN &THRDATN &LOADN +
                          &COPYN &SALESP &CUST &EXPORTYN &UOM +
                          &LOADT &QTR &RPTSLT &OTREAS &OTRDEPT +
                          &OTRGROUP)
/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)     /* RETURN CODE            */
        DCL    &APPLIC     *CHAR   (06)    /* APPLICATION CODE       */
        DCL    &PGM        *CHAR   (10)    /* CL TO SUBMIT           */
        DCL    &DESC       *CHAR   (25)    /* PRT PGM DESCRIPTION    */
        DCL    &COPYN      *DEC            /* NO. OF COPIES          */
        DCL    &OUTQ       *CHAR   (10)    /* OUTPUT QUEUE           */
        DCL    &HOLD       *CHAR   (4)     /* HOLD STATUS            */
        DCL    &SAVE       *CHAR   (4)     /* SAVE STATUS            */
        DCL    &NQSTAT     *CHAR   (1)     /* IF MUST RUN IN NIGHT Q */
        DCL    &NIGHTQ     *CHAR   (10)    /* NIGHT JOB QUEUE        */
        DCL    &COMPNYN    *DEC            /* COMPANY NUMBER (PASSED)*/
        DCL    &QTR        *CHAR   (1)     /* QUARTER                */
        DCL    &YEARN      *DEC            /* YEAR           (PASSED)*/
        DCL    &FRMDATN    *DEC            /* FROM DATE      (PASSED)*/
        DCL    &THRDATN    *DEC            /* THRU DATE      (PASSED)*/
        DCL    &LOADN      *DEC            /* LOAD NUMBER    (PASSED)*/
        DCL    &CUST       *DEC            /* SHIP TO CUST   (PASSED)*/
        DCL    &SALESP     *CHAR   (3)     /* SALESPERSON CODE       */
        DCL    &CARRIER    *CHAR   (3)     /* CARRIER CODE           */
        DCL    &LOADT      *CHAR   (1)     /* LOAD TYPE              */
        DCL    &OTRGROUP   *CHAR   (1)     /* ON TIME REASON GROUP   */
        DCL    &COPYN      *DEC            /* NO. OF COPIES          */
        DCL    &DESC       *CHAR   (25)    /* PRT PGM DESCRIPTION    */
        DCL    &OUTQ       *CHAR   (10)    /* OUTPUT QUEUE           */
        DCL    &HOLD       *CHAR   (4)     /* HOLD STATUS            */
        DCL    &SAVE       *CHAR   (4)     /* SAVE STATUS            */
        DCL    &NQSTAT     *CHAR   (1)     /* IF MUST RUN IN NIGHT Q */
        DCL    &NIGHTQ     *CHAR   (10)    /* NIGHT JOB QUEUE        */
        DCL    &RPTSLT     *CHAR   (1)     /* REPORT SELECT          */
        DCL    &UOM        *CHAR   (2)     /* UNIT OF MEASURE        */
        DCL    &EXPORTYN   *CHAR   (1)     /* EXCLUDE EXPORTS Y/N    */
        DCL    &OTREAS     *CHAR   (3)     /* ON TIME REASON         */
        DCL    &OTRDEPT    *CHAR   (2)     /* ON TIME REASON DEPT    */
        DCL    &DISTSCH    *CHAR    30    /* SCHED EMAIL DIST          LIST */
        DCL    &COMPNY     *DEC    (3)     /* COMPANY                */
        DCL    &EMXLS      *CHAR   (1)     /* EMAIL XLS             */
E4076:  DCL    &EMAIL1     *CHAR   (25)    /* EMAIL 1                */
        DCL    &EMAIL2     *CHAR   (25)    /* EMAIL 2                */

/* INTERNAL FIELDS                                                   */
        DCL    &COPY       *DEC    (2 0)   /* NUMBER OF COPIES       */

/* ERROR MESSAGE FIELDS                                              */
        DCL    &MSG        *CHAR   (512)   /* MESSAGE                */
        DCL    &MSGID      *CHAR   (7)     /* MESSAGE ID             */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/
             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&COMPNY) VALUE(&COMPNYN)
             CHGVAR     VAR(&EMXLS)  VALUE('N')

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PRINT CL PROGRAM   PPY4UPC                                */
/********************************************************************/
/* E004076                                                          */
 DONTSBM:      CALL PGM(&PGM) PARM(&COMPNYN &FRMDATN +
                          &THRDATN &LOADN &QTR &YEARN &SALESP &CUST +
                          &CARRIER &LOADT &OUTQ &HOLD &SAVE &COPYN +
                          &NIGHTQ &RPTSLT &UOM &OTRGROUP &EXPORTYN +
                          &OTREAS &OTRDEPT &EMXLS &EMAIL1 &EMAIL2)

/*-------------------------EMAIL TO RECIPS ------------------------*/
/* GET EMAIL DIST LIST FOR LATE DELIVERY            */
 GETEMLSC:   CALL       PGM(POMTXFR) PARM(' ' &COMPNY 'LATEDLVRPT' +
                          &DISTSCH)

             ESNDMAIL   RECIPIENT(&DISTSCH) SUBJECT('Late Delivery +
                          Report') ATTLIST((* *PDF *N PPY4PFR$))
             MONMSG     MSGID(CPF0000)
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')


/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
