/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING ACCRUAL CODE DEPENDENCIES    */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  9/15/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &STATUS        -  PASSED ACCRUAL CODE RECORD STATUS      */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
/* RMC 10/21/04 REMOVE COMPANY FROM PARM AND CALLED PGM - PROJ EXPCOM */
/* RMC 15/29/07 ADDED ACCRUAL CODE PARM, SBML FROM WW INSTEAD OF     */
/*              CALLING AN INTERMEDIATE CL PGM  */
/********************************************************************/
 PDYZCLP:    PGM        PARM(&RTN  &ACCRL &STATUS &OUTQ +
                          &HOLD &SAVE &NIGHTQ &COPYN)
        DCL    &STATUS      *CHAR  (01)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &ACCRL       *CHAR  (03)
        DCL    &RTN         *CHAR  (7)

        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDYZPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDYZPFR) PARM(' '  &ACCRL &STATUS)

 ENDPGM:     ENDPGM
