/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING CUSTOMER  LISTING            */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  3/03/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &CUSTN         - CUSTOMER NUMBER (PASSED)                */
/*         &CITY          - CITY(PASSED)                            */
/*         &STATE         - STATE(PASSED)                           */
/*         &STATUS        - STATUS(PASSED)                          */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 CABXCLP:    PGM        PARM(&APPLIC &CUSTN &CITY &STATE &STATUS +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ &EDITPN)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &CUSTN       *DEC
        DCL    &CUST        *DEC   (7 0)
        DCL    &CITY        *CHAR  (20)
        DCL    &STATE       *CHAR  (2)
        DCL    &STATUS      *CHAR  (1)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &EDITPN      *CHAR  (6)

        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)
             CHGVAR     VAR(&CUST)  VALUE(&CUSTN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDCOPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDCOPFR) PARM(' ' &CUST &CITY &STATE +
                          &STATUS &EDITPN)

 ENDPGM:     ENDPGM
