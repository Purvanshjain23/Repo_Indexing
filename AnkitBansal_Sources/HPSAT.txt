      * FUNCTION:  Given an incoming date, this program retrieves the date of the
      *            previous Saturday.

      ****************************************************************

      * 99/99/99
      *
      /EJECT

      *
      ****************************************************************

      ****************************************************************

      *
      ****************************************************************

      ****************************************************************

     D no              c                   'N'

      *
      ****************************************************************
      * ARRAYS AND TABLES
      ****************************************************************
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      *
      * Work fields for date manipulation
      *
     D wksatdt         s              8  0
     D wkyymmdd        s              6  0
     D wkcymdiso       s               D   datfmt(*iso)
      *
      *
      * Parms in and out of this program
      *
     D xxccyymmdd      s              8  0
     D xxsatdt         s              8  0
      *
      *
      * Parms for call to Synon program to get date
      *
     D xxcocd          s              3  0
     D xxreturn        s              7a
     D xxcyymmddi      s              7  0
     D xxcyymmddo      s              7  0
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************
      *
     C                   z-add     0             xxsatdt
      *
      *
      * Get the date into the SYNON format.
      *
     C                   z-add     xxccyymmdd    wkyymmdd
     C                   if        xxccyymmdd >= 20000000
     C     wkyymmdd      add       1000000       xxcyymmddi
     C                   else
     C                   z-add     wkyymmdd      xxcyymmddi
     C                   endif
      *
      * Call SYNON program to get previous Saturday
      *
     C                   z-add     0             xxsatdt
     C                   call      'POGSXFR'
     C                   parm      *blank        xxreturn
     C                   parm      350           xxcocd
     C                   parm                    xxcyymmddi
     C                   parm      0             xxcyymmddo
      *
     C                   if        xxreturn = *blank                            If got Saturday
     C                   z-add     xxcyymmddo    wkyymmdd
     C                   z-add     wkyymmdd      xxsatdt
     C                   if        xxcyymmddo >= 1000000
     C                   add       20000000      xxsatdt
     C                   else
     C                   add       19000000      xxsatdt
     C                   endif
     C                   endif                                                  If got Saturday
      *
     C                   seton                                        lr
      /EJECT
      *---------------------------------------------------------------
      * Initialization subroutine
      *---------------------------------------------------------------

     C     *inzsr        begsr
      *
      * Parm lists
      *
     C     *entry        plist
     C                   parm                    xxccyymmdd
     C                   parm                    xxsatdt
      *
     C                   endsr
      /EJECT
