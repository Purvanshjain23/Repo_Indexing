/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PRE-INVOICE REGISTER                           */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT PRE-INVOICE REGISTER                          */
/*  DATE      -  7/21/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPNYN    - COMPANY NUMBER - BLANK FOR ALL            */
/*          &LOADN      - PROCESS ORDERS ON LOAD                    */
/*          &THRDATN    - PROCESS ORDERS THRU ACTUAL SHIP DATE      */
/*          &ORDRN      - NOT USED HERE; PASSED TO MATCH PARMS OF   */
/*                        PDR4PCLP.                                 */
/*          &RFLAG      - NOT USED HERE; PASSED TO MATCH PARMS OF   */
/*                        PDR4PCLP.                                 */
/*          &BILTYP     - BILLING TYPE (INVOICE,MEMO)               */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/*          &NIGHTQ     - NIGHT OUTPUT QUEUE                        */
/*          &PRTREG     - PRINT PRE-INVOICE REGISTER                */
/*          &PRTEXC     - PRINT PRE-INVOICE EXCEPTION REPORT        */
/*          &JOBNME     - NOT USED HERE--FOR AUTOMATIC FAXING       */
/********************************************************************/
/* RMC 4/25/05 PRINT INV EXC ON PRKPRTO5 FOR DAMON.. PER JEFF SHERBONDY   */
/*             BUT NOT IF ISRCENT, ISPPATE, OR ISSMASO USER         */
/* SLM 10/25/05 THIS CLP WAS ADD TO THE OMSDEVMDL FUNCTIONS         */
/*              THIS CLP WAS DEFINED TO QCLSRC, BUT NOT OMSDEVMDL   */
/* SLM 10/25/05 PASS IN NEW PARM AND REARRANGE PARM TO MATCH PGM IF */
/*              USED.  THIS WILL CHANGE ON THE CL AND ALSO ON THE   */
/*              PGM THAT IS BEING PASSED IN AND ALSO PASS TO        */
/*              PDULPFR PDJ7PFR                                     */
/* SLM 10/27/05 ALSO ADD PRKTEST AND ALICEB                         */
/*              THEN CHANGE THE 2ND PDJ7PFR TO PASS THE PRO CUST TYP*/
/* LJB 02/10/06 REMOVED CODE TO PRINT 2ND EXCEPTION REPORT ON       */
/*              PRKPRT05 SINCE DAMON CAN NOW PRINT THE REPORT WHEN  */
/*              HE NEEDS IT. REMOVED ASSOCIATED CODE THAT SKIPPED   */
/*              THAT CODE FOR SELECTED USER PROFILES. THIS CODE     */
/*              IS CURRENTLY AFTER THE ENDPGM - DELETE WHEN IT IS   */
/*              DETERMINED THAT THIS WON'T CHANGE BACK.             */
/********************************************************************/
 PDULPCLP:   PGM        PARM(&COMPNYN &ORDRN &THRDATN &LOADN &BILTYP +
                          &OUTQ &HOLD &SAVE &COPYN &NQSTAT &NIGHTQ +
                          &RFLAG &PRTREG &PRTEXC &JOBNME &JOBNUM +
                          &ARCUSTP &PRCUSTT)

/* PARMS                                                                      */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &ORDRN      *DEC                  /* Order Number   (Passed)   */
        DCL    &THRDATN    *DEC                  /* Thru Ship Date (Passed)   */
        DCL    &LOADN      *DEC                  /* Load  Number   (Passed)   */
        DCL    &BILTYP     *CHAR   (1)           /* Bill Type (Invoice,Memo)  */

        DCL    &RFLAG      *CHAR   (1)           /* NOT USED                  */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COPYN      *DEC                  /* No. of Copies  (Passed)   */
        DCL    &NQSTAT     *CHAR   (1)           /* If MUST run in night q.   */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &PRTREG     *CHAR   (1)           /* Print Pre-Invoice Register*/
        DCL    &PRTEXC     *CHAR   (1)           /* Print Pre-Invoice Except. */
        DCL    &JOBNME     *CHAR   (10)          /* Job Name -- Not Used      */
        DCL    &JOBNUM     *CHAR   (6)          /* Job Name -- Not Use d      */
        DCL    &ARCUSTP    *DEC                  /* No. of Copies  (Passed)   */
        DCL    &PRCUSTT    *CHAR   (02)          /* PROCESS CUST TYPE (PASSED)*/
        DCL    &USER       *CHAR   (10)          /* USER                     */

/* INTERNAL FIELDS                                                            */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */
        DCL    &LOAD       *DEC    (7 0)         /* Load Number               */
        DCL    &ORDER      *DEC    (7 0)         /* ORDER NUMBER              */
        DCL    &THRDAT     *DEC    (7 0)         /* Thru Actual Ship Date     */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &PGM1O      *CHAR   (10)          /* Printer File              */
        DCL    &PGM2O      *CHAR   (10)          /* Printer File              */
        DCL    &COMMAND    *CHAR   (512)         /* Printer override field    */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&LOAD) VALUE(&LOADN)
             CHGVAR     VAR(&ORDER) VALUE(&ORDRN)
             CHGVAR     VAR(&THRDAT) VALUE(&THRDATN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&PGM1O) VALUE('PDULPFR$')
             CHGVAR     VAR(&PGM2O) VALUE('PDJ7PFR$')

/********************************************************************/
/* OVERRIDE PRINTER FILES WITH USER SPECIFIED SELECTION VALUES      */
/********************************************************************/
             IF         COND(&PRTREG *EQ 'Y') THEN(DO)
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGM1O) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)
             ENDDO

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGM2O) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*********************************************************************/
/*   CALL PROGRAMS                                                   */
/*********************************************************************/

/*      PRE-INVOICE REGISTER                                         */
             IF         COND(&PRTREG *EQ 'Y') THEN(DO)
             CALL       PGM(PDULPFR) PARM(' ' &COMPANY &ORDER +
                          &THRDAT &LOAD &BILTYP &PRCUSTT)
             ENDDO

             CALL       PGM(PDJ7PFR) PARM(' ' &COMPANY &ORDER +
                          &THRDAT &LOAD &BILTYP &PRCUSTT)

 END:        ENDPGM
