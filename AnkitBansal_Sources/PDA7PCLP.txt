/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING LOAD LISTING                 */
/*                                                                  */
/*  SYSTEM    - TRAFFIC                                             */
/*  DATE      - 10/18/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &DATEN         -  PASSED SCHEDULED SHIP DATE             */
/*         &SEQN          -  PASSED SCHEDULE SEQUENCE NUMBER        */
/*         &LOADSTS       -  PASSED LOAD STATUS                     */
/*         &CARRIER       -  PASSED CARRIER CODE                    */
/*         &TYPE          -  PASSED LOAD TYPE (REGULAR, BULK)       */
/*         &STATE         -  PASSED STATE CODE                      */
/*         &INCEXC        -  PASSED INCLUDE/EXCLUDE OPTION          */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 PDA7PCLP:   PGM        PARM(&COMPNYN &DATEN &SEQN &LOADSTS &CARRIER +
                          &TYPE &STATE &INCEXC &APPLIC &OUTQ &HOLD +
                          &SAVE &COPYN &NIGHTQ)
        DCL    &COMPNYN     *DEC
        DCL    &DATEN       *DEC
        DCL    &SEQN        *DEC
        DCL    &LOADSTS     *CHAR  (01)
        DCL    &CARRIER     *CHAR  (03)
        DCL    &TYPE        *CHAR  (01)
        DCL    &STATE       *CHAR  (02)
        DCL    &INCEXC      *CHAR  (03)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &DATE        *DEC   (7 0)
        DCL    &SEQ         *DEC   (5 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&DATE)     VALUE(&DATEN)
             CHGVAR     VAR(&SEQ)      VALUE(&SEQN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDA7PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDA7PFR) PARM(' ' &COMPANY &DATE &LOADSTS +
                          &CARRIER &TYPE &SEQ &STATE &INCEXC)

 ENDPGM:     ENDPGM
