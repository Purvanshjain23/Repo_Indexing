/*  PROGRAM NUMBER:  PKQTUPR                                       */
/*  PROGRAM NAME:    PRINT DOCUMENT SCREEN                        */
/*  PROGRAMMER:      ROSE MARTIN                                   */
/*  CREATE DATE:     04/17/95                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM WORK WITH #OFFICE                        */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  04/20/95                                                       */
/*  Rose M. Martin   Added SBDPRTOVR code and command.             */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &FUN &FILE &PDEV &HOLD &SAVE +
                          &XCOPIES)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* SYSTEM NAME PARM                                              */
 /*          DCL        VAR(&OMS) TYPE(*CHAR) LEN(40)          +
             DCL        VAR(&XOMS) TYPE(*CHAR) LEN(33)       */

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&FUN) TYPE(*CHAR) LEN(25)
             DCL        VAR(&FILE) TYPE(*CHAR) LEN(25)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
    /*       CHGVAR     VAR(&OMS) VALUE(&XOMS)                  */

/* ADD LIBRARY Y2SY */
             ADDLIBLE   LIB(Y2SY) POSITION(*LAST)
             ADDLIBLE   LIB(Y1SY) POSITION(*LAST)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(YDOCMDLFN$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT                                                       */
        YDOCMDLFUN MDLFILE(&FILE) MDLFUN(&FUN) PRTTEXT(*NONE) +
                          PRTDEVDSN(*YES)

             RMVLIBLE   LIB(Y1SY)
             MONMSG     MSGID(CPF0000)
             RMVLIBLE   LIB(Y2SY)
             MONMSG     MSGID(CPF0000)

             DLTOVR *ALL
             ENDPGM
