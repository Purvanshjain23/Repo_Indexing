/********************************************************************/
/*                                                                  */
/*  PROGRAM   - PMVTUPC - PRINT CUSTOMER PRICE CODE REPORRTS        */
/*  SYNOPSIS  - OVERRIDE PRINTER FILE TO PASSED IN PARAMETERS       */
/*              SELECT REPORT BASED SELECTION FIELD THAT IS         */
/*              PASSED IN FROM THE SCREEN                           */
/*                                                                  */
/*  SYSTEM    - OMS                                                 */
/*  DATE      - 06/16/2008                                          */
/*            - LARA BUSER                                          */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*  &RETURN     - RETURN CODE                                       */
/*  &APPLIC     - APPLICATION                                       */
/*  &OUTQ       - OUTPUT QUEUE                                      */
/*  &HOLD       - HOLD STATUS                                       */
/*  &SAVE       - SAVE STATUS                                       */
/*  &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q             */
/*  &NIGHTQ     - NIGHT JOB QUEUE                                   */
/*  &COPIES     NUMBER OF COPIES (DEFAULTS TO 1 INITIALLY)           */
/*  &CUSTN      CUSTOMER NUMBER 15.5                                 */
/*  &FDATEN     START DATE 15.5                                      */
/*  &TDATEN     END DATE 15.5                                        */
/*  &PRCDE      PRICE CODE 15.5                                      */
/*  &USER       ENTERED BY USER PROFILE                              */
/*  &SELECT     SELECTED REPORT TYPE                                 */
/*                                                                  */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/*------------------------------------------------------------------*/
/*                                                                  */
/********************************************************************/
 PMVTUPC:    PGM        PARM(&RETURN &APPLIC &OUTQ &HOLD &SAVE +
                          &NQSTAT &NIGHTQ &CUSTN &FDATEN &TDATEN +
                          &COPYN &FPRCDEN &USER &SELECT &TPRCDEN)

/* PARMS                                                             */
             DCL  VAR(&RETURN)  TYPE(*CHAR) LEN(7)
             DCL  VAR(&APPLIC)  TYPE(*CHAR) LEN(6)
             DCL  VAR(&CUSTN)   TYPE(*DEC)
             DCL  VAR(&CUST)    TYPE(*DEC) LEN(7 0)
             DCL  VAR(&DESC)    TYPE(*CHAR) LEN(25)
             DCL  VAR(&OUTQ)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&HOLD)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&SAVE)    TYPE(*CHAR) LEN(4)
             DCL  VAR(&COPYN)   TYPE(*DEC)
             DCL  VAR(&COPY)    TYPE(*DEC)  LEN(2 0)
             DCL  VAR(&NQSTAT)  TYPE(*CHAR) LEN(01)
             DCL  VAR(&NIGHTQ)  TYPE(*CHAR) LEN(10)
             DCL  VAR(&FDATEN)  TYPE(*DEC)
             DCL  VAR(&FDATE)   TYPE(*DEC) LEN(7 0)
             DCL  VAR(&TDATEN)  TYPE(*DEC)
             DCL  VAR(&TDATE)   TYPE(*DEC) LEN(7)
             DCL  VAR(&FPRCDEN) TYPE(*DEC)
             DCL  VAR(&FPRCDE)  TYPE(*DEC) LEN(6)
             DCL  VAR(&TPRCDEN) TYPE(*DEC)
             DCL  VAR(&TPRCDE)  TYPE(*DEC) LEN(6)
             DCL  VAR(&USER)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&SELECT)  TYPE(*CHAR) LEN(1)

/*------------------------------------------------------------------*/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/*------------------------------------------------------------------*/
             CHGVAR     VAR(&RETURN) VALUE(' ')
             CHGVAR     VAR(&DESC) VALUE('Customer Price Code Reports')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&CUST) VALUE(&CUSTN)
             CHGVAR     VAR(&FDATE) VALUE(&FDATEN)
             CHGVAR     VAR(&TDATE) VALUE(&TDATEN)
             CHGVAR     VAR(&FPRCDE) VALUE(&FPRCDEN)
             CHGVAR     VAR(&TPRCDE) VALUE(&TPRCDEN)

/*------------------------------------------------------------------*/
/* OVERRIDE PRINTER FILE TO PARMS PASSED IN FROM CALLING PROGRAM    */
/* CALL THE SELECTED CUSTOMER PRICE CODE PROGRAM                    */
/*-- REPORT BY CUSTOMER NUMBER -------------------------------------*/
             IF         COND(&SELECT *EQ '1') THEN(DO)
             OVRPRTF    FILE(PMVOPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)
 CUSTOMER:   CALL       PGM(PMVOPFR) PARM('' &CUST &FPRCDE &FDATE +
                          &TDATE &USER &TPRCDE)
             GOTO       CMDLBL(END)
             ENDDO

/*-- REPORT BY START DATE  -----------------------------------------*/
             IF         COND(&SELECT *EQ '2') THEN(DO)
             OVRPRTF    FILE(PMVQPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)
 FROMDATE:   CALL       PGM(PMVQPFR) PARM('' &CUST &FPRCDE &FDATE +
                          &TDATE &USER &TPRCDE)
             GOTO       CMDLBL(END)
             ENDDO

/*-- REPORT BY PRICE CODE  -----------------------------------------*/
             IF         COND(&SELECT *EQ '3') THEN(DO)
             OVRPRTF    FILE(PMVPPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)
 PRICECODE:  CALL       PGM(PMVPPFR) PARM('' &CUST &FPRCDE &FDATE +
                          &TDATE &USER &TPRCDE)
             GOTO       CMDLBL(END)
             ENDDO

/*-- REPORT BY ENTERED BY USER  ------------------------------------*/
             IF         COND(&SELECT *EQ '4') THEN(DO)
             OVRPRTF    FILE(PMVRPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)
 ENTEREDBY:  CALL       PGM(PMVRPFR) PARM('' &CUST &FPRCDE &FDATE +
                          &TDATE &USER &TPRCDE)
             GOTO       CMDLBL(END)
             ENDDO

/*-- REPORT IF ALL SELECTIONS WERE BLANKS  -------------------------*/
             IF         COND(&SELECT *EQ '5') THEN(DO)
             OVRPRTF    FILE(PPANPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)
 ALL:        CALL       PGM(PPANPFR) PARM('')
             ENDDO

 END:        ENDPGM
