SELECT
  ALL       B7ELDT, SUM(B7A5QT), SUM(B7AHWG)
  FROM      PRKFLIB/CAB7CPP T01
  WHERE     B7BSST IN ('RR', 'TW')
    AND     B7ELDT BETWEEN 1010917 AND 1011006
    AND     DIGITS(B7HGCD) NOT LIKE '%0'
    AND     DIGITS(B7HGCD) NOT LIKE '009%'
    AND     B7AIC3 = 360
    AND     B7AJCD = 'GP1'
  GROUP BY  B7ELDT
  ORDER BY  B7ELDT ASC
