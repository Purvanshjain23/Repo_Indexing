/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM WITH SELECTIONS FOR LATE DELIVERY RPT. */
/*             COMPANY, DATES, LOAD, CARRIER                        */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT DROP TRAILER REPORT                           */
/*  DATE      - 12/19/97                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &CARRIER    - CARRIER CODE                              */
/*          &ONYARD     - ON YARD STATUS                            */
/*          &ASSIGN     - ASSIGNED STATUS                           */
/*          &AVAIL      - AVAILABLE STATUS                          */
/*          &RCDSTS     - RECORD STATUS                             */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/********************************************************************/

 PDGPUCLP:   PGM        PARM(&CARRIER &COMPNYN &ONYARD &ASSIGN +
                          &AVAIL &RCDSTS &OUTQ &HOLD &SAVE &COPYN)
/* PARMS                                                                      */
        DCL    &CARRIER    *CHAR   (3)           /* CARRIER CODE              */
        DCL    &COMPNYN    *DEC                  /* COMPANY NUMBER            */
        DCL    &ONYARD     *CHAR   (1)           /* LOAD TYPE                 */
        DCL    &ASSIGN     *CHAR   (1)           /* ASSIGNED STS              */
        DCL    &AVAIL      *CHAR   (1)           /* AVAILABLE STS             */
        DCL    &RCDSTS     *CHAR   (1)           /* RECORD STATUS             */
        DCL    &DESC       *CHAR   (25)          /* DESCRIPTION               */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMPANY    *DEC    (3 0)         /* Company number            */


/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)


/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDGRPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)


/*********************************************************************/
/*   CALL PRINT PROGRAM -
/*********************************************************************/

             CALL       PGM(PDGRPFR) PARM(' ' &CARRIER  &COMPANY +
                          &ONYARD &ASSIGN  &AVAIL &RCDSTS)

 END:        ENDPGM
