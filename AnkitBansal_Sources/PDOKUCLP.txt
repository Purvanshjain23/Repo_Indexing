/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM WITH SELECTIONS FOR PRINT EXP DOC DFT. */
/*             COMPANY, DOC TYPE, COUNTRY                          */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT EXPORT DOCUMENT DEFAULTS                      */
/*  DATE      - 05/28/98                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPNYN    - COMPANY NUMBER                            */
/*          &DOCTYP     - DOCUMENT TYPE                             */
/*          &COUNTR     - COUNTRY                                   */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/********************************************************************/

 PDF5PCLP:   PGM   PARM(&COMPNYN &DOCTYP &COUNTR &PGM2 &PGM2$ &OUTQ  +
                 &HOLD    &SAVE &COPYN)

/* PARMS                                                                      */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &DOCTYP     *CHAR   (1)           /* DOC TYPE                  */
        DCL    &COUNTR     *CHAR   (6)           /* COUNTRY                   */
        DCL    &PGM2       *CHAR   (10)          /* PRTFIL TO CALL            */
        DCL    &PGM2$      *CHAR   (10)          /* PRTFIL TO OVR             */
        DCL    &DESC       *CHAR   (25)          /* DESCRIPTION               */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/


     SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGM2$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)


/*********************************************************************/
/*   CALL PRINT PROGRAM -
/*********************************************************************/

             CALL       PGM(&PGM2) PARM(' ' &COMPANY  +
                          &DOCTYP &COUNTR)

 END:        ENDPGM
