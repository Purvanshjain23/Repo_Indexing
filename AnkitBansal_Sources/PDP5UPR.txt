/*       Title: Display Spool File, Then Delete the Spool File       */
/*        Date: 06/17/98                                             */
/* Description: - DSPSPLF                                            */
/*              - DLTSPLF                                            */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &FILE       Spool File Name                              */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*                                                                   */
/*   Date     Pgmr     Description                                   */
/* -------- ---------- --------------------------------------------  */
/*                                                                   */
/*********************************************************************/
             PGM  PARM(&FILE)

   /*-------------------*/
   /* Declare Variables */
   /*-------------------*/

             /* Parameter Fields */
             DCL  VAR(&FILE) TYPE(*CHAR) LEN(10)

   /*--------------*/
   /* Display SPLF */
   /*--------------*/
             DSPSPLF FILE(&FILE) SPLNBR(*LAST)
             MONMSG MSGID(CPF3309 CPF3303)

   /*-------------*/
   /* Delete SPLF */
   /*-------------*/
             DLTSPLF FILE(&FILE) SPLNBR(*LAST)
             MONMSG MSGID(CPF3309 CPF3303)

END:
             ENDPGM
