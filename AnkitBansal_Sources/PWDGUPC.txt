             PGM        PARM(&PCODE &LCODE &HPEFILELIB)

             DCL        VAR(&GRPPRF) TYPE(*CHAR) LEN(30)

             DCL        VAR(&PCODE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&LCODE) TYPE(*CHAR) LEN(6)
             DCL        VAR(&HPEFILELIB) TYPE(*CHAR) LEN(10)

             DCL        VAR(&PRODLOC) TYPE(*CHAR) LEN(12)
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(40) VALUE('OMS +
                          Earlier release')

             IF         COND(&HPEFILELIB *EQ 'PRKFTSTGUP') +
                          THEN(CHGVAR VAR(&HPEFILELIB) VALUE(PRKFGUP))
             IF         COND(&HPEFILELIB *EQ 'PRKFTSTTFP') +
                          THEN(CHGVAR VAR(&HPEFILELIB) VALUE(PRKFTFP))
             IF         COND(&HPEFILELIB *EQ 'PRKFTSTSTP') +
                          THEN(CHGVAR VAR(&HPEFILELIB) VALUE(PRKFSTP))

             OVRDBF     FILE(PLAPREL1) TOFILE(&HPEFILELIB/PLAPREL1)
             OVRDBF     FILE(PLAPREL0) TOFILE(&HPEFILELIB/PLAPREL0)

             OVRDSPF    FILE(PLEFEFR#) TOFILE(PRKPLIB/PLEFEFR#)
             CHGVAR     VAR(&PRODLOC) VALUE(&PCODE *CAT &LCODE)
             CALL       PGM(PRKPLIB/PLEFEFR) PARM(' ' &PRODLOC &OMS)

             DLTOVR     FILE(*ALL)

ENDPGM
