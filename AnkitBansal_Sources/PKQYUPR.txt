/*                                                                 */
/*  FUNCTION:        THIS CL IS CALLED FROM THE EXTERNAL SCALE     */
/*                   PRINT FUNCTION TO RETRIEVE THE DTAARA THAT    */
/*                   SIGNALS A USER HAS REQUESTED TO INITIALIZE     */
/*                   THE SCALE TICKET TO BLANKS AND ZEROS.         */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&INTSCL) /* Retrieve dtaara */
             DCL        VAR(&INTSCL) TYPE(*CHAR) LEN(1)
             RTVDTAARA  DTAARA(INITSCALE (1 1)) RTNVAR(&INTSCL)
             CHGDTAARA  DTAARA(INITSCALE (1 1)) VALUE(' ')
             ENDPGM
