/*                                                                 */
/*  PROGRAM NUMBER:  PKQXUPR                                       */
/*  PROGRAM NAME:    PRINT COMPARISON CARCASS SUMMARY/DETAIL       */
/*  PROGRAMMER:      PURVA K DROGE                                 */
/*  CREATE DATE:     03/19/99                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED FROM A PROMPT FOR        */
/*                   PRINTING A COMPARISON CARCASS SUMMARY & DETAIL*/
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  SLM     04/16/04  TFM - REPLACE PKMLPFR WITH PLEMPFR           */
/*******************************************************************/
             PGM        PARM(&PDEV &HOLD &SAVE &XCOPIES &COMP &CODE +
                          &FDTE &TDTE &TTO &FKDTE &TKDTE &PRT)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(2)
             DCL        VAR(&CODE) TYPE(*CHAR) LEN(10)
/*           DCL        VAR(&XFDTE) TYPE(*CHAR) LEN(15) */
/*           DCL        VAR(&XTDTE) TYPE(*CHAR) LEN(15) */
/*           DCL        VAR(&XTTO) TYPE(*CHAR) LEN(5)   */
/*           DCL        VAR(&XKDTE) TYPE(*CHAR) LEN(15) */
             DCL        VAR(&PRT) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TTO) TYPE(*DEC) LEN(5 0)
             DCL        VAR(&FKDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TKDTE) TYPE(*DEC) LEN(7 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
/*           CHGVAR     VAR(&COMP) VALUE(&XCOMP)     */
/*           CHGVAR     VAR(&FDTE) VALUE(&XFDTE)     */
/*           CHGVAR     VAR(&TDTE) VALUE(&XTDTE)     */
/*           CHGVAR     VAR(&BO) VALUE(&XBO)         */
/*           CHGVAR     VAR(&LD) VALUE(&XLD)         */
/*           CHGVAR     VAR(&TTO) VALUE(&XTTO)       */
/*           CHGVAR     VAR(&KDTE) VALUE(&XKDTE)     */

/* CREATE TATTOO HEADER DRIVER FILE IN QTEMP                    */
 THPHY:      CRTDUPOBJL OBJ(PKA1CPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 THUPD:      CRTDUPOBJL OBJ(PKA1CPL1) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 THRTV:      CRTDUPOBJL OBJ(PKA1CPL0) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 THRSQ:      CRTDUPOBJL OBJ(PKA1CPLF) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 THRSQ2:     CRTDUPOBJL OBJ(PKA1CPLI) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 CDPHY:      CRTDUPOBJL OBJ(PKB1CPP) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 CDUPD:      CRTDUPOBJL OBJ(PKB1CPLD) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
 CDRSQ:      CRTDUPOBJL OBJ(PKB1CPL2) FROMLIB(*LIBL) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO) /*RSQ*/

/* BUILD TATTOO HEADER & CARCASS DETAIL */
             CALL       PGM(PKNZXFR) PARM(' ' &COMP &CODE &FDTE +
                          &TDTE &TTO &FKDTE &TKDTE &PRT)

/*******************************************************************/
/*     PRINT CARCASS SUMMARY AND DETAIL                            */
/*******************************************************************/

/*           SBDPRTOVR  CMDVAR(&CMD) FILE(PKMLPFR$) OUTQ(&PDEV) +   */
/*                        COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) + */
/*                        LETTER(*YES)                              */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PLEMPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE) +
                          LETTER(*YES)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/*          CALL       PGM(PKMLPFR) PARM(' ' &COMP &BO &LD &TTO 'N' +*/
/*                        'SUMM' 'DTL' &PRT) */
             CALL       PGM(PKQYXFR) PARM(' ' &COMP &CODE &PRT)

END:         DLTOVR *ALL
             ENDPGM
