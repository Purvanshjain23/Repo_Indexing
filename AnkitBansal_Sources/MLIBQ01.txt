SELECT
  ALL       T01.ODOBNM, T01.ODOBTP, T01.ODOBAT, T01.ODLBNM, T01.ODDDAT,
            T01.ODDTIM, T01.ODDCEN
  FROM      HPEGEN/TLIB T01 LEFT EXCEPTION JOIN
            HPEGEN/FLIB T02
  ON        T01.ODOBNM = T02.ODOBNM
    AND     T01.ODOBTP = T02.ODOBTP
  WHERE     T01.ODOBTP <> '*PGM'
    AND     T02.ODOBTP <> '*PGM'
    AND     T01.ODOBAT <> 'DSPF'
    AND     T02.ODOBAT <> 'DSPF'
