/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PDJNUPR                                       */
/*  PROGRAM NAME:    Print Projected Cut Floor Items               */
/*  PROGRAMMER:      Purva Patel                                   */
/*  CREATE DATE:     03/22/95                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from Display Projected Cut Floor Items.       */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*  CENTONZE -- 10/08/98  ADD BUY ORDER NUMBER AND LOAD TO PARMS   */
/*                        AND PASS INTO THE PRINT FILE.           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES &PGM +
                          &PGMPRTF &APP &XCOMP &XBO &XBOL &XTATTOO +
                          &XITEM &PRDCR &BUYER &BYRTYPE &PAYTYPE +
                          &XKILLDTE)
/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGMPRTF) TYPE(*CHAR) LEN(10)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(6)

/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XKILLDTE) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XTATTOO) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XITEM) TYPE(*CHAR) LEN(15)
             DCL        VAR(&PRDCR) TYPE(*CHAR) LEN(6)
             DCL        VAR(&BUYER) TYPE(*CHAR) LEN(6)
             DCL        VAR(&BYRTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PAYTYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&XBO) TYPE(*CHAR) LEN(15)
             DCL        VAR(&XBOL) TYPE(*CHAR) LEN(15)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

             DCL        VAR(&COMMAND) TYPE(*CHAR) LEN(512)

             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KILLDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TATTOO) TYPE(*DEC) LEN(5 0)
             DCL        VAR(&ITEM) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&BO) TYPE(*DEC) LEN(5 0)
             DCL        VAR(&BOL) TYPE(*DEC) LEN(5 0)

/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)
             CHGVAR     VAR(&KILLDTE) VALUE(&XKILLDTE)
             CHGVAR     VAR(&TATTOO) VALUE(&XTATTOO)
             CHGVAR     VAR(&BO) VALUE(&XBO)
             CHGVAR     VAR(&BOL) VALUE(&XBOL)
             IF COND(&XITEM *EQ ' ') +
              THEN(DO)
                 CHGVAR     VAR(&ITEM) VALUE('0')
              ENDDO
             ELSE +
              CMD(DO)
                 CHGVAR     VAR(&ITEM) VALUE(&XITEM)
              ENDDO


/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(&PGMPRTF) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(&PGM) PARM(' ' &COMP &KILLDTE &BO &BOL +
                  &TATTOO &ITEM &PRDCR &BUYER &BYRTYPE &PAYTYPE &APP)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
