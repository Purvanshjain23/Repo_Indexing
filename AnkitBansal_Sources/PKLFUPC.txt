/*  FUNCTION:        THIS CL IS SUBMITTED WITH EXECUTE MESSAGE     */
/*                   FROM PROMPT FOR CREATE DAMAGE FILE STEP I.    */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*******************************************************************/
             PGM        PARM(&RTN &ACO &ADFROM &ADTO &CCODE &SRCE +
                          &MKTCD &PRCGRP &DTMETH &PROPT &CMPPGM +
                          &PAYTYP &PAYSTS &DSTEP)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)

/* SYSTEM NAME PARM                                              */

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&ADTO) TYPE(*CHAR) LEN(7)
             DCL        VAR(&ADFROM) TYPE(*CHAR) LEN(7)
             DCL        VAR(&ACO) TYPE(*CHAR) LEN(3)
             DCL        VAR(&CCODE) TYPE(*CHAR) LEN(5)
             DCL        VAR(&SRCE) TYPE(*CHAR) LEN(3)
             DCL        VAR(&MKTCD) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PRCGRP) TYPE(*CHAR) LEN(7)
             DCL        VAR(&DTMETH) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PROPT) TYPE(*CHAR) LEN(3)
             DCL        VAR(&CMPPGM) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PAYTYP) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PAYSTS) TYPE(*CHAR) LEN(2)
             DCL        VAR(&DSTEP) TYPE(*CHAR) LEN(1)


/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&DFROM) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&DTO) TYPE(*DEC) LEN(7 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&CO) VALUE(&ACO)
             CHGVAR     VAR(&DFROM) VALUE(&ADFROM)
             CHGVAR     VAR(&DTO) VALUE(&ADTO)

/* INITIALIZE BOH AND TH AND CREATE DAMAGE FILE WITH ACTUALS  */
STEP1:       IF         COND(&DSTEP = '1')        +
        THEN(CALL      PGM(PKKTXFR) PARM(' ' &CCODE &CO &DFROM &DTO +
                          &SRCE &MKTCD &PRCGRP &DTMETH &PROPT +
                          &CMPPGM &PAYTYP &PAYSTS))

/* UPDATE DAMAGE FILE WITH SELECTED CONTRACT'S INFO           */
STEP4:       IF         COND(&DSTEP = '4')        +
        THEN(CALL      PGM(PKKWXFR) PARM(' ' &CCODE &CO &DTO +
                          &SRCE  &DFROM))
             ENDPGM
