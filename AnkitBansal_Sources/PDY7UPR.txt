/********************************************************************/
/* PRINT CASH JOURNAL                                               */
/* RMC 3/24/05 ADDED DEPOSIT TIME AND UPDATE STS PARMS,RMV COMPANY  */
/********************************************************************/
 PDY7UPR:    PGM        PARM(&COMPNYN &BANK &DATNUM &TIME &UPDSTS &OUTQ +
                          &HOLD &SAVE &NIGHTQ &COPYN)
/* PARMS                                                                      */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &PGM        *CHAR   (10)          /* CL to Submit              */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &BANK       *CHAR   (3)           /* BANK CODE */
        DCL    &DEPDAT     *DEC    (7 0)         /* DEPOSIT DATE  */
        DCL    &UPDSTS     *CHAR   (1)           /* UPDATE STS     */
        DCL    &TIME       *CHAR   (4)           /* DEPOSIT TIME   */
        DCL    &TIMEN      *DEC    (4 0)         /* DEPOSIT TIME   */
        DCL    &DATNUM     *DEC
        DCL    &COMMAND    *CHAR   (512)   /* PRINTFILE OVERRIDE */
/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM                                        */
/********************************************************************/

             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&DEPDAT) VALUE(&DATNUM)
             CHGVAR     VAR(&TIMEN) VALUE(&TIME)
             CLRPFM     FILE(PDJ5CPP)
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDVKPFR$) OUTQ(&OUTQ) +
                          LANDSCAPE(*YES) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&command 512)
             CALL       PGM(PDWOXFR) PARM(' ' &BANK &DEPDAT &TIMEN +
                          &UPDSTS)
             GOTO       CMDLBL(END)
 END:        ENDPGM
