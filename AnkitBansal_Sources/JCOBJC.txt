/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  PROGRAM TO CHECK FOR EXISTENCE OF OBJECT TO         */
/*              VALIDATE PROMPT FROM RPG PROGRAM                    */
/*                                                                  */
/*                                                                  */
/*  COMPANY   - DATAWRIGHT INC.                                     */
/*  SYSTEM    - IDEAS/400 JOB CONTROL                               */
/*  USER NAME - LINDA MEINKE                                        */
/*  DATE      - 11/27/89                                            */
/*  COPYRIGHT 1989 DATAWRIGHT INC.                                  */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &OBJ        - OBJECT NAME                               */
/*          &TYPE       - OBJECT TYPE                               */
/*                                                                  */
/*      PARAMETERS USED:                                            */
/*                                                                  */
/*          &RETURN     - RETURN CODE IF ERROR                      */
/*                                                                  */
/********************************************************************/

 JCOBJC:     PGM        PARM(&OBJ &TYPE &RETURN)

      DCL      &OBJ         *CHAR   10           /*  OBJECT NAME    */
      DCL      &TYPE        *CHAR   10           /*  OBJECT TYPE    */
      DCL      &RETURN      *CHAR   7            /*  RETURN CODE    */

/********************************************************************/
/*  CHECK OBJECT TO SEE IF IT EXISTS                                */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE(' ')
             CHKOBJ     OBJ(&OBJ) OBJTYPE(&TYPE)

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(CHGVAR VAR(&RETURN) +
                          VALUE(USR0280))

 END:        ENDPGM
