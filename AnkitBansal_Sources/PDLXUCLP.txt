/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM WITH SELECTIONS FOR LOAD SCHEDULING   */
/*             POSITION REPORT. PARMS CO,LOAD, SHIP DATE,DDLN DPRT  */
/*             DATE AND TIME, LOAD GRADE                            */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PRINT LOAD SCHEDULING DETAIL REPORT                 */
/*  DATE      - 03/27/98                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPNYN    - COMPANY NUMBER - BLANK FOR ALL            */
/*          &SHIPDTN    - SCHED SHIP DATE                            */
/*          &LOADIDN     - LOAD ID                                   */
/*          &DPRTDTN    - DEADLINE DEPARTURE DATE                    */
/*          &DPRTTMN    - DEADLINE DEPARTURE TIME                   */
/*          &LOADG      - LOAD GRADE                                */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD PRINT                                */
/*          &SAVE       - SAVE PRINT                                */
/*          &COPYN      - NUMBER OF COPIES                          */
/********************************************************************/
/* RMC 04/26/07 DONT PASS LOAD GRADE PARM, OR USE AS THE KEY IN THE */
/*              PRTF                                                */
/* SLM 02/02/09 FILE FROM &DATAS AND PULL TO QTEMP                  */
/* CREATE FILES FOR REPORT IN QTEMP                                 */
/* FROM THE USER LIBRARY &DATAS                                     */
/********************************************************************/

 PDLXUCLP:   PGM        PARM(&COMPNYN &LOADIDN &SHIPDTN &DPRTDTN +
                          &DPRTTMN &OUTQ &HOLD &SAVE &COPYN)

/* PARMS                                                                      */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &LOADIDN    *DEC                  /* LOAD ID        (Passed)   */
        DCL    &SHIPDTN    *DEC                  /* SCHD SHIP DATE (Passed)   */
        DCL    &DPRTDTN    *DEC                  /* DDLN DEPRT DTE (Passed)   */
        DCL    &DPRTTMN    *DEC                  /* DDLN DEPRT TM (Pas sed)   */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &LOADG      *CHAR  LEN(1)       /* LOAD GRADE PASSED  */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COMMAND    *CHAR   (512)

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */
        DCL    &COMPANY    *DEC    (3 0)         /* Company Number            */
        DCL    &SHIPDT     *DEC    (7 0)         /* SCHD SHIP DATE            */
        DCL    &LOADID     *DEC    (7 0)       /* LOAD ID                     */
        DCL    &DPRTDT     *DEC    (7 0)         /* DDLN DPRT DATE            */
        DCL    &DPRTTM     *DEC    (6 0)         /* DDLN DPRT TIME            */
             DCL        VAR(&USER) TYPE(*CHAR) LEN(10) /* USER */
             DCL        VAR(&DATAS) TYPE(*CHAR) LEN(10)

/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/********************************************************************/

             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&SHIPDT) VALUE(&SHIPDTN)
             CHGVAR     VAR(&LOADID) VALUE(&LOADIDN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
             CHGVAR     VAR(&DPRTDT) VALUE(&DPRTDTN)
             CHGVAR     VAR(&DPRTTM) VALUE(&DPRTTMN)


/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDLWPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*********************************************************************/
/*    PRINT THE REPORT                  */
/*********************************************************************/
/*******************************************************************/
/* Create Duplicate verion of file to QTEMP.                       */
/*******************************************************************/

/*           GET THE USER WORK LIBRARY                             */
             CALL       PGM(PMD9XFR) PARM(' ' &DATAS)


/*  IF  FILES  ARE IN QTEMP THEN GET RID OF THESE                  */
/*  ANY ERROR MESSAGE WOULD BE THEY ARE NOT THERE                 */
             DLTF       FILE(QTEMP/PDMHCP*)
             MONMSG     MSGID(CPF0000) EXEC(DO)
             GOTO       CMDLBL(START)
             ENDDO

START:

             CRTDUPOBJ  OBJ(PDMHCPP) FROMLIB(&DATAS) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*YES)

             CRTDUPOBJ  OBJ(PDMHCPL3) FROMLIB(&DATAS) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

             CALL       PGM(PDLWPFR) PARM(' ' &COMPANY &LOADID       +
                          &SHIPDT &DPRTDT &DPRTTM)

 END:        ENDPGM
