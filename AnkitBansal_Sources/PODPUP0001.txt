/* ***************************************************************** */
/* PROGRAM - PODPUPR EXC FAX LOAD CONFIRM UP                         */
/* ***************************************************************** */
/*  MODIFIED:  RMC 5/24/02  ADDED 'FAX TO CARRIER' PARM &FAXCAR AND  */
/*                            'FAX TO DISPATCH OFFICE' PARM &FAXDIS  */
/*  MODIFIED:  LJB 5/16/06  CHANGED &FAX# TO 25 CHARACTERS TO        */
/*                            ACCOMMODATE INTERNATIONAL FAX NUMBERS  */
/* ***************************************************************** */
             PGM        PARM(&FAXR &FAXC &FAXA &FAXP &USER &PGM +
                          &COMPL &ORDERL &CARRIER &JOBNM &PRTF +
                          &FAXCAR &FAXDIS)

             DCL        VAR(&FAXR) TYPE(*CHAR) LEN(25)
             DCL        VAR(&FAXC) TYPE(*CHAR) LEN(30)
             DCL        VAR(&FAXA) TYPE(*CHAR) LEN(3)
             DCL        VAR(&FAXP) TYPE(*CHAR) LEN(12)
/*** LJB     DCL        VAR(&FAX#) TYPE(*CHAR) LEN(15)  5/16/06     */
             DCL        VAR(&FAX#) TYPE(*CHAR) LEN(25)
             DCL        VAR(&FTX1) TYPE(*CHAR) LEN(75)
             DCL        VAR(&FTX2) TYPE(*CHAR) LEN(75)
             DCL        VAR(&FTX3) TYPE(*CHAR) LEN(75)
             DCL        VAR(&FTX4) TYPE(*CHAR) LEN(75)
             DCL        VAR(&FTX5) TYPE(*CHAR) LEN(75)
             DCL        VAR(&FTX6) TYPE(*CHAR) LEN(75)
             DCL        VAR(&USER) TYPE(*CHAR) LEN(10)
             DCL        VAR(&USER) TYPE(*CHAR) LEN(10)
             DCL        VAR(&PGM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&COMPL) TYPE(*DEC)
             DCL        VAR(&ORDERL) TYPE(*DEC)
             DCL        VAR(&JOBNM) TYPE(*CHAR) LEN(10)
             DCL        VAR(&CARRIER) TYPE(*CHAR) LEN(3)
             DCL        VAR(&PRTF)    TYPE(*CHAR) LEN(10)
             DCL        VAR(&FAXCAR) TYPE(*CHAR) LEN(1)
             DCL        VAR(&FAXDIS) TYPE(*CHAR) LEN(1)

             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(3)

             DCL        VAR(&DATALIB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HISTLIB) TYPE(*CHAR) LEN(10)

             CALL       PGM(PDVVUPC) PARM(&RETURN &PGM &COMPL +
                          &ORDERL &CARRIER &JOBNM &FAXCAR &FAXDIS)

             ENDPGM
