SELECT
  ALL       T01.GLMCU,
            GLMCU CONCAT '.' CONCAT SUBSTR(GLOBJ,1,4) CONCAT '.' CONCAT GLSUB
            AS ACCT, T02.RPAN8, ((GLAA/100)), T01.GLCO, T01.GLPN, T03.MCDL01
  FROM      JFOODPROD/F0911 T01 LEFT OUTER JOIN
            JFOODPROD/F0411 T02
  ON        T01.GLDCT = T02.RPDCT
    AND     T01.GLDOC = T02.RPDOC LEFT OUTER JOIN
            JFOODPROD/F0006 T03
  ON        T01.GLMCU = T03.MCMCU
  WHERE     GLCO = '00360'
    AND     GLDCT IN ('PV', 'PM')
    AND     GLPN = 3
  ORDER BY  001 ASC, 002 ASC, 003 ASC, 005 ASC, 006 ASC, 007 ASC, 004 ASC
