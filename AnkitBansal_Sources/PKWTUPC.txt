/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PKWTUPC                                       */
/*  PROGRAM NAME:    RECALCULATE HITCH PAYMENT                     */
/*  PROGRAMMER:      PURVA DROGE                                   */
/*  CREATE DATE:     02/09/01                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL IS SUBMITTED FROM A PROMPT THAT       */
/*                   WILL RECALCUALTE PAYMENT FOR HITCH TATTOOS    */
/*                   AND PRINT A REPORT WITH THE DIFFERENCE BETWEEN*/
/*                   THE AMOUNT THAT WAS PAID BEFORE AND THE       */
/*                   NEW AMOUNT.                                   */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&CO &FDTE &TDTE &PRCTYP &PRICE &PRD +
                          &PDEV &HOLD &SAVE &XCOPIES)

/*******************************************************************/
/*  VARIABLE DEFINITION                                            */
/*******************************************************************/

/* INPUT PARMS                                                    */
             DCL        VAR(&CO) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&FDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&TDTE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&PRCTYP) TYPE(*CHAR) LEN(2)
             DCL        VAR(&PRICE) TYPE(*DEC) LEN(5 2)
             DCL        VAR(&PRD) TYPE(*CHAR) LEN(6)

/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(2)
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)

/* SYSTEM NAME PARM                                               */

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/*******************************************************************/
/*  INITIALIZATION                                                 */
/*******************************************************************/
/* CHANGE ALPHA PARMS TO NUMERIC VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)

/*******************************************************************/
/*  MAINLINE                                                       */
/*******************************************************************/
/* RECALCULATE HITCH PAYMENT                                   */
             CALL       PGM(PKWRXFR) PARM(' ' &CO &FDTE &TDTE +
                          &PRCTYP &PRICE &PRD)

/* CREATE PRINTER FILE OVERRIDE CODE                            */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PKWSPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PRINT VARIANCE REPORT                                       */
             CALL       PGM(PKWSPFR) PARM(' ' &CO &FDTE &PRD &TDTE)

             GOTO       CMDLBL(COMPLETION)

/*******************************************************************/
/*  ERROR HANDLING                                                 */
/*******************************************************************/

/*******************************************************************/
/*  END PROGRAM                                                    */
/*******************************************************************/
COMPLETION:

ENDPGM:
             DLTOVR *ALL

             ENDPGM
