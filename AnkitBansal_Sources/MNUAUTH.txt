SELECT
  ALL       T01.MNAME, T01.MOPTN, T02.MDESC, T01."USERID"
  FROM      WM400/WMAUTH T01 LEFT OUTER JOIN
            WM400/WMOPTN T02
  ON        T01.COID = T02.COID
    AND     T01.MCLASS = T02.MCLASS
    AND     T01.MAPLCN = T02.MAPLCN
    AND     T01.MNAME = T02.MNAME
    AND     T01.MOPTN = T02.MOPTN
  WHERE     T01.COID = 'CHF'
  ORDER BY  T01."USERID" ASC, T01.MNAME ASC, T01.MOPTN ASC
