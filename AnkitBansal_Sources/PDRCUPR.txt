/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR PRINTER OVERRIDE INFORMATION */
/*              GENERIC PROMPTING FUNCTION. CAUSE PRINTER OVERRIDE  */
/*              PROMPTING AND PASSES COMPANY, WAREHOUSE, ITEM,      */
/*              ITEM STR TYPE, ITEM STR GROUP, ITEM STR CLASS,      */
/*              FROM DATE, TO DATE, TRANSACTION TYPES, REASON CODES,*/
/*              USER ID, AND REPORT TYPE                            */
/*              INVENTORY TRANSACTION JOURNAL.                      */
/*                                                                  */
/*  SYSTEM    - INVENTORY CONTROL                                   */
/*  PROGRAM   - INVENTORY TRANSACTION JOURNAL                       */
/*  DATE      - 06/14/95                                            */
/********************************************************************/
/*                                                                  */
/* MODIFICATIONS:                                                   */
/*  CENTONZE 11/05/98  ADDED FROM/THRU TIME PARMS                   */
/*  DROGE    10/26/04  ADDED INVENTORY TRANS. ID                    */
/*  BUSER    01/24/07  ADDED "INCLUDE ZERO QUANTITIES"              */
/*  CENTONZE 01/15/09  SBMJOB TO SBDWORK1 FOR NOW                   */
/*                     SCAN: FORNOW                                */
/*  MASON S  03/04/09  PASS THE EMAIL ADDRESS                       */
/*  CENTONZE 03/21/13  SBMJOB TO &LONGQ INSTEAD OF SBDWORK1 -    */
/*                     FROM COMPANY VALUE "JOBQLONG"            */
/*                     SCAN: FORNOW        E2516                   */
/********************************************************************/
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &CLP        - CL PROGRAM TO SUBMIT                      */
/*     I    &PGM        - RPG PROGRAM TO SUBMIT                     */
/*     I    &DESC       - REPORT DESCRIPTION                        */
/*     B    &OUTQ       -  OUTPUT QUEUE                             */
/*                         IF THIS IS BLANK AT START, PROMPT        */
/*                         PGM IS CALLED; OTHERWISE, THE PRINT CL   */
/*                         IS SUBMITTED WITHOUT PROMPTING           */
/*     B    &HOLD       -  HOLD STATUS                              */
/*     B    &SAVE       -  SAVE STATUS                              */
/*     B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q    */
/*     B    &NIGHTQ     -  NIGHT JOB QUEUE                          */
/*     B    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     B    &FRMDATN    - FROM DATE      (PASSED)                   */
/*     B    &THRDATN    - THRU DATE      (PASSED)                   */
/*     B    &FRMTIMN    - FROM TIME      (PASSED)                   */
/*     B    &THRTIMN    - THRU TIME      (PASSED)                   */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     B    &WHSE       - WAREHOUSE CODE (PASSED)                   */
/*     B    &ITEM       - ITEM CODE      (PASSED)                   */
/*     B    &ITMTYP     - ITEM STR TYPE  (PASSED)                   */
/*     B    &ITMGRP     - ITEM STR GROUP (PASSED)                   */
/*     B    &ITMCLS     - ITEM STR CLASS (PASSED)                   */
/*     B    &TTYPE1     - TRANSACTION TYPE1 (PASSED)                */
/*     B    &TTYPE2     - TRANSACTION TYPE2 (PASSED)                */
/*     B    &TTYPE3     - TRANSACTION TYPE3 (PASSED)                */
/*     B    &TTYPE4     - TRANSACTION TYPE4 (PASSED)                */
/*     B    &RSN1       - REASON CODE 1   (PASSED)                  */
/*     B    &RSN2       - REASON CODE 2   (PASSED)                  */
/*     B    &RSN3       - REASON CODE 3   (PASSED)                  */
/*     B    &RSN4       - REASON CODE 4   (PASSED)                  */
/*     B    &USER       - USER ID         (PASSED)                  */
/*     B    &RPTTYP     - REPORT TYPE     (PASSED)                  */
/*     B    &DATETYP    - DATE TYPE       (PASSED)                  */
/*     B    &SHFT       - SHIFT NUMBER    (PASSED)                  */
/*     B    &EXPDOM     - EXPORT/DOMESTIC (PASSED)                  */
/*     B    &AVLGRP     - PA GROUP        (PASSED)                  */
/*     B    &TRNID      - INVENTORY TRANS ID (PASSED)               */
/*     B    &ZEROQTY    - INCLUDE ZERO QUANTITIES (PASSED)          */
/*                                                                  */
/********************************************************************/
 PDRCUPR:    PGM        PARM(&RETURN &APPLIC &CLP &PGM &DESC &OUTQ +
                          &HOLD &SAVE &NQSTAT &LONGQ &COMPNYN +
                          &FRMDATN &THRDATN &COPYN &WHSE &ITEM +
                          &TTYPE1 &TTYPE2 &TTYPE3 &TTYPE4 &RSN1 +
                          &RSN2 &RSN3 &RSN4 &USER &ITMCLS &ITMGRP +
                          &ITMTYP &RPTTYP &ZEROQTY &DATETYP &SHFT +
                          &EXPDOM &AVLGRP &FRMTIMN &THRTIMN &TRNID +
                          &EMAIL)

/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &CLP        *CHAR   (10)          /* CL to Submit              */
        DCL    &PGM        *CHAR   (10)          /* RPG TO SUBMIT             */
        DCL    &DESC       *CHAR   (25)          /* REPORT DESC               */
        DCL    &EMAIL      *CHAR   (50)          /* EMAIL ADDRESS             */
        DCL    &COPYN      *DEC                  /* No. of Copies             */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &NQSTAT     *CHAR   (01)          /* If MUST run in Night Q    */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue           */
        DCL    &LONGQ      *CHAR   (10)          /* LONG RUNNING JOBS Q       */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &FRMDATN    *DEC                  /* FROM DATE      (Passed)   */
        DCL    &THRDATN    *DEC                  /* THROUGH DATE   (Passed)   */
        DCL    &FRMTIMN    *DEC                  /* FROM TIME      (Passed)   */
        DCL    &THRTIMN    *DEC                  /* THROUGH TIME   (Passed)   */
        DCL    &WHSE       *CHAR   (3)           /* WARHOUSE       (Passed)   */
        DCL    &ITEM        *DEC                 /* ITEM           (Passed)   */
        DCL    &ITMTYP      *DEC                 /* ITEM STR TYPE  (Passed)   */
        DCL    &ITMGRP      *DEC                 /* ITEM STR GROUP (Passed)   */
        DCL    &ITMCLS      *DEC                 /* ITEM STR CLASS (Passed)   */
        DCL    &TTYPE1      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &TTYPE2      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &TTYPE3      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &TTYPE4      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &RSN1        *CHAR   3           /*  REASON CODE     */
        DCL    &RSN2        *CHAR   3           /*  REASON CODE     */
        DCL    &RSN3        *CHAR   3           /*  REASON CODE     */
        DCL    &RSN4        *CHAR   3           /*  REASON CODE     */
        DCL    &RPTTYP      *CHAR   1           /*  REPORT TYPE     */
        DCL    &DATETYP     *CHAR   1           /*  DATE TYPE       */
        DCL    &USER        *CHAR   10          /*  USER ID SEL.    */
        DCL    &SHFT        *CHAR   1           /*  SHIFT NUMBER    */
        DCL    &EXPDOM      *CHAR   1           /*  EXPORT/DOMESTIC */
        DCL    &AVLGRP      *CHAR   3           /*  AVAIL GROUP     */
        DCL    &TRNID       *CHAR   1           /*  TRANS ID        */
        DCL    &ZEROQTY     *CHAR   1           /*  INCL ZERO QTY   */

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

             CHGVAR     VAR(&RETURN) VALUE('       ')
             CHGVAR     VAR(&COPY) VALUE(&COPYN)
/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PROMPT PROGRAM                                            */
/********************************************************************/

/*           IF         COND(&OUTQ *NE '   ') THEN(GOTO CMDLBL(SUBMIT))*/

             CALL       PGM(PDG0PVR) PARM(&RETURN &DESC &OUTQ &HOLD +
                          &SAVE &COPY &NQSTAT &NIGHTQ &APPLIC)
             CHGVAR     VAR(&COPYN) VALUE(&COPY)
             IF         COND(&RETURN *NE ' ') THEN(GOTO +
                          CMDLBL(USERQUIT))

/********************************************************************/
/*   SUBMIT PRINT CL PROGRAM                                        */
/********************************************************************/
/* FORNOW:      CHGVAR     VAR(&NIGHTQ) VALUE('SBDWORK1')     */
FORNOW:      IF COND(&LONGQ *GT '      ') THEN(DO)
      CHGVAR     VAR(&NIGHTQ) VALUE(&LONGQ)
      ENDDO
 SUBMIT:     SBMJOB     CMD(CALL PGM(&CLP) PARM(&PGM &COMPNYN +
                          &FRMDATN &THRDATN &WHSE &ITEM &TTYPE1 +
                          &TTYPE2 &TTYPE3 &TTYPE4 &RSN1 &RSN2 &RSN3 +
                          &RSN4 &USER &RPTTYP &DATETYP &ITMTYP +
                          &ITMGRP &ITMCLS &ZEROQTY &SHFT &EXPDOM +
                          &AVLGRP &FRMTIMN &THRTIMN &TRNID &OUTQ +
                          &HOLD &SAVE &COPYN &EMAIL)) JOB(&PGM) +
                          JOBQ(&NIGHTQ)

             GOTO       CMDLBL(END)

/*-------------------------USER QUIT--------------------------------*/
 USERQUIT:   CHGVAR     VAR(&RETURN) VALUE('Y2U0005')
             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
