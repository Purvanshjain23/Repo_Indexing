/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR CANCELING A PHYSICAL COUNT            */
/*                                                                  */
/*  SYSTEM    - INVENTORY MANAGEMENT                                */
/*  DATE      -  8/18/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  COMPANY                                */
/*         &DATE          -  DATE                                   */
/*         &WHSE          -  WAREHOUSE                              */
/*         &CYCLE         -  CYCLE                                  */
/*                                                                  */
/********************************************************************/
 ICPOCLP:    PGM        PARM(&COMPNYN &DATE &WHSE &CYCLE)

        DCL    &COMPNYN     *DEC
        DCL    &DATE        *DEC
        DCL    &WHSE        *CHAR  (03)
        DCL    &CYCLE       *CHAR  (03)
        DCL    &RETURN      *CHAR  (7)

        DCL    &COMP        *DEC   (3 0)
        DCL    &DATEL       *DEC   (7 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE                                     */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)
             CHGVAR     VAR(&DATEL)  VALUE(&DATE)

 CALL:       CALL       PGM(ICP0XFR) PARM(&RETURN &COMP &WHSE &CYCLE +
                                          &DATEL)

 ENDPGM:     ENDPGM
