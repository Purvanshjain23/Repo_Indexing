/*  PROGRAM  -  PBC6UPC                                             */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  DATE      - 04/20/2012                                          */
/*  PROGRAMMER- LARA BUSER                                          */
/*  SYNOPSIS -  THIS PROGRAM MATCHES THE RECORD STATUS IN FILES     */
/*              ASSOCIATED TO THE CUSTOMER FILE TO THE RECORD       */
/*              STATUS IN THE CUSTOMER FILE                         */
/*              FILES ARE: COMPANY CUSTOMER, COMPANY CUSTOMER ITEM, */
/*                         CUSTOMER EXT, CUSTOMER FREIGHT RATE,     */
/*                         CUSTOMER ITEM REFERENCE                  */
/*                                                                  */
/*              THIS IS A ONE TIME CONVERSION                       */
/*                                                                  */
/*              THE 6 FILES ARE OVERRIDDEN TO PRKFLIB LIBRARY.      */
/*              THEN THE FILES ARE BACKED UP TO PRKFSAV. THEY       */
/*              WILL STAY THERE FOR 1 MONTH. FINALLY, THE           */
/*              CUSTOMER RECORD STATUS IS COMPARED TO EACH          */
/*              RECORD IN EACH FILE AND IF IT IS DIFFERENT THEN     */
/*              THE ASSOCIATED RECORD IS CHANGED TO MATCH THE       */
/*              CUSTOMER RECORD.                                    */
/*                                                                  */
/********************************************************************/
/* XXX NN/NN/NNNN XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX    */
/*                                                                  */
/********************************************************************/
 PBC6UPC:    PGM

             MONMSG     MSGID(CPF0000)

/**-------------------------PRKFLIB---------------------------------**/
/* EXECUTE FOR EACH COMPANY                                          */
/* OVERRIDE DATABASE FILES TO PRKFLIB                                */
             OVRDBF     FILE(CABBREP) TOFILE(PRKFLIB/CABBREP)
             OVRDBF     FILE(CAEOREP) TOFILE(PRKFLIB/CAEOREP)
             OVRDBF     FILE(PDNEREP) TOFILE(PRKFLIB/PDNEREP)
             OVRDBF     FILE(PDISREP) TOFILE(PRKFLIB/PDISREP)
             OVRDBF     FILE(PNGKREP) TOFILE(PRKFLIB/PNGKREP)
             OVRDBF     FILE(POBIREP) TOFILE(PRKFLIB/POBIREP)

/* BACKUP FILES TO PRKFSAV                                           */
             CPYF       FROMFILE(CABBREP) TOFILE(PRKFSAV/CABBREP_P) +
                          MBROPT(*REPLACE) CRTFILE(*YES)
             CPYF       FROMFILE(CAEOREP) TOFILE(PRKFSAV/CAEOREP_P) +
                          MBROPT(*REPLACE) CRTFILE(*YES)
             CPYF       FROMFILE(PDNEREP) TOFILE(PRKFSAV/PDNEREP_P) +
                          MBROPT(*REPLACE) CRTFILE(*YES)
             CPYF       FROMFILE(PDISREP) TOFILE(PRKFSAV/PDISREP_P) +
                          MBROPT(*REPLACE) CRTFILE(*YES)
             CPYF       FROMFILE(PNGKREP) TOFILE(PRKFSAV/PNGKREP_P) +
                          MBROPT(*REPLACE) CRTFILE(*YES)
             CPYF       FROMFILE(POBIREP) TOFILE(PRKFSAV/POBIREP_P) +
                          MBROPT(*REPLACE) CRTFILE(*YES)

/* CALL PROGRAM TO MATCH ASSOCIATED FILES TO THE CUSTOMER FILE       */
             CALL       PGM(PBC5XFR) PARM(' ')

             ENDPGM
