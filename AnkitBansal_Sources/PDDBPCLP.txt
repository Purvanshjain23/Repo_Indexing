/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING DESTINATION-ZONE LISTING     */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      - 10/02/97                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
PDDBPCLP:    PGM        PARM(&COMPNYN +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)
        DCL    &COMPNYN     *DEC
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COMPANY     *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND     *CHAR  (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMPANY)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)     VALUE(&COPYN)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDDBPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDDBPFR) PARM(' ' &COMPANY)

 ENDPGM:     ENDPGM
