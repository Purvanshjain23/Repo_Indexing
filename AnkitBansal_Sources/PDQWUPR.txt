/*  SYNOPSIS:                                                      */
/*              CL program that automatically start a Physical &    */
/*              read and update item balance & Item balance detail  */
/*              files.  This will make the physical ready to edit   */
/*              & Post on the AS400.                                */
/*                                                                  */
/*  SYSTEM:                                                         */
/*              Order Management System                             */
/*  DATE:                                                           */
/*              2/8/96                                              */
/*  PARAMETERS:                                                     */
/********************************************************************/
PGM
             DCL        VAR(&RETURN) TYPE(*CHAR) LEN(10)
             DCL        VAR(&FLIB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HLIB) TYPE(*CHAR) LEN(10)

 /*         RETREIVE FILE LIBRARY INTO FLIB                          */
             CALL       PGM(CAQXXFR) PARM(&RETURN &FLIB &HLIB)


 /*         * Start Physical Count                                   */

             CALL       PGM(PDY6XFR) PARM(' ')


 /*         * read physical entries & update 400 files               */

             CALL       PGM(PDZAXFR) PARM(' ')

 /*         COPY UPLOAD FILE INTO "AS" (HISTORY) FILE                */
             IF         COND(&FLIB *NE '          ') THEN(DO)

                    CPYF       FROMFILE(&FLIB/PDKACPP) +
                               TOFILE(&FLIB/ASKACPP) MBROPT(*ADD) +
                               CRTFILE(*YES)
             MONMSG     MSGID(CPF0000)

             ENDDO


 END: ENDPGM
