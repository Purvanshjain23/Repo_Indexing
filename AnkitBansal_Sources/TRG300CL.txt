/*                        CHGJRNOBJ command for HSP085.                   */
/**************************************************************************/

             PGM

/* CHANGE JOURNALED OBJECT TO SET IMAGES=*BOTH FOR HSP038 FILE. */

             CHGJRNOBJ  OBJ((PRKFLIB/HSP038 *FILE)) ATR(*IMAGES) +
                          IMAGES(*BOTH)

/* CHANGE JOURNALED OBJECT TO SET IMAGES=*BOTH FOR HSP085 FILE. */

             CHGJRNOBJ  OBJ((PRKFLIB/HSP085 *FILE)) ATR(*IMAGES) +
                          IMAGES(*BOTH)


/* APPLY PHYSICAL FILE TRIGGER (INSERT/UPDATE/DELETE) TO PRKFLIB LIB. +
   FOR HPS FEED TICKET DTL (HSP038) FILE. */

/****        CALL       PGM(TRG001CL) PARM('A' 'PRKFLIB')      ***/
             CALL       PGM(TRG038CL) PARM('A' 'PRKFLIB')


/* APPLY PHYSICAL FILE TRIGGER (INSERT/UPDATE/DELETE) TO PRKFLIB LIB. +
   FOR HPS SALES MOVEMENT DTL (HSP085) FILE. */

/****        CALL       PGM(TRG002CL) PARM('A' 'PRKFLIB')     ***/
             CALL       PGM(TRG085CL) PARM('A' 'PRKFLIB')


 END:        ENDPGM
