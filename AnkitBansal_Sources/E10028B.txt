       //----------------------------------------------------------------------
       // Program To Get the YMD from Julian date passed in to it
       //
       //
       // Create using: CRTRPGMOD MODULE(e1idevgen/e10028b)
       //                         SRCFILE(e1idevgen/QRPGLESRC)
       //                         SRCMBR(e10028b)
       //                         REPLACE(*NO)
       //
       //   R Centonze    5/26/2020
       //----------------------------------------------------------------------
      * Parms
     d datejul         s              6  0
     d yymmdd          s              6  0
      *
     d*
      *****************************************************************
      * Entry parameters
     C     *ENTRY        PLIST
     C                   PARM                    datejul
     C                   PARM                    yymmdd
      *****************************************************************

      /free
           yymmdd  = %dec(%char(%date(datejul:*jul):*ymd):6:0);
       return;
      /end-free
      *
