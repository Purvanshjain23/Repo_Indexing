      * Description:                                                 *
      * This program receives the Pallet Type and Quantity fields.   *
      * The values are concatenated together to create one long      *
      * string which is returned to the calling program.             *
      *                                                              *
      ****************************************************************

      * incoming parms
     d iptype1         s              6                                         Pallet Type 1
     d ipqty1          s              5  0                                      Pallet Qty 1
     d iptype2         s              6                                         Pallet Type 2
     d ipqty2          s              5  0                                      Pallet Qty 2
     d iptype3         s              6                                         Pallet Type 3
     d ipqty3          s              5  0                                      Pallet Qty 3
     d iptype4         s              6                                         Pallet Type 4
     d ipqty4          s              5  0                                      Pallet Qty 4

      * outgoing parm
     d opstring        s             60                                         Concatenated values

      * work fields
     d wkqtya1         s              5
     d wkqtya2         s              5
     d wkqtya3         s              5
     d wkqtya4         s              5
     d wktypeqty1      s             12
     d wktypeqty2      s             12
     d wktypeqty3      s             12
     d wktypeqty4      s             12

     c     *entry        plist
     c                   parm                    ipqty1
     c                   parm                    ipqty2
     c                   parm                    ipqty3
     c                   parm                    ipqty4
     c                   parm                    iptype1
     c                   parm                    iptype2
     c                   parm                    iptype3
     c                   parm                    iptype4
     c     opstring      parm                    opstring

      // convert decimal values to alpha
        wkqtya1 = %triml(%editc( ipqty1 : 'Z'));
        wkqtya2 = %triml(%editc( ipqty2 : 'Z'));
        wkqtya3 = %triml(%editc( ipqty3 : 'Z'));
        wkqtya4 = %triml(%editc( ipqty4 : 'Z'));

      // concatenate type1 and qty1 values
        if iptype1 <> *blanks;
          eval wktypeqty1 = iptype1 + ':' + wkqtya1;
        endif;

      // concatenate type2 and qty2 values
        if iptype2 <> *blanks;
          eval wktypeqty2 = iptype2 + ':' + wkqtya2;
        endif;

      // concatenate type3 and qty3 values
        if iptype3 <> *blanks;
          eval wktypeqty3 = iptype3 + ':' + wkqtya3;
        endif;

      // concatenate type4 and qty4 values
        if iptype4 <> *blanks;
          eval wktypeqty4 = iptype4 + ':' + wkqtya4;
        endif;

      // create output parm
        opstring = wktypeqty1 + wktypeqty2 + wktypeqty3 + wktypeqty4;

        *inlr = *on;
        return;
