/* THIS PROGRAM SENDS A PROGRAM MESSAGE                              */
/*  PARMS:  I  MESSAGE ID                                            */
/*          I  ORDER NUMBER (PASS)                                   */
/*********************************************************************/
PDTRUPR:     PGM        PARM(&MSG1 &ORDERN)
             DCL        VAR(&MSG1)   TYPE(*CHAR) LEN(10)
             DCL        VAR(&ORDERN) TYPE(*DEC)
             DCL        VAR(&ORDER)  TYPE(*DEC)  LEN(7)
             DCL        VAR(&ORDERA) TYPE(*CHAR)  LEN(7)

             CHGVAR     VAR(&ORDER)  VALUE(&ORDERN)
             CHGVAR     VAR(&ORDERA) VALUE(&ORDER)
             SNDUSRMSG  MSGID(&MSG1) MSGF(QUSRMSG) MSGDTA(&ORDERA) +
                          MSGTYPE(*INQ) TOMSGQ(*)
/*           SNDPGMMSG  MSGID(USR2033) MSGF(OMSDEVGEN/QUSRMSG) +    */
/*                     MSGDTA(&ORDERA) TOPGMQ(*EXT) MSGTYPE(*NOTIFY)*/
END:         ENDPGM
