/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING PHYSICAL COUNT SHEETS        */
/*                                                                  */
/*  SYSTEM    - INVENTORY MANAGEMENT                                */
/*  DATE      -  8/18/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  COMPANY                                */
/*         &WHSE          -  WAREHOUSE                              */
/*         &CYCLE         -  CYCLE                                  */
/*         &PRTZR         -  PRINT NON ZERO QTYS                    */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 ICPOCLP:    PGM        PARM(&COMPNYN &WHSE &CYCLE &PRTZR &APPLIC +
                             &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)

        DCL    &COMPNYN     *DEC
        DCL    &WHSE        *CHAR  (03)
        DCL    &CYCLE       *CHAR  (03)
        DCL    &PRTZR       *CHAR  (01) /* NOT USED IN PROGRAM */
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)
        DCL    &COMMAND     *CHAR  (512)
        DCL    &RETURN      *CHAR  (7)

        DCL    &COMP        *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/* CALL PRINT FUNCTION USING CONVERTED VARIABLES                           */
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(ICPOPFR$) +
                          OUTQ(&OUTQ) COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(ICPOPFR) PARM(&RETURN &COMP &WHSE &CYCLE)

 ENDPGM:     ENDPGM
