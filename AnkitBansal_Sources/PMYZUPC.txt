/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  SUBMIT JOB TO CREATE THE LOAD LIST REPORT AS AN     */
/*              EXCEL SPREADSHEET AND EMAIL IT                      */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - DOWNLOAD LOAD LIST REPORT TO EXCEL                  */
/*  DATE      - 09/22/2008                                          */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*                                                                  */
/*  &RETURN     - RETURN CODE                                       */
/*  &LCOMPN     - COMPANY NUMBER (15.5)   - LOAD START COMPANY      */
/*  &OCOMPN     - COMPANY NUMBER (15.5)   - ORDER COMPANY           */
/*  &SSDATEN    - SCHEDULED SHIP DATE (15.5)                        */
/*  &LOADSTS    - LOAD STATUS - SHIPPED/UNSHIPPED/BLANKS            */
/*  &WHSE1      - WAREHOUSE 1    (PASSED)                           */
/*  &WHSE2      - WAREHOUSE 2    (PASSED)                           */
/*  &WHSE3      - WAREHOUSE 3    (PASSED)                           */
/*  &WHSE4      - WAREHOUSE 4    (PASSED)                           */
/*  &WHSE5      - WAREHOUSE 5    (PASSED)                           */
/*  &WHSE6      - WAREHOUSE 6    (PASSED)                           */
/*  &WHSE7      - WAREHOUSE 7    (PASSED)                           */
/*  &WHSE8      - WAREHOUSE 8    (PASSED)                           */
/*  &WHSE9      - WAREHOUSE 9    (PASSED)                           */
/*  &WHSE10     - WAREHOUSE 10   (PASSED)                           */
/*  &LDTYP      - LOAD TYPE      (PASSED)                           */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/* MM/DD/YYYY  XXX  DESCRIPTION                                     */
/*                                                                  */
/********************************************************************/
 PMYZUPC:    PGM        PARM(&RETURN &LCOMPN &OCOMPN &SSDATEN +
                          &LOADSTS &LOADTYP &WHSE1 &WHSE2 &WHSE3 +
                          &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 &WHSE9 +
                          &WHSE10 &EMAILADDR)
/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)
        DCL    &LCOMPN     *DEC
        DCL    &OCOMPN     *DEC
        DCL    &SSDATEN    *DEC
        DCL    &LOADSTS    *CHAR   (1)
        DCL    &LOADTYP    *CHAR   (1)
        DCL    &WHSE1      *CHAR   (3)
        DCL    &WHSE2      *CHAR   (3)
        DCL    &WHSE3      *CHAR   (3)
        DCL    &WHSE4      *CHAR   (3)
        DCL    &WHSE5      *CHAR   (3)
        DCL    &WHSE6      *CHAR   (3)
        DCL    &WHSE7      *CHAR   (3)
        DCL    &WHSE8      *CHAR   (3)
        DCL    &WHSE9      *CHAR   (3)
        DCL    &WHSE10     *CHAR   (3)
        DCL    &EMAILADDR  *CHAR   (50)

/* ERROR MESSAGE FIELDS                                              */
        DCL    &MSG        *CHAR   (512)
        DCL    &MSGID      *CHAR   (7)
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/*   SUBMIT CL PROGRAM TO DOWNLOAD REPORT DATA TO EXCEL             */

             SBMJOB     CMD(CALL PGM(PMY0UPC) PARM(&LCOMPN &OCOMPN +
                          &SSDATEN &LOADSTS &LOADTYP &WHSE1 &WHSE2 +
                          &WHSE3 &WHSE4 &WHSE5 &WHSE6 &WHSE7 &WHSE8 +
                          &WHSE9 &WHSE10 &EMAILADDR)) JOB(LDLSTDWNLD)

             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/
 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
