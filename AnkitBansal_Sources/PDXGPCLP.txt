        DCL    &WHSE3     *CHAR   3               /*  WAREHOUSE      */
        DCL    &WHSE4     *CHAR   3               /*  WAREHOUSE      */
        DCL    &WHSE5     *CHAR   3               /*  WAREHOUSE      */
        DCL    &WHSE6     *CHAR   3               /*  WAREHOUSE      */

        DCL    &WHSE8     *CHAR   3               /*  WAREHOUSE      */
        DCL    &WHSE9     *CHAR   3               /*  WAREHOUSE      */
        DCL    &WHSE10    *CHAR   3               /*  WAREHOUSE      */
        DCL    &SOFLAG    *CHAR   1               /*  SELECT/OMIT    */
        DCL    &COMMAND    *CHAR   (512)

/* SEND PROGRAM MESSAGE PARAMETERS  */

        DCL     &D        *CHAR   25              /* QRY REASON CODE */

        DCL     &F        *CHAR   25              /* QRY SALES ROUTE */

        DCL     &L        *CHAR   25
        DCL     &M        *CHAR   25
        DCL     &N        *CHAR   25
        DCL     &O        *CHAR   25
        DCL     &P        *CHAR   25

             CHGVAR    VAR(&COMP) VALUE(&COMPN) /* Change to length +

             CHGVAR    VAR(&FDATEA) VALUE(&FDATE) /* Change to alpha +

             CHGVAR     VAR(&TDATE) VALUE(&TDATEN) /* Change to +

             CHGVAR    VAR(&TDATEA) VALUE(&TDATE) /* Change to alpha +

             CHGVAR    VAR(&CUST#) VALUE(&CUSTN) /* Change to length +
                          for print program */
             CHGVAR    VAR(&CUSTA) VALUE(&CUST#) /* Change to alpha +
                          for query data */
             CHGVAR    VAR(&ITEM) VALUE(&ITEMN) /* Change to length +
                          for print program */
             CHGVAR    VAR(&ITEMA) VALUE(&ITEM) /* Change to alpha +
                          for query data */
             CHGVAR    VAR(&COPIES) VALUE(&COPIESL) /* change to +

             IF        COND(&SELRPT *EQ '5') THEN(DO)

             ADDLIBLE  LIB(QTEMP)
             MONMSG    MSGID(CPF0000)

             CLRPFM    FILE(PDMICPP)
             CRTDUPOBJ OBJ(PDMICPP) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJ OBJ(PDMICPL0) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJ OBJ(PDMICPL1) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             CRTDUPOBJ OBJ(PDMICPL2) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP)
             ENDDO
/********************************************************************/
/* LOAD PGM NAME INTO THE CALLED VARIABLE                           */
/********************************************************************/
             IF COND(&SELRPT *EQ '1') THEN(DO)
                CHGVAR VAR(&PGM)  VALUE('PDXGPFR')
             ENDDO
             IF COND(&SELRPT *EQ '2') THEN(DO)
                CHGVAR VAR(&PGM)  VALUE('PDXJPFR')
             ENDDO
             IF COND(&SELRPT *EQ '3') THEN(DO)
                CHGVAR VAR(&PGM)  VALUE('PDXKPFR')
             ENDDO
             IF COND(&SELRPT *EQ '4') THEN(DO)
                CHGVAR VAR(&PGM)  VALUE('PDXLPFR')

             IF COND(&SELRPT *EQ '5') THEN(DO)
                CHGVAR VAR(&PGM)  VALUE('PDLLPFR')

/* F4AWDT=INVOICE DATE, F4GNDT=ACTUAL SHIPPED DATE                   */
             IF         COND(&SELRPT *NE '5') THEN(DO)
  /*PIO      CHGVAR     VAR(&A) VALUE('F4AWDT *GE ' *CAT +  +

  /*PIO*/    CHGVAR     VAR(&A) VALUE('F4AWDT >=  ' *CAT +

  /*PIO      CHGVAR     VAR(&B) VALUE('*AND F4AWDT *LE ' *CAT + +

  /*PIO*/    CHGVAR     VAR(&B) VALUE(' AND F4AWDT <=  ' *CAT +
                         &TDATEA)
             ENDDO
             IF         COND(&SELRPT *EQ '5') THEN(DO)
  /*PIO      CHGVAR     VAR(&A) VALUE('F4GNDT *GE ' *CAT + +

  /*PIO*/    CHGVAR     VAR(&A) VALUE('F4GNDT  >= ' *CAT +
                         &FDATEA)
  /*PIO      CHGVAR     VAR(&B) VALUE('*AND F4GNDT *LE ' *CAT + +
                         &TDATEA)                             */
  /*PIO*/    CHGVAR     VAR(&B) VALUE(' AND F4GNDT  <= ' *CAT +
                         &TDATEA)
             ENDDO

  /*PIO      IF         COND(&COMPA *NE '000') THEN(CHGVAR VAR(&C) + +
                          VALUE('*AND F4AIC3 *EQ ' *CAT &COMPA))   */

                          VALUE(' AND F4AIC3  =  ' *CAT &COMPA))

             IF         COND(&REASCD *NE '   ' *AND &SELRPT *NE '5') +
                          THEN(DO)
  /*PIO      CHGVAR     VAR(&D) VALUE('*AND F4RMCD *EQ "' *CAT + +
                         &REASCD *CAT '"')                      */
  /*PIO*/    CHGVAR     VAR(&D) VALUE(' AND F4RMCD =   "' *CAT +
                         &REASCD *CAT '"')
             ENDDO

             IF         COND(&CUSTA *NE '0000000') THEN(DO)
  /*PIO      CHGVAR     VAR(&E) VALUE('*AND F4ANC7 *EQ ' *CAT + +

             IF         COND(&ITEMA *NE '0000000') THEN(DO)

  /*PIO*/    CHGVAR     VAR(&H) VALUE(' AND F4S5NB  =  ' *CAT +

             ENDDO

             IF         COND(&SLSRTE *NE '   ') THEN(DO)

  /*PIO*/    CHGVAR     VAR(&F) VALUE(' AND F4R4CD  =  "' *CAT +
                          &SLSRTE *CAT '"')
             ENDDO

             CHGVAR     VAR(&OPEN) VALUE('      ')
             CHGVAR     VAR(&CLOSED) VALUE(' ')

/* SORT REPORT 1 BY: SHIPPED ITEM CODE, REASON CODE, COMPANY,       */
/*                   ATTACH TO ORDER NUMBER                         */
             IF         COND(&SELRPT *EQ '1') THEN(DO)
   /*PIO     OVRDBF     FILE(OMHSTPL4) SHARE(*YES)                     +
             OPNQRYF    FILE((OMHSTPL4)) QRYSLT(&G) KEYFLD((F4S5NB) +  +
                          (F4RMCD) (F4AIC3) (F4T3NB))                  */

  /*PIO*/    OVRDBF     FILE(OMHSTPP) TOFILE(OMHSTPP) LVLCHK(*NO)
             OPNSQLF    SQL('Select * from OMHSTPP Where  ' *BCAT +
                          &G *BCAT ' ORDER BY F4S5NB ASC, F4RMCD +
                          ASC, F4AIC3 ASC, F4T3NB ASC') +
                          OPNID(OMHSTPL4) OPTION(*ALL) SEQONLY(*NO) +

  /*PIO*/    OVRDBF     FILE(OMHSTPL4) TOFILE(OMHSTPP) LVLCHK(*NO) +

             IF COND(&SELRPT *EQ '2') THEN(DO)

  /*PIO*/       OVRDBF     FILE(OMHSTPP) TOFILE(OMHSTPP) LVLCHK(*NO)
             OPNSQLF    SQL('Select * from OMHSTPP Where  ' *BCAT &G +

  /*PIO*/       OVRDBF     FILE(OMHSTPL6) TOFILE(OMHSTPP) LVLCHK(*NO) +

     /*PIO      DLTOVR FILE(OMHSTPM4)   */
