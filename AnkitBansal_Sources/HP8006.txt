      * Function:  This function receives an accounting period and returns:
      *                1) the quarter associated with this period
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      *
      /eject
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *
      ****************************************************************
      * STANDALONE FIELDS
      ****************************************************************
      *
      * Parm fields
      *
     D xxperiod        s              2  0
     D xxquarter       s              1  0
      *
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************
      *
      * Hardcode the quarter associated with the period
      *
     C                   select
     C                   when      xxperiod <= 3
     C                   z-add     1             xxquarter
      *
     C                   when      xxperiod <= 6
     C                   z-add     2             xxquarter
      *
     C                   when      xxperiod <= 9
     C                   z-add     3             xxquarter
     C                   other
      *
     C                   z-add     4             xxquarter
     C                   endsl
      *
     C                   seton                                        lr
      /eject
      *---------------------------------------------------------------
      * Initialization subroutine
      *---------------------------------------------------------------

     C     *inzsr        begsr
      *
     C     *entry        plist
     C                   parm                    xxperiod
     C                   parm                    xxquarter
      *
     C                   endsr
      /eject
