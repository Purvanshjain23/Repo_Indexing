/********************************************************************/
/*                                                                  */
/*  PBFUUPC -   CALL PRINT TPM LOCATION MASTER REPORT               */
/*  SYNOPSIS -  PASSES THE TPM FACILITY TYPE                        */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  PROGRAM   - PBFUUPC                                             */
/*  DATE      - 10/05/2012 L. BUSER                                 */
/*                                                                  */
/*  PARAMETERS:                                                     */
/*  O    &RETURN     - RETURN CODE                                  */
/*  I    &PGM        - PROGRAM TO CALL                              */
/*  I    &DESC       - PRINT PROGRAM DESCRIPTION                    */
/*  B    &OUTQ       -  OUTPUT QUEUE                                */
/*  B    &HOLD       -  HOLD STATUS                                 */
/*  B    &SAVE       -  SAVE STATUS                                 */
/*  B    &NQSTAT     -  INDICATOR WHETHER MUST RUN IN NIGHT Q       */
/*  B    &NIGHTQ     -  NIGHT JOB QUEUE                             */
/*  B    &COPYN      - NO. OF COPIES  (PASSED)                      */
/********************************************************************/
/* MAINTENANCE                                                      */
/* 99/99/99 XXX X999999                                             */
/*   DESCRIPTION OF CHANGES                                         */
/********************************************************************/
 PBFUUPC:    PGM        PARM(&DESC &OUTQ &HOLD &SAVE &NQSTAT &NIGHTQ +
                          &PGM &COPYN &FACILITY)

        DCL    &PGM        *CHAR   (10)
        DCL    &PRTF       *CHAR   (10)
        DCL    &DESC       *CHAR   (25)
        DCL    &FACILITY   *CHAR   (24)
        DCL    &OUTQ       *CHAR   (10)
        DCL    &HOLD       *CHAR   (4)
        DCL    &SAVE       *CHAR   (4)
        DCL    &NQSTAT     *CHAR   (1)
        DCL    &NIGHTQ     *CHAR   (10)
        DCL    &COPY       *DEC    (2 0)
        DCL    &COPYN      *DEC

             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
             CHGVAR     VAR(&PRTF) VALUE(&PGM *TCAT '$')
             OVRPRTF    FILE(&PRTF) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(&PGM) PARM(' ' &FACILITY)

 END:        ENDPGM
