SELECT
  ALL       COID, MCLASS, MAPLCN, MNAME, MOPTN, MCL
  FROM      WM400/WMCL T01
  WHERE     MCL LIKE '%RMTOPT%'
    AND     COID <> 'COR'
    AND     COID <> 'CHF'
    AND     COID <> 'IS'
    AND     COID <> 'SML'
    AND     COID <> 'SSM'
  ORDER BY  COID ASC, MCLASS ASC, MAPLCN ASC, MNAME ASC, MOPTN ASC
