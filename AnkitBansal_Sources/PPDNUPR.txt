/*              CHANGE THIS PROCESS TO PASS THIS INFORMATON         */
/*              FROM THE PARM LINE TO THE PPDLPFR                   */
/*                                                                  */
/* 10/17/19 JBB E15650 Multi-Stop Seaboard Warehouse Report         */
/*   WI359      Added parameters for warehouses 1 thru 6.  These    */
/*              values are passed in from screen file program       */
/*              PUOMDFR.  The warehouse values will be passed to    */
/*              the report program PPDLPFR.                         */
/*                                                                  */
/********************************************************************/

 PPDNUPR:    PGM        PARM(&OUTQ &HOLD &SAVE &COPIES &COMPN &FDTE +
                          &TDTE &WHSE1 &WHSE2 &WHSE3 &WHSE4 &WHSE5 +
                          &WHSE6)

/* INPUT PARMS */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COPIES     *DEC  (2 0)             /*  # OF COPIES   */
        DCL    &COMPN      *DEC   (3 0)          /* Company Number (PASSED)   */
        DCL    &FDTE       *DEC   (7 0)          /* PROD DATE    */
        DCL    &TDTE       *DEC   (7 0)          /* PROD DATE    */

        DCL    &WHSE1      *CHAR  (3)            /* Stop 1 Warehouse */
        DCL    &WHSE2      *CHAR  (3)            /* Stop 2 Warehouse */
        DCL    &WHSE3      *CHAR  (3)            /* Stop 3 Warehouse */
        DCL    &WHSE4      *CHAR  (3)            /* Stop 4 Warehouse */
        DCL    &WHSE5      *CHAR  (3)            /* Stop 5 Warehouse */
        DCL    &WHSE6      *CHAR  (3)            /* Stop 6 Warehouse */

/* OVERRIDE PRINT FILE */
        DCL    &COMMAND    *CHAR   (512)

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/********************************************************************/
/*   INITIALIZE VARIABLES                                           */
/********************************************************************/

/********************************************************************/
/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
/********************************************************************/

             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL PRINT CL PROGRAM                                        */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PPDLPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/* 10/17/19 JBB WI359 Add 6 Warehouses                              */
             CALL       PGM(PPDLPFR) PARM(' ' &COMPN &FDTE &TDTE +
                          &WHSE1 &WHSE2 &WHSE3 &WHSE4 &WHSE5 &WHSE6)

             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/

 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
