/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR COMPANY, WAREHOUSE, ITEM     */
/*              FROM DATE, TO DATE, TRANSACTION TYPES, REASON CODES */
/*                                                                  */
/*                                                                  */
/*  SYSTEM    - INVENTORY CONTROL                                   */
/*  PROGRAM   - INVENTORY TRANSACTION JOURNAL DOWNLOAD              */
/*  DATE      - 06/02/2008                                          */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/* NN/NN/NN XXX XNNNNN                                              */
/********************************************************************/
/* PARAMETERS:                                                      */
/*                                                                  */
/*     I    &APPLIC     - APPLICATION CODE                          */
/*     I    &CLP        - CL PROGRAM TO SUBMIT                      */
/*     I    &PGM        - RPG PROGRAM TO SUBMIT                     */
/*     I    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     I    &FRMDATN    - FROM DATE      (PASSED)                   */
/*     I    &THRDATN    - THRU DATE      (PASSED)                   */
/*     I    &FRMTIMN    - FROM TIME      (PASSED)                   */
/*     I    &THRTIMN    - THRU TIME      (PASSED)                   */
/*     I    &WHSE       - WAREHOUSE CODE (PASSED)                   */
/*     I    &ITEM       - ITEM CODE      (PASSED)                   */
/*     I    &TTYPE1     - TRANSACTION TYPE1 (PASSED)                */
/*     I    &TTYPE2     - TRANSACTION TYPE2 (PASSED)                */
/*     I    &TTYPE3     - TRANSACTION TYPE3 (PASSED)                */
/*     I    &TTYPE4     - TRANSACTION TYPE4 (PASSED)                */
/*     I    &RSN1       - REASON CODE 1   (PASSED)                  */
/*     I    &RSN2       - REASON CODE 2   (PASSED)                  */
/*     I    &RSN3       - REASON CODE 3   (PASSED)                  */
/*     I    &RSN4       - REASON CODE 4   (PASSED)                  */
/*     I    &SHFT       - SHIFT NUMBER    (PASSED)                  */
/*     I    &AVLGRP     - PA GROUP        (PASSED)                  */
/*     I    &TRNID      - INVENTORY TRANS ID (PASSED)               */
/*     I    &EMAIL      - EMAIL ADDRESS                             */
/*     I    &FORMAT     - EMAIL FORMAT                              */
/*                                                                  */
/********************************************************************/
 PMU8UPC:    PGM        PARM(&APPLIC &CLP &PGM &COMPNYN &FRMDATN +
                          &THRDATN &WHSE &ITEM &TTYPE1 &TTYPE2 +
                          &TTYPE3 &TTYPE4 &RSN1 &RSN2 &RSN3 &RSN4 +
                          &SHFT &AVLGRP &FRMTIMN &THRTIMN &TRNID +
                          &EMAIL &FORMAT)

/* PARMS                                                                      */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &CLP        *CHAR   (10)          /* CL to Submit              */
        DCL    &PGM        *CHAR   (10)          /* RPG TO SUBMIT             */
        DCL    &DESC       *CHAR   (25)          /* REPORT DESC               */
        DCL    &COMPNYN    *DEC                  /* Company Number (Passed)   */
        DCL    &FRMDATN    *DEC                  /* FROM DATE      (Passed)   */
        DCL    &THRDATN    *DEC                  /* THROUGH DATE   (Passed)   */
        DCL    &FRMTIMN    *DEC                  /* FROM TIME      (Passed)   */
        DCL    &THRTIMN    *DEC                  /* THROUGH TIME   (Passed)   */
        DCL    &WHSE       *CHAR   (3)           /* WARHOUSE       (Passed)   */
        DCL    &ITEM        *DEC                 /* ITEM           (Passed)   */
        DCL    &TTYPE1      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &TTYPE2      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &TTYPE3      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &TTYPE4      *CHAR   2           /*  TRANS. TYPE     */
        DCL    &RSN1        *CHAR   3           /*  REASON CODE     */
        DCL    &RSN2        *CHAR   3           /*  REASON CODE     */
        DCL    &RSN3        *CHAR   3           /*  REASON CODE     */
        DCL    &RSN4        *CHAR   3           /*  REASON CODE     */
        DCL    &SHFT        *CHAR   1           /*  SHIFT NUMBER    */
        DCL    &AVLGRP      *CHAR   3           /*  AVAIL GROUP     */
        DCL    &TRNID       *CHAR   1           /*  TRANS ID        */
        DCL    &EMAIL       *CHAR   50          /*  EMAIL ADDRESS   */
        DCL    &FORMAT      *CHAR   5           /*  EMAIL FORMAT    */

/* INTERNAL FIELDS                                                            */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL THE PRINT PROGRAM                                         */
/********************************************************************/
 CALL:       CALL       PGM(&CLP) PARM(&PGM &COMPNYN &FRMDATN +
                          &THRDATN &WHSE &ITEM &TTYPE1 &TTYPE2 +
                          &TTYPE3 &TTYPE4 &RSN1 &RSN2 &RSN3 &RSN4 +
                          &SHFT &AVLGRP &FRMTIMN &THRTIMN &TRNID +
                          &EMAIL &FORMAT)

             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/
 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
