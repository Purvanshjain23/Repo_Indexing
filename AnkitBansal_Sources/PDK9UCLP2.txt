 PDK9UCLP2:  PGM

             DCL        VAR(&DATA) TYPE(*CHAR) LEN(10)
             DCL        VAR(&DATAH) TYPE(*CHAR) LEN(10)
             DCL        VAR(&DATAS) TYPE(*CHAR) LEN(10)
             DCL        VAR(&REPROCS) TYPE(*CHAR) LEN(1) VALUE(' ')
             DCL        VAR(&USER) TYPE(*CHAR) LEN(10)

/*  GET THE DATA LIBRARY (DTALIB = PRKFLIB, HSTLIB = PRKFHST)      */
             CALL       PGM(CAQXXFR) PARM(' ' &DATA &DATAH)

             RTVJOBA    USER(&USER)

             DLTF       FILE(QTEMP/PDMHCP*)
             MONMSG     MSGID(CPF0000)

START:
             CHGVAR     VAR(&REPROCS) VALUE(' ')

/* Create Duplicate verion of file to QTEMP.                       */

             CRTDUPOBJ OBJ(PDMHCPP) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*YES)

             CRTDUPOBJ OBJ(PDMHCPL0) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL1) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL3) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL4) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL5) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL6) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL7) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL8) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPL9) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPLA) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPLB) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPLC) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)
             CRTDUPOBJ  OBJ(PDMHCPLD) FROMLIB(&DATA) OBJTYPE(*FILE) +
                          TOLIB(QTEMP) DATA(*NO)

             CALL       PGM(PDLCEFR) PARM(' ' &REPROCS)

DLQTEMP:
             DLTF       FILE(QTEMP/PDMHCP*)

             IF         COND(&REPROCS *EQ 'Y') THEN(GOTO START)

 ENDJOB:     ENDPGM
