             PGM        PARM(&RETURN &SRCCDE &COMP &KILLDATE &LIBRARY)

             DCL        VAR(&SRCCDE)   TYPE(*CHAR) LEN(3)
             DCL        VAR(&COMP)     TYPE(*DEC) LEN(3 0)
             DCL        VAR(&KILLDATE) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&LIBRARY)  TYPE(*CHAR) LEN(10)
             DCL        VAR(&RTVLIB)   TYPE(*CHAR) LEN(10)
             DCL        VAR(&RETURN)   TYPE(*CHAR) LEN(7)

             /* Validate library requested is first in library list */
             RTVOBJD    OBJ(PKBLCPP) OBJTYPE(*FILE) RTNLIB(&RTVLIB)
             IF         COND(&RTVLIB *NE &LIBRARY) THEN(DO)
             CHGVAR     VAR(&RETURN) VALUE('ERROR ')
             GOTO       CMDLBL(ENDIT)
             ENDDO

             /* Call program to print and delete pushed records */
             CALL       PGM(PWE2PFR) PARM(&RETURN &SRCCDE &COMP &KILLDATE)

 ENDIT:      ENDPGM
