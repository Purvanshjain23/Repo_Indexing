      * Function:    This function deletes the Header and Detail records for a
      *              specified Farm Budget.
      *
      ****************************************************************
      * MODIFICATIONS:
      ****************************************************************
      * DATE      PROGRAMMER
      *
      * 10/13/04  LeAnne Fedor
      *           Recompile only. Cost Per Unit Amount was changed from
      *           9,2 to 9,4.
      /eject
      ****************************************************************
      * FILE SPECIFICATIONS
      ****************************************************************
      *
     Fhsp188    uf   e           k disk
      *    Farm budget header
      *
      *
     Fhsp189    uf   e           k disk
      *    Farm budget detail
      *
      /eject
      ****************************************************************
      * DEFINITION SPECIFICATIONS
      ****************************************************************
      *
      *----------------------------------------------------------------
      * CONSTANTS
      *----------------------------------------------------------------
      *
     D yes             c                   'Y'
     D no              c                   'N'
      *
      *----------------------------------------------------------------
      * STANDALONE FIELDS
      *----------------------------------------------------------------
      *
      *
      * Parms
      *
     D xxfbsn          s                   like(jhfbsn)
      *
      *
      ****************************************************************
      * DATA STRUCTURES
      ****************************************************************
      *
      *---------------------------------------------------------------
      * Standard program status data structure
      *---------------------------------------------------------------
      *    externally defined as UTPGFR (record format: PGMDSR)
     D pgmds         esds                  extname(utpgfr)
      *
      *
      * The following is a dummy C-spec to keep the D-specs from
      * running over into the C-specs on the compile listing. It
      * serves no other purpose and can be deleted at any time.
      *
     C                   if        sdpgm = *blank
     C                   endif
      /EJECT
      ****************************************************************
      * Mainline
      ****************************************************************
      *
      * Delete each Farm Budget detail record for the incoming Farm Budget
      * number.
      *
     C     xxfbsn        setll     hsp189
     C                   dou       *in91 = *on                                  Do detail
     C     xxfbsn        reade     hsp189                                 91
     C                   if        *in91 = *off                                 If not EOF
     C                   delete    jdrec
     C                   endif                                                  If not EOF
     C                   enddo                                                  Do Detail
      *
      * Delete the Farm Budget Header record for the incoming Farm Budget
      * number.
      *
     C     xxfbsn        chain     hsp188                             92
     C                   if        *in92 = *off                                 If got it
     C                   delete    jhrec
     C                   endif                                                  If got it
      *
     C                   seton                                        lr
      /eject
      *---------------------------------------------------------------
      * Initialization subroutine
      *---------------------------------------------------------------
      *
     C     *inzsr        begsr
      *
     C     *entry        plist
     C                   parm                    xxfbsn
      *
     C                   endsr
      /eject
