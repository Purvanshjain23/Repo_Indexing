/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING CARRIER-DESTINATION RATES    */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  2/06/97                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &CARRIER       -  CARRIER                                */
/*         &CITY          -  CITY                                   */
/*         &STATE         -  STATE                                  */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/********************************************************************/
/*  RMC 10/25/06 ADDED WHSE PARM                                    */
/********************************************************************/
 PDWTPCLP:   PGM        PARM(&APPLIC &CARRIER &CITY &STATE +
                          &OUTQ &HOLD &SAVE &COPYN &NIGHTQ &WHSE)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &CARRIER     *CHAR  (03)
        DCL    &WHSE        *CHAR  (03)
        DCL    &CITY        *CHAR  (20)
        DCL    &STATE       *CHAR  (02)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDWTPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CALL       PGM(PDWTPFR) PARM(' ' &CARRIER &WHSE &CITY &STATE)

 ENDPGM:     ENDPGM
