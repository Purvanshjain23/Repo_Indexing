/*       Title: Execute the command OVRDBF                           */
/*        Date: 04/20/2012                                           */
/* Description: - Construct the command string.                      */
/*              - Execute the command.                               */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &FROMFILE   FROM FILE NAME                               */
/*    I     &FROMLIB    FROM LIBRARY                                 */
/*    I     &TOFILE     TO FILE NAME                                 */
/*    I     &TOLIB      TO LIBRARY                                   */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/* -------- ---------- --------------------------------------------  */
/*                                                                   */
/*********************************************************************/
             PGM  PARM(&FROMFILE &FROMLIB &TOFILE &TOLIB)

/* VARIABLES                                                         */
             DCL  VAR(&FROMFILE) TYPE(*CHAR) LEN(10)
             DCL  VAR(&FROMLIB)  TYPE(*CHAR) LEN(10)
             DCL  VAR(&TOFILE)   TYPE(*CHAR) LEN(10)
             DCL  VAR(&TOLIB)    TYPE(*CHAR) LEN(10)

             MONMSG     MSGID(CPF0000)

/* Override DATABASE FILE                                            */
             OVRDBF     FILE(&FROMFILE) TOFILE(&TOLIB/&TOFILE)

/* End of Program                                                    */
 ENDPGM:     ENDPGM
