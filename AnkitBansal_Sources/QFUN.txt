SELECT
  ALL       T01.REQSYN, T01.DSPSEQ, T04.APPCDE, T02.FIL, T01.MSG
  FROM      HPEMDL/YMSGDTARFP T01 LEFT OUTER JOIN
            HPEMDL/YFILDTARFP T02
  ON        T01.@@FIL = T02.@@FIL LEFT OUTER JOIN
            HPEMDL/YAPPFILRFP T03
  ON        T02.@@FIL = T03.@@FIL LEFT OUTER JOIN
            HPEMDL/YAPPDTARFP T04
  ON        T03.@@APP = T04.@@APP
  WHERE     T01.OBJATR = 'RPG'
    AND     T02.FIL NOT LIKE '*%'
