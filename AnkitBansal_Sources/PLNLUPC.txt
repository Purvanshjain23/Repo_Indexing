/***************************************************************** */
/*                                                                 */
/*  PROGRAM NUMBER:  PLNLUPC                                       */
/*  PROGRAM NAME:    Print BUY ORDER LOAD HELD OVERS               */
/*  PROGRAMMER:      SUSAN MASON                                   */
/*  CREATE DATE:     11/10/08                                      */
/*                                                                 */
/*  FUNCTION:        This CLP is submitted with an Execute Message */
/*                   from YGO PRKRPT                               */
/*                                                                 */
/*                   USER OPTIONS ARE PASSED THROUGH THE PARMS.    */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*                                                                 */
/*******************************************************************/
             PGM        PARM(&RTN &PDEV &HOLD &SAVE &XCOPIES +
                           &XCOMP &XBO &XBL &XTO)

/*-----------------------------------------------------------------*/
/*                      PARAMETER                                  */
/*-----------------------------------------------------------------*/
/* Print Options                                                   */
             DCL        VAR(&RTN) TYPE(*CHAR) LEN(7)
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*CHAR) LEN(02)
/* Function Specific                                               */
             DCL        VAR(&XCOMP) TYPE(*CHAR) LEN(03)
             DCL        VAR(&XBO) TYPE(*CHAR) LEN(05)
             DCL        VAR(&XBL) TYPE(*CHAR) LEN(05)
             DCL        VAR(&XTO) TYPE(*CHAR) LEN(05)

/*-----------------------------------------------------------------*/
/*                    PROGRAM VARIABLES                            */
/*-----------------------------------------------------------------*/
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&BO) TYPE(*DEC) LEN(5 0)
             DCL        VAR(&BL) TYPE(*DEC) LEN(5 0)
             DCL        VAR(&TO) TYPE(*DEC) LEN(5 0)
/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             IF COND(&XCOMP *EQ ' ') +
              THEN(DO)
                 CHGVAR     VAR(&COMP) VALUE('0')
              ENDDO
             ELSE +
              CMD(DO)
                 CHGVAR     VAR(&COMP) VALUE(&XCOMP)
              ENDDO
             CHGVAR     VAR(&BO) VALUE(&XBO)
             CHGVAR     VAR(&BL) VALUE(&XBL)
             CHGVAR     VAR(&TO) VALUE(&XTO)

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/

             OVRPRTF    FILE(PLNKPFR$) TOFILE(PLNKPFR$) DEV(&PDEV) +
                          OUTQ(&PDEV) COPIES(&COPIES) HOLD(&HOLD) +
                          SAVE(&SAVE)

             CALL       PGM(PLNKPFR) PARM(' ' &COMP &BO &BL &TO)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
