       //----------------------------------------------------------------------
       // Program To Get the julian date from a ccyymmdd date passed in to it
       //   Add 10000 to the result for the century digit
       //
       // Create using: CRTRPGMOD MODULE(e1idevgen/e10028a)
       //                         SRCFILE(e1idevgen/QRPGLESRC)
       //                         SRCMBR(e10028a)
       //                         REPLACE(*NO)
       //
       //   R Centonze    5/21/2020
       //----------------------------------------------------------------------
      * Parms
     d ccyymmdd        s              8  0
     d datejul         s              6  0
      *
     d*
      *****************************************************************
      * Entry parameters
     C     *ENTRY        PLIST
     C                   PARM                    ccyymmdd
     C                   PARM                    datejul
      *****************************************************************

      /free
           datejul = %dec(%char(%date(ccyymmdd:*iso):*jul0):6:0);
           datejul = datejul + 100000;
       return;
      /end-free
      *
