/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - INITIAL PROGRAM FOR YIELD REPORTING SYSTEM           */
/*                                                                  */
/*  SYSTEM   - YIELD REPORTING                                      */
/*  DATE     - 07/17/03                                             */
/*                                                                  */
/********************************************************************/
             PGM

             DCL        VAR(&USER) TYPE(*CHAR) LEN(10)
             DCL        VAR(&MENU) TYPE(*CHAR) LEN(10)
             DCL        VAR(&MENULIB) TYPE(*CHAR) LEN(10)
             DCL        VAR(&MSG) TYPE(*CHAR) LEN(132)

/*-------------------------------------------------------------------*/
/* RETRIEVE USER MENU                                                */
/*-------------------------------------------------------------------*/

             RTVJOBA    USER(&USER)
/*           RTVUSRPRF  INLMNU(&MENU) INLMNULIB(&MENULIB) */

             CHGVAR     VAR(&MENULIB) VALUE(*LIBL)
             CALL       PGM(POXNXFR) PARM(' ' &USER &MENU)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(ERROR))

/*-------------------------------------------------------------------*/
/* EXECUTE MENU                                                      */
/*-------------------------------------------------------------------*/

             GO         MENU(&MENULIB/&MENU)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(ERROR))
             GOTO ENDPGM

/*-------------------------------------------------------------------*/
/* ERROR PROCESSING                                                  */
/*-------------------------------------------------------------------*/
ERROR:
             CHGVAR     VAR(&MSG) VALUE('YIELD REPORTING SYSTEM: +
                          Error occured while calling menu' *bcat +
                          &menu *bcat 'for user' *bcat &user)
             SNDPGRMSG  TOPGR(PRKDEV) MSG(&MSG)
/*-------------------------------------------------------------------*/
/* ERROR PROCESSING                                                  */
/*-------------------------------------------------------------------*/
ENDPGM:
             SIGNOFF
