/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - OVRDBF TO FILE LIBRARY FOR THE COMPANY BEFORE        */
/*             CALLING THE HPE PROGRAM                              */
/*                                                                  */
/*                                                                  */
/*  COMPANY   - SEABOARD COPRORATION                                */
/*  SYSTEM    - ORDER MANAGEMENT                                    */
/*  USER NAME - ROSE CENTONZE                                      */
/*  DATE      - 10/13/05                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &LIB        - OUTPUT QUEUE                             */
/*          &COMPL      - COMPANY NUMBER                            */
/*          &PDATEL     - PRODUCTION DATE                           */
/*          &CUTEST     - CUT HEAD ESTIMATE                        */
/********************************************************************/
/*                                                                  */
/********************************************************************/
 PPT7UPC:    PGM        PARM(&RTN &LIB &COMPL &PDATEL &CUTESTL)

        DCL    &RTN       *CHAR   (7)            /*  RETURN CODE   */
        DCL    &LIB       *CHAR   (10)            /*  FILE LIB      */
        DCL    &COMP      *DEC    (3 0)           /*  COMPANY #     */
        DCL    &COMPL     *DEC                    /*  FOR PASSING   */
        DCL    &PDATE     *DEC    (7 0)           /*  FROM DATE     */
        DCL    &PDATEL    *DEC                    /*  FOR PASSING   */
        DCL    &CUTEST    *DEC    (5 0)           /*  CUT HEAD EST  */
        DCL    &CUTESTL   *DEC                    /*  CUT HEAD EST */

/* CHANGE VAR TO INTERNAL LENGTH    */

             CHGVAR    VAR(&COMP) VALUE(&COMPL) /* Change to length */
             CHGVAR    VAR(&PDATE) VALUE(&PDATEL) /* Change to length */
             CHGVAR    VAR(&CUTEST) VALUE(&CUTESTL) /* Change to length */

/* OVERRIDE DB FILE TO CORRECT LIBRARY                               */

             OVRDBF     FILE(PKAOCPP)  TOFILE(&LIB/PKAOCPP)
             OVRDBF     FILE(PKAOCPL0) TOFILE(&LIB/PKAOCPL0)
             OVRDBF     FILE(PKAOCPL1) TOFILE(&LIB/PKAOCPL1)

             CALL       PGM(PKY6XFR) PARM('' &COMP &PDATE &CUTEST)
             CHGVAR    VAR(&CUTESTL) VALUE(&CUTEST) /* Change to length */

             DLTOVR     FILE(PKAOCPP)
             DLTOVR     FILE(PKAOCPL0)
             DLTOVR     FILE(PKAOCPL1)

 END:        ENDPGM
