/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - PRINT PROGRAM FOR A/R SUMMARY INQUIRY INFORMATION    */
/*                                                                  */
/*                                                                  */
/*                                                                  */
/*  COMPANY   - DATAWRIGHT INC.                                     */
/*  SYSTEM    - IDeAS/400 ACCOUNTS RECEIVABLE                       */
/*  USER NAME - DEBBIE HAMNING                                      */
/*  DATE      - 03/20/90                                            */
/*  COPYRIGHT 1990 DATAWRIGHT INC.                                  */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*                                                                  */
/*          &COMPANYN   - PASSED COMPANY NUMBER                     */
/*          &CUSTOMRN   - PASSED CUSTOMER NUMBER                    */
/*          &APPLIC     - APPLICATION CODE                          */
/*          &OUTQ       - OUTPUT QUEUE                              */
/*          &HOLD       - HOLD FLAG                                 */
/*          &SAVE       - SAVE FLAG                                 */
/*          &COPIESN    - PASSED NUMBER OF COPIES                   */
/*          &NIGHTQ     - NIGHT JOB QUEUE                           */
/*          &ACCR       - ACCRUAL CODE -- NOT USED IN THIS CL!      */
/********************************************************************/

 ARJVPCLP:   PGM        PARM(&COMPANYN &CUSTOMRN &APPLIC &OUTQ +
                             &HOLD &SAVE &COPIESN &NIGHTQ &ACCR)

        DCL    &COMPANYN   *DEC                  /* Passed Company Number    */
        DCL    &CUSTOMRN   *DEC                  /* Passed Customer Number   */
        DCL    &APPLIC     *CHAR   (10)          /* Application Code         */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue             */
        DCL    &COPIESN    *DEC                  /* Passed Number of Copies  */
        DCL    &HOLD       *CHAR   (04)          /* Hold Flag                */
        DCL    &SAVE       *CHAR   (04)          /* Save Flag                */
        DCL    &NIGHTQ     *CHAR   (10)          /* Night Job Queue          */
        DCL    &ACCR       *CHAR   (03)          /* ACCRUAL -- NOT USED HERE */

        DCL    &COPIES     *DEC    (3 0)         /* no. of copies            */
        DCL    &CUST       *DEC    (7 0)         /* Customer number          */
        DCL    &COMP       *DEC    (3 0)         /* company number           */
        DCL    &COMMAND    *CHAR   (512)

/********************************************************************/
/* APPEND $ TO THE PROGRAM NAME                                     */
/********************************************************************/

             CHGVAR     VAR(&COMP) VALUE(&COMPANYN) /* Change to length +
                          for print program      */
             CHGVAR     VAR(&CUST) VALUE(&CUSTOMRN) /* Change to length +
                          for print program */
             CHGVAR     VAR(&COPIES) VALUE(&COPIESN)

/********************************************************************/
/* Override printer file with user specified values                 */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(ARJVPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

/*********************************************************************/
/*   Call Listing Program                                            */
/*********************************************************************/

             CALL       PGM(ARJVPFR) PARM('' &COMP &CUST)

 END:        ENDPGM
