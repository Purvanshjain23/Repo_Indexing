/*  PROGRAM:         PWJXUPC - Refresh GUP Delv Files CL           */
/*  PROGRAMMER:      Brad Baden                                    */
/*  CREATE DATE:      9/10/2019                                    */
/*                                                                 */
/*  FUNCTION:        This CL is submitted from the Job Scheduler   */
/*    E15452         as REFDELVGUP.  It will copy records from the */
/*                   Deliver Sched Head (PLADCPP), Delivery Sched  */
/*                   Detail (PLAECPP), and Delivery Tracking       */
/*                   (PLAFCPP) files                               */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/* 10/07/2019 JBB [SR] Dedicated lane request - Refresh test tables*/
/*    E15586      Add file PLAGCPP to this program.  Also,         */
/*                scheduled job RFSHDLVGUP that executes this      */
/*                program has been held at this time.              */
/*******************************************************************/

             PGM

/* Copy PLADCPP - Delivery Sched Header */
             CPYF       FROMFILE(PRKFGUP/PLADCPP) +
                          TOFILE(PRKFTSTGUP/PLADCPP) MBROPT(*REPLACE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(CPYERR))

/* Copy PLAECPP - Delivery Sched Detail */
             CPYF       FROMFILE(PRKFGUP/PLAECPP) +
                          TOFILE(PRKFTSTGUP/PLAECPP) MBROPT(*REPLACE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(CPYERR))

/* Copy PLAFCPP - Delivery tRACKING     */
             CPYF       FROMFILE(PRKFGUP/PLAFCPP) +
                          TOFILE(PRKFTSTGUP/PLAFCPP) MBROPT(*REPLACE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(CPYERR))

/* Copy PLAGCPP - Delivery Sched Split  */
             CPYF       FROMFILE(PRKFGUP/PLAGCPP) +
                          TOFILE(PRKFTSTGUP/PLAGCPP) MBROPT(*REPLACE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(CPYERR))
             GOTO       CMDLBL(ENDIT)

 CPYERR:     SNDPGMMSG  MSG('Copy GUP Delivery records to ISJBADELIB +
                          failed')




 ENDIT:      ENDPGM
