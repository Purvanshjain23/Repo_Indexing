 //?USER: Processing if Data record not found

 //?USER: Process Data record

 //?** Accumulate freight for Reference Required & Incl in Act Frt
 CASE;

 // IF DB1.LFC Act Frt Include Sts is Yes
 IF DB1.LFC_Act_Frt_Include_Sts = 'Y';
