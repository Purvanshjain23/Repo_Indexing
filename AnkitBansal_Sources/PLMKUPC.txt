/********************************************************************/
/*                                                                  */
/*  SYNOPSIS -  JOB CONTROL PROMPT FOR COMPANY, PRODUCER, LOCATION, */
/*              FROM DATE, TO DATE, AND EMAIL ADDRESS               */
/*                                                                  */
/*  SYSTEM    - HPE                                                 */
/*  PROGRAM   - BUY ORDER AND LOAD DOWNLOAD                         */
/*  DATE      - 08/04/2008                                          */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/* NN/NN/NN XXX XNNNNN                                              */
/********************************************************************/
/* PARAMETERS:                                                      */
/*                                                                  */
/*     I    &COMPNYN    - COMPANY NUMBER 15.5                       */
/*     I    &FROMDTA    - FROM DATE - ALPHA                         */
/*     I    &THRUDTA    - THRU DATE - ALPHA                         */
/*     I    &CLP        - CL PROGRAM TO SUBMIT                      */
/*     I    &TYPE       - REPORT TYPE                               */
/*     I    &PROD       - PRODUCER                                  */
/*     I    &LOCAT      - LOCATION                                  */
/*     I    &EMAIL      - EMAIL ADDRESS                             */
/*                                                                  */
/********************************************************************/
 PLMKUPC:    PGM        PARM(&COMPNYN &FROMDTA &THRUDTA &CLP &TYPE +
                          &PROD &LOCAT &EMAIL)

/* PARMS                                                                      */
             DCL        VAR(&COMPNYN) TYPE(*DEC)         /* Co# 15.5 */
             DCL        VAR(&FROMDTA) TYPE(*CHAR) LEN(7)
             DCL        VAR(&THRUDTA) TYPE(*CHAR) LEN(7)
             DCL        VAR(&CLP) TYPE(*CHAR) LEN(10)
             DCL        VAR(&TYPE) TYPE(*CHAR) LEN(1)
             DCL        VAR(&PROD) TYPE(*CHAR) LEN(6)
             DCL        VAR(&LOCAT) TYPE(*CHAR) LEN(6)
             DCL        VAR(&EMAIL) TYPE(*CHAR) LEN(50)

/* ERROR MESSAGE FIELDS                                                       */
        DCL    &MSG        *CHAR   (512)         /* message                   */
        DCL    &MSGID      *CHAR   (7)           /* message id                */
        DCL    &MSGL       *DEC    (5 0)

/*   MONITOR FOR ALL MESSAGES AT PROGRAM LEVEL                      */
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(MSGTRAP))

/********************************************************************/
/*   CALL THE PRINT PROGRAM                                         */
/********************************************************************/
 CALL:       CALL       PGM(&CLP) PARM(&COMPNYN &FROMDTA &THRUDTA +
                          &TYPE &PROD &LOCAT &EMAIL)

             GOTO       CMDLBL(END)

/*----------------------RECEIVE MESSAGES----------------------------*/
 MSGTRAP:    RCVMSG     MSGTYPE(*EXCP) MSGDTA(&MSG) MSGDTALEN(&MSGL) +
                          MSGID(&MSGID)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

             SNDPGMMSG  MSGID(&MSGID) MSGF(QCPFMSG) MSGDTA(%SST(&MSG +
                          1 &MSGL)) MSGTYPE(*ESCAPE)
             MONMSG     MSGID(CPF0000) EXEC(GOTO CMDLBL(END))

 END:        ENDPGM
