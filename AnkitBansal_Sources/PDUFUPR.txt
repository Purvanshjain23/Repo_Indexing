/*********************************************************************/
/*     Program: PDUFUPR                                              */
/*       Title: Generic program to execute an copy file command      */
/*        Date: 11/10/98                                             */
/* Description: - Execute a copy file command.                       */
/*                                                                   */
/*  Parameters:                                                      */
/*  Usage   Field       Field Description                            */
/*  -----   ---------   ------------------------------------------   */
/*    I     &FROMLIB    Library to copy the file from                */
/*    I     &FILE       File to copy                                 */
/*    I     &TOLIB      Library to copy the file to                  */
/*                                                                   */
/*********************************************************************/
/* Modifications:                                                    */
/*                                                                   */
/*   Date     Pgmr     Description                                   */
/* -------- ---------- --------------------------------------------  */
/*                                                                   */
/*********************************************************************/
             PGM  PARM(&FROMLIB &FILE &TOLIB)

   /*-------------------*/
   /* Declare Variables */
   /*-------------------*/

             /* Parameter Fields */
             DCL  VAR(&FROMLIB) TYPE(*CHAR) LEN(10)
             DCL  VAR(&FILE)    TYPE(*CHAR) LEN(10)
             DCL  VAR(&TOLIB)   TYPE(*CHAR) LEN(10)

   /*-------------------------------*/
   /* Execute the Copy File Command */
   /*-------------------------------*/
             CPYF FROMFILE(&FROMLIB/&FILE) +
                  TOFILE(&TOLIB/&FILE) +
                  MBROPT(*REPLACE)
             MONMSG MSGID(CPF0000)

 ENDPGM:
             ENDPGM
