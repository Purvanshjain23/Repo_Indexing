       //----------------------------------------------------------------------
       // Program To Get the MDY from Julian date passed in to it
       //
       //
       // Create using: CRTRPGMOD MODULE(e1idevgen/e10028b)
       //                         SRCFILE(e1idevgen/QRPGLESRC)
       //                         SRCMBR(e10028b)
       //                         REPLACE(*NO)
       //
       //   R Centonze    5/26/2020
       //----------------------------------------------------------------------
      * Parms
     d datejul         s              5  0
     d cyymmdd         s              7  0
      *
     d*
      *****************************************************************
      * Entry parameters
     C     *ENTRY        PLIST
     C                   PARM                    datejul
     C                   PARM                    cyymmdd
      *****************************************************************

      /free
        if datejul <> 0;
             cyymmdd  = %dec(%char(%date(datejul:*jul):*cymd0):7:0);
        endif;
       return;
      /end-free
      *
