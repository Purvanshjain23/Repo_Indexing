/* 08/05/05 SLM PASS IN THE ITEM TF CLASSIFICATION FOR SELECTION   */
/*              TFCLSS NEW PARM, AND REMOVE ITEM LEVEL - ITMLVL    */
/*                                                                 */
/*                                                                 */
/*                                                                 */
/********************************************************************/
 PDHOUPR:    PGM        PARM(&RETURN &APPLIC &CONUM &STRCLS &STRGRP +
                          &STRTYP &ITMTYP &BSCV &CRDITM &GLSUB +
                          &PRTRLZ &PCCNT &VALADD &TFCLSS &RCDSTS +
                          &OUTQ &HOLD &SAVE &COPY)

/* PARMS                                                                      */
        DCL    &RETURN     *CHAR   (7)           /* Return Code               */
        DCL    &APPLIC     *CHAR   (06)          /* Application Code          */
        DCL    &CONUM      *DEC   LEN(3 0)       /* COMPANY NUMBER */
        DCL    &STRCLS     *DEC   LEN(3 0)       /* ITEM STR CLASS */
        DCL    &STRGRP     *DEC   LEN(3 0)       /* ITEM STR GROUP */
        DCL    &STRTYP     *DEC   LEN(3 0)       /* ITEM STR TYPE  */
        DCL    &ITMTYP     *CHAR  LEN(2)         /* ITEM TYPE CODE*/
        DCL    &BSCV       *CHAR  LEN(2)         /* BASE CONVERTED*/
        DCL    &CRDITM     *CHAR  LEN(1)         /* CREDIT ITEM   */
        DCL    &GLSUB      *CHAR  LEN(3)         /* GL SUB CODE   */
/*****  DCL    &PRTRLZ     *CHAR  LEN(1)            PRINT ON REALIZATION */
        DCL    &PRTRLZ     *CHAR  LEN(2)         /* PRINT ON REALIZATION */
        DCL    &PCCNT      *CHAR  LEN(1)         /* PIECE COUNT ITEM     */
        DCL    &VALADD     *CHAR  LEN(1)         /* VALUE ADDED STS */
        DCL    &TFCLSS     *CHAR  LEN(3)         /* ITEM TF CLASSIFICATION    */
        DCL    &RCDSTS     *CHAR  LEN(1)         /* RECORD STATUS   */
        DCL    &OUTQ       *CHAR   (10)          /* Output Queue              */
        DCL    &HOLD       *CHAR   (4)           /* Hold Status               */
        DCL    &SAVE       *CHAR   (4)           /* Save Status               */
        DCL    &COPY       *DEC    (2 0)         /* Number of copies          */

/* INTERNAL FIELDS                                                            */
        DCL    &COMMAND    *CHAR   (512)         /* message                   */
/*-----------------------------------------------------------------*/
/*                     INITIALIZATION                              */
/*-----------------------------------------------------------------*/

/*-----------------------------------------------------------------*/
/*                           MAIN                                  */
/*-----------------------------------------------------------------*/
             SBDPRTOVR  CMDVAR(&COMMAND) FILE(POU2PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(POU2PFR) PARM(' ' &CONUM &STRCLS &STRGRP +
                          &STRTYP &ITMTYP &BSCV &CRDITM &GLSUB +
                          &PRTRLZ &PCCNT &VALADD &TFCLSS &RCDSTS)

/*-----------------------------------------------------------------*/
/*                      PROGRAM END                                */
/*-----------------------------------------------------------------*/
 ENDPGM:
             DLTOVR     FILE(*ALL)

             ENDPGM
 END:        ENDPGM
