/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING EXPORT A/R RECEIPTS REPORT   */
/*                                                                  */
/*  SYSTEM    - ACCOUNTS RECEIVABLE                                 */
/*  DATE      - 01/13/97                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPNYN       -  PASSED COMPANY NUMBER                  */
/*         &FROMDATN      -  PASSED FROM DATE                       */
/*         &THRUDATN      -  PASSED THRU DATE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*                                                                  */
/********************************************************************/
 PDGBPBLP:   PGM        PARM(&COMPNYN &FROMDATN &THRUDATN  +
                             &OUTQ &HOLD &SAVE &COPYN &NIGHTQ)

        DCL    &COMPNYN     *DEC
        DCL    &FROMDATN    *DEC
        DCL    &THRUDATN    *DEC
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &COMP        *DEC   (3 0)
        DCL    &FROMDATE    *DEC   (7 0)
        DCL    &THRUDATE    *DEC   (7 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COMP)  VALUE(&COMPNYN)
             CHGVAR     VAR(&COPY)  VALUE(&COPYN)
             CHGVAR     VAR(&FROMDATE)  VALUE(&FROMDATN)
             CHGVAR     VAR(&THRUDATE)  VALUE(&THRUDATN)

/********************************************************************/
/* OVERRIDE PRINTER FILE WITH USER SPECIFIED SELECTION VALUES       */
/********************************************************************/

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDVUPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) SAVE(&SAVE)

             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

 CALL:       CLRPFM     FILE(PDK5REP)
             CALL       PGM(PDVNXFR) PARM(' ' &COMP &FROMDATE +
                          &THRUDATE)

             CALL       PGM(PDVUPFR) PARM(' ' &COMP &FROMDATE +
                             &THRUDATE)

 ENDPGM:     ENDPGM
