/*  PROGRAM NUMBER:  PWBKUPC                                       */
/*  PROGRAM NAME:    TRUCKER M3 SUPPLIER EDIT REPORT               */
/*  PROGRAMMER:      SUSAN L. MASON                                */
/*  CREATE DATE:     04/11/17                                      */
/*                                                                 */
/*  FUNCTION:        THIS CL USED FRO THE PROMPT TO PERFORM A      */
/*                   OVERRIDE ON THE PRINTER                       */
/*                                                                 */
/*******************************************************************/
/*  Modification History                                           */
/*                                                                 */
/*  CREATED          04/11/17  PROJECT HPE044                      */
/*                   PRINT A TRUCKER VALIDATION REPORT             */
/*******************************************************************/
             PGM        PARM(&PDEV &HOLD &SAVE &XCOPIES &XCOMP &OMS +
                          &APP &XADAT &XRCUSR &XRTRUCK)
/* PRINT OPTION PARMS                                             */
             DCL        VAR(&PDEV) TYPE(*CHAR) LEN(10)
             DCL        VAR(&HOLD) TYPE(*CHAR) LEN(4)
             DCL        VAR(&SAVE) TYPE(*CHAR) LEN(4)
             DCL        VAR(&XCOPIES) TYPE(*DEC) LEN(2 0)

/* FUNCTION SPECIFIC PARMS                                        */
             DCL        VAR(&XCOMP) TYPE(*DEC) LEN(3 0)
             DCL        VAR(&OMS) TYPE(*CHAR) LEN(30)
             DCL        VAR(&APP) TYPE(*CHAR) LEN(40)

             DCL        VAR(&XADAT) TYPE(*DEC) LEN(7)
             DCL        VAR(&XRCUSR) TYPE(*CHAR) LEN(7)
             DCL        VAR(&XRTRUCK) TYPE(*CHAR) LEN(1)

             DCL        VAR(&LPP) TYPE(*CHAR) LEN(40)

             DCL        VAR(&LDT) TYPE(*DEC) LEN(7 0)
             DCL        VAR(&LUS) TYPE(*CHAR) LEN(7)
             DCL        VAR(&LLTRUCK) TYPE(*CHAR) LEN(1)

/* NUMERIC VARIABLES                                             */
             DCL        VAR(&COPIES) TYPE(*DEC) LEN(2 0)
             DCL        VAR(&COMP) TYPE(*DEC) LEN(3 0)

/* SBDPRTOVR VARIABLE                                          */
             DCL        VAR(&CMD) TYPE(*CHAR) LEN(512)

/* CHANGE INPUT PARMS TO PARK VARIABLES                       */
             CHGVAR     VAR(&COPIES) VALUE(&XCOPIES)
             CHGVAR     VAR(&COMP) VALUE(&XCOMP)

             CHGVAR     VAR(&LPP) VALUE(&OMS)
             CHGVAR     VAR(&LDT) VALUE(&XADAT)

/* *********************************************************/

/* CREATE PRINTER FILE OVERRIDE CODE    TRUCKER PAYMENT EDIT    */
             SBDPRTOVR  CMDVAR(&CMD) FILE(PWBJPFR$) OUTQ(&PDEV) +
                          COPIES(&COPIES) HOLD(&HOLD) SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&CMD 512)

/* PROCESS TRUCKER PAYMENT EDIT                               */

             CALL       PGM(PWBJPFR) PARM(' ' &COMP &LPP &LDT &LUS +
                          &LLTRUCK)

             CHGVAR     VAR(&XRCUSR) VALUE(&LUS)
             CHGVAR     VAR(&XRTRUCK) VALUE(&LLTRUCK)

             DLTOVR *ALL
             ENDPGM
