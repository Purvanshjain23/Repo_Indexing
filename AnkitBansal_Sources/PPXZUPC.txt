/********************************************************************/
/*                                                                  */
/*  PROGRAM   - PPXZUPC - PRT INVOICE EXCPT RPT  UP                 */
/*  SYNOPSIS  - PROMPT JOB CONTROL FOR PRINTER OVERRIDE INFORMATION */
/*              TO PRINT INVOICE EXCEPTION REPORT FOR THIRD         */
/*              PARTY OR INTER-COMPANY CUSTOMERS.                   */
/*                                                                  */
/*  SYSTEM    - ORDER MANAGEMENT SYSTEM                             */
/*  DATE      - 04/27/2006                                          */
/*            - LARA BUSER                                          */
/*                                                                  */
/*      PARAMETERS:                                                 */
/*                                                                  */
/*     O    &RETURN     - RETURN CODE                               */
/*     I    &APPLIC     - APPLICATION                               */
/*     I    &COMPNYN    - COMPANY NUMBER (PASSED)                   */
/*     I    &ORDRN      - ORDER NO. (PASSED)                        */
/*     I    &THRDATN    - THRU SHIP DATE (PASSED)                   */
/*     I    &LOADIDN    - LOAD ID   (PASSED)                        */
/*     I    &BILTYP     - BILLING ACTIVITY TYPE (INVOICE, MEMO)     */
/*     I    &DESC       - PRINT PROGRAM DESCRIPTION (FOR PROMPT     */
/*                         SCREEN)                                  */
/*     B    &OUTQ       - OUTPUT QUEUE                              */
/*     B    &HOLD       - HOLD STATUS                               */
/*     B    &SAVE       - SAVE STATUS                               */
/*     B    &COPYN      - NO. OF COPIES  (PASSED)                   */
/*     B    &NQSTAT     - INDICATOR WHETHER MUST RUN IN NIGHT Q     */
/*     B    &NIGHTQ     - NIGHT JOB QUEUE                           */
/*     I    &JOBNME     - JOB NAME (PASSED)                         */
/*     I    &PRCUSTT    - THRID PARTY OR INTER-COMPANY              */
/*     I    &STATUSYN   - PRINT AMOUNTS ON REPORT-YES OR NO         */
/********************************************************************/
/* MODIFICATIONS:                                                   */
/*------------------------------------------------------------------*/
/*                                                                  */
/********************************************************************/

 PPXZUPC:    PGM        PARM(&RETURN &APPLIC &BILTYP &DESC &OUTQ +
                          &HOLD &SAVE &NQSTAT &NIGHTQ &COMPNYN +
                          &THRDATN &ORDRN &LOADN &COPYN &STATUSYN +
                          &PRCUSTT &JOBNME)
/* PARMS                                                             */
        DCL    &RETURN     *CHAR   (7)   /* RETURN CODE              */
        DCL    &APPLIC     *CHAR   (6)   /* APPLICATION              */
        DCL    &COMPNYN    *DEC          /* COMPANY NUMBER (PASSED)  */
        DCL    &ORDRN      *DEC          /* ORDER NUMBER   (PASSED)  */
        DCL    &THRDATN    *DEC          /* ACTUAL SHIP DATE (PASSED)*/
        DCL    &LOADN      *DEC          /* LOAD ID        (PASSED)  */
        DCL    &BILTYP     *CHAR   (01)  /* BILL TYPE (INVOICE,MEMO) */
        DCL    &DESC       *CHAR   (25)  /* PRT PGM DESCRIPTION      */
        DCL    &OUTQ       *CHAR   (10)  /* OUTPUT QUEUE             */
        DCL    &HOLD       *CHAR   (4)   /* HOLD STATUS              */
        DCL    &SAVE       *CHAR   (4)   /* SAVE STATUS              */
        DCL    &NQSTAT     *CHAR   (01)  /* IF MUST RUN IN NIGHT Q   */
        DCL    &NIGHTQ     *CHAR   (10)  /* NIGHT JOB QUEUE          */
        DCL    &COPYN      *DEC          /* NO. OF COPIES  (PASSED)  */
        DCL    &JOBNME     *CHAR   (10)  /* JOB NAME                 */
        DCL    &PRCUSTT    *CHAR   (02)  /* TP OR IC (PASSED)        */
        DCL    &STATUSYN   *CHAR   (01)  /* PRINT AMOUNT STATUS Y/N  */
        DCL    &USRSTATUS  *CHAR   (01)  /* ERROR CONTROL FLAG       */

/* INTERNAL FIELDS                                                   */
        DCL    &COMPANY    *DEC    (3 0) /* COMPANY NUMBER           */
        DCL    &LOAD       *DEC    (7 0) /* LOAD NUMBER              */
        DCL    &ORDER      *DEC    (7 0) /* ORDER NUMBER             */
        DCL    &THRDAT     *DEC    (7 0) /* THRU ACTUAL SHIP DATE    */
        DCL    &USRSTATUS  *CHAR   (01)  /* ERROR CONTROL FLAG       */
        DCL    &COPY       *DEC    (2 0) /* NUMBER OF COPIES         */
        DCL    &COMMAND    *CHAR   (512) /* PRINTER OVERRIDE FIELD   */


/********************************************************************/
/* CHANGE VARIABLES FOR USE IN PROGRAM                              */
/*------------------------------------------------------------------*/
             CHGVAR     VAR(&COMPANY) VALUE(&COMPNYN)
             CHGVAR     VAR(&LOAD) VALUE(&LOADN)
             CHGVAR     VAR(&ORDER) VALUE(&ORDRN)
             CHGVAR     VAR(&THRDAT) VALUE(&THRDATN)
             CHGVAR     VAR(&COPY) VALUE(&COPYN)

/*------------------------------------------------------------------*/
             OVRPRTF    FILE(PPXSPFR$) OUTQ(&OUTQ) COPIES(&COPY) +
                          HOLD(&HOLD) SAVE(&SAVE)

/********************************************************************/
/* CALL THE INVOICE EXCEPTION REPORT PROGRAM                        */
/********************************************************************/
 CALL:       CALL       PGM(PPXSPFR) PARM(&RETURN &COMPANY &ORDER +
                          &THRDAT &LOAD &BILTYP &DESC &OUTQ &HOLD +
                          &SAVE &NQSTAT &NIGHTQ &COPYN &JOBNME +
                          &PRCUSTT &STATUSYN &USRSTATUS)

 END:        ENDPGM
