//?USER: Load subfile record from DBF record

//?Select Only Items with Translated Desc.
//?DO2642 DN 06/02/22-Added 'ED Language Xlate Desc 2'.
EXECUTE FUNCTION(RTV Translated Item DscRT) TYPE(RTVOBJ) FILE(PBBTREP)          AC2162905;
PARAMETER(DB1.Item_Code);
PARAMETER(PAR.ED_Xlate_Desc_to_Country);
PARAMETER(RCD.ED_Language_Xlate_Desc_2);
PARAMETER(RCD.ED_Language_Xlate_Desc);
{
 //?USER: Initialize routine

 MOVE *ALL (To: PAR From: CON);

 //?USER: Process Data record

 CASE;

 // IF PAR.ED Xlate Desc to Country is Spanish
 IF PAR.ED_Xlate_Desc_to_Country = 'S';

 // PAR.ED Language Xlate Desc = DB1.Spanish Description
 PAR.ED_Language_Xlate_Desc = DB1.Spanish_Description;

 //?DO2642 DN 04/19/22-Get Spanish Desc Line 2.
 // PAR.ED Language Xlate Desc 2 = DB1.Spanish Description Line2
 PAR.ED_Language_Xlate_Desc_2 = DB1.Spanish_Description_Line2;

 ENDIF;

}


CASE;

// IF RCD.ED Language Xlate Desc is Not Entered
IF RCD.ED_Language_Xlate_Desc = *BLANK;

// AND RCD.ED Language Xlate Desc 2 is Not Entered
AND RCD.ED_Language_Xlate_Desc_2 = *BLANK;

// PGM.*Record selected = CND.*NO
PGM.*Record_selected = 'N';

ENDIF;

//?USER: Process selected line

// PAR.ED Language Xlate Desc = RCD.ED Language Xlate Desc
PAR.ED_Language_Xlate_Desc = RCD.ED_Language_Xlate_Desc;

// PAR.ED Language Xlate Desc 2 = RCD.ED Language Xlate Desc 2
PAR.ED_Language_Xlate_Desc_2 = RCD.ED_Language_Xlate_Desc_2;
