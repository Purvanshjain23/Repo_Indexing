/**************************************************************************/
/*  PROGRAM:       PUQ8UPC                                                */
/*  PROGRAM DESC:  APPLY TRIGGER TO TRAILER (PDLYREP) FILE.               */
/*  PROGRAMMER:    DANNY NGUYEN                                           */
/*  CREATE DATE:   06/01/2020                                             */
/*  PROJECT:       S16606                                                 */
/*                                                                        */
/*  PURPOSE:       THIS WILL APPLY PHYSICAL FILE TRIGGER TO THE TRAILER   */
/*                 (PDLYREP) FILE FOR ALL TRIGGER DATABASE EVENTS:        */
/*                 EVENTS:                                                */
/*                   INSERT, UPDATE & DELETE.                             */
/* -----------------------------------------------------------------------*/
/*  PARMS:                                                                */
/*                 &ADDRMV = 'A' (ADD) OR 'R' (REMOVE)                    */
/*                 &OBJLIB = OBJECT LIBRARY                               */
/**************************************************************************/
/* MODIFICATION HISTORY                                                   */
/**************************************************************************/
/* DATE     INT  PROJECT DESCRIPTION                                      */
/* -------- ---- ------- -------------------------------------------------*/
/* 03/03/23 DN   W122889 ADDED MONITOR MESSAGE FOR 'RMVPFTRG' TO PREVENT  */
/*                       ERROR.                                           */
/**************************************************************************/

             PGM        PARM(&ADDRMV &OBJLIB)

             DCL        VAR(&ADDRMV) TYPE(*CHAR) LEN(1)
             DCL        VAR(&OBJLIB) TYPE(*CHAR) LEN(10)

/* ADD TRIGGER */
             IF         COND(&ADDRMV *EQ 'A') THEN(DO)
/* ADD TRIGGER FOR INSERT */
             ADDPFTRG   FILE(&OBJLIB/PDLYREP) TRGTIME(*AFTER) +
                          TRGEVENT(*INSERT) PGM(PURGUPR) +
                          RPLTRG(*YES) TRG(SEB_PDLYREP_INSERT)

/* ADD TRIGGER FOR UPDATE */
             ADDPFTRG   FILE(&OBJLIB/PDLYREP) TRGTIME(*AFTER) +
                          TRGEVENT(*UPDATE) PGM(PURGUPR) +
                          RPLTRG(*YES) TRG(SEB_PDLYREP_UPDATE) +
                          TRGUPDCND(*CHANGE)

/* ADD TRIGGER FOR DELETE */
             ADDPFTRG   FILE(&OBJLIB/PDLYREP) TRGTIME(*AFTER) +
                          TRGEVENT(*DELETE) PGM(PURGUPR) +
                          RPLTRG(*YES) TRG(SEB_PDLYREP_DELETE)
             ENDDO      /* &ADDRMV = 'A' */


/* REMOVE TRIGGER */
             IF         COND(&ADDRMV *EQ 'R') THEN(DO)
             RMVPFTRG   FILE(&OBJLIB/PDLYREP) TRGEVENT(*ALL)
             MONMSG     MSGID(CPF32C6)      /* NOT SUCCESSFUL */
             ENDDO

 END:        ENDPGM
