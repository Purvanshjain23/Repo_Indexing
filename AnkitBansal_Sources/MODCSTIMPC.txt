/* Desc   : ModDdlSrcC - Synon Pre-Compilation Override Program to Update     */
/*        : Synon Generated DDL Source Before Compilation                     */
/* Caller : Implementer - Option 42, F10=*ALL Spec Cmds, IEXCRQSDTL Command   */
/* Issue# : DDS To DDL Conversion                                             */
/* Date   : 05/01/2024                                                        */

             Pgm Parm(&ObjCod &Object &TrgEnv &EnvLib &PgmLib)

             Dcl &ObjCod     *Char 10
             Dcl &Object     *Char 10
             Dcl &TrgEnv     *Char 10
             Dcl &EnvLib     *Char 10
             Dcl &PgmLib     *Char 10
             Dcl &SysName    *Char 10
             Dcl &ReturnCode *Char 10

             /*‚Do not process anything if command is not YEXCSQL,     */
             /*‚And SourceFile is not QSQLSRC                          */
             /*‚And SourceLibrary not in OMSDEVGEN/HPEDEVGEN/E1IDEVGEN */
             If Cond(&ObjCod *NE 'YSQLPF') Then(Do)
               Return
             EndDo

             RTVNETA    SYSNAME(&SYSNAME)

             If Cond(&SysName *EQ 'SBFDEV') Then(Do)
               OvrDbf PVBVREL2 OmsDevSqc/PVBVREL2
               OvrDbf PVBVREL3 OmsDevSqc/PVBVREL3
               OvrDbf PVBVREL4 OmsDevSqc/PVBVREL4
               OvrDbf PVBVREL6 OmsDevSqc/PVBVREL6
             EndDo
             Else If Cond(&SysName *EQ 'SBFPROD') Then(Do)

               /*‚Copy SBFPROD/QGPL/PVBVREPDDM file data to PVBVREP file */
               /*‚Which actually contains constraint details             */
               CPYF FROMFILE(QGPL/PVBVREPDDM) TOFILE(PRKFLIB/PVBVREP) +
                    MBROPT(*REPLACE) CRTFILE(*NO)
               MonMsg MsgID(CPF0000)

               OvrDbf PVBVREL2 PRKFLIB/PVBVREL2
               OvrDbf PVBVREL3 PRKFLIB/PVBVREL3
               OvrDbf PVBVREL4 PRKFLIB/PVBVREL4
               OvrDbf PVBVREL6 PRKFLIB/PVBVREL6

               If Cond(&PgmLib *NE 'QGPL') Then(ChgVar &PgmLib 'PRKPLIB')
             EndDo
             Else Cmd(Do)
               Return
             EndDo

             Call &PgmLib/ModCstImpR (&ObjCod &Object &TrgEnv &EnvLib)
             MonMsg CPF0000

             DltOvr PVBVREL2
             DltOvr PVBVREL3
             DltOvr PVBVREL4
             DltOvr PVBVREL6

             EndPgm
