/*  SYSTEM    - OMS                                                 */
/*  DATE      - 08/29/06                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*               DISTL1    - DISTR LIST STORED IN COMPANY VALUE     */
/*                           CLMEXPPND, AND DIST LIST OF            */
/*                           ESEND                                  */
/*               SUBJECT   - TITLE OF WHAT WE ARE SENDING           */
/********************************************************************/
/********************************************************************/
 START:      PGM        PARM(&DISTL1 &SUBJECT)


             DCL        VAR(&DISTL1) TYPE(*CHAR) LEN(40)
             DCL        VAR(&SUBJECT) TYPE(*CHAR) LEN(50)


/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CALL       PGM(PLA0PFR) PARM(' ')

             IF         COND(&DISTL1 *NE ' ') THEN(DO)
             ESNDMAIL   RECIPIENT(&DISTL1) SUBJECT(&SUBJECT) +
                          ATTLIST((* *PDF *N PLA0PFR$))

             DLTSPLF    FILE(PLA0PFR$) SPLNBR(*LAST)

             MONMSG     MSGID(CFP0000) EXEC(GOTO CMDLBL(ENDPGM))

             ENDDO


ENDPGM:     ENDPGM
