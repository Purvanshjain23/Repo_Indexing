// Execute function EUS Repl SpcChar 3 BlnkUS.
EXECUTE FUNCTION(EUS Repl SpcChar 3 BlnkUS) TYPE(EXCUSRSRC) ('PUY7UFR');
PARAMETER(PAR.String_Text_Old_USR);
PARAMETER(PAR.String_Text_New_USR);
{
 * Scan and Replace Special Characters to Blank.
 d $string_data    s             80a   inz(*blank)
 d $new_string     s             80a   inz(*blank)
 d i               s              3  0
 d char            s              1a
 *
 /free
 $string_data = #II2U1;
 $new_string = #II2U1;
 #OI3U1 = *blank;

 // Read each character in '$string_data' and translate the special charc to a blank.
 for i = 1 to %len($string_data);
 char = %subst($string_data:i:1);
 if char = '*';
 $new_string = %xlate('*' : ' ' : $string_data);
 elseif char = '~';
 $new_string = %xlate('~' : ' ' : $string_data);
 elseif char = '<';
 $new_string = %xlate('<' : ' ' : $string_data);
 elseif char = '>';
 $new_string = %xlate('>' : ' ' : $string_data);
 elseif char = '!';
 $new_string = %xlate('!' : ' ' : $string_data);
 elseif char = '^';
 $new_string = %xlate('^' : ' ' : $string_data);
 endif;
 $string_data = $new_string;
 endfor;

 // If $new_string is blank then default to 'blank'.
 if $new_string = *blank;
 #OI3U1 = *blank;
 else;
 #OI3U1 = $new_string;
 endif;

 /end-free
 *
}
