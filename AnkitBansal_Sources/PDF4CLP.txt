/********************************************************************/
/*                                                                  */
/*  SYNOPSIS - CL PROGRAM FOR PRINTING COMMODITY MKT PRICES         */
/*                                                                  */
/*  SYSTEM    - CROSS APPLICATION                                   */
/*  DATE      -  3/09/95                                            */
/*                                                                  */
/*      PARAMETERS RECEIVED:                                        */
/*         &COMPN         -  COMPANY DFLT  (PASSED)                 */
/*         &MRKTCD        -  MARKET CODE   (PASSED)                 */
/*         &COMGRP        -  COMMODITY GROUP (PASSED)               */
/*         &DATEN         -  DATE    (PASSED)                       */
/*         &RPT1          -  REPORT SEL 1  (PASSED)                 */
/*         &RPT2          -  REPORT SEL 2  (PASSED)                 */
/*         &RPT3          -  REPORT SEL 3  (PASSED)                 */
/*         &APPLIC        -  APPLICATION CODE                       */
/*         &OUTQ          -  OUTPUT QUEUE                           */
/*         &HOLD          -  HOLD STATUS                            */
/*         &SAVE          -  SAVE STATUS                            */
/*         &COPYN         -  PASSED NUMBER OF COPIES                */
/*         &NIGHTQ        -  NIGHT JOB QUEUE                        */
/*         &TYPEN         -  ITEM TYPE (PASSED)                     */
/*         &GROUPN        -  ITEM GROUP (PASSED)                    */
/*         &CLASSN        -  ITEM CLASS (PASSED)                    */
/*                                                                  */
/********************************************************************/
 PDF4CLP:    PGM        PARM(&COMPN &MRKTCD &COMGRP &DATEN &RPT1 +
                          &RPT2 &RPT3 &APPLIC &OUTQ &HOLD &SAVE +
                          &COPYN &NIGHTQ &TYPEN &GROUPN &CLASSN)
        DCL    &MRKTCD      *CHAR  (03)
        DCL    &COMGRP      *CHAR  (07)
        DCL    &DATEN       *DEC
        DCL    &COMPN       *DEC
        DCL    &TYPEN       *DEC
        DCL    &GROUPN      *DEC
        DCL    &CLASSN      *DEC
        DCL    &RPT1        *CHAR  (01)
        DCL    &RPT2        *CHAR  (01)
        DCL    &RPT3        *CHAR  (01)
        DCL    &APPLIC      *CHAR  (06)
        DCL    &OUTQ        *CHAR  (10)
        DCL    &HOLD        *CHAR  (04)
        DCL    &SAVE        *CHAR  (04)
        DCL    &COPYN       *DEC
        DCL    &NIGHTQ      *CHAR  (10)

        DCL    &DATE        *DEC   (7 0)
        DCL    &COMPANY     *DEC   (3 0)
        DCL    &TYPE        *DEC   (3 0)
        DCL    &GROUP       *DEC   (3 0)
        DCL    &CLASS       *DEC   (3 0)
        DCL    &COPY        *DEC   (2 0)
        DCL    &COMMAND    *CHAR   (512)

/* CONVERT PARAMETERS TO REQUIRED SIZE FOR PRINT FUNCTION                  */

             CHGVAR     VAR(&COPY)  VALUE(&COPYN)
             CHGVAR     VAR(&DATE)  VALUE(&DATEN)
             CHGVAR     VAR(&COMPANY)  VALUE(&COMPN)
             CHGVAR     VAR(&TYPE) VALUE(&TYPEN)
             CHGVAR     VAR(&GROUP) VALUE(&GROUPN)
             CHGVAR     VAR(&CLASS) VALUE(&CLASSN)

/*********************************************************************/
/*   CALL LISTING PROGRAM                                            */
/*********************************************************************/
  /* COMMODITY MARKET PRICE LIST                   */

     IF COND(&RPT1 *EQ 'Y') THEN(DO)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDF4PFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PDF4PFR) PARM(' ' &MRKTCD +
                          &COMGRP &DATE)
     ENDDO

     IF COND(&RPT2 *EQ 'Y') THEN(DO)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDFRPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PDFRPFR) PARM(' ' &COMPANY &MRKTCD +
                          &COMGRP &DATE)
     ENDDO

     IF COND(&RPT3 *EQ 'Y') THEN(DO)

             SBDPRTOVR  CMDVAR(&COMMAND) FILE(PDMXPFR$) OUTQ(&OUTQ) +
                          COPIES(&COPY) HOLD(&HOLD) +
                          SAVE(&SAVE)
             CALL       PGM(QCMDEXC) PARM(&COMMAND 512)

             CALL       PGM(PDMXPFR) PARM(' ' &COMPANY &DATE &TYPE +
                          &GROUP &CLASS)
     ENDDO

 ENDPGM:     ENDPGM
